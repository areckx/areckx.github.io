86 int FlateStream::fill_buffer() {
87  /* only fill the buffer if it has all been used */
88  if (out_pos >= out_buf_len) {
89  /* check if the flatestream has been exhausted */
90  if (status == Z_STREAM_END) {
91  return -1;
92  }
93 
94  /* set to the begining of out_buf */
95  d_stream.avail_out = sizeof(out_buf);
96  d_stream.next_out = out_buf;
97  out_pos = 0;
98 
99  while (1) {
100  /* buffer is empty so we need to fill it */
101  if (d_stream.avail_in == 0) {
102  int c;
103  /* read from the source stream */
104  while (d_stream.avail_in < sizeof(in_buf) && (c = str->getChar()) != EOF) {
105  in_buf[d_stream.avail_in++] = c;
106  }
107  d_stream.next_in = in_buf;
108  }
109 
110  /* keep decompressing until we can't anymore */
111  if (d_stream.avail_out == 0 || d_stream.avail_in == 0 || (status != Z_OK && status != Z_BUF_ERROR))
112  break;
113  status = inflate(&d_stream, Z_SYNC_FLUSH);
114  }
115 
116  out_buf_len = sizeof(out_buf) - d_stream.avail_out;
117  if (status != Z_OK && status != Z_STREAM_END)
118  return -1;
119  if (!out_buf_len)
120  return -1;
121  }
122 
123  return 0;
124 }
