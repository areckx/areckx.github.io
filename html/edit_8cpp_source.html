205 Chord* Score::addChord(int tick, TDuration d, Chord* oc, bool genTie, Tuplet* tuplet)
206  {
207  Measure* measure = tick2measure(tick);
208  if (measure->endTick() <= tick) {
209  qDebug("Score::addChord(): end of score?");
210  return 0;
211  }
212 
213  Chord* chord = new Chord(this);
214  chord->setTuplet(tuplet);
215  chord->setTrack(oc->track());
216  chord->setDurationType(d);
217  chord->setDuration(d.fraction());
218 
219  foreach(Note* n, oc->notes()) {
220  Note* nn = new Note(this);
221  nn->setPitch(n->pitch());
222  nn->setTpc1(n->tpc1());
223  nn->setTpc2(n->tpc2());
224  chord->add(nn);
225  }
226  undoAddCR(chord, measure, tick);
227 
228  //
229  // now as both chords are in place
230  // (have segments as parent) we can add ties:
231  //
232  if (genTie) {
233  int n = oc->notes().size();
234  for(int i = 0; i < n; ++i) {
235  Note* n = oc->notes()[i];
236  Note* nn = chord->notes()[i];
237  Tie* tie = new Tie(this);
238  tie->setStartNote(n);
239  tie->setEndNote(nn);
240  tie->setTrack(n->track());
241  undoAddElement(tie);
242  }
243  }
244 
245  return chord;
246  }

272 // create one or more rests to fill "l"

275 Rest* Score::setRest(int tick, int track, Fraction l, bool useDots, Tuplet* tuplet, bool useFullMeasureRest)
276  {
277  Measure* measure = tick2measure(tick);
278  Rest* r = 0;
279  Staff* staff = Score::staff(track / VOICES);
280 
281  while (!l.isZero()) {
282  //
283  // divide into measures
284  //
285  Fraction f;
286  if (tuplet) {
287  int ticks = (tuplet->tick() + tuplet->actualTicks()) - tick;
288 
289  f = Fraction::fromTicks(ticks);
290  for (Tuplet* t = tuplet; t; t = t->tuplet())
291  f *= t->ratio();
292  //
293  // restrict to tuplet len
294  //
295  if (f < l)
296  l = f;
297  }
298  else if (measure->tick() < tick)
299  f = Fraction::fromTicks(measure->tick() + measure->ticks() - tick);
300  else
301  f = measure->len();
302  f *= staff->timeStretch(tick);
303  f.reduce();
304 
305  if (f > l)
306  f = l;
307 
308  if ((track % VOICES) && !measure->hasVoice(track) && (tick == measure->tick())) {
309  l -= f;
310  measure = measure->nextMeasure();
311  if (!measure)
312  break;
313  tick = measure->tick();
314  continue;
315  }
316 
317  if ((measure->timesig() == measure->len()) // not in pickup measure
318  && (measure->tick() == tick)
319  && (measure->stretchedLen(staff) == f)
320  && !tuplet
321  && (useFullMeasureRest)) {
322  Rest* rest = addRest(tick, track, TDuration(TDuration::DurationType::V_MEASURE), tuplet);
323  tick += rest->actualTicks();
324  if (r == 0)
325  r = rest;
326  }
327  else {
328  //
329  // compute list of durations which will fit l
330  //
331  std::vector<TDuration> dList = toDurationList(f, useDots);
332  if (dList.empty())
333  return 0;
334 
335  Rest* rest = 0;
336  if (((tick - measure->tick()) % dList[0].ticks()) == 0) {
337  for (const TDuration& d : dList) {
338  rest = addRest(tick, track, d, tuplet);
339  if (r == 0)
340  r = rest;
341  tick += rest->actualTicks();
342  }
343  }
344  else {
345  for (int i = dList.size() - 1; i >= 0; --i) {
346  rest = addRest(tick, track, dList[i], tuplet);
347  if (r == 0)
348  r = rest;
349  tick += rest->actualTicks();
350  }
351  }
352  }
353  l -= f;
354 
355  measure = measure->nextMeasure();
356  if (!measure)
357  break;
358  tick = measure->tick();
359  }
360  return r;
361  }

385 //---------------------------------------------------------
386 // rewriteMeasures
387 // rewrite all measures from fm to lm (including)
388 // If staffIdx is valid (>= 0), then rewrite a local
389 // timesig change.
390 //---------------------------------------------------------
391 
392 bool Score::rewriteMeasures(Measure* fm, Measure* lm, const Fraction& ns, int staffIdx)
393  {
394  if (staffIdx >= 0) {
395  // local timesig
396  // don't actually rewrite, just update measure rest durations
397  // abort if there is anything other than measure rests in range
398  int strack = staffIdx * VOICES;
399  int etrack = strack + VOICES;
400  for (Measure* m = fm; ; m = m->nextMeasure()) {
401  for (Segment* s = m->first(SegmentType::ChordRest); s; s = s->next(SegmentType::ChordRest)) {
402  for (int track = strack; track < etrack; ++track) {
403  ChordRest* cr = toChordRest(s->element(track));
404  if (!cr)
405  continue;
406  if (cr->isRest() && cr->durationType() == TDuration::DurationType::V_MEASURE)
407  cr->undoChangeProperty(P_ID::DURATION, QVariant::fromValue(ns));
408  else
409  return false;
410  }
411  }
412  if (m == lm)
413  break;
414  }
415  return true;
416  }
417  int measures = 1;
418  bool fmr = true;
419  for (Measure* m = fm; m; m = m->nextMeasure()) {
420  if (!m->isFullMeasureRest())
421  fmr = false;
422  if (m == lm)
423  break;
424  ++measures;
425  }
426 
427  if (!fmr) {
428  // check for local time signatures
429  for (Measure* m = fm; m; m = m -> nextMeasure()) {
430  for (int staffIdx = 0; staffIdx < nstaves(); ++staffIdx) {
431  if (staff(staffIdx)->timeStretch(m->tick()) != Fraction(1,1)) {
432  // we cannot change a staff with a local time signature
433  return false;
434  }
435  if (m == lm)
436  break;
437  }
438  }
439  }
440 
441  ScoreRange range;
442  range.read(fm->first(), lm->last());
443 
444  //
445  // calculate number of required measures = nm
446  //
447  Fraction k = range.duration() / ns;
448  int nm = (k.numerator() + k.denominator() - 1)/ k.denominator();
449 
450  Fraction nd(nm * ns.numerator(), ns.denominator()); // total new duration
451 
452  // evtl. we have to fill the last measure
453  Fraction fill = nd - range.duration();
454  range.fill(fill);
455 
456  for (Score* s : scoreList()) {
457  Measure* m1 = s->tick2measure(fm->tick());
458  Measure* m2 = s->tick2measure(lm->tick());
459 
460  int tick1 = m1->tick();
461  int tick2 = m2->endTick();
462  auto spanners = s->spannerMap().findContained(tick1, tick2);
463  for (auto i : spanners)
464  undo(new RemoveElement(i.value));
465  s->undoRemoveMeasures(m1, m2);
466 
467  Measure* nfm = 0;
468  Measure* nlm = 0;
469  int tick = 0;
470 // bool endBarGenerated = m1->endBarLineGenerated();
471  for (int i = 0; i < nm; ++i) {
472  Measure* m = new Measure(s);
473  m->setPrev(nlm);
474  if (nlm)
475  nlm->setNext(m);
476  m->setTimesig(ns);
477  m->setLen(ns);
478  m->setTick(tick);
479 // m->setEndBarLineType(BarLineType::NORMAL, endBarGenerated);
480  tick += m->ticks();
481  nlm = m;
482  if (nfm == 0)
483  nfm = m;
484  }
485 // nlm->setEndBarLineType(m2->endBarLineType(), m2->endBarLineGenerated(),
486 // m2->endBarLineVisible(), m2->endBarLineColor());
487  //
488  // insert new calculated measures
489  //
490  nfm->setPrev(m1->prev());
491  nlm->setNext(m2->next());
492  s->undo(new InsertMeasures(nfm, nlm));
493  }
494  if (!fill.isZero())
495  undoInsertTime(lm->endTick(), fill.ticks());
496 
497  if (!range.write(masterScore(), fm->tick()))
498  return false;
499  connectTies(true);
500 
501  if (noteEntryMode()) {
502  // set input cursor to possibly re-written segment
503  int icTick = inputPos();
504  Segment* icSegment = tick2segment(icTick, false, SegmentType::ChordRest);
505  if (!icSegment) {
506  // this can happen if cursor was on a rest
507  // and in the rewriting it got subsumed into a full measure rest
508  Measure* icMeasure = tick2measure(icTick);
509  if (!icMeasure) // shouldn't happen, but just in case
510  icMeasure = firstMeasure();
511  icSegment = icMeasure->first(SegmentType::ChordRest);
512  }
513  inputState().setSegment(icSegment);
514  }
515 
516  return true;
517  }

519 //---------------------------------------------------------
520 // rewriteMeasures
521 // rewrite all measures up to the next time signature or section break
522 //---------------------------------------------------------
523 
524 bool Score::rewriteMeasures(Measure* fm, const Fraction& ns, int staffIdx)
525  {
526  Measure* lm = fm;
527  Measure* fm1 = fm;
528  Measure* nm = nullptr;
529  LayoutBreak* sectionBreak = nullptr;
530 
531  // disable local time sig modifications in linked staves
532  if (staffIdx != -1 && excerpts().size() > 0) {
533  MScore::setError(CANNOT_CHANGE_LOCAL_TIMESIG);
534  return false;
535  }
536 
537  //
538  // split into Measure segments fm-lm
539  //
540  for (MeasureBase* m = fm; ; m = m->next()) {
541 
542  if (!m || !m->isMeasure() || lm->sectionBreak()
543  || (toMeasure(m)->first(SegmentType::TimeSig) && m != fm))
544  {
545 
546  // save section break to reinstate after rewrite
547  if (lm->sectionBreak())
548  sectionBreak = new LayoutBreak(*lm->sectionBreakElement());
549 
550  if (!rewriteMeasures(fm1, lm, ns, staffIdx)) {
551  if (staffIdx >= 0) {
552  MScore::setError(CANNOT_CHANGE_LOCAL_TIMESIG);
553  // restore measure rests that were prematurely modified
554  Fraction fr(staff(staffIdx)->timeSig(fm->tick())->sig());
555  for (Measure* m = fm1; m; m = m->nextMeasure()) {
556  ChordRest* cr = m->findChordRest(m->tick(), staffIdx * VOICES);
557  if (cr && cr->isRest() && cr->durationType() == TDuration::DurationType::V_MEASURE)
558  cr->undoChangeProperty(P_ID::DURATION, QVariant::fromValue(fr));
559  else
560  break;
561  }
562  }
563  else {
564  // this can be hit for local time signatures as well
565  // (if we are rewriting all staves, but one has a local time signature)
566  // TODO: detect error conditions better, have clearer error messages
567  // and perform necessary fixups
568  MScore::setError(TUPLET_CROSSES_BAR);
569  }
570  for (Measure* m = fm1; m; m = m->nextMeasure()) {
571  if (m->first(SegmentType::TimeSig))
572  break;
573  Fraction fr(ns);
574  undoChangeProperty(m, P_ID::TIMESIG_NOMINAL, QVariant::fromValue(fr));
575  }
576  return false;
577  }
578 
579  // after rewrite, lm is not necessarily valid
580  // m is first MeasureBase after rewritten range
581  // m->prevMeasure () is new last measure of range
582  // set nm to first true Measure after rewritten range
583  // we may use this to reinstate time signatures
584  if (m && m->prevMeasure())
585  nm = m->prevMeasure()->nextMeasure();
586  else
587  nm = nullptr;
588 
589  if (sectionBreak) {
590  // reinstate section break, then stop rewriting
591  if (m && m->prevMeasure()) {
592  sectionBreak->setParent(m->prevMeasure());
593  undoAddElement(sectionBreak);
594  }
595  else if (!m) {
596  sectionBreak->setParent(lastMeasure());
597  undoAddElement(sectionBreak);
598  }
599  else {
600  qDebug("unable to restore section break");
601  nm = nullptr;
602  sectionBreak = nullptr;
603  }
604  break;
605  }
606 
607  // stop rewriting at end of score
608  // or at a measure (which means we found a time signature segment)
609  if (!m || m->isMeasure())
610  break;
611 
612  // skip frames
613  while (!m->isMeasure()) {
614  if (m->sectionBreak()) {
615  // frame has a section break; we can stop skipping ahead
616  sectionBreak = m->sectionBreakElement();
617  break;
618  }
619  m = m->next();
620  if (!m)
621  break;
622  }
623  // stop rewriting if we encountered a section break on a frame
624  // or if there is a time signature on first measure after the frame
625  if (sectionBreak || (m && toMeasure(m)->first(SegmentType::TimeSig)))
626  break;
627 
628  // set up for next range to rewrite
629  fm1 = toMeasure(m);
630  if (fm1 == 0)
631  break;
632  }
633 
634  // if we didn't break the loop already,
635  // we must have an ordinary measure
636  // add measure to range to rewrite
637  lm = toMeasure(m);
638  }
639 
640  // if any staves don't have time signatures at the point where we stopped,
641  // we need to reinstate their previous time signatures
642  if (!nm)
643  return true;
644  Segment* s = nm->undoGetSegment(SegmentType::TimeSig, nm->tick());
645  for (int i = 0; i < nstaves(); ++i) {
646  if (!s->element(i * VOICES)) {
647  TimeSig* ots = staff(i)->timeSig(nm->tick());
648  if (ots) {
649  TimeSig* nts = new TimeSig(*ots);
650  nts->setParent(s);
651  if (sectionBreak) {
652  nts->setGenerated(false);
653  nts->setShowCourtesySig(false);
654  }
655  undoAddElement(nts);
656  }
657  }
658  }
659 
660  return true;
661  }

663 //---------------------------------------------------------
664 // cmdAddTimeSig
665 //
666 // Add or change time signature at measure in response
667 // to gui command (drop timesig on measure or timesig)
668 //---------------------------------------------------------
669 
670 void Score::cmdAddTimeSig(Measure* fm, int staffIdx, TimeSig* ts, bool local)
671  {
672  deselectAll();
673 
674  Fraction ns = ts->sig();
675  int tick = fm->tick();
676  TimeSig* lts = staff(staffIdx)->timeSig(tick);
677  if (local) {
678  Fraction stretch = (ns / fm->timesig()).reduced();
679  ts->setStretch(stretch);
680  }
681 
682  Fraction stretch;
683  Fraction lsig; // last signature
684  if (lts) {
685  stretch = lts->stretch();
686  lsig = lts->sig();
687  }
688  else {
689  stretch.set(1,1);
690  lsig.set(4,4); // set to default
691  }
692 
693  int track = staffIdx * VOICES;
694  Segment* seg = fm->undoGetSegment(SegmentType::TimeSig, tick);
695  TimeSig* ots = toTimeSig(seg->element(track));
696 
697  if (ots && (*ots == *ts)) {
698  //
699  // ignore if there is already a timesig
700  // with same values
701  //
702  delete ts;
703  return;
704  }
705 
706  if (ots && ots->sig().identical(ns) && ots->stretch() == ts->stretch()) {
707  ots->undoChangeProperty(P_ID::TIMESIG, QVariant::fromValue(ns));
708  ots->undoChangeProperty(P_ID::GROUPS, QVariant::fromValue(ts->groups()));
709  if (ts->hasCustomText()) {
710  ots->undoChangeProperty(P_ID::NUMERATOR_STRING, ts->numeratorString());
711  ots->undoChangeProperty(P_ID::DENOMINATOR_STRING, ts->denominatorString());
712  }
713  foreach (Score* score, scoreList()) {
714  Measure* fm = score->tick2measure(tick);
715  for (Measure* m = fm; m; m = m->nextMeasure()) {
716  if ((m != fm) && m->first(SegmentType::TimeSig))
717  break;
718  bool changeActual = m->len() == m->timesig();
719  undoChangeProperty(m, P_ID::TIMESIG_NOMINAL, QVariant::fromValue(ns));
720  if (changeActual)
721  undoChangeProperty(m, P_ID::TIMESIG_ACTUAL, QVariant::fromValue(ns));
722  // undoChangeProperty(ots, P_ID::GROUPS, QVariant::fromValue(ts->groups()));
723  }
724  }
725  int n = nstaves();
726  for (int staffIdx = 0; staffIdx < n; ++staffIdx) {
727  TimeSig* nsig = toTimeSig(seg->element(staffIdx * VOICES));

955 //---------------------------------------------------------
956 // regroupNotesAndRests
957 // * combine consecutive rests into fewer rests of longer duration.
958 // * combine tied notes/chords into fewer notes of longer duration.
959 // Only operates on one voice - protects manual layout adjustment, etc.
960 //---------------------------------------------------------
961 
962 void Score::regroupNotesAndRests(int startTick, int endTick, int track)
963  {
964  Segment* inputSegment = _is.segment(); // store this so we can get back to it later.
965  Segment* seg = tick2segment(startTick, true, SegmentType::ChordRest);
966  for (Measure* msr = seg->measure(); msr && msr->tick() < endTick; msr = msr->nextMeasure()) {
967  int maxTick = endTick > msr->endTick() ? msr->endTick() : endTick;
968  if (!seg || seg->measure() != msr)
969  seg = msr->first(SegmentType::ChordRest);
970  for (; seg && seg->tick() + seg->ticks() < maxTick; seg = seg->next(SegmentType::ChordRest)) {
971  ChordRest* curr = seg->cr(track);
972  if (!curr)
973  continue; // voice is empty here
974  if (curr->isRest()) {
975  // combine consecutive rests
976  ChordRest* lastRest = curr;
977  for (Segment* s = seg->next(SegmentType::ChordRest); s && s->tick() + s->ticks() <= maxTick; s = s->next(SegmentType::ChordRest)) {
978  ChordRest* cr = s->cr(track);
979  if (!cr || !cr->isRest())
980  break;
981  lastRest = cr;
982  }
983  int restTicks = lastRest->tick() + lastRest->duration().ticks() - curr->tick();
984  seg = setNoteRest(seg, curr->track(), NoteVal(), Fraction::fromTicks(restTicks), Direction::AUTO);
985  }
986  else {
987  // combine tied chords
988  Chord* chord = toChord(curr);
989  Chord* lastTiedChord = chord;
990  for (Chord* next = chord->nextTiedChord(); next && next->tick() + next->duration().ticks() <= maxTick; next = next->nextTiedChord()) {
991  lastTiedChord = next;
992  }
993  if (!lastTiedChord)
994  lastTiedChord = chord;
995  int numNotes = chord->notes().size();
996  int pitches[numNotes];
997  Tie* tieBack[numNotes];
998  Tie* tieFor[numNotes];
999  for (int i = 0; i < numNotes; i++) {
1000  Note* n = chord->notes()[i];
1001  Note* nn = lastTiedChord->notes()[i];
1002  pitches[i] = n->pitch();
1003  tieBack[i] = n->tieBack();
1004  tieFor[i] = nn->tieFor();
1005  n->setTieBack(0);
1006  nn->setTieFor(0);
1007  }
1008  int noteTicks = lastTiedChord->tick() + lastTiedChord->duration().ticks() - chord->tick();
1009  seg = setNoteRest(seg, curr->track(), NoteVal(pitches[0]), Fraction::fromTicks(noteTicks), Direction::AUTO);
1010  Chord* newChord = toChord(seg->cr(track));
1011  Note* n = newChord->notes().front();
1012  undoRemoveElement(n);
1013  for (int i = 0; i < numNotes; i++) {
1014  NoteVal nval = NoteVal(pitches[i]);
1015  n = addNote(newChord, nval);
1016  if (tieBack[i]) {
1017  n->setTieBack(tieBack[i]);
1018  tieBack[i]->setEndNote(n);
1019  }
1020  if (tieFor[i]) {
1021  n->setTieFor(tieFor[i]);
1022  tieFor[i]->setStartNote(n);
1023  }
1024  }
1025  }
1026  }
1027  }
1028  // now put the input state back where it was before
1029  _is.setSegment(inputSegment);
1030  }
