82 void Asection::set_size (float time)
83 {
84  int i, d;
85 
86  float r = (time * _fsam);
87  if (r > N - PERIOD)
88  r = N - PERIOD;
89  for (i = 0; i < 16; i++)
90  {
91  d = (int)(r * _refl [i]);
92  d = (_offs0 - d * PERIOD) & (N - 1);
93  _offs [i] = d + (i >> 2) * N;
94  }
95 }

102 void Asection::process (float vol, float *W, float *X, float *Y, float *R)
103  {
104  float x[PERIOD];
105  float y[PERIOD];
106 
107  float gw = vol * _apar [DIRECT].fval();
108  float g = 0.45f * _apar [STWIDTH].fval();
109  float s = 0.5f + g * (1 - g);
110  float d = g - 0.5f;
111  float gx1 = gw * (s - d);
112  float gy1 = gw * (s + d);
113  g = 0.25f * _apar [STWIDTH].fval();
114  s = 0.5f + g * (1 - g);
115  d = g - 0.5f;
116  float gx2 = gw * (s - d);
117  float gy2 = gw * (s + d);
118  float* p = _base + _offs0;
119  float gr = 0.5f * _apar [REVERB].fval();
120 
121  for (int i = 0; i < PERIOD; i++) {
122  float t0 = p [0 * N];
123  float t1 = p [1 * N];
124  float t2 = p [2 * N];
125  float t3 = p [3 * N];
126  p++;
127  s = t0 + t1 + t2 + t3;
128  R [i] += gr * s;
129  W [i] += gw * s;
130  x [i] = gx1 * (t3 + t0) + gx2 * (t2 + t1);
131  y [i] = gy1 * (t3 - t0) + gy2 * (t2 - t1);
132  }
133 
134  gr = vol * _apar [REFLECT].fval();
135  p = _base;
136 
137  for (int i = 0; i < PERIOD; i++) {
138  float t0 = _dif0.process (p[_offs[1]] + p[_offs[5]] + p [_offs [11]] + p [_offs [15]] + 1e-20f);
139  float t1 = _dif1.process (p[_offs[0]] + p[_offs[4]] + p [_offs [10]] + p [_offs [14]] + 1e-20f);
140  float t2 = _dif2.process (p[_offs[2]] + p[_offs[6]] + p [_offs [8]] + p [_offs [12]] + 2e-20f);
141  float t3 = _dif3.process (p[_offs[3]] + p[_offs[7]] + p [_offs [9]] + p [_offs [13]] + 2e-20f);
142  p++;
143  s = t0 + t1 + t2 + t3;
144  _sw += 0.5f * (s - _sw);
145  _sx += 0.5f * (0.4f * (t0 + t3) + 0.6f * (t2 + t1) - _sx);
146  _sy += 0.5f * (0.9f * (t0 - t3) + 0.8f * (t2 - t1) - _sy);
147  W [i] += gr * _sw;
148  x [i] += gr * _sx;
149  y [i] += gr * _sy;
150  }
151 
152  g = 6.283184f * _apar [AZIMUTH].fval();
153  gx1 = cosf (g);
154  gy1 = sinf (g);
155  for (int i = 0; i < PERIOD; i++) {
156  X [i] += gx1 * x [i] + gy1 * y [i];
157  Y [i] += gx1 * y [i] - gy1 * x [i];
158  }
159  _offs0 = (_offs0 + PERIOD) & (N - 1);
160  for (int i = 0; i < 16; i++)
161  _offs [i] = ((_offs [i] + PERIOD) & (N - 1)) + (i >> 2) * N;
162  p = _base + _offs0;
163  memset(p + 0 * N, 0, PERIOD * sizeof(float));
164  memset(p + 1 * N, 0, PERIOD * sizeof(float));
165  memset(p + 2 * N, 0, PERIOD * sizeof(float));
166  memset(p + 3 * N, 0, PERIOD * sizeof(float));
167  }
