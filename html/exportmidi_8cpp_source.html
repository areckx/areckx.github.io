85  TimeSigMap* sigmap = cs->sigmap();

87  int startTick = rs->tick;
88  int endTick = startTick + rs->len;
89  int tickOffset = rs->utick - rs->tick;
90 
91  auto bs = sigmap->lower_bound(startTick);
92  auto es = sigmap->lower_bound(endTick);
93 
94  for (auto is = bs; is != es; ++is) {
95  SigEvent se = is->second;
96  unsigned char* data = new unsigned char[4];
97  Fraction ts(se.timesig());
98  data[0] = ts.numerator();
99  int n;
100  switch (ts.denominator()) {
101  case 1: n = 0; break;
102  case 2: n = 1; break;
103  case 4: n = 2; break;
104  case 8: n = 3; break;
105  case 16: n = 4; break;
106  case 32: n = 5; break;
107  default:
108  n = 2;
109  qDebug("ExportMidi: unknown time signature %s",
110  qPrintable(ts.print()));
111  break;
112  }
113  data[1] = n;
114  data[2] = 24;
115  data[3] = 8;
116 
117  MidiEvent ev;
118  ev.setType(ME_META);
119  ev.setMetaType(META_TIME_SIGNATURE);
120  ev.setEData(data);
121  ev.setLen(4);
122  track.insert(is->first + tickOffset, ev);
123  }
124  }

171  data[1] = 0; // major

186  int startTick = rs->tick;
187  int endTick = startTick + rs->len;
188  int tickOffset = rs->utick - rs->tick;
189 
190  auto se = tempomap->lower_bound(startTick);
191  auto ee = tempomap->lower_bound(endTick);
192  for (auto it = se; it != ee; ++it) {
193  MidiEvent ev;
194  ev.setType(ME_META);

196  // compute midi tempo: microseconds / quarter note
197  //
198  int tempo = lrint((1.0 / (it->second.tempo * relTempo)) * 1000000.0);

202  unsigned char* data = new unsigned char[3];
203  data[0] = tempo >> 16;
204  data[1] = tempo >> 8;
205  data[2] = tempo;
206  ev.setEData(data);
207  track.insert(it->first + tickOffset, ev);

218 bool ExportMidi::write(const QString& name, bool midiExpandRepeats)
219  {
220  f.setFileName(name);
221  if (!f.open(QIODevice::WriteOnly))
222  return false;
223 
224  mf.setDivision(MScore::division);
225  mf.setFormat(1);
226  QList<MidiTrack>& tracks = mf.tracks();
227 
228  for (int i = 0; i < cs->nstaves(); ++i)
229  tracks.append(MidiTrack());
230 
231  cs->updateSwing();
232  cs->createPlayEvents();
233  cs->updateRepeatList(midiExpandRepeats);
234  writeHeader();
235 
236  int staffIdx = 0;
237  for (auto &track: tracks) {
238  Staff* staff = cs->staff(staffIdx);
239  Part* part = staff->part();
240 
241  track.setOutPort(part->midiPort());
242  track.setOutChannel(part->midiChannel());
243 
244  // Render each staff only once
245  EventMap events;
246  cs->renderStaff(&events, staff);
247  cs->renderSpanners(&events, staffIdx);
248 
249  // Pass throught the all instruments in the part
250  const InstrumentList* il = part->instruments();
251  for(auto j = il->begin(); j!= il->end(); j++) {
252  // Pass throught the all channels of the instrument
253  // "normal", "pizzicato", "tremolo" for Strings,
254  // "normal", "mute" for Trumpet
255  foreach(const Channel* ch, j->second->channel()) {
256  char port = part->masterScore()->midiPort(ch->channel);
257  char channel = part->masterScore()->midiChannel(ch->channel);
258 
259  if (staff->isTop()) {
260  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_RESET_ALL_CTRL, 0));
261  // We need this to get the correct pitch of bends
262  // Hidden under preferences because some software
263  // crashes when receiving RPNs: https://musescore.org/en/node/37431
264  if (channel != 9 && preferences.midiExportRPNs) {
265  // set pitch bend sensitivity to 12 semitones:
266  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 0));
267  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 0));
268  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HDATA, 12));
269 
270  // reset fine tuning
271  /*track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 1));
272  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 0));
273  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HDATA, 64));*/
274 
275  // deactivate rpn
276  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 127));
277  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 127));
278  }
279 
280  if (ch->program != -1)
281  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_PROGRAM, ch->program));
282  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_VOLUME, ch->volume));
283  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_PANPOT, ch->pan));
284  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_REVERB_SEND, ch->reverb));
285  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_CHORUS_SEND, ch->chorus));
286  }
287 
288  // Export port to MIDI META event
289  if (track.outPort() >= 0 && track.outPort() <= 127) {
290  MidiEvent ev;
291  ev.setType(ME_META);
292  ev.setMetaType(META_PORT_CHANGE);
293  ev.setLen(1);
294  unsigned char* data = new unsigned char[1];
295  data[0] = int(track.outPort());
296  ev.setEData(data);
297  track.insert(0, ev);
298  }
299 
300  for (auto i = events.begin(); i != events.end(); ++i) {
301  NPlayEvent event(i->second);
302  char eventPort = cs->masterScore()->midiPort(event.channel());
303  char eventChannel = cs->masterScore()->midiChannel(event.channel());
304  if (port != eventPort || channel != eventChannel)
305  continue;
306 
307  if (event.type() == ME_NOTEON) {
308  track.insert(i->first, MidiEvent(ME_NOTEON, channel,
309  event.pitch(), event.velo()));
310  }
311  else if (event.type() == ME_CONTROLLER) {
312  track.insert(i->first, MidiEvent(ME_CONTROLLER, channel,
313  event.controller(), event.value()));
314  }
315  else if(event.type() == ME_PITCHBEND) {
316  track.insert(i->first, MidiEvent(ME_PITCHBEND, channel,
317  event.dataA(), event.dataB()));
318  }
319  else {
320  qDebug("writeMidi: unknown midi event 0x%02x", event.type());
321  }
322  }
323  }
324  }
325  ++staffIdx;
326  }
327  return !mf.write(&f);
328  }
