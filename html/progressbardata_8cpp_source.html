53 bool ProgressBarData::eventFilter( QObject* object, QEvent* event ) {
54 
55  if ( !( enabled() && object && object == target().data() ) ) return AnimationData::eventFilter( object, event );
56  switch ( event->type() ) {
57  case QEvent::Show: {
58 
59  // reset start and end value
60  QProgressBar* progress = static_cast<QProgressBar*>( target().data() );
61  setStartValue( progress->value() );
62  setEndValue( progress->value() );
63  break;
64 
65  }
66 
67  case QEvent::Hide: {
68  if ( animation().data()->isRunning() ) {
69  animation().data()->stop();
70  }
71  break;
72  }
73 
74  default:
75  break;
76 
77  }
78 
79  return AnimationData::eventFilter( object, event );
80 
81  }

84 void ProgressBarData::valueChanged( int value ) {
85 
86  // do nothing if not enabled
87  if ( !enabled() ) return;
88 
89  // do nothing if progress is invalid
90  QProgressBar* progress = static_cast<QProgressBar*>( target().data() );
91  if ( !( progress && progress->maximum() != progress->minimum() ) ) return;
92 
93  // update start and end values
94  bool isRunning( animation().data()->isRunning() );
95  if ( isRunning ) {
96 
97  // in case next value arrives while animation is running,
98  // end animation, set value immediately
99  // and trigger target update. This increases responsiveness of progressbars
100  setStartValue( value );
101  setEndValue( value );
102  animation().data()->stop();
103  setOpacity(0);
104 
105  if ( target() ) target().data()->update();
106 
107  return;
108 
109  }
110 
111  setStartValue( endValue() );
112  setEndValue( value );
113 
114  // start animation only if target is enabled, visible, not running,
115  // and if end and start values are different enough
116  // (with end value being larger than start value)
117  if ( !(target() && target().data()->isEnabled() && target().data()->isVisible()) ) return;
118  if ( isRunning || endValue() - startValue() < 2 ) return;
119 
120  animation().data()->start();
121 
122  }
