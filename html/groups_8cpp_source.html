69 Beam::Mode Groups::endBeam(ChordRest* cr, ChordRest* prev)
70  {
71  if (cr->isGrace() || cr->beamMode() != Beam::Mode::AUTO)
72  return cr->beamMode();
73  Q_ASSERT(cr->staff());
74 
75  TDuration d = cr->durationType();
76  const Groups& g = cr->staff()->group(cr->tick());
77  Fraction stretch = cr->staff()->timeStretch(cr->tick());
78  int tick = (cr->rtick() * stretch.numerator()) / stretch.denominator();
79 
80  Beam::Mode val = g.beamMode(tick, d.type());
81 
82  // context-dependent checks
83  if (val == Beam::Mode::AUTO && tick) {
84  // if current or previous cr is in tuplet (but not both in same tuplet):
85  // consider it as if this were next shorter duration
86  if (prev && (cr->tuplet() != prev->tuplet()) && (d == prev->durationType())) {
87  if (d >= TDuration::DurationType::V_EIGHTH)
88  val = g.beamMode(tick, TDuration::DurationType::V_16TH);
89  else if (d == TDuration::DurationType::V_16TH)
90  val = g.beamMode(tick, TDuration::DurationType::V_32ND);
91  else
92  val = g.beamMode(tick, TDuration::DurationType::V_64TH);
93  }
94  // if there is a hole between previous and current cr, break beam
95  // exclude tuplets from this check; tick calculations can be unreliable
96  // and they seem to be handled well anyhow
97  if (cr->voice() && prev && !prev->tuplet() && prev->tick() + prev->actualTicks() < cr->tick())
98  val = Beam::Mode::BEGIN;
99  }
100 
101  return val;
102  }

109 Beam::Mode Groups::beamMode(int tick, TDuration::DurationType d) const
110  {
111  int shift;
112  switch (d) {
113  case TDuration::DurationType::V_EIGHTH: shift = 0; break;
114  case TDuration::DurationType::V_16TH: shift = 4; break;
115  case TDuration::DurationType::V_32ND: shift = 8; break;
116  default:
117  return Beam::Mode::AUTO;
118  }
119  const int dm = MScore::division / 8;
120  for (const GroupNode& e : *this) {
121  if (e.pos * dm < tick)
122  continue;
123  if (e.pos * dm > tick)
124  break;
125 
126  int action = (e.action >> shift) & 0xf;
127  switch (action) {
128  case 0: return Beam::Mode::AUTO;
129  case 1: return Beam::Mode::BEGIN;
130  case 2: return Beam::Mode::BEGIN32;
131  case 3: return Beam::Mode::BEGIN64;
132  default:
133  qDebug(" Groups::beamMode: bad action %d", action);
134  return Beam::Mode::AUTO;
135  }
136  }
137  return Beam::Mode::AUTO;
138  }

144 const Groups& Groups::endings(const Fraction& f)
145  {
146  for (const NoteGroup& g : noteGroups) {
147  if (g.timeSig.identical(f)) {
148  return g.endings;
149  }
150  }
151  NoteGroup g;
152  g.timeSig = f;
153  noteGroups.push_back(g);
154 
155  int pos = 0;
156  switch (f.denominator()) {
157  case 2: pos = 16; break;
158  case 4: pos = 8; break;
159  case 8: pos = 4; break;
160  case 16: pos = 2; break;
161  case 32: pos = 1; break;
162  break;
163  }
164  for (int i = 1; i < f.numerator(); ++i) {
165  GroupNode n;
166  n.pos = pos * i;
167  n.action = 0x111;
168  g.endings.push_back(n);
169  }
170  return noteGroups.back().endings;
171  }

210 void Groups::addStop(int pos, TDuration::DurationType d, Beam::Mode bm)
211  {
212  int shift;
213  switch (d) {
214  case TDuration::DurationType::V_EIGHTH: shift = 0; break;
215  case TDuration::DurationType::V_16TH: shift = 4; break;
216  case TDuration::DurationType::V_32ND: shift = 8; break;
217  default:
218  return;
219  }
220  int action;
221  if (bm == Beam::Mode::BEGIN)
222  action = 1;
223  else if (bm == Beam::Mode::BEGIN32)
224  action = 2;
225  else if (bm == Beam::Mode::BEGIN64)
226  action = 3;
227  else
228  return;
229 
230  pos /= 60;
231  action <<= shift;
232 
233  auto i = begin();
234  for (; i != end(); ++i) {
235  if (i->pos == pos) {
236  i->action = (i->action & ~(0xf << shift)) | action;
237  return;
238  }
239  if (i->pos > pos)
240  break;
241  }
242  insert(i, GroupNode({ pos, action }));
243  }
