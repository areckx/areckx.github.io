43 Sample* ZInstrument::readSample(const QString& s, MQZipReader* uz)
44  {
45  if (uz) {
46  QList<MQZipReader::FileInfo> fi = uz->fileInfoList();
47 
48  buf = uz->fileData(s);
49  if (buf.isEmpty()) {
50  printf("Sample::read: cannot read sample data <%s>\n", qPrintable(s));
51  return 0;
52  }
53  }
54  else {
55  QFile f(s);
56  if (!f.open(QIODevice::ReadOnly)) {
57  printf("Sample::read: open <%s> failed\n", qPrintable(s));
58  return 0;
59  }
60  buf = f.readAll();
61  }
62 
63  AudioFile a;
64  if (!a.open(buf)) {
65  printf("open <%s> failed: %s\n", qPrintable(s), a.error());
66  return 0;
67  }
68 
69  int channel = a.channels();
70  int frames = a.frames();
71  int sr = a.samplerate();
72 
73  short* data = new short[(frames + 3) * channel];
74  Sample* sa = new Sample(channel, data, frames, sr);
75  sa->setLoopStart(a.loopStart());
76  sa->setLoopEnd(a.loopEnd());
77  sa->setLoopMode(a.loopMode());
78 
79  if (frames != a.read(data + channel, frames)) {
80  qDebug("Sample read failed: %s\n", a.error());
81  delete sa;
82  sa = 0;
83  }
84  for (int i = 0; i < channel; ++i) {
85  data[i] = data[channel + i];
86  data[(frames-1) * channel + i] = data[(frames-3) * channel + i];
87  data[(frames-2) * channel + i] = data[(frames-3) * channel + i];
88  }
89  return sa;
90  }
