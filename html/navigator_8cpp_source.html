145 void Navigator::setScoreView(ScoreView* v)
146  {
147  if (_cv) {
148  disconnect(this, SIGNAL(viewRectMoved(const QRectF&)), _cv, SLOT(setViewRect(const QRectF&)));
149  disconnect(_cv, SIGNAL(viewRectChanged()), this, SLOT(updateViewRect()));
150  }
151  _cv = QPointer<ScoreView>(v);
152  if (v) {
153  _score = v->score();
154  rescale();
155  connect(this, SIGNAL(viewRectMoved(const QRectF&)), v, SLOT(setViewRect(const QRectF&)));
156  connect(_cv, SIGNAL(viewRectChanged()), this, SLOT(updateViewRect()));
157  rescale();
158  updateViewRect();
159  }
160  else {
161  _score = 0;
162  updateViewRect();
163  //update() should be enough... see #21841
164  repaint();
165  }

186 void Navigator::rescale()
187  {
188  if (!_score || _score->pages().isEmpty()) {
189  setMaximumSize(QWIDGETSIZE_MAX, QWIDGETSIZE_MAX);
190  setMinimumSize(0, 0);
191  return;
192  }
193  Page* lp = _score->pages().back();
194 
195  // reset the layout before setting fix size
196  setMaximumSize(QWIDGETSIZE_MAX, QWIDGETSIZE_MAX);
197  setMinimumSize(0, 0);
198 
199  if (MScore::verticalOrientation() && !_previewOnly) {
200  qreal scoreWidth = lp->width();
201  qreal scoreHeight = lp->y() + lp->height();
202  qreal m = width() / scoreWidth;
203  setFixedHeight(int(scoreHeight * m));
204  matrix = QTransform(m, 0, 0, m, 0, 0);
205  }
206  else {
207  qreal scoreWidth = lp->x() + lp->width();
208  qreal scoreHeight = lp->height();
209  if (_previewOnly)
210  scoreWidth = lp->width() * _score->pages().size();
211  qreal m = height() / scoreHeight;
212  setFixedWidth(int(scoreWidth * m));
213  matrix = QTransform(m, 0, 0, m, 0, 0);
214  }
215  }

233 void Navigator::mousePressEvent(QMouseEvent* ev)
234  {
235  if (_cv == 0)
236  return;
237  startMove = ev->pos();
238  if (!viewRect->geometry().contains(startMove)) {
239  QPointF p = matrix.inverted().map(QPointF(ev->pos()));
240  QRectF r(_cv->toLogical(QRectF(0.0, 0.0, _cv->width(), _cv->height())));
241  double dx = p.x() - (r.x() + (r.width() * .5));
242  double dy = p.y() - (r.y() + (r.height() * .5));
243  r.translate(dx, dy);
244  setViewRect(r);
245  emit viewRectMoved(matrix.inverted().mapRect(viewRect->geometry()));
246  }
247  }

253 void Navigator::mouseMoveEvent(QMouseEvent* ev)
254  {
255  QPoint delta = ev->pos() - startMove;
256  QRect r(viewRect->geometry().translated(delta));
257  startMove = ev->pos();
258 
259  if (r.width() == width())
260  r.moveLeft(0);
261  else if (r.width() < width()) {
262  if (r.x() < 0)
263  r.moveLeft(0);
264  else if (r.right() > width())
265  r.moveRight(width());
266  }
267  else {
268  if (r.right() < width())
269  r.moveRight(width());
270  else if (r.left() > 0)
271  r.moveLeft(0);
272  }
273 
274  if (r.height() == height())
275  r.moveTop(0);
276  else if (r.height() < height()) {
277  if (r.y() < 0)
278  r.moveTop(0);
279  else if (r.bottom() > height())
280  r.moveBottom(height());
281  }
282  else {
283  if (r.bottom() < height())
284  r.moveBottom(height());
285  else if (r.top() > 0)
286  r.moveTop(0);
287  }
288 
289  viewRect->setGeometry(r);
290 
291  emit viewRectMoved(matrix.inverted().mapRect(r));
292  if (MScore::verticalOrientation() && !_previewOnly) {
293  int y = delta.y() > 0 ? r.y() + r.height() : r.y();
294  scrollArea->ensureVisible(width()/2, y, 0, 0);
295  }
296  else {
297  int x = delta.x() > 0 ? r.x() + r.width() : r.x();
298  scrollArea->ensureVisible(x, height()/2, 0, 0);
299  }
300  }

342 
343 void Navigator::paintEvent(QPaintEvent* ev)
344  {
345  QPainter p(this);
346  QRect r(ev->rect());
347  p.fillRect(r, palette().color(QPalette::Window));
348 
349 // qDebug("navigator paint %d %d", r.width(), r.height());
350 
351  if (!_score)
352  return;
353  if (_score->pages().size() <= 0)
354  return;
355 
356  // compute optimal size of page number
357  QFont font("FreeSans", 4000);
358  QFontMetrics fm (font);
359  Page* firstPage = _score->pages()[0];
360  qreal factor = (firstPage->width() * 0.5) / fm.width(QString::number(_score->pages().size()));
361  font.setPointSizeF(font.pointSizeF() * factor);
362 
363  p.setTransform(matrix);
364  QRectF fr = matrix.inverted().mapRect(QRectF(r));
365  int i = 0;
366  for (Page* page : _score->pages()) {
367  QPointF pos(page->pos());
368  if (_previewOnly)
369  pos = QPointF(i*page->width(), 0);
370  QRectF pr(page->abbox().translated(pos));
371  if (pr.right() < fr.left())
372  continue;
373  if (pr.left() > fr.right())
374  break;
375 
376  p.fillRect(pr, Qt::white);
377  p.translate(pos);
378  for (System* s : page->systems()) {
379  for (MeasureBase* m : s->measures())
380  m->scanElements(&p, paintElement, false);
381  }
382  page->scanElements(&p, paintElement, false);
383  if (page->score()->layoutMode() == LayoutMode::PAGE) {
384  p.setFont(font);
385  p.setPen(MScore::layoutBreakColor);
386  p.drawText(page->bbox(), Qt::AlignCenter, QString("%1").arg(page->no() + 1 + _score->pageNumberOffset()));
387  }
388  p.translate(-pos);
389  i++;
390  }
391  }
