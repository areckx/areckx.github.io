85 // layoutChords1
86 // - layout upstem and downstem chords
87 // - offset as necessary to avoid conflict
88 //---------------------------------------------------------
89 
90 void Score::layoutChords1(Segment* segment, int staffIdx)
91  {
92  Staff* staff = Score::staff(staffIdx);
93 
94  if (staff->isTabStaff(segment->tick()))
95  return;
96 
97  std::vector<Note*> upStemNotes;
98  std::vector<Note*> downStemNotes;
99  int upVoices = 0;
100  int downVoices = 0;
101  int startTrack = staffIdx * VOICES;
102  int endTrack = startTrack + VOICES;
103  qreal nominalWidth = noteHeadWidth() * staff->mag(segment->tick());
104  qreal maxUpWidth = 0.0;
105  qreal maxDownWidth = 0.0;
106  qreal maxUpMag = 0.0;
107  qreal maxDownMag = 0.0;
108 
109  // dots and hooks can affect layout of notes as well as vice versa
110  int upDots = 0;
111  int downDots = 0;
112  bool upHooks = false;
113  bool downHooks = false;
114 
115  // also check for grace notes
116  bool upGrace = false;
117  bool downGrace = false;
118 
119  for (int track = startTrack; track < endTrack; ++track) {
120  Element* e = segment->element(track);
121  if (e && e->isChord()) {
122  Chord* chord = toChord(e);
123  bool hasGraceBefore = false;
124  for (Chord* c : chord->graceNotes()) {
125  if (c->isGraceBefore())
126  hasGraceBefore = true;
127  layoutChords2(c->notes(), c->up()); // layout grace note noteheads
128  layoutChords3(c->notes(), staff, 0); // layout grace note chords
129  }
130  if (chord->up()) {
131  ++upVoices;
132  upStemNotes.insert(upStemNotes.end(), chord->notes().begin(), chord->notes().end());
133  upDots = qMax(upDots, chord->dots());
134  maxUpMag = qMax(maxUpMag, chord->mag());
135  if (!upHooks)
136  upHooks = chord->hook();
137  if (hasGraceBefore)
138  upGrace = true;
139  }
140  else {
141  ++downVoices;
142  downStemNotes.insert(downStemNotes.end(), chord->notes().begin(), chord->notes().end());
143  downDots = qMax(downDots, chord->dots());
144  maxDownMag = qMax(maxDownMag, chord->mag());
145  if (!downHooks)
146  downHooks = chord->hook();
147  if (hasGraceBefore)
148  downGrace = true;
149  }
150  }
151  }
152 
153  if (upVoices + downVoices) {
154  // TODO: use track as secondary sort criteria?
155  // otherwise there might be issues with unisons between voices
156  // in some corner cases
157 
158  maxUpWidth = nominalWidth * maxUpMag;
159  maxDownWidth = nominalWidth * maxDownMag;
160 
161  // layout upstem noteheads
162  if (upVoices > 1) {
163  qSort(upStemNotes.begin(), upStemNotes.end(),
164  [](Note* n1, const Note* n2) ->bool {return n1->line() > n2->line(); } );
165  }
166  if (upVoices) {
167  qreal hw = layoutChords2(upStemNotes, true);
168  maxUpWidth = qMax(maxUpWidth, hw);
169  }
170 
171  // layout downstem noteheads
172  if (downVoices > 1) {
173  qSort(downStemNotes.begin(), downStemNotes.end(),
174  [](Note* n1, const Note* n2) ->bool {return n1->line() > n2->line(); } );
175  }
176  if (downVoices) {
177  qreal hw = layoutChords2(downStemNotes, false);
178  maxDownWidth = qMax(maxDownWidth, hw);
179  }
180 
181  qreal sp = staff->spatium(segment->tick());
182  qreal upOffset = 0.0; // offset to apply to upstem chords
183  qreal downOffset = 0.0; // offset to apply to downstem chords
184  qreal dotAdjust = 0.0; // additional chord offset to account for dots
185  qreal dotAdjustThreshold = 0.0; // if it exceeds this amount
186 
187  // centering adjustments for whole note, breve, and small chords
188  qreal centerUp = 0.0; // offset to apply in order to center upstem chords
189  qreal oversizeUp = 0.0; // adjustment to oversized upstem chord needed if laid out to the right
190  qreal centerDown = 0.0; // offset to apply in order to center downstem chords
191  qreal centerAdjustUp = 0.0; // adjustment to upstem chord needed after centering donwstem chord
192  qreal centerAdjustDown = 0.0; // adjustment to downstem chord needed after centering upstem chord
193 
194  // only center chords if they differ from nominal by at least this amount
195  // this avoids unnecessary centering on differences due only to floating point roundoff
196  // it also allows for the possibility of disabling centering
197  // for notes only "slightly" larger than nominal, like half notes
198  // but this will result in them not being aligned with each other between voices
199  // unless you change to left alignment as described in the comments below
200  qreal centerThreshold = 0.01 * sp;
201 
202  // amount by which actual width exceeds nominal, adjusted for staff mag() only
203  qreal headDiff = maxUpWidth - nominalWidth;
204  // amount by which actual width exceeds nominal, adjusted for staff & chord/note mag()
205  qreal headDiff2 = maxUpWidth - nominalWidth * (maxUpMag / staff->mag(segment->tick()));
206  if (headDiff > centerThreshold) {
207  // larger than nominal
208  centerUp = headDiff * -0.5;
209  // maxUpWidth is true width, but we no longer will care about that
210  // instead, we care only about portion to right of origin
211  maxUpWidth += centerUp;
212  // to left align rather than center, delete both of the above
213  if (headDiff2 > centerThreshold) {
214  // if max notehead is wider than nominal with chord/note mag() applied
215  // then noteheads extend to left of origin
216  // because stemPosX() is based on nominal width
217  // so we need to correct for that too
218  centerUp += headDiff2;
219  oversizeUp = headDiff2;
220  }
221  }
222  else if (-headDiff > centerThreshold) {
223  // smaller than nominal
224  centerUp = -headDiff * 0.5;
225  if (headDiff2 > centerThreshold) {
226  // max notehead is wider than nominal with chord/note mag() applied
227  // perform same adjustment as above
228  centerUp += headDiff2;
229  oversizeUp = headDiff2;
230  }
231  centerAdjustDown = centerUp;
232  }
233 
234  headDiff = maxDownWidth - nominalWidth;
235  if (headDiff > centerThreshold) {
236  // larger than nominal
237  centerDown = headDiff * -0.5;
238  // to left align rather than center, change the above to
239  //centerAdjustUp = headDiff;
240  maxDownWidth = nominalWidth - centerDown;
241  }
242  else if (-headDiff > centerThreshold) {
243  // smaller than nominal
244  centerDown = -headDiff * 0.5;
245  centerAdjustUp = centerDown;
246  }
247 
248  // handle conflict between upstem and downstem chords
249 
250  if (upVoices && downVoices) {
251  Note* bottomUpNote = upStemNotes.front();
252  Note* topDownNote = downStemNotes.back();
253  int separation;
254  if (bottomUpNote->chord()->staffMove() == topDownNote->chord()->staffMove())
255  separation = topDownNote->line() - bottomUpNote->line();
256  else
257  separation = 2; // no conflict
258  QVector<Note*> overlapNotes;
259  overlapNotes.reserve(8);
260 
261  if (separation == 1) {
262  // second
263  downOffset = maxUpWidth;
264  // align stems if present, leave extra room if not
265  if (topDownNote->chord()->stem() && bottomUpNote->chord()->stem())
266  downOffset -= topDownNote->chord()->stem()->lineWidth();
267  else
268  downOffset += 0.1 * sp;
269  }
270 
271  else if (separation < 1) {
272 
273  // overlap (possibly unison)
274 
275  // build list of overlapping notes
276  for (int i = 0, n = upStemNotes.size(); i < n; ++i) {
277  if (upStemNotes[i]->line() >= topDownNote->line() - 1)
278  overlapNotes.append(upStemNotes[i]);
279  else
280  break;
281  }
282  for (int i = downStemNotes.size() - 1; i >= 0; --i) {
283  if (downStemNotes[i]->line() <= bottomUpNote->line() + 1)
284  overlapNotes.append(downStemNotes[i]);
285  else
286  break;
287  }
288  qSort(overlapNotes.begin(), overlapNotes.end(),
289  [](Note* n1, const Note* n2) ->bool {return n1->line() > n2->line(); } );
290 
291  // determine nature of overlap
292  bool shareHeads = true; // can all overlapping notes share heads?
293  bool matchPending = false; // looking for a unison match
294  bool conflictUnison = false; // unison found
295  bool conflictSecondUpHigher = false; // second found
296  bool conflictSecondDownHigher = false; // second found
297  int lastLine = 1000;
298  Note* p = overlapNotes[0];
299  for (int i = 0, count = overlapNotes.size(); i < count; ++i) {
300  Note* n = overlapNotes[i];
301  NoteHead::Type nHeadType;
302  NoteHead::Type pHeadType;
303  Chord* nchord = n->chord();
304  Chord* pchord = p->chord();
305  if (n->mirror()) {
306  if (separation < 0) {
307  // don't try to share heads if there is any mirroring
308  shareHeads = false;
309  // don't worry about conflicts involving mirrored notes
310  continue;
311  }
312  }
313  int line = n->line();
314  int d = lastLine - line;
315  switch (d) {
316  case 0:
317  // unison
318  conflictUnison = true;
319  matchPending = false;
320  nHeadType = (n->headType() == NoteHead::Type::HEAD_AUTO) ? n->chord()->durationType().headType() : n->headType();
321  pHeadType = (p->headType() == NoteHead::Type::HEAD_AUTO) ? p->chord()->durationType().headType() : p->headType();
322  // the most important rules for sharing noteheads on unisons between voices are
323  // that notes must be one same line with same tpc
324  // noteheads must be unmirrored and of same group
325  // and chords must be same size (or else sharing code won't work)
326  if (n->headGroup() != p->headGroup() || n->tpc() != p->tpc() || n->mirror() || p->mirror() || nchord->small() != pchord->small()) {
327  shareHeads = false;
328  }
329  else {
330  // noteheads are potentially shareable
331  // it is more subjective at this point
332  // current default is to require *either* of the following:
333  // 1) both chords have same number of dots, both have stems, and both noteheads are same type and are full size (automatic match)
334  // or 2) one or more of the noteheads is not of type AUTO, but is explicitly set to match the other (user-forced match)
335  // or 3) exactly one of the noteheads is invisible (user-forced match)
336  // thus user can force notes to be shared despite differing number of dots or either being stemless
337  // by setting one of the notehead types to match the other or by making one notehead invisible
338  // TODO: consider adding a style option, staff properties, or note property to control sharing
339  if ((nchord->dots() != pchord->dots() || !nchord->stem() || !pchord->stem() || nHeadType != pHeadType || n->small() || p->small()) &&
340  ((n->headType() == NoteHead::Type::HEAD_AUTO && p->headType() == NoteHead::Type::HEAD_AUTO) || nHeadType != pHeadType) &&
341  (n->visible() == p->visible())) {
342  shareHeads = false;
343  }
344  }
345  break;
346  case 1:
347  // second
348  // trust that this won't be a problem for single unison
349  if (separation < 0) {
350  if (n->chord()->up())
351  conflictSecondUpHigher = true;
352  else
353  conflictSecondDownHigher = true;
354  shareHeads = false;
355  }
356  break;
357  default:
358  // no conflict
359  if (matchPending)
360  shareHeads = false;
361  matchPending = true;
362  }
363  p = n;
364  lastLine = line;
365  }
366  if (matchPending)
367  shareHeads = false;
368 
369  // calculate offsets
370  if (shareHeads) {
371  for (int i = overlapNotes.size() - 1; i >= 1; i -= 2) {
372  Note* p = overlapNotes[i-1];
373  Note* n = overlapNotes[i];
374  if (!(p->chord()->isNudged() || n->chord()->isNudged())) {
375  if (p->chord()->dots() == n->chord()->dots()) {
376  // hide one set dots
377  bool onLine = !(p->line() & 1);
378  if (onLine) {
379  // hide dots for lower voice
380  if (p->voice() & 1)
381  p->setDotsHidden(true);
382  else
383  n->setDotsHidden(true);
384  }
385  else {
386  // hide dots for upper voice
387  if (!(p->voice() & 1))
388  p->setDotsHidden(true);
389  else
390  n->setDotsHidden(true);
391  }
392  }
393  // formerly we hid noteheads in an effort to fix playback
394  // but this doesn't work for cases where noteheads cannot be shared
395  // so better to solve the problem elsewhere
396  }
397  }
398  }
399  else if (conflictUnison && separation == 0 && (!downGrace || upGrace))
400  downOffset = maxUpWidth + 0.3 * sp;
401  else if (conflictUnison)
402  upOffset = maxDownWidth + 0.3 * sp;
403  else if (conflictSecondUpHigher)
404  upOffset = maxDownWidth + 0.2 * sp;
405  else if ((downHooks && !upHooks) && !(upDots && !downDots))
406  downOffset = maxUpWidth + 0.3 * sp;
407  else if (conflictSecondDownHigher) {
408  if (downDots && !upDots)
409  downOffset = maxUpWidth + 0.3 * sp;
410  else {
411  upOffset = maxDownWidth - 0.2 * sp;
412  if (downHooks)
413  upOffset += 0.3 * sp;
414  }
415  }
416  else {
417  // no direct conflict, so parts can overlap (downstem on left)
418  // just be sure that stems clear opposing noteheads
419  qreal clearLeft = 0.0, clearRight = 0.0;
420  if (topDownNote->chord()->stem())
421  clearLeft = topDownNote->chord()->stem()->lineWidth() + 0.3 * sp;
422  if (bottomUpNote->chord()->stem())
423  clearRight = bottomUpNote->chord()->stem()->lineWidth() + qMax(maxDownWidth - maxUpWidth, 0.0) + 0.3 * sp;
424  else
425  downDots = 0; // no need to adjust for dots in this case
426  upOffset = qMax(clearLeft, clearRight);
427  if (downHooks) {
428  // we will need more space to avoid collision with hook
429  // but we won't need as much dot adjustment
430  upOffset = qMax(upOffset, maxDownWidth + 0.1 * sp);
431  dotAdjustThreshold = maxUpWidth - 0.3 * sp;
432  }
433  // if downstem chord is small, don't center
434  // and we might not need as much dot adjustment either
435  if (centerDown > 0.0) {
436  centerDown = 0.0;
437  centerAdjustUp = 0.0;
438  dotAdjustThreshold = (upOffset - maxDownWidth) + maxUpWidth - 0.3 * sp;
439  }
440  }
441 
442  }
443 
444  // adjust for dots
445  if ((upDots && !downDots) || (downDots && !upDots)) {
446  // only one sets of dots
447  // place between chords
448  int dots;
449  qreal mag;
450  if (upDots) {
451  dots = upDots;
452  mag = maxUpMag;
453  }
454  else {
455  dots = downDots;
456  mag = maxDownMag;
457  }
458  qreal dotWidth = segment->symWidth(SymId::augmentationDot);
459  // first dot
460  dotAdjust = styleP(StyleIdx::dotNoteDistance) + dotWidth;
461  // additional dots
462  if (dots > 1)
463  dotAdjust += styleP(StyleIdx::dotDotDistance) * (dots - 1);
464  dotAdjust *= mag;
465  // only by amount over threshold
466  dotAdjust = qMax(dotAdjust - dotAdjustThreshold, 0.0);
467  }
468  if (separation == 1)
469  dotAdjust += 0.1 * sp;
470 
471  }
472 
473  // apply chord offsets
474  for (int track = startTrack; track < endTrack; ++track) {
475  Element* e = segment->element(track);
476  if (e && e->isChord()) {
477  Chord* chord = toChord(e);
478  if (chord->up()) {
479  if (upOffset != 0.0) {
480  chord->rxpos() += upOffset + centerAdjustUp + oversizeUp;
481  if (downDots && !upDots)
482  chord->rxpos() += dotAdjust;
483  }
484  else
485  chord->rxpos() += centerUp;
486  }
487  else {
488  if (downOffset != 0.0) {
489  chord->rxpos() += downOffset + centerAdjustDown;
490  if (upDots && !downDots)
491  chord->rxpos() += dotAdjust;
492  }
493  else
494  chord->rxpos() += centerDown;
495  }
496  }
497  }
498 
499  // layout chords
500  std::vector<Note*> notes;
501  if (upVoices)
502  notes.insert(notes.end(), upStemNotes.begin(), upStemNotes.end());
503  if (downVoices)
504  notes.insert(notes.end(), downStemNotes.begin(), downStemNotes.end());
505  if (upVoices + downVoices > 1)
506  qSort(notes.begin(), notes.end(),
507  [](Note* n1, const Note* n2) ->bool {return n1->line() > n2->line(); } );
508  layoutChords3(notes, staff, segment);
509  }
510 
511  for (int track = startTrack; track < endTrack; ++track) {
512  Element* e = segment->element(track);
513  if (e)
514  e->layout();
515  }
516  }

518 //---------------------------------------------------------
519 // layoutChords2
520 // - determine which notes need mirroring
521 // - this is called once for each stem direction
522 // eg, once for voices 1&3, once for 2&4
523 // with all notes combined and sorted to resemble one chord
524 // - return maximum non-mirrored notehead width
525 //---------------------------------------------------------
526 
527 qreal Score::layoutChords2(std::vector<Note*>& notes, bool up)
528  {
529  int startIdx, endIdx, incIdx;
530  qreal maxWidth = 0.0;
531 
532  // loop in correct direction so that first encountered notehead wins conflict
533  if (up) {
534  // loop bottom up
535  startIdx = 0;
536  endIdx = notes.size();
537  incIdx = 1;
538  }
539  else {
540  // loop top down
541  startIdx = notes.size() - 1;
542  endIdx = -1;
543  incIdx = -1;
544  }
545 
546  int ll = 1000; // line of previous notehead
547  // hack: start high so first note won't show as conflict
548  bool lvisible = false; // was last note visible?
549  bool mirror = false; // should current notehead be mirrored?
550  // value is retained and may be used on next iteration
551  // to track mirror status of previous note
552  bool isLeft = notes[startIdx]->chord()->up(); // is notehead on left?
553  int lmove = notes[startIdx]->chord()->staffMove(); // staff offset of last note (for cross-staff beaming)
554 
555  for (int idx = startIdx; idx != endIdx; idx += incIdx) {
556  Note* note = notes[idx]; // current note
557  int line = note->line(); // line of current note
558  Chord* chord = note->chord();
559  int move = chord->staffMove(); // staff offset of current note
560 
561  // there is a conflict
562  // if this is same or adjacent line as previous note (and chords are on same staff!)
563  // but no need to do anything about it if either note is invisible
564  bool conflict = (qAbs(ll - line) < 2) && (lmove == move) && note->visible() && lvisible;
565 
566  // this note is on opposite side of stem as previous note
567  // if there is a conflict
568  // or if this the first note *after* a conflict
569  if (conflict || (chord->up() != isLeft))
570  isLeft = !isLeft;
571 
572  // determine if we would need to mirror current note
573  // to get it to the correct side
574  // this would be needed to get a note to left or downstem or right of upstem
575  // whether or not we actually do this is determined later (based on user mirror property)
576  bool nmirror = (chord->up() != isLeft);
577 
578  // by default, notes and dots are not hidden
579  // this may be changed later to allow unisons to share noteheads
580  note->setHidden(false);
581  note->setDotsHidden(false);
582 
583  // be sure chord position is initialized
584  // chord may be moved to the right later
585  // if there are conflicts between voices
586  chord->rxpos() = 0.0;
587 
588  // let user mirror property override the default we calculated
589  if (note->userMirror() == MScore::DirectionH::AUTO) {
590  mirror = nmirror;
591  }
592  else {
593  mirror = note->chord()->up();
594  if (note->userMirror() == MScore::DirectionH::LEFT)
595  mirror = !mirror;
596  }
597  note->setMirror(mirror);
598 
599  // accumulate return value
600  if (!mirror)
601  maxWidth = qMax(maxWidth, note->headWidth());
602 
603  // prepare for next iteration
604  lvisible = note->visible();
605  lmove = move;
606  ll = line;
607  }
608 
609  return maxWidth;
610  }

616 struct AcEl {
617  Note* note;
618  qreal x; // actual x position of this accidental relative to origin
619  qreal top; // top of accidental bbox relative to staff
620  qreal bottom; // bottom of accidental bbox relative to staff
621  int line; // line of note
622  int next; // index of next accidental of same pitch class (ascending list)
623  qreal width; // width of accidental
624  qreal ascent; // amount (in sp) vertical strokes extend above body
625  qreal descent; // amount (in sp) vertical strokes extend below body
626  qreal rightClear; // amount (in sp) to right of last vertical stroke above body
627  qreal leftClear; // amount (in sp) to left of last vertical stroke below body
628  };

630 //---------------------------------------------------------
631 // resolveAccidentals
632 // lx = calculated position of rightmost edge of left accidental relative to origin
633 //---------------------------------------------------------
634 
635 static bool resolveAccidentals(AcEl* left, AcEl* right, qreal& lx, qreal pd, qreal sp)
636  {
637  AcEl* upper;
638  AcEl* lower;
639  if (left->line >= right->line) {
640  upper = right;
641  lower = left;
642  }
643  else {
644  upper = left;
645  lower = right;
646  }
647 
648  qreal gap = lower->top - upper->bottom;
649 
650  // no conflict at all if there is sufficient vertical gap between accidentals
651  // the arrangement of accidentals into columns assumes accidentals an octave apart *do* clear
652  if (gap >= pd || lower->line - upper->line >= 7)
653  return false;
654 
655  qreal allowableOverlap = qMax(upper->descent, lower->ascent) - pd;
656 
657  // accidentals that are "close" (small gap or even slight overlap)
658  if (qAbs(gap) <= 0.33 * sp) {
659  // acceptable with slight offset
660  // if one of the accidentals can subsume the overlap
661  // and both accidentals allow it
662  if (-gap <= allowableOverlap && qMin(upper->descent, lower->ascent) > 0.0) {
663  qreal align = qMin(left->width, right->width);
664  lx = qMin(lx, right->x + align - pd);
665  return true;
666  }
667  }
668 
669  // amount by which overlapping accidentals will be separated
670  // for example, the vertical stems of two flat signs
671  // these need more space than we would need between non-overlapping accidentals
672  qreal overlapShift = pd * 1.41;
673 
674  // accidentals with more significant overlap
675  // acceptable if one accidental can subsume overlap
676  if (left == lower && -gap <= allowableOverlap) {
677  qreal offset = qMax(left->rightClear, right->leftClear);
678  offset = qMin(offset, left->width) - overlapShift;
679  lx = qMin(lx, right->x + offset);
680  return true;
681  }
682 
683  // accidentals with even more overlap
684  // can work if both accidentals can subsume overlap
685  if (left == lower && -gap <= upper->descent + lower->ascent - pd) {
686  qreal offset = qMin(left->rightClear, right->leftClear) - overlapShift;
687  if (offset > 0.0) {
688  lx = qMin(lx, right->x + offset);
689  return true;
690  }
691  }
692 
693  // otherwise, there is real conflict
694  lx = qMin(lx, right->x - pd);
695  return true;
696  }

702 static qreal layoutAccidental(AcEl* me, AcEl* above, AcEl* below, qreal colOffset, QVector<Note*>& leftNotes, qreal pnd, qreal pd, qreal sp)
703  {
704  qreal lx = colOffset;
705  Accidental* acc = me->note->accidental();
706  qreal mag = acc->mag();
707  pnd *= mag;
708  pd *= mag;
709 
710  // extra space for ledger lines
711  if (me->line <= -2 || me->line >= me->note->staff()->lines(me->note->chord()->tick()) * 2)
712  lx = qMin(lx, -0.2 * sp);
713 
714  // clear left notes
715  int lns = leftNotes.size();
716  for (int i = 0; i < lns; ++i) {
717  Note* ln = leftNotes[i];
718  int lnLine = ln->line();
719  qreal lnTop = (lnLine - 1) * 0.5 * sp;
720  qreal lnBottom = lnTop + sp;
721  if (me->top - lnBottom <= pnd && lnTop - me->bottom <= pnd) {
722  // undercut note above if possible
723  if (lnBottom - me->top <= me->ascent - pnd)
724  lx = qMin(lx, ln->x() + ln->chord()->x() + me->rightClear);
725  else
726  lx = qMin(lx, ln->x() + ln->chord()->x());
727  }
728  else if (lnTop > me->bottom)
729  break;
730  }
731 
732  // clear other accidentals
733  bool conflictAbove = false;
734  bool conflictBelow = false;
735 
736  if (above)
737  conflictAbove = resolveAccidentals(me, above, lx, pd, sp);
738  if (below)
739  conflictBelow = resolveAccidentals(me, below, lx, pd, sp);
740  if (conflictAbove || conflictBelow)
741  me->x = lx - acc->width() - acc->bbox().x();
742  else if (colOffset != 0.0)
743  me->x = lx - pd - acc->width() - acc->bbox().x();
744  else
745  me->x = lx - pnd - acc->width() - acc->bbox().x();
746 
747  return me->x;
748  }

800  acel.rightClear = bboxNE.x() - cutOutNE.x();

804  acel.rightClear = 0.0;
805  }

807  acel.descent = bboxSW.y() - cutOutSW.y();
808  acel.leftClear = cutOutSW.x() - bboxSW.x();

810  else {
811  acel.descent = 0.0;
812  acel.leftClear = 0.0;
813  }

818  ++nAcc;

825

827  Stem* stem = chord->stem();
828  if (stem)
829  overlapMirror = stem->lineWidth();
830  else if (chord->durationType().headType() == NoteHead::Type::HEAD_WHOLE)
831  overlapMirror = styleP(StyleIdx::stemWidth) * chord->mag();
832  else
833  overlapMirror = 0.0;

836  if (note->mirror()) {
837  if (_up)
838  x = stemX - overlapMirror;
839  else
840  x = stemX - hw + overlapMirror;
841  }
842  else {
843  if (_up)
844  x = stemX - hw;
845  else
846  x = 0.0;

852  // or else note pos / readPos / userOff will be out of sync
853  // and we rely on note->x() throughout the layout process
854  note->adjustReadPos();

856  // find leftmost non-mirrored note to set as X origin for accidental layout
857  // a mirrored note that extends to left of segment X origin
858  // will displace accidentals only if there is conflict
859  qreal sx = x + chord->x(); // segment-relative X position of note
860  if (note->mirror() && !chord->up() && sx < 0.0)
861  leftNotes.append(note);
862  else if (sx < lx)

864

866

869  if (chord->up())
870  upDotPosX = qMax(upDotPosX, xx);
871  else
872  downDotPosX = qMax(downDotPosX, xx);
873 
874  if (dotPosition == Direction::AUTO && nNotes > 1 && note->visible() && !note->dotsHidden()) {
875  // resolve dot conflicts
876  int line = note->line();
877  Note* above = (i < nNotes - 1) ? notes[i+1] : 0;
878  if (above && (!above->visible() || above->dotsHidden()))
879  above = 0;
880  int intervalAbove = above ? line - above->line() : 1000;
881  Note* below = (i > 0) ? notes[i-1] : 0;
882  if (below && (!below->visible() || below->dotsHidden()))
883  below = 0;
884  int intervalBelow = below ? below->line() - line : 1000;
885  if ((line & 1) == 0) {
886  // line
887  if (intervalAbove == 1 && intervalBelow != 1)
888  dotPosition = Direction::DOWN;
889  else if (intervalBelow == 1 && intervalAbove != 1)
890  dotPosition = Direction::UP;
891  else if (intervalAbove == 0 && above->chord()->dots()) {
892  // unison
893  if (((above->voice() & 1) == (note->voice() & 1))) {
894  above->setDotY(Direction::UP);
895  dotPosition = Direction::DOWN;
896  }
897  }
898  }
899  else {
900  // space
901  if (intervalAbove == 0 && above->chord()->dots()) {
902  // unison
903  if (!(note->voice() & 1))
904  dotPosition = Direction::UP;
905  else {
906  if (!(above->voice() & 1))
907  above->setDotY(Direction::UP);
908  else
909  dotPosition = Direction::DOWN;
910  }
911  }
912  }
913  }
914  }
915  note->setDotY(dotPosition); // also removes invalid dots
916  }
917 
918  if (segment) {
919  // align all dots for segment/staff
920  // it would be possible to dots for up & down chords separately
921  // this would require space to have been allocated previously
922  // when calculating chord offsets
923  segment->setDotPosX(staff->idx(), qMax(upDotPosX, downDotPosX));
924  }
925 
926  if (nAcc == 0)
927  return;
928 
929  QVector<int> umi;
930  qreal pd = styleP(StyleIdx::accidentalDistance);
931  qreal pnd = styleP(StyleIdx::accidentalNoteDistance);
932  qreal colOffset = 0.0;
933 
934  if (nAcc >= 2 && aclist[nAcc-1].line - aclist[0].line >= 7) {
935 
936  // accidentals spread over an octave or more
937  // set up columns for accidentals with octave matches
938  // these will start at right and work to the left
939  // unmatched accidentals will use zig zag approach (see below)
940  // starting to the left of the octave columns
941 
942  qreal minX = 0.0;
943  int columnTop[7] = { -1, -1, -1, -1, -1, -1, -1 };
944 
945  // find columns of octaves
946  for (int pc = 0; pc < 7; ++pc) {
947  if (columnBottom[pc] == -1)
948  continue;
949  // calculate column height
950  for (int j = columnBottom[pc]; j != -1; j = aclist[j].next)
951  columnTop[pc] = j;
952  }
953 
954  // compute reasonable column order
955  // use zig zag
956  QVector<int> column;
957  QVector<int> unmatched;
958  int n = nAcc - 1;
959  for (int i = 0; i <= n; ++i, --n) {
960  int pc = (aclist[i].line + 700) % 7;
961  if (aclist[columnTop[pc]].line != aclist[columnBottom[pc]].line) {
962  if (!column.contains(pc))
963  column.append(pc);
964  }
965  else
966  unmatched.append(i);
967  if (i == n)
968  break;
969  pc = (aclist[n].line + 700) % 7;
970  if (aclist[columnTop[pc]].line != aclist[columnBottom[pc]].line) {
971  if (!column.contains(pc))
972  column.append(pc);
973  }
974  else
975  unmatched.append(n);
976  }
977  int nColumns = column.size();
978  int nUnmatched = unmatched.size();
979 
980  // handle unmatched accidentals
981  for (int i = 0; i < nUnmatched; ++i) {
982  // first try to slot it into an existing column
983  AcEl* me = &aclist[unmatched[i]];
984  // find column
985  bool found = false;
986  for (int j = 0; j < nColumns; ++j) {
987  int pc = column[j];
988  int above = -1;
989  int below = -1;
990  // find slot within column
991  for (int k = columnBottom[pc]; k != -1; k = aclist[k].next) {
992  if (aclist[k].line < me->line) {
993  above = k;
994  break;
995  }
996  below = k;
997  }
998  // check to see if accidental can fit in slot
999  qreal myPd = pd * me->note->accidental()->mag();
1000  bool conflict = false;
1001  if (above != -1 && me->top - aclist[above].bottom < myPd)
1002  conflict = true;
1003  else if (below != -1 && aclist[below].top - me->bottom < myPd)
1004  conflict = true;
1005  if (!conflict) {
1006  // insert into column
1007  found = true;
1008  me->next = above;
1009  if (above == -1)
1010  columnTop[pc] = unmatched[i];
1011  if (below != -1)
1012  aclist[below].next = unmatched[i];
1013  else
1014  columnBottom[pc] = unmatched[i];
1015  break;
1016  }
1017  }
1018  // if no slot found, then add to list of unmatched accidental indices
1019  if (!found)
1020  umi.push_back(unmatched[i]);
1021  }
1022  nAcc = umi.size();
1023  if (nAcc > 1)
1024  qSort(umi);
1025 
1026  // lay out columns
1027  for (int i = 0; i < nColumns; ++i) {
1028  int pc = column[i];
1029  AcEl* below = 0;
1030  // lay out accidentals
1031  for (int j = columnBottom[pc]; j != -1; j = aclist[j].next) {
1032  qreal x = layoutAccidental(&aclist[j], 0, below, colOffset, leftNotes, pnd, pd, sp);
1033  minX = qMin(minX, x);
1034  below = &aclist[j];
1035  }
1036  // align within column
1037  int next = -1;
1038  for (int j = columnBottom[pc]; j != -1; j = next) {
1039  next = aclist[j].next;
1040  if (next != -1 && aclist[j].line == aclist[next].line)
1041  continue;
1042  aclist[j].x = minX;
1043  }
1044  // move to next column
1045  colOffset = minX;
1046  }
1047 
1048  }
1049 
1050  else {
1051  for (int i = 0; i < nAcc; ++i)
1052  umi.push_back(i);
1053  }
1054 
1055  if (nAcc) {
1056 
1057  // for accidentals with no octave matches, use zig zag approach
1058  // layout right to left in pairs, (next) highest then lowest
1059 
1060  AcEl* me = &aclist[umi[0]];
1061  AcEl* above = 0;
1062  AcEl* below = 0;
1063 
1064  // layout top accidental
1065  layoutAccidental(me, above, below, colOffset, leftNotes, pnd, pd, sp);
1066 
1067  // layout bottom accidental
1068  int n = nAcc - 1;
1069  if (n > 0) {
1070  above = me;
1071  me = &aclist[umi[n]];
1072  layoutAccidental(me, above, below, colOffset, leftNotes, pnd, pd, sp);
1073  }
1074 
1075  // layout middle accidentals
1076  if (n > 1) {
1077  for (int i = 1; i < n; ++i, --n) {
1078  // next highest
1079  below = me;
1080  me = &aclist[umi[i]];
1081  layoutAccidental(me, above, below, colOffset, leftNotes, pnd, pd, sp);
1082  if (i == n - 1)
1083  break;
1084  // next lowest
1085  above = me;
1086  me = &aclist[umi[n-1]];
1087  layoutAccidental(me, above, below, colOffset, leftNotes, pnd, pd, sp);
1088  }
1089  }
1090 
1091  }
1092 
1093  for (const AcEl& e : aclist) {
1094  // even though we initially calculate accidental position relative to segment
1095  // we must record pos for accidental relative to note,
1096  // since pos is always interpreted relative to parent
1097  Note* note = e.note;
1098  qreal x = e.x + lx - (note->x() + note->chord()->x());
1099  note->accidental()->setPos(x, 0);
1100  note->accidental()->adjustReadPos();
1101  }
1102  }

1461  }

1464  }

1507 //---------------------------------------------------------
1508 // sff2
1509 // compute 1/Force for a given Extend
1510 //---------------------------------------------------------
1511 
1512 static qreal sff2(qreal width, qreal xMin, const SpringMap& springs)
1513  {
1514  if (width <= xMin)
1515  return 0.0;
1516  auto i = springs.begin();
1517  qreal c = i->second.stretch;
1518  if (c == 0.0) //DEBUG
1519  c = 1.1;
1520  qreal f = 0.0;
1521  for (; i != springs.end();) {
1522  xMin -= i->second.fix;
1523  f = (width - xMin) / c;
1524  ++i;
1525  if (i == springs.end() || f <= i->first)
1526  break;
1527  c += i->second.stretch;
1528  }
1529  return f;
1530  }

1533 // respace
1534 //---------------------------------------------------------
1535 
1536 void Score::respace(std::vector<ChordRest*>* elements)
1537  {
1538  ChordRest* cr1 = elements->front();
1539  ChordRest* cr2 = elements->back();
1540  int n = elements->size();
1541  qreal x1 = cr1->segment()->pos().x();
1542  qreal x2 = cr2->segment()->pos().x();
1543 
1544  qreal width[n-1];
1545  int ticksList[n-1];
1546  int minTick = 100000;
1547 
1548  for (int i = 0; i < n-1; ++i) {
1549  ChordRest* cr = (*elements)[i];
1550  ChordRest* ncr = (*elements)[i+1];
1551  width[i] = cr->shape().minHorizontalDistance(ncr->shape());
1552  ticksList[i] = cr->duration().ticks();
1553  minTick = qMin(ticksList[i], minTick);
1554  }
1555 
1556  //---------------------------------------------------
1557  // compute stretches
1558  //---------------------------------------------------
1559 
1560  SpringMap springs;
1561  qreal minimum = 0.0;
1562  for (int i = 0; i < n-1; ++i) {
1563  qreal w = width[i];
1564  int t = ticksList[i];
1565  qreal str = 1.0 + 0.865617 * log(qreal(t) / qreal(minTick));
1566  qreal d = w / str;
1567 
1568  springs.insert(std::pair<qreal, Spring>(d, Spring(i, str, w)));
1569  minimum += w;
1570  }
1571 
1572  //---------------------------------------------------
1573  // distribute stretch to elements
1574  //---------------------------------------------------
1575 
1576  qreal force = sff2(x2 - x1, minimum, springs);
1577  for (auto i = springs.begin(); i != springs.end(); ++i) {
1578  qreal stretch = force * i->second.stretch;
1579  if (stretch < i->second.fix)
1580  stretch = i->second.fix;
1581  width[i->second.seg] = stretch;
1582  }
1583  qreal x = x1;
1584  for (int i = 1; i < n-1; ++i) {
1585  x += width[i-1];
1586  ChordRest* cr = (*elements)[i];
1587  qreal dx = x - cr->segment()->pos().x();
1588  cr->rxpos() += dx;
1589  }
1590  }
