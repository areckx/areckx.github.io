190 void OmrState::importPdfPage(OmrPage* omrPage, qreal top)
191  {
192  TDuration d(TDuration::DurationType::V_MEASURE);
193 
194  int nsystems = omrPage->systems().size();
195  if(nsystems == 0) return;
196 
197  //add top margin for alignment
198  MeasureBase* first_measure = score->first();
199  if (first_measure == 0 || first_measure->isVBox()) {
200  VBox* vbox = new VBox(score);
201  vbox->setNext(score->first());
202  vbox->setTick(0);
203  vbox->setBoxHeight(Spatium(top));
204  vbox->setBottomMargin(0);
205  vbox->setBottomGap(0);
206  score->measures()->add(vbox);
207  }
208 
209  for (int k = 0; k < nsystems; ++k) {
210  importPdfSystem(omrPage->system(nsystems - k - 1));
211  }
212 
213  Measure* measure = score->lastMeasure();
214  if (measure) {
215  LayoutBreak* b = new LayoutBreak(score);
216  b->setLayoutBreakType(LayoutBreak::Type::PAGE);
217  measure->add(b);
218  }
219 
220  measure = score->firstMeasure();
221  if (measure) {
222  if (!measure->vspacerUp(0)){
223  Spacer* spacer = new Spacer(score);
224  spacer->setSpacerType(SpacerType::UP);
225  spacer->setTrack(0);
226  measure->add(spacer);
227  }
228  Spacer* sp = measure->vspacerUp(0);
229  sp->layout();
230  sp->setPos(sp->rxpos(), top);
231  }
232  }

238 Score::FileError importPdf(MasterScore* score, const QString& path)
239  {
240  Omr* omr = new Omr(path, score);
241  if (!omr->readPdf()) {
242  delete omr;
243  return Score::FileError::FILE_BAD_FORMAT;
244  }
245 
246  score->setOmr(omr);
247  qreal sp = omr->spatiumMM();
248  if (sp == 0.0)
249  sp = 1.5;
250  score->setSpatium(sp * DPMM);
251  score->style().set(StyleIdx::lastSystemFillLimit, 0.0);
252  score->style().set(StyleIdx::staffLowerBorder, 0.0);
253  score->style().set(StyleIdx::measureSpacing, 1.0);
254  score->style().set(StyleIdx::frameSystemDistance, 0);
255  score->style().set(StyleIdx::pageEvenLeftMargin, 5.0 * DPMM / DPI);
256  score->style().set(StyleIdx::pageEvenTopMargin, 0);
257  score->style().set(StyleIdx::pageEvenBottomMargin, 0);
258  score->style().set(StyleIdx::pageOddLeftMargin, 5.0 * DPMM / DPI);
259  score->style().set(StyleIdx::pageOddTopMargin, 0);
260  score->style().set(StyleIdx::pageOddBottomMargin, 0);
261  score->style().set(StyleIdx::minSystemDistance, Spatium(omr->systemDistance()));
262  score->style().set(StyleIdx::maxSystemDistance, Spatium(omr->systemDistance()));
263  score->style().set(StyleIdx::akkoladeDistance, Spatium(omr->staffDistance()));
264 
265  Part* part = new Part(score);
266  OmrPage* omrPage = omr->pages().front();
267 
268  //may need to identify maximal number of staff per system and then high some of those
269  if (omrPage->systems().size() > 0) {
270  for (int i = 0; i < omrPage->systems().front().staves().size(); i++) {
271  Staff* staff = new Staff(score);
272  staff->setPart(part);
273  part->insertStaff(staff, -1);
274  score->staves().append(staff);
275  }
276  }
277  score->appendPart(part);
278 
279  OmrState state;
280  state.score = score;
281  foreach (OmrPage* omrPage, omr->pages()) {
282  OmrStaff staff = omrPage->systems().last().staves().first();
283  qreal top = staff.top()/omr->spatium();
284  state.importPdfPage(omrPage, top);
285  }
286 
287  //---create bracket
288 
289  //score->staff(0)->setBracket(0, BracketType::BRACE);
290  //score->staff(0)->setBracketSpan(0, 2);
291 
292  score->setShowOmr(true);
293  omr->page(0)->readHeader(score);
294  score->rebuildMidiMapping();
295  return Score::FileError::FILE_NO_ERROR;
296  }
297 }
