594 XmlStreamReader::TokenType XmlStreamReader::readNext()
595 {
596  Q_D(XmlStreamReader);
597  if (d->type != Invalid) {
598  if (!d->hasCheckedStartDocument)
599  if (!d->checkStartDocument())
600  return d->type; // synthetic StartDocument or error
601  d->parse();
602  if (d->atEnd && d->type != EndDocument && d->type != Invalid)
603  d->raiseError(PrematureEndOfDocumentError);
604  else if (!d->atEnd && d->type == EndDocument)
605  d->raiseWellFormedError(XmlStream::tr("Extra content at end of document."));
606  } else if (d->error == PrematureEndOfDocumentError) {
607  // resume error
608  d->type = NoToken;
609  d->atEnd = false;
610  d->token = -1;
611  return readNext();
612  }
613  return d->type;

965 bool XmlStreamReaderPrivate::scanUntil(const char *str, short tokenToInject)
966 {
967  int pos = textBuffer.size();
968  int oldLineNumber = lineNumber;
969 
970  while (uint c = getChar()) {
971  /* First, we do the validation & normalization. */
972  switch (c) {
973  case '\r':
974  if ((c = filterCarriageReturn()) == 0)
975  break;
976  // fall through
977  case '\n':
978  ++lineNumber;
979  lastLineStart = characterOffset + readBufferPos;
980  // fall through
981  case '\t':
982  textBuffer += QChar(c);
983  continue;
984  default:
985  if (c < 0x20 || (c > 0xFFFD && c < 0x10000) || c > QChar::LastValidCodePoint ) {
986  raiseWellFormedError(XmlStream::tr("Invalid XML character."));
987  lineNumber = oldLineNumber;
988  return false;
989  }
990  textBuffer += QChar(c);
991  }
992 
993 
994  /* Second, attempt to lookup str. */
995  if (c == uint(*str)) {
996  if (!*(str + 1)) {
997  if (tokenToInject >= 0)
998  injectToken(tokenToInject);
999  return true;
1000  } else {
1001  if (scanString(str + 1, tokenToInject, false))
1002  return true;
1003  }
1004  }
1005  }
1006  putString(textBuffer, pos);
1007  textBuffer.resize(pos);
1008  lineNumber = oldLineNumber;
1009  return false;
1010 }

1012 bool XmlStreamReaderPrivate::scanString(const char *str, short tokenToInject, bool requireSpace)
1013 {
1014  int n = 0;
1015  while (str[n]) {
1016  ushort c = getChar();
1017  if (c != ushort(str[n])) {
1018  if (c)
1019  putChar(c);
1020  while (n--) {
1021  putChar(ushort(str[n]));
1022  }
1023  return false;
1024  }
1025  ++n;
1026  }
1027  for (int i = 0; i < n; ++i)
1028  textBuffer += QChar(ushort(str[i]));
1029  if (requireSpace) {
1030  int s = fastScanSpace();
1031  if (!s || atEnd) {
1032  int pos = textBuffer.size() - n - s;
1033  putString(textBuffer, pos);
1034  textBuffer.resize(pos);
1035  return false;
1036  }
1037  }
1038  if (tokenToInject >= 0)
1039  injectToken(tokenToInject);
1040  return true;
1041 }
