73 QString FontInfo::typeName() const
74 {
75  switch (type()) {
76  case unknown:
77  return QObject::tr("unknown");
78  case Type1:
79  return QObject::tr("Type 1");
80  case Type1C:
81  return QObject::tr("Type 1C");
82  case Type3:
83  return QObject::tr("Type 3");
84  case TrueType:
85  return QObject::tr("TrueType");
86  case CIDType0:
87  return QObject::tr("CID Type 0");
88  case CIDType0C:
89  return QObject::tr("CID Type 0C");
90  case CIDTrueType:
91  return QObject::tr("CID TrueType");
92  case Type1COT:
93  return QObject::tr("Type 1C (OpenType)");
94  case TrueTypeOT:
95  return QObject::tr("TrueType (OpenType)");
96  case CIDType0COT:
97  return QObject::tr("CID Type 0C (OpenType)");
98  case CIDTrueTypeOT:
99  return QObject::tr("CID TrueType (OpenType)");
100  }
101  return QObject::tr("Bug: unexpected font type. Notify poppler mailing list!");
102 }

124 QList<FontInfo> FontIterator::next()
125 {
126  ++d->currentPage;
127 
128  QList<FontInfo> fonts;
129  GooList *items = d->fontInfoScanner.scan( 1 );
130  if ( !items )
131  return fonts;
132  fonts.reserve( items->getLength() );
133  for ( int i = 0; i < items->getLength(); ++i ) {
134  fonts.append( FontInfo( FontInfoData( ( ::FontInfo* )items->get( i ) ) ) );
135  }
136  deleteGooList( items, ::FontInfo );
137  return fonts;
138 }
