329 //---------------------------------------------------------
330 // getNoteList
331 // return notes as list of indices in BagpipeNoteInfoList
332 //---------------------------------------------------------
333 
334 noteList BagpipeEmbellishment::getNoteList() const
335  {
336  noteList nl;
337  if (_embelType >= 0 && _embelType < nEmbellishments()) {
338  QStringList notes = BagpipeEmbellishmentList[_embelType].notes.split(' ');
339  int noteInfoSize = sizeof(BagpipeNoteInfoList) / sizeof(*BagpipeNoteInfoList);
340  foreach (const QString note, notes) {
341  // search for note in BagpipeNoteInfoList
342  for (int i = 0; i < noteInfoSize; ++i) {
343  if (BagpipeNoteInfoList[i].name == note) {
344  // found it, append to list
345  nl << i;
346  break;
347  }
348  }
349  }
350  }
351  return nl;
352  }

369 void BagpipeEmbellishment::read(XmlReader& e)
370  {
371  while (e.readNextStartElement()) {
372  const QStringRef& tag(e.name());
373  if (tag == "subtype")
374  _embelType = e.readInt();
375  else
376  e.unknown();
377  }
