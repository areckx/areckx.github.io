96 QString MScoreTextToMXML::toPlainText(const QString& text)
97  {
98  QString res;
99  bool inElem = false;
100  foreach(QChar ch, text) {
101  if (ch == '<')
102  inElem = true;
103  else if (ch == '>')
104  inElem = false;
105  else {
106  if (!inElem)
107  res += ch;
108  }
109  }
110  //qDebug("MScoreTextToMXML::toPlainText('%s') res '%s'", qPrintable(text), qPrintable(res));
111  return res;
112  }

160 bool MScoreTextToMXML::split(const QList<TextFragment>& in, const int pos, const int len,
161  QList<TextFragment>& left, QList<TextFragment>& mid, QList<TextFragment>& right)
162  {
163  //qDebug("MScoreTextToMXML::split in size %d pos %d len %d", plainTextPlusSymbolsListSize(in), pos, len);
164  //qDebug("-> in");
165  //dumpText(in);
166 
167  if (pos < 0 || len < 0)
168  return false;
169 
170  // ensure output is empty at start
171  left.clear();
172  mid.clear();
173  right.clear();
174 
175  // set pos to begin of first fragment
176  int fragmentNr = 0;
177  TextFragment fragment;
178  if (fragmentNr < in.size()) fragment = in.at(fragmentNr);
179  QList<TextFragment>* currentDest = &left;
180  int currentMaxSize = pos;
181 
182  // while text left
183  while (fragmentNr < in.size()) {
184  int destSize = plainTextPlusSymbolsListSize(*currentDest);
185  int fragSize = plainTextPlusSymbolsFragmentSize(fragment);
186  // if no room left in current destination (check applies only to left and mid)
187  if ((currentDest != &right && destSize >= currentMaxSize)
188  || currentDest == &right) {
189  // move to next destination
190  if (currentDest == &left) {
191  currentDest = &mid;
192  currentMaxSize = len;
193  }
194  else if (currentDest == &mid) {
195  currentDest = &right;
196  }
197  }
198  // if current fragment fits in current destination (check applies only to left and mid)
199  if ((currentDest != &right && destSize + fragSize <= currentMaxSize)
200  || currentDest == &right) {
201  // add it
202  currentDest->append(fragment);
203  // move to next fragment
204  fragmentNr++;
205  if (fragmentNr < in.size()) fragment = in.at(fragmentNr);
206  }
207  else {
208  // split current fragment
209  TextFragment rightPart = fragment.split(currentMaxSize - plainTextPlusSymbolsListSize(*currentDest));
210  // add first part to current destination
211  currentDest->append(fragment);
212  fragment = rightPart;
213  }
214  }
215 
216  /*
217  qDebug("-> left");
218  dumpText(left);
219  qDebug("-> mid");
220  dumpText(mid);
221  qDebug("-> right");
222  dumpText(right);
223  */
224 
225  return true;
226  }

267 QString MScoreTextToMXML::updateFormat()
268  {
269  QString res;
270  res += attribute(newFormat.bold() != oldFormat.bold(), newFormat.bold(), "font-weight=\"bold\"", "font-weight=\"normal\"");
271  res += attribute(newFormat.italic() != oldFormat.italic(), newFormat.italic(), "font-style=\"italic\"", "font-style=\"normal\"");
272  res += attribute(newFormat.underline() != oldFormat.underline(), newFormat.underline(), "underline=\"1\"", "underline=\"0\"");
273  res += attribute(newFormat.fontFamily() != oldFormat.fontFamily(), true, QString("font-family=\"%1\"").arg(newFormat.fontFamily()), "");
274  bool needSize = newFormat.fontSize() < 0.99 * oldFormat.fontSize() || newFormat.fontSize() > 1.01 * oldFormat.fontSize();
275  res += attribute(needSize, true, QString("font-size=\"%1\"").arg(newFormat.fontSize()), "");
276  //qDebug("updateFormat() res '%s'", qPrintable(res));
277  oldFormat = newFormat;
278  return res;
279  }
