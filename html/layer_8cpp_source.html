125  QString tag(tags->item(i, 0)->text());

144  }
145  }
146  QTableWidgetItem* wi = layers->item(row, 1);

180 void LayerManager:: accept()
181  {
182  score->startCmd();
183  for (int i = 0; i < 31; ++i) {
184  QString tag(tags->item(i, 0)->text());
185  QString comment(tags->item(i, 1)->text());
186  score->layerTags()[i+1] = tag;
187  score->layerTagComments()[i+1] = comment;
188  }
189  int row = layers->currentRow();
190  if (row != -1)
191  score->setCurrentLayer(row);
192 
193  QList<Layer>& layer = score->layer();
194  layer.clear();
195 
196  int n = layers->rowCount();
197  for (int i = 0; i < n; ++i) {
198  Layer l;
199  l.name = layers->item(i, 0)->text();
200  l.tags = 1;
201  QString ts = layers->item(i, 1)->text();
202  QStringList tags = ts.split(",");
203  foreach (QString tag, tags) {
204  for (int idx = 0; idx < 32; ++idx) {
205  if (tag == score->layerTags()[idx]) {
206  l.tags |= 1 << idx;
207  break;
208  }
209  }
210  }
211  if (i == 0) // hardwired default tag
212  l.tags |= 1;
213  layer.append(l);
214  }
215  score->setLayoutAll();
216  score->endCmd();
217  if (enableExperimental)
218  mscore->updateLayer();
219  QDialog::accept();
220  }
