73 void PDFRectangle::clipTo(PDFRectangle *rect) {
74  if (x1 < rect->x1) {
75  x1 = rect->x1;
76  } else if (x1 > rect->x2) {
77  x1 = rect->x2;
78  }
79  if (x2 < rect->x1) {
80  x2 = rect->x1;
81  } else if (x2 > rect->x2) {
82  x2 = rect->x2;
83  }
84  if (y1 < rect->y1) {
85  y1 = rect->y1;
86  } else if (y1 > rect->y2) {
87  y1 = rect->y2;
88  }
89  if (y2 < rect->y1) {
90  y2 = rect->y1;
91  } else if (y2 > rect->y2) {
92  y2 = rect->y2;
93  }
94 }

100 PageAttrs::PageAttrs(PageAttrs *attrs, Dict *dict) {
101  Object obj1;
102  PDFRectangle mBox;
103  const GBool isPage = dict->is("Page");
104 
105  // get old/default values
106  if (attrs) {
107  mediaBox = attrs->mediaBox;
108  cropBox = attrs->cropBox;
109  haveCropBox = attrs->haveCropBox;
110  rotate = attrs->rotate;
111  attrs->resources.copy(&resources);
112  } else {
113  // set default MediaBox to 8.5" x 11" -- this shouldn't be necessary
114  // but some (non-compliant) PDF files don't specify a MediaBox
115  mediaBox.x1 = 0;
116  mediaBox.y1 = 0;
117  mediaBox.x2 = 612;
118  mediaBox.y2 = 792;
119  cropBox.x1 = cropBox.y1 = cropBox.x2 = cropBox.y2 = 0;
120  haveCropBox = gFalse;
121  rotate = 0;
122  resources.initNull();
123  }
124 
125  // media box
126  if (readBox(dict, "MediaBox", &mBox)) {
127  mediaBox = mBox;
128  }
129 
130  // crop box
131  if (readBox(dict, "CropBox", &cropBox)) {
132  haveCropBox = gTrue;
133  }
134  if (!haveCropBox) {
135  cropBox = mediaBox;
136  }
137 
138  if (isPage) {
139  // cropBox can not be bigger than mediaBox
140  if (cropBox.x2 - cropBox.x1 > mediaBox.x2 - mediaBox.x1)
141  {
142  cropBox.x1 = mediaBox.x1;
143  cropBox.x2 = mediaBox.x2;
144  }
145  if (cropBox.y2 - cropBox.y1 > mediaBox.y2 - mediaBox.y1)
146  {
147  cropBox.y1 = mediaBox.y1;
148  cropBox.y2 = mediaBox.y2;
149  }
150  }
151 
152  // other boxes
153  bleedBox = cropBox;
154  readBox(dict, "BleedBox", &bleedBox);
155  trimBox = cropBox;
156  readBox(dict, "TrimBox", &trimBox);
157  artBox = cropBox;
158  readBox(dict, "ArtBox", &artBox);
159 
160  // rotate
161  dict->lookup("Rotate", &obj1);
162  if (obj1.isInt()) {
163  rotate = obj1.getInt();
164  }
165  obj1.free();
166  while (rotate < 0) {
167  rotate += 360;
168  }
169  while (rotate >= 360) {
170  rotate -= 360;
171  }
172 
173  // misc attributes
174  dict->lookup("LastModified", &lastModified);
175  dict->lookup("BoxColorInfo", &boxColorInfo);
176  dict->lookup("Group", &group);
177  dict->lookup("Metadata", &metadata);
178  dict->lookup("PieceInfo", &pieceInfo);
179  dict->lookup("SeparationInfo", &separationInfo);
180 
181  // resource dictionary
182  dict->lookup("Resources", &obj1);
183  if (obj1.isDict()) {
184  resources.free();
185  obj1.copy(&resources);
186  }
187  obj1.free();
188 }

207 GBool PageAttrs::readBox(Dict *dict, const char *key, PDFRectangle *box) {
208  PDFRectangle tmp;
209  double t;
210  Object obj1, obj2;
211  GBool ok;
212 
213  dict->lookup(key, &obj1);
214  if (obj1.isArray() && obj1.arrayGetLength() == 4) {
215  ok = gTrue;
216  obj1.arrayGet(0, &obj2);
217  if (obj2.isNum()) {
218  tmp.x1 = obj2.getNum();
219  } else {
220  ok = gFalse;
221  }
222  obj2.free();
223  obj1.arrayGet(1, &obj2);
224  if (obj2.isNum()) {
225  tmp.y1 = obj2.getNum();
226  } else {
227  ok = gFalse;
228  }
229  obj2.free();
230  obj1.arrayGet(2, &obj2);
231  if (obj2.isNum()) {
232  tmp.x2 = obj2.getNum();
233  } else {
234  ok = gFalse;
235  }
236  obj2.free();
237  obj1.arrayGet(3, &obj2);
238  if (obj2.isNum()) {
239  tmp.y2 = obj2.getNum();
240  } else {
241  ok = gFalse;
242  }
243  obj2.free();
244  if (tmp.x1 == 0 && tmp.x2 == 0 && tmp.y1 == 0 && tmp.y2 == 0)
245  ok = gFalse;
246  if (ok) {
247  if (tmp.x1 > tmp.x2) {
248  t = tmp.x1; tmp.x1 = tmp.x2; tmp.x2 = t;
249  }
250  if (tmp.y1 > tmp.y2) {
251  t = tmp.y1; tmp.y1 = tmp.y2; tmp.y2 = t;
252  }
253  *box = tmp;
254  }
255  } else {
256  ok = gFalse;
257  }
258  obj1.free();
259  return ok;
260 }

266 Page::Page(PDFDoc *docA, int numA, Dict *pageDict, Ref pageRefA, PageAttrs *attrsA, Form *form) {
267  Object tmp;
268 
269 #if MULTITHREADED
270  gInitMutex(&mutex);
271 #endif
272  ok = gTrue;
273  doc = docA;
274  xref = doc->getXRef();
275  num = numA;
276  duration = -1;
277  annots = NULL;
278 
279  pageObj.initDict(pageDict);
280  pageRef = pageRefA;
281 
282  // get attributes
283  attrs = attrsA;
284  attrs->clipBoxes();
285 
286  // transtion
287  pageDict->lookupNF("Trans", &trans);
288  if (!(trans.isRef() || trans.isDict() || trans.isNull())) {
289  error(errSyntaxError, -1, "Page transition object (page {0:d}) is wrong type ({1:s})",
290  num, trans.getTypeName());
291  trans.free();
292  }
293 
294  // duration
295  pageDict->lookupNF("Dur", &tmp);
296  if (!(tmp.isNum() || tmp.isNull())) {
297  error(errSyntaxError, -1, "Page duration object (page {0:d}) is wrong type ({1:s})",
298  num, tmp.getTypeName());
299  } else if (tmp.isNum()) {
300  duration = tmp.getNum();
301  }
302  tmp.free();
303 
304  // annotations
305  pageDict->lookupNF("Annots", &annotsObj);
306  if (!(annotsObj.isRef() || annotsObj.isArray() || annotsObj.isNull())) {
307  error(errSyntaxError, -1, "Page annotations object (page {0:d}) is wrong type ({1:s})",
308  num, annotsObj.getTypeName());
309  annotsObj.free();
310  goto err2;
311  }
312 
313  // contents
314  pageDict->lookupNF("Contents", &contents);
315  if (!(contents.isRef() || contents.isArray() ||
316  contents.isNull())) {
317  error(errSyntaxError, -1, "Page contents object (page {0:d}) is wrong type ({1:s})",
318  num, contents.getTypeName());
319  contents.free();
320  goto err1;
321  }
322 
323  // thumb
324  pageDict->lookupNF("Thumb", &thumb);
325  if (!(thumb.isStream() || thumb.isNull() || thumb.isRef())) {
326  error(errSyntaxError, -1, "Page thumb object (page {0:d}) is wrong type ({1:s})",
327  num, thumb.getTypeName());
328  thumb.free();
329  thumb.initNull();
330  }
331 
332  // actions
333  pageDict->lookupNF("AA", &actions);
334  if (!(actions.isDict() || actions.isNull())) {
335  error(errSyntaxError, -1, "Page additional action object (page {0:d}) is wrong type ({1:s})",
336  num, actions.getTypeName());
337  actions.free();
338  actions.initNull();
339  }
340 
341  return;
342 
343  trans.initNull();
344  err2:
345  annotsObj.initNull();
346  err1:
347  contents.initNull();
348  ok = gFalse;
349 }

375 void Page::replaceXRef(XRef *xrefA) {
376  Object obj1;
377  Dict *pageDict = pageObj.getDict()->copy(xrefA);
378  xref = xrefA;
379  trans.free();
380  pageDict->lookupNF("Trans", &trans);
381  annotsObj.free();
382  pageDict->lookupNF("Annots", &annotsObj);
383  contents.free();
384  pageDict->lookupNF("Contents", &contents);
385  if (contents.isArray()) {
386  contents.free();
387  pageDict->lookupNF("Contents", &obj1)->getArray()->copy(xrefA, &contents);
388  obj1.free();
389  }
390  thumb.free();
391  pageDict->lookupNF("Thumb", &thumb);
392  actions.free();
393  pageDict->lookupNF("AA", &actions);
394  pageDict->lookup("Resources", &obj1);
395  if (obj1.isDict()) {
396  attrs->replaceResource(obj1);
397  }
398  obj1.free();
399  delete pageDict;
400 }

412 void Page::addAnnot(Annot *annot) {
413  Object obj1;
414  Object tmp;
415  Ref annotRef = annot->getRef ();
416 
417  // Make sure we have annots before adding the new one
418  // even if it's an empty list so that we can safely
419  // call annots->appendAnnot(annot)
420  pageLocker();
421  getAnnots();
422 
423  if (annotsObj.isNull()) {
424  Ref annotsRef;
425  // page doesn't have annots array,
426  // we have to create it
427 
428  obj1.initArray(xref);
429  obj1.arrayAdd(tmp.initRef (annotRef.num, annotRef.gen));
430  tmp.free();
431 
432  annotsRef = xref->addIndirectObject (&obj1);
433  annotsObj.initRef(annotsRef.num, annotsRef.gen);
434  pageObj.dictSet ("Annots", &annotsObj);
435  xref->setModifiedObject (&pageObj, pageRef);
436  } else {
437  getAnnots(&obj1);
438  if (obj1.isArray()) {
439  obj1.arrayAdd (tmp.initRef (annotRef.num, annotRef.gen));
440  if (annotsObj.isRef())
441  xref->setModifiedObject (&obj1, annotsObj.getRef());
442  else
443  xref->setModifiedObject (&pageObj, pageRef);
444  }
445  obj1.free();
446  }
447 
448  // Popup annots are already handled by markup annots,
449  // so add to the list only Popup annots without a
450  // markup annotation associated.
451  if (annot->getType() != Annot::typePopup ||
452  static_cast<AnnotPopup*>(annot)->getParentNF()->isNull()) {
453  annots->appendAnnot(annot);
454  }
455  annot->setPage(num, gTrue);
456 
457  AnnotMarkup *annotMarkup = dynamic_cast<AnnotMarkup*>(annot);
458  if (annotMarkup) {
459  AnnotPopup *annotPopup = annotMarkup->getPopup();
460  if (annotPopup)
461  addAnnot(annotPopup);
462  }
463 }

465 void Page::removeAnnot(Annot *annot) {
466  Ref annotRef = annot->getRef();
467  Object annArray;
468 
469  pageLocker();
470  getAnnots(&annArray);
471  if (annArray.isArray()) {
472  int idx = -1;
473  // Get annotation position
474  for (int i = 0; idx == -1 && i < annArray.arrayGetLength(); ++i) {
475  Object tmp;
476  if (annArray.arrayGetNF(i, &tmp)->isRef()) {
477  Ref currAnnot = tmp.getRef();
478  if (currAnnot.num == annotRef.num && currAnnot.gen == annotRef.gen) {
479  idx = i;
480  }
481  }
482  tmp.free();
483  }
484 
485  if (idx == -1) {
486  error(errInternal, -1, "Annotation doesn't belong to this page");
487  annArray.free();
488  return;
489  }
490  annots->removeAnnot(annot); // Gracefully fails on popup windows
491  annArray.arrayRemove(idx);
492  xref->removeIndirectObject(annotRef);
493 
494  if (annotsObj.isRef()) {
495  xref->setModifiedObject (&annArray, annotsObj.getRef());
496  } else {
497  xref->setModifiedObject (&pageObj, pageRef);
498  }
499  }
500  annArray.free();
501  annot->removeReferencedObjects(); // Note: Might recurse in removeAnnot again
502  annot->setPage(0, gFalse);
503 }

526 Gfx *Page::createGfx(OutputDev *out, double hDPI, double vDPI,
527  int rotate, GBool useMediaBox, GBool crop,
528  int sliceX, int sliceY, int sliceW, int sliceH,
529  GBool printing,
530  GBool (*abortCheckCbk)(void *data),
531  void *abortCheckCbkData, XRef *xrefA) {
532  PDFRectangle *mediaBox, *cropBox;
533  PDFRectangle box;
534  Gfx *gfx;
535 
536  rotate += getRotate();
537  if (rotate >= 360) {
538  rotate -= 360;
539  } else if (rotate < 0) {
540  rotate += 360;
541  }
542 
543  makeBox(hDPI, vDPI, rotate, useMediaBox, out->upsideDown(),
544  sliceX, sliceY, sliceW, sliceH, &box, &crop);
545  cropBox = getCropBox();
546  mediaBox = getMediaBox();
547 
548  if (globalParams->getPrintCommands()) {
549  printf("***** MediaBox = ll:%g,%g ur:%g,%g\n",
550  mediaBox->x1, mediaBox->y1, mediaBox->x2, mediaBox->y2);
551  printf("***** CropBox = ll:%g,%g ur:%g,%g\n",
552  cropBox->x1, cropBox->y1, cropBox->x2, cropBox->y2);
553  printf("***** Rotate = %d\n", attrs->getRotate());
554  }
555 
556  if (!crop) {
557  crop = (box == *cropBox) && out->needClipToCropBox();
558  }
559  gfx = new Gfx(doc, out, num, attrs->getResourceDict(),
560  hDPI, vDPI, &box, crop ? cropBox : (PDFRectangle *)NULL,
561  rotate, abortCheckCbk, abortCheckCbkData, xrefA);
562 
563  return gfx;
564 }

566 void Page::displaySlice(OutputDev *out, double hDPI, double vDPI,
567  int rotate, GBool useMediaBox, GBool crop,
568  int sliceX, int sliceY, int sliceW, int sliceH,
569  GBool printing,
570  GBool (*abortCheckCbk)(void *data),
571  void *abortCheckCbkData,
572  GBool (*annotDisplayDecideCbk)(Annot *annot, void *user_data),
573  void *annotDisplayDecideCbkData,
574  GBool copyXRef) {
575  Gfx *gfx;
576  Object obj;
577  Annots *annotList;
578  int i;
579 
580  if (!out->checkPageSlice(this, hDPI, vDPI, rotate, useMediaBox, crop,
581  sliceX, sliceY, sliceW, sliceH,
582  printing,
583  abortCheckCbk, abortCheckCbkData,
584  annotDisplayDecideCbk, annotDisplayDecideCbkData)) {
585  return;
586  }
587  pageLocker();
588  XRef *localXRef = (copyXRef) ? xref->copy() : xref;
589  if (copyXRef) {
590  replaceXRef(localXRef);
591  }
592 
593  gfx = createGfx(out, hDPI, vDPI, rotate, useMediaBox, crop,
594  sliceX, sliceY, sliceW, sliceH,
595  printing,
596  abortCheckCbk, abortCheckCbkData, localXRef);
597 
598  contents.fetch(localXRef, &obj);
599  if (!obj.isNull()) {
600  gfx->saveState();
601  gfx->display(&obj);
602  gfx->restoreState();
603  } else {
604  // empty pages need to call dump to do any setup required by the
605  // OutputDev
606  out->dump();
607  }
608  obj.free();
609 
610  // draw annotations
611  annotList = getAnnots();
612 
613  if (annotList->getNumAnnots() > 0) {
614  if (globalParams->getPrintCommands()) {
615  printf("***** Annotations\n");
616  }
617  for (i = 0; i < annotList->getNumAnnots(); ++i) {
618  Annot *annot = annotList->getAnnot(i);
619  if ((annotDisplayDecideCbk &&
620  (*annotDisplayDecideCbk)(annot, annotDisplayDecideCbkData)) ||
621  !annotDisplayDecideCbk) {
622  annotList->getAnnot(i)->draw(gfx, printing);
623  }
624  }
625  out->dump();
626  }
627 
628  delete gfx;
629  if (copyXRef) {
630  replaceXRef(doc->getXRef());
631  delete localXRef;
632  }
633 }

647 GBool Page::loadThumb(unsigned char **data_out,
648  int *width_out, int *height_out,
649  int *rowstride_out)
650 {
651  unsigned int pixbufdatasize;
652  int width, height, bits;
653  Object obj1, fetched_thumb;
654  Dict *dict;
655  GfxColorSpace *colorSpace;
656  GBool success = gFalse;
657  Stream *str;
658  GfxImageColorMap *colorMap;
659 
660  /* Get stream dict */
661  pageLocker();
662  thumb.fetch(xref, &fetched_thumb);
663  if (!fetched_thumb.isStream()) {
664  fetched_thumb.free();
665  return gFalse;
666  }
667 
668  dict = fetched_thumb.streamGetDict();
669  str = fetched_thumb.getStream();
670 
671  if (!dict->lookupInt("Width", "W", &width))
672  goto fail1;
673  if (!dict->lookupInt("Height", "H", &height))
674  goto fail1;
675  if (!dict->lookupInt("BitsPerComponent", "BPC", &bits))
676  goto fail1;
677 
678  /* Check for invalid dimensions and integer overflow. */
679  if (width <= 0 || height <= 0)
680  goto fail1;
681  if (width > INT_MAX / 3 / height)
682  goto fail1;
683  pixbufdatasize = width * height * 3;
684 
685  /* Get color space */
686  dict->lookup ("ColorSpace", &obj1);
687  if (obj1.isNull ()) {
688  obj1.free ();
689  dict->lookup ("CS", &obj1);
690  }
691  colorSpace = GfxColorSpace::parse(NULL, &obj1, NULL, NULL);
692  obj1.free();
693  if (!colorSpace) {
694  fprintf (stderr, "Error: Cannot parse color space\n");
695  goto fail1;
696  }
697 
698  dict->lookup("Decode", &obj1);
699  if (obj1.isNull()) {
700  obj1.free();
701  dict->lookup("D", &obj1);
702  }
