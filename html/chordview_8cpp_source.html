61 void GripItem::mouseMoveEvent(QGraphicsSceneMouseEvent* event)
62  {
63  QPointF np = mapToScene(event->pos());
64  if (_event) {
65  NoteEvent* ne = _event->event();
66  if (_gripType == 0) {
67  qreal x = ChordView::pix2pos(np.x());
68  int x2 = ne->ontime() + ne->len();
69  ne->setOntime(x);
70  ne->setLen(x2 - x);
71  _event->setPos(ChordView::pos2pix(x), _event->pos().y());
72  setPos(0.0, pos().y());
73  GripItem* rg = _view->rightGrip();
74  rg->setPos(x2-x, rg->pos().y());
75  }
76  else {
77  qreal x = np.x() - _event->pos().x();
78  setPos(x, pos().y());
79  _event->event()->setLen(x);
80  }
81  _view->setDirty(true);
82  _event->update();
83  }
84  }

177 void ChordView::drawBackground(QPainter* p, const QRectF& r)
178  {
179  if (chord == 0)
180  return;
181  QRectF r1(-1000000.0, 0.0, 1000000.0+CHORD_MAP_OFFSET, 1000000.0);
182  QRectF r2(ticks + CHORD_MAP_OFFSET, 0.0, 1000000.0, 1000000.0);
183  QRectF r3(-1000000.0, 127*keyHeight, 1000000.0+CHORD_MAP_OFFSET, keyHeight);
184  QRectF r4(ticks + CHORD_MAP_OFFSET, 127*keyHeight, 1000000.0, keyHeight);
185 
186  QColor bg(0x71, 0x8d, 0xbe);
187  QColor bg1 = bg.darker(150);
188  QColor bg2 = bg.lighter(150);
189  QColor bg3 = bg.darker(75);
190 
191  p->fillRect(r, bg);
192  p->fillRect(r.intersected(r1), bg1);
193  p->fillRect(r.intersected(r2), bg1);
194 
195  foreach (const Note* n, chord->notes()) {
196  p->fillRect(QRect(CHORD_MAP_OFFSET, (127 - n->pitch()) * keyHeight,
197  1000, keyHeight), bg2);
198  }
199 
200  p->fillRect(r.intersected(r3), bg3);
201  p->fillRect(r.intersected(r4), bg3);
202 
203  //---------------------------------------------------
204  // draw horizontal grid lines
205  //---------------------------------------------------
206 
207  qreal y1 = r.y();
208  qreal y2 = y1 + r.height();
209  qreal x1 = r.x();
210  qreal x2 = x1 + r.width();
211 
212  int key = floor(y1 / keyHeight);
213  qreal y = key * keyHeight;
214 
215  for (; key < 127; ++key, y += keyHeight) {
216  if (y < y1)
217  continue;
218  if (y > y2)
219  break;
220  p->setPen(QPen((key % 6) == 5 ? Qt::lightGray : Qt::gray));
221  p->drawLine(QLineF(x1, y, x2, y));
222  }
223 
224  //---------------------------------------------------
225  // draw raster
226  //---------------------------------------------------
227 
228  if (_evenGrid) {
229  for (int x = 0; x <= 1000; x += 50) {
230  if (x % 250)
231  p->setPen(Qt::lightGray);
232  else
233  p->setPen(Qt::black);
234  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
235  }
236  }
237  else {
238  double step1 = 1000.0 / 3;
239  double step2 = step1 / 4;
240  for (int i = 0; i < 3; ++i) {
241  p->setPen(Qt::lightGray);
242  for (int k = 1; k < 4; ++k) {
243  int x = lrint(i * step1 + k * step2);
244  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
245  }
246  p->setPen(Qt::black);
247  int x = lrint(i * step1);
248  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
249  }
250  }
251  }

257 void ChordView::setChord(Chord* c)
258  {
259  chord = c;
260  _dirty = false;
261  _pos = 0;
262  _locator = 0;
263 
264  scene()->blockSignals(true);
265  scene()->clear();
266  locatorLine = new QGraphicsLineItem(QLineF(0.0, 0.0, 0.0, keyHeight * 127.0 * 5));
267  QPen pen(Qt::red);
268  pen.setWidth(2);
269  locatorLine->setPen(pen);
270  locatorLine->setZValue(1000); // set stacking order
271  locatorLine->setFlag(QGraphicsItem::ItemIgnoresTransformations, true);
272  scene()->addItem(locatorLine);
273 
274  curEvent = 0;
275  _curNote = 0;
276  foreach(Note* note, c->notes()) {
277  if (note->selected() && _curNote == 0)
278  _curNote = note;
279  int n = note->playEvents().size();
280  for (int i = 0; i < n; ++i) {
281  NoteEvent* e = &note->playEvents()[i];
282  ChordItem* item = new ChordItem(this, note, e);
283  if (_curNote == note && curEvent == 0)
284  curEvent = item;
285  scene()->addItem(item);
286  }
287  }
288  lg = new GripItem(0, this);
289  rg = new GripItem(1, this);
290  lg->setZValue(100);
291  rg->setZValue(101);
292  lg->setVisible(false);
293  rg->setVisible(false);
294  scene()->addItem(lg);
295  scene()->addItem(rg);
296 
297  scene()->blockSignals(false);
298  setCurItem(curEvent);
299  // selectionChanged();
300 
301  scene()->setSceneRect(0.0, 0.0, double(ticks + CHORD_MAP_OFFSET * 2), keyHeight * 127);
302 
303  moveLocator();
304 
305  //
306  // move to something interesting
307  //
308  QList<QGraphicsItem*> items = scene()->selectedItems();
309  QRectF boundingRect;
310  foreach(QGraphicsItem* item, items) {
311  Note* note = static_cast<Note*>(item->data(0).value<void*>());
312  if (note)
313  boundingRect |= item->mapToScene(item->boundingRect()).boundingRect();
314  }
315  centerOn(boundingRect.center());
316  scene()->update();
317  }

338 void ChordView::wheelEvent(QWheelEvent* event)
339  {
340  int step = event->delta() / 120;
341  double xmag = transform().m11();
342  double ymag = transform().m22();
343 
344  if (event->modifiers() == Qt::ControlModifier) {
345  if (step > 0) {
346  for (int i = 0; i < step; ++i) {
347  if (xmag > 10.0)
348  break;
349  scale(1.1, 1.0);
350  xmag *= 1.1;
351  }
352  }
353  else {
354  for (int i = 0; i < -step; ++i) {
355  if (xmag < 0.001)
356  break;
357  scale(.9, 1.0);
358  xmag *= .9;
359  }
360  }
361  emit magChanged(xmag, ymag);
362 
363  int tpix = 1000 * xmag;
364  magStep = -5;
365  if (tpix <= 4000)
366  magStep = -4;
367  if (tpix <= 2000)
368  magStep = -3;
369  if (tpix <= 1000)
370  magStep = -2;
371  if (tpix <= 500)
372  magStep = -1;
373  if (tpix <= 128)
374  magStep = 0;
375  if (tpix <= 64)
376  magStep = 1;
377  if (tpix <= 32)
378  magStep = 2;
379  if (tpix <= 16)
380  magStep = 3;
381  if (tpix <= 8)
382  magStep = 4;
383  if (tpix <= 4)
384  magStep = 5;
385  if (tpix <= 2)
386  magStep = 6;
387 
388  //
389  // if xpos <= 0, then the scene is centered
390  // there is no scroll bar anymore sending
391  // change signals, so we have to do it here:
392  //
393  double xpos = -(mapFromScene(QPointF()).x());
394  if (xpos <= 0)
395  emit xposChanged(xpos);
396  }
397  else if (event->modifiers() == Qt::ShiftModifier) {
398  QWheelEvent we(event->pos(), event->delta(), event->buttons(), 0, Qt::Horizontal);
399  QGraphicsView::wheelEvent(&we);
400  }
401  else if (event->modifiers() == 0) {
402  QGraphicsView::wheelEvent(event);
403  }
404  else if (event->modifiers() == (Qt::ShiftModifier | Qt::ControlModifier)) {
405  if (step > 0) {
406  for (int i = 0; i < step; ++i) {
407  if (ymag > 3.0)
408  break;
409  scale(1.0, 1.1);
410  ymag *= 1.1;
411  }
412  }
413  else {
414  for (int i = 0; i < -step; ++i) {
415  if (ymag < 0.4)
416  break;
417  scale(1.0, .9);
418  ymag *= .9;
419  }
420  }
421  emit magChanged(xmag, ymag);
422  }
423  }

447 void ChordView::mousePressEvent(QMouseEvent* event)
448  {
449  if (event->modifiers() & Qt::ControlModifier) {
450  QPointF p(mapToScene(event->pos()));
451  int pitch = y2pitch(int(p.y()));
452  int tick = int(p.x()) - CHORD_MAP_OFFSET;
453  int ticks = 1000 - tick;
454  foreach(const NoteEvent& e, _curNote->playEvents()) {
455  if (e.pitch() != pitch)
456  continue;
457  if (tick >= e.ontime() && tick < e.offtime()) {
458  return; // cannot place an event here
459  }
460  int nticks = e.ontime() - tick;
461  if (nticks > 0)
462  ticks = qMin(ticks, nticks);
463  }
464 
465  NoteEvent ne;
466  ne.setPitch(pitch);
467  ne.setOntime(tick);
468  ne.setLen(ticks);
469  _curNote->playEvents().append(ne);
470  NoteEvent* pne = &_curNote->playEvents()[_curNote->playEvents().size()-1];
471  ChordItem* item = new ChordItem(this, _curNote, pne);
472  scene()->addItem(item);
473  setCurItem(item);
474  _dirty = true;
475  }
476  QGraphicsView::mousePressEvent(event);
477  }

598  rg->setPos(item->event()->len(), keyHeight / 4);
599  item->setCurrent(true);
600  if (item->note() != _curNote) {
601  _curNote = item->note();
