35 void splitChord(
36  std::multimap<ReducedFraction, MidiChord>::iterator &chordIt,
37  const QSet<int> &notesToMove,
38  std::multimap<ReducedFraction, MidiChord> &chords)
39  {
40  MidiChord &chord = chordIt->second;
41  auto &notes = chord.notes;
42 
43  if (notesToMove.size() == notes.size()) { // just move chord to another voice
44  chord.voice = 1;
45  }
46  else { // split chord
47  MidiChord newChord(chord);
48  newChord.notes.clear();
49  newChord.voice = 1;
50  QList<MidiNote> updatedOldNotes;
51 
52  for (int i = 0; i != notes.size(); ++i) {
53  if (notesToMove.contains(i))
54  newChord.notes.append(notes[i]);
55  else
56  updatedOldNotes.append(notes[i]);
57  }
58  notes = updatedOldNotes;
59 
60  Q_ASSERT_X(!notes.isEmpty(),
61  "MidiDrum::splitChord", "Old chord notes are empty");
62  Q_ASSERT_X(!newChord.notes.isEmpty(),
63  "MidiDrum::splitChord", "New chord notes are empty");
64 
65  chordIt = chords.insert({chordIt->first, newChord});
66  }

70  {
71  for (auto &track: tracks) {
72  MTrack &mtrack = track.second;
73  auto &chords = mtrack.chords;
74  if (chords.empty())
75  continue;
76  const Drumset* const drumset = mtrack.mtrack->drumTrack() ? smDrumset : 0;
77  if (!drumset)
78  continue;
79  // all chords of drum track should have voice == 0
80  // because allowedVoices == V_1 (see MidiImportOperations)
81  // also, all chords should have different onTime values
82  Q_ASSERT_X(MChord::areOnTimeValuesDifferent(chords),
83  "MidiDrum::splitDrumVoices",
84  "onTime values of chords are equal but should be different");
85  Q_ASSERT_X(haveNonZeroVoices(chords),
86  "MidiDrum::splitDrumVoices",
87  "All voices of drum track should be zero here");
88 
89  for (auto chordIt = chords.begin(); chordIt != chords.end(); ++chordIt) {
90  auto &notes = chordIt->second.notes;
91  // search for the drumset pitches with voice = 1
92  QSet<int> notesToMove;

94  const int pitch = notes[i].pitch;
95  if (drumset->isValid(pitch) && drumset->voice(pitch) == 1)
96  notesToMove.insert(i);

98  if (notesToMove.isEmpty())

100 
101  splitChord(chordIt, notesToMove, chords);

126  {
