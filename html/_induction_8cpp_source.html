32 AgentList Induction::beatInduction(const AgentParameters &params,
33  const EventList &events)
34  {
35  int i, j, b, bestCount;
36  bool submult;
37  int intervals = 0; // number of interval clusters
38  std::vector<int> bestn; // count of high-scoring clusters
39  bestn.resize(topN);
40 
41  double ratio, err;
42  int degree;
43  int maxClusterCount = (int) ceil((maxIOI - minIOI) / clusterWidth);
44  std::vector<double> clusterMean;
45  clusterMean.resize(maxClusterCount);
46  std::vector<int> clusterSize;
47  clusterSize.resize(maxClusterCount);
48  std::vector<int> clusterScore;
49  clusterScore.resize(maxClusterCount);
50 
51  EventList::const_iterator ptr1, ptr2;
52  Event e1, e2;
53  ptr1 = events.begin();
54 
55  while (ptr1 != events.end()) {
56  e1 = *ptr1;
57  ++ptr1;
58  ptr2 = events.begin();
59  e2 = *ptr2;
60  ++ptr2;
61  while (e2 != e1 && ptr2 != events.end()) {
62  e2 = *ptr2;
63  ++ptr2;
64  }
65  while (ptr2 != events.end()) {
66  e2 = *ptr2;
67  ++ptr2;
68  double ioi = e2.time - e1.time;
69  if (ioi < minIOI) // skip short intervals
70  continue;
71  if (ioi > maxIOI) // ioi too long
72  break;
73  for (b = 0; b < intervals; b++) // assign to nearest cluster
74  if (std::fabs(clusterMean[b] - ioi) < clusterWidth) {
75  if ((b < intervals - 1) && (std::fabs(clusterMean[b + 1] - ioi)
76  < std::fabs(clusterMean[b] - ioi))) {
77  b++; // next cluster is closer
78  }
79  clusterMean[b] = (clusterMean[b] * clusterSize[b] + ioi)
80  / (clusterSize[b] + 1);
81  clusterSize[b]++;
82  break;
83  }
84  if (b == intervals) { // no suitable cluster; create new one
85  if (intervals == maxClusterCount) {
86  // System.err.println("Warning: Too many clusters");
87  continue; // ignore this IOI
88  }
89  intervals++;
90  for ( ; (b > 0) && (clusterMean[b - 1] > ioi); b--) {
91  clusterMean[b] = clusterMean[b - 1];
92  clusterSize[b] = clusterSize[b - 1];
93  }
94  clusterMean[b] = ioi;
95  clusterSize[b] = 1;
96  }
97  }
98  }
99  for (b = 0; b < intervals; b++) // merge similar intervals
100  // TODO: they are now in order, so don't need the 2nd loop
101  // TODO: check BOTH sides before averaging or upper gps don't work
102  for (i = b + 1; i < intervals; i++)
103  if (std::fabs(clusterMean[b] - clusterMean[i]) < clusterWidth) {
104  clusterMean[b] = (clusterMean[b] * clusterSize[b] +
105  clusterMean[i] * clusterSize[i]) /
106  (clusterSize[b] + clusterSize[i]);
107  clusterSize[b] = clusterSize[b] + clusterSize[i];
108  --intervals;
109  for (j = i + 1; j <= intervals; j++) {
110  clusterMean[j - 1] = clusterMean[j];
111  clusterSize[j - 1] = clusterSize[j];
112  }
113  }
114  if (intervals == 0)
115  return AgentList();
116  for (b = 0; b < intervals; b++)
117  clusterScore[b] = 10 * clusterSize[b];
118  bestn[0] = 0;
119  bestCount = 1;
120 
121  for (b = 0; b < intervals; b++) {
122  for (i = 0; i <= bestCount; i++) {
123  if (i < topN && (i == bestCount || clusterScore[b] > clusterScore[bestn[i]])) {
124  if (bestCount < topN)
125  bestCount++;
126  for (j = bestCount - 1; j > i; j--)
127  bestn[j] = bestn[j - 1];
128  bestn[i] = b;
129  break;
130  }
131  }
132  }
133  for (b = 0; b < intervals; b++) { // score intervals
134  for (i = b + 1; i < intervals; i++) {
135  ratio = clusterMean[b] / clusterMean[i];
136  submult = ratio < 1;
137  if (submult)
138  degree = (int) nearbyint(1 / ratio);
139  else
140  degree = (int) nearbyint(ratio);
141  if ((degree >= 2) && (degree <= 8)) {
142  if (submult)
143  err = std::fabs(clusterMean[b] * degree - clusterMean[i]);
144  else
145  err = std::fabs(clusterMean[b] - clusterMean[i] * degree);
146  if (err < (submult ? clusterWidth : clusterWidth * degree)) {
147  if (degree >= 5)
148  degree = 1;
149  else
150  degree = 6 - degree;
151  clusterScore[b] += degree * clusterSize[i];
152  clusterScore[i] += degree * clusterSize[b];
153  }
154  }
155  }
156  }
157 
158  AgentList a;
159  for (int index = 0; index < bestCount; index++) {
160  b = bestn[index];
161  // Adjust it, using the size of super- and sub-intervals
162  double newSum = clusterMean[b] * clusterScore[b];
163  int newCount = clusterSize[b];
164  int newWeight = clusterScore[b];
165 
166  for (i = 0; i < intervals; i++) {
167  if (i == b)
168  continue;
169  ratio = clusterMean[b] / clusterMean[i];
170  if (ratio < 1) {
171  degree = (int) nearbyint(1 / ratio);
172  if ((degree >= 2) && (degree <= 8)) {
173  err = std::fabs(clusterMean[b] * degree - clusterMean[i]);
174  if (err < clusterWidth) {
175  newSum += clusterMean[i] / degree * clusterScore[i];
176  newCount += clusterSize[i];
177  newWeight += clusterScore[i];
178  }
179  }
180  }
181  else {
182  degree = (int) nearbyint(ratio);
183  if ((degree >= 2) && (degree <= 8)) {
184  err = std::fabs(clusterMean[b] - degree * clusterMean[i]);
185  if (err < clusterWidth * degree) {
186  newSum += clusterMean[i] * degree * clusterScore[i];
187  newCount += clusterSize[i];
188  newWeight += clusterScore[i];
189  }
190  }
191  }
192  }
193  double beat = newSum / newWeight;
194  // Scale within range ... hope the grouping isn't ternary :(
195  while (beat < minIBI) // Maximum speed
196  beat *= 2.0;
197  while (beat > maxIBI) // Minimum speed
198  beat /= 2.0;
199  if (beat >= minIBI) {
200  a.push_back(new Agent(params, beat));
201  }
202  }
203  return a;
204  }
