110 QByteArray EmbeddedFile::data()
111 {
112  if (!isValid())
113  return QByteArray();
114  Stream *stream = m_embeddedFile->embFile() ? m_embeddedFile->embFile()->stream() : NULL;
115  if (!stream)
116  return QByteArray();
117 
118  stream->reset();
119  int dataLen = 0;
120  QByteArray fileArray;
121  int i;
122  while ( (i = stream->getChar()) != EOF) {
123  fileArray[dataLen] = (char)i;
124  ++dataLen;
125  }
126  fileArray.resize(dataLen);
127  return fileArray;
128 }
