234 static void readAccidental(Accidental* a, XmlReader& e)
235  {
236  while (e.readNextStartElement()) {
237  const QStringRef& tag(e.name());
238  if (tag == "bracket") {
239  int i = e.readInt();
240  if (i == 0 || i == 1)
241  a->setBracket(AccidentalBracket(i));
242  }
243  else if (tag == "subtype") {
244  QString text(e.readElementText());
245  bool isInt;
246  int i = text.toInt(&isInt);
247  if (isInt) {
248  a->setBracket(i & 0x8000 ? AccidentalBracket::PARENTHESIS : AccidentalBracket::BRACKET);
249  i &= ~0x8000;
250  AccidentalType at;
251  switch (i) {
252  case 0:
253  at = AccidentalType::NONE;
254  break;
255  case 6:
256  a->setBracket(AccidentalBracket::PARENTHESIS);
257  case 1:
258  case 11:
259  at = AccidentalType::SHARP;
260  break;
261  case 7:
262  a->setBracket(AccidentalBracket::PARENTHESIS);
263  case 2:
264  case 12:
265  at = AccidentalType::FLAT;
266  break;
267  case 8:
268  a->setBracket(AccidentalBracket::PARENTHESIS);
269  case 3:
270  case 13:
271  at = AccidentalType::SHARP2;
272  break;
273  case 9:
274  a->setBracket(AccidentalBracket::PARENTHESIS);
275  case 4:
276  case 14:
277  at = AccidentalType::FLAT2;
278  break;
279  case 10:
280  a->setBracket(AccidentalBracket::PARENTHESIS);
281  case 5:
282  case 15:
283  at = AccidentalType::NATURAL;
284  break;
285  case 16:
286  at = AccidentalType::FLAT_SLASH;
287  break;
288  case 17:
289  at = AccidentalType::FLAT_SLASH2;
290  break;
291  case 18:
292  at = AccidentalType::MIRRORED_FLAT2;
293  break;
294  case 19:
295  at = AccidentalType::MIRRORED_FLAT;
296  break;
297  case 20:
298  at = AccidentalType::NONE;//AccidentalType::MIRRORED_FLAT_SLASH;
299  break;
300  case 21:
301  at = AccidentalType::NONE;//AccidentalType::FLAT_FLAT_SLASH;
302  break;
303  case 22:
304  at = AccidentalType::SHARP_SLASH;
305  break;
306  case 23:
307  at = AccidentalType::SHARP_SLASH2;
308  break;
309  case 24:
310  at = AccidentalType::SHARP_SLASH3;
311  break;
312  case 25:
313  at = AccidentalType::SHARP_SLASH4;
314  break;
315  case 26:
316  at = AccidentalType::SHARP_ARROW_UP;
317  break;
318  case 27:
319  at = AccidentalType::SHARP_ARROW_DOWN;
320  break;
321  case 28:
322  at = AccidentalType::NONE;//AccidentalType::SHARP_ARROW_BOTH;
323  break;
324  case 29:
325  at = AccidentalType::FLAT_ARROW_UP;
326  break;
327  case 30:
328  at = AccidentalType::FLAT_ARROW_DOWN;
329  break;
330  case 31:
331  at = AccidentalType::NONE;//AccidentalType::FLAT_ARROW_BOTH;
332  break;
333  case 32:
334  at = AccidentalType::NATURAL_ARROW_UP;
335  break;
336  case 33:
337  at = AccidentalType::NATURAL_ARROW_DOWN;
338  break;
339  case 34:
340  at = AccidentalType::NONE;//AccidentalType::NATURAL_ARROW_BOTH;
341  break;
342  default:
343  at = AccidentalType::NONE;
344  break;
345  }
346  a->setAccidentalType(at);
347  }
348  else {
349  const static std::map<QString, AccidentalType> accMap = {
350  {"none", AccidentalType::NONE}, {"sharp", AccidentalType::SHARP},
351  {"flat", AccidentalType::FLAT}, {"natural", AccidentalType::NATURAL},
352  {"double sharp", AccidentalType::SHARP2}, {"double flat", AccidentalType::FLAT2},
353  {"flat-slash", AccidentalType::FLAT_SLASH}, {"flat-slash2", AccidentalType::FLAT_SLASH2},
354  {"mirrored-flat2", AccidentalType::MIRRORED_FLAT2}, {"mirrored-flat", AccidentalType::MIRRORED_FLAT},
355  {"sharp-slash", AccidentalType::SHARP_SLASH}, {"sharp-slash2", AccidentalType::SHARP_SLASH2},
356  {"sharp-slash3", AccidentalType::SHARP_SLASH3}, {"sharp-slash4", AccidentalType::SHARP_SLASH4},
357  {"sharp arrow up", AccidentalType::SHARP_ARROW_UP}, {"sharp arrow down", AccidentalType::SHARP_ARROW_DOWN},
358  {"flat arrow up", AccidentalType::FLAT_ARROW_UP}, {"flat arrow down", AccidentalType::FLAT_ARROW_DOWN},
359  {"natural arrow up", AccidentalType::NATURAL_ARROW_UP}, {"natural arrow down", AccidentalType::NATURAL_ARROW_DOWN},
360  {"sori", AccidentalType::SORI}, {"koron", AccidentalType::KORON}
361  };
362  auto it = accMap.find(text);
363  if (it == accMap.end()) {
364  qDebug("invalid type %s", qPrintable(text));
365  a->setAccidentalType(AccidentalType::NONE);
366  }
367  else
368  a->setAccidentalType(it->second);
369  }
370  }
371  else if (tag == "role") {
372  AccidentalRole r = AccidentalRole(e.readInt());
373  if (r == AccidentalRole::AUTO || r == AccidentalRole::USER)
374  a->setRole(r);
375  }
376  else if (tag == "small")
377  a->setSmall(e.readInt());
378  else if (tag == "offset")
379  e.skipCurrentElement(); // ignore manual layout in older scores
380  else if (a->Element::readProperties(e))
381  ;
382  else
383  e.unknown();
384  }
385  }

444 static void readNote(Note* note, XmlReader& e)
445  {
446  e.hasAccidental = false; // used for userAccidental backward compatibility
447 
448  note->setTpc1(Tpc::TPC_INVALID);
449  note->setTpc2(Tpc::TPC_INVALID);
450 
451  if (e.hasAttribute("pitch")) // obsolete
452  note->setPitch(e.intAttribute("pitch"));
453  if (e.hasAttribute("tpc")) // obsolete
454  note->setTpc1(e.intAttribute("tpc"));
455 
456  while (e.readNextStartElement()) {
457  const QStringRef& tag(e.name());
458  if (tag == "Accidental") {
459  // on older scores, a note could have both a <userAccidental> tag and an <Accidental> tag
460  // if a userAccidental has some other property set (like for instance offset)
461  Accidental* a;
462  if (e.hasAccidental) // if the other tag has already been read,
463  a = note->accidental(); // re-use the accidental it constructed
464  else
465  a = new Accidental(note->score());
466  // the accidental needs to know the properties of the
467  // track it belongs to (??)
468  a->setTrack(note->track());
469  readAccidental(a, e);
470  if (!e.hasAccidental) // only the new accidental, if it has been added previously
471  note->add(a);
472  e.hasAccidental = true; // we now have an accidental
473  }
474  else if (tag == "Text") {
475  Fingering* f = new Fingering(note->score());
476  f->read(e);
477  note->add(f);
478  }
479  else if (tag == "onTimeType") {
480  if (e.readElementText() == "offset")
481  note->setOnTimeType(2);
482  else
483  note->setOnTimeType(1);
484  }
485  else if (tag == "offTimeType") {
486  if (e.readElementText() == "offset")
487  note->setOffTimeType(2);
488  else
489  note->setOffTimeType(1);
490  }
491  else if (tag == "onTimeOffset") {
492  if (note->onTimeType() == 1)
493  note->setOnTimeOffset(e.readInt() * 1000 / note->chord()->actualTicks());
494  else
495  note->setOnTimeOffset(e.readInt() * 10);
496  }
497  else if (tag == "offTimeOffset") {
498  if (note->offTimeType() == 1)
499  note->setOffTimeOffset(1000 + (e.readInt() * 1000 / note->chord()->actualTicks()));
500  else
501  note->setOffTimeOffset(1000 + (e.readInt() * 10));
502  }
503  else if (tag == "userAccidental") {
504  QString val(e.readElementText());
505  bool ok;
506  int k = val.toInt(&ok);
507  if (ok) {
508  // on older scores, a note could have both a <userAccidental> tag and an <Accidental> tag
509  // if a userAccidental has some other property set (like for instance offset)
510  // only construct a new accidental, if the other tag has not been read yet
511  // (<userAccidental> tag is only used in older scores: no need to check the score mscVersion)
512  if (!e.hasAccidental) {
513  Accidental* a = new Accidental(note->score());
514  note->add(a);
515  }
516  // TODO: for backward compatibility
517  bool bracket = k & 0x8000;
518  k &= 0xfff;
519  AccidentalType at = AccidentalType::NONE;
520  switch(k) {
521  case 0: at = AccidentalType::NONE; break;
522  case 1: at = AccidentalType::SHARP; break;
523  case 2: at = AccidentalType::FLAT; break;
524  case 3: at = AccidentalType::SHARP2; break;
525  case 4: at = AccidentalType::FLAT2; break;
526  case 5: at = AccidentalType::NATURAL; break;
527 
528  case 6: at = AccidentalType::FLAT_SLASH; break;
529  case 7: at = AccidentalType::FLAT_SLASH2; break;
530  case 8: at = AccidentalType::MIRRORED_FLAT2; break;
531  case 9: at = AccidentalType::MIRRORED_FLAT; break;
532  case 10: at = AccidentalType::NONE; break; // AccidentalType::MIRRORED_FLAT_SLASH
533  case 11: at = AccidentalType::NONE; break; // AccidentalType::FLAT_FLAT_SLASH
534 
535  case 12: at = AccidentalType::SHARP_SLASH; break;
536  case 13: at = AccidentalType::SHARP_SLASH2; break;
537  case 14: at = AccidentalType::SHARP_SLASH3; break;
538  case 15: at = AccidentalType::SHARP_SLASH4; break;
539 
540  case 16: at = AccidentalType::SHARP_ARROW_UP; break;
541  case 17: at = AccidentalType::SHARP_ARROW_DOWN; break;
542  case 18: at = AccidentalType::NONE; break; // AccidentalType::SHARP_ARROW_BOTH
543  case 19: at = AccidentalType::FLAT_ARROW_UP; break;
544  case 20: at = AccidentalType::FLAT_ARROW_DOWN; break;
545  case 21: at = AccidentalType::NONE; break; // AccidentalType::FLAT_ARROW_BOTH
546  case 22: at = AccidentalType::NATURAL_ARROW_UP; break;
547  case 23: at = AccidentalType::NATURAL_ARROW_DOWN; break;
548  case 24: at = AccidentalType::NONE; break; // AccidentalType::NATURAL_ARROW_BOTH
549  case 25: at = AccidentalType::SORI; break;
550  case 26: at = AccidentalType::KORON; break;
551  }
552  note->accidental()->setAccidentalType(at);
553 
554  note->accidental()->setBracket(AccidentalBracket(bracket));
555 
556  note->accidental()->setRole(AccidentalRole::USER);
557  e.hasAccidental = true; // we now have an accidental
558  }
559  }
560  else if (tag == "onTimeType")
561  e.skipCurrentElement(); // _onTimeType = readValueType(e);
562  else if (tag == "offTimeType")
563  e.skipCurrentElement(); // _offTimeType = readValueType(e);
564  else if (tag == "offset")
565  e.skipCurrentElement(); // ignore manual layout in older scores
566  else if (tag == "move")
567  note->chord()->setStaffMove(e.readInt());
568  else if (tag == "head") {
569  int i = e.readInt();
570  NoteHead::Group val = convertHeadGroup(i);
571  note->setHeadGroup(val);
572  }
573  else if (tag == "headType") {
574  int i = e.readInt();
575  NoteHead::Type val;
576  switch (i) {
577  case 1:
578  val = NoteHead::Type::HEAD_WHOLE;
579  break;
580  case 2:
581  val = NoteHead::Type::HEAD_HALF;
582  break;
583  case 3:
584  val = NoteHead::Type::HEAD_QUARTER;
585  break;
586  case 4:
587  val = NoteHead::Type::HEAD_BREVIS;
588  break;
589  default:
590  val = NoteHead::Type::HEAD_AUTO;
591  }
592  note->setHeadType(val);
593  }
594  else if (note->readProperties(e))
595  ;
596  else
597  e.unknown();
598  }
599  // ensure sane values:
600  note->setPitch(limit(note->pitch(), 0, 127));
601 
602  if (note->concertPitch())
603  note->setTpc2(Tpc::TPC_INVALID);
604  else {
605  note->setPitch(note->pitch() + note->transposition());
606  note->setTpc2(note->tpc1());
607  note->setTpc1(Tpc::TPC_INVALID);
608  }
609  if (!tpcIsValid(note->tpc1()) && !tpcIsValid(note->tpc2())) {
610  Key key = (note->staff() && note->chord()) ? note->staff()->key(note->chord()->tick()) : Key::C;
611  int tpc = pitch2tpc(note->pitch(), key, Prefer::NEAREST);
612  if (note->concertPitch())
613  note->setTpc1(tpc);
614  else
615  note->setTpc2(tpc);
616  }
617  if (!(tpcIsValid(note->tpc1()) && tpcIsValid(note->tpc2()))) {
618  int tick = note->chord() ? note->chord()->tick() : -1;
619  Interval v = note->staff() ? note->part()->instrument(tick)->transpose() : Interval();
620  if (tpcIsValid(note->tpc1())) {
621  v.flip();
622  if (v.isZero())
623  note->setTpc2(note->tpc1());
624  else
625  note->setTpc2(Ms::transposeTpc(note->tpc1(), v, true));
626  }
627  else {
628  if (v.isZero())
629  note->setTpc1(note->tpc2());
630  else
631  note->setTpc1(Ms::transposeTpc(note->tpc2(), v, true));
632  }
633  }
634 
635  // check consistency of pitch, tpc1, tpc2, and transposition
636  // see note in InstrumentChange::read() about a known case of tpc corruption produced in 2.0.x
637  // but since there are other causes of tpc corruption (eg, https://musescore.org/en/node/74746)
638  // including perhaps some we don't know about yet,
639  // we will attempt to fix some problems here regardless of version
640 
641  if (!e.pasteMode() && !MScore::testMode) {
642  int tpc1Pitch = (tpc2pitch(note->tpc1()) + 12) % 12;
643  int tpc2Pitch = (tpc2pitch(note->tpc2()) + 12) % 12;
644  int concertPitch = note->pitch() % 12;
645  if (tpc1Pitch != concertPitch) {
646  qDebug("bad tpc1 - concertPitch = %d, tpc1 = %d", concertPitch, tpc1Pitch);
647  note->setPitch(note->pitch() + tpc1Pitch - concertPitch);
648  }
649  Interval v = note->staff()->part()->instrument(e.tick())->transpose();
650  int transposedPitch = (note->pitch() - v.chromatic) % 12;
651  if (tpc2Pitch != transposedPitch) {
652  qDebug("bad tpc2 - transposedPitch = %d, tpc2 = %d", transposedPitch, tpc2Pitch);
653  // just in case the staff transposition info is not reliable here,
654  v.flip();
655  note->setTpc2(Ms::transposeTpc(note->tpc1(), v, true));
656  }
657  }
658  }

806 //---------------------------------------------------------
807 // readMeasure
808 //---------------------------------------------------------
809 
810 static void readMeasure(Measure* m, int staffIdx, XmlReader& e)
811  {
812  Segment* segment = 0;
813  qreal _spatium = m->spatium();
814 
815  QList<Chord*> graceNotes;
816 
817  //sort tuplet elements. needed for nested tuplets #22537
818  for (Tuplet* t : e.tuplets())
819  t->sortElements();
820  e.tuplets().clear();
821  e.setTrack(staffIdx * VOICES);
822 
823  m->createStaves(staffIdx);
824 
825  // tick is obsolete
826  if (e.hasAttribute("tick"))
827  e.initTick(m->score()->fileDivision(e.intAttribute("tick")));
828 
829  if (e.hasAttribute("len")) {
830  QStringList sl = e.attribute("len").split('/');
831  if (sl.size() == 2)
832  m->setLen(Fraction(sl[0].toInt(), sl[1].toInt()));
833  else
834  qDebug("illegal measure size <%s>", qPrintable(e.attribute("len")));
835  m->score()->sigmap()->add(m->tick(), SigEvent(m->len(), m->timesig()));
836  m->score()->sigmap()->add(m->endTick(), SigEvent(m->timesig()));
837  }
838 
839  Staff* staff = m->score()->staff(staffIdx);
840  Fraction timeStretch(staff->timeStretch(m->tick()));
841 
842  // keep track of tick of previous element
843  // this allows markings that need to apply to previous element to do so
844  // even though we may have already advanced to next tick position
845  int lastTick = e.tick();
846 
847  while (e.readNextStartElement()) {
848  const QStringRef& tag(e.name());
849 
850  if (tag == "move")
851  e.initTick(e.readFraction().ticks() + m->tick());
852  else if (tag == "tick") {
853  e.initTick(m->score()->fileDivision(e.readInt()));
854  lastTick = e.tick();
855  }
856  else if (tag == "BarLine") {
857  BarLine* barLine = new BarLine(m->score());
858  barLine->setTrack(e.track());
859  barLine->resetProperty(P_ID::BARLINE_SPAN);
860  barLine->resetProperty(P_ID::BARLINE_SPAN_FROM);
861  barLine->resetProperty(P_ID::BARLINE_SPAN_TO);
862 
863  while (e.readNextStartElement()) {
864  const QStringRef& tag(e.name());
865  if (tag == "subtype") {
866  BarLineType t;
867  switch (e.readInt()) {
868  default:
869  case 0:
870  t = BarLineType::NORMAL;
871  break;
872  case 1:
873  t = BarLineType::DOUBLE;
874  break;
875  case 2:
876  t = BarLineType::START_REPEAT;
877  break;
878  case 3:
879  t = BarLineType::END_REPEAT;
880  break;
881  case 4:
882  t = BarLineType::BROKEN;
883  break;
884  case 5:
885  t = BarLineType::END;
886  break;
887  case 6:
888  // TODO t = BarLineType::END_START_REPEAT;
889  break;
890  }
891  barLine->setBarLineType(t);
892  }
893  else if (!barLine->Element::readProperties(e))
894  e.unknown();
895  }
896 
897  //
898  // StartRepeatBarLine: always at the beginning tick of a measure, always BarLineType::START_REPEAT
899  // BarLine: in the middle of a measure, has no semantic
900  // EndBarLine: at the end tick of a measure
901  // BeginBarLine: first segment of a measure
902 
903  SegmentType st;
904  if ((e.tick() != m->tick()) && (e.tick() != m->endTick()))
905  st = SegmentType::BarLine;
906  else if (barLine->barLineType() == BarLineType::START_REPEAT && e.tick() == m->tick())
907  st = SegmentType::StartRepeatBarLine;
908  else if (e.tick() == m->tick() && segment == 0)
909  st = SegmentType::BeginBarLine;
910  else
911  st = SegmentType::EndBarLine;
912  segment = m->getSegment(st, e.tick());
913  segment->add(barLine);
914  }
915  else if (tag == "Chord") {
916  Chord* chord = new Chord(m->score());
917  chord->setTrack(e.track());
918  readChord(chord, e);
919  segment = m->getSegment(SegmentType::ChordRest, e.tick());
920  if (chord->noteType() != NoteType::NORMAL) {
921  graceNotes.push_back(chord);
922  if (chord->tremolo() && chord->tremolo()->tremoloType() < TremoloType::R8) {
923  // old style tremolo found
924  Tremolo* tremolo = chord->tremolo();
925  TremoloType st;
926  switch (tremolo->tremoloType()) {
927  default:
928  case TremoloType::OLD_R8: st = TremoloType::R8; break;
929  case TremoloType::OLD_R16: st = TremoloType::R16; break;
930  case TremoloType::OLD_R32: st = TremoloType::R32; break;
931  case TremoloType::OLD_C8: st = TremoloType::C8; break;
932  case TremoloType::OLD_C16: st = TremoloType::C16; break;
933  case TremoloType::OLD_C32: st = TremoloType::C32; break;
934  }
935  tremolo->setTremoloType(st);
936  }
937  }
938  else {
939  segment->add(chord);
940  Q_ASSERT(segment->segmentType() == SegmentType::ChordRest);
941 
942  for (int i = 0; i < graceNotes.size(); ++i) {
943  Chord* gc = graceNotes[i];
944  gc->setGraceIndex(i);
945  chord->add(gc);
946  }
947  graceNotes.clear();
948  int crticks = chord->actualTicks();
949 
950  if (chord->tremolo() && chord->tremolo()->tremoloType() < TremoloType::R8) {
951  // old style tremolo found
952 
953  Tremolo* tremolo = chord->tremolo();
954  TremoloType st;
955  switch (tremolo->tremoloType()) {
956  default:
957  case TremoloType::OLD_R8: st = TremoloType::R8; break;
958  case TremoloType::OLD_R16: st = TremoloType::R16; break;
959  case TremoloType::OLD_R32: st = TremoloType::R32; break;
960  case TremoloType::OLD_C8: st = TremoloType::C8; break;
961  case TremoloType::OLD_C16: st = TremoloType::C16; break;
962  case TremoloType::OLD_C32: st = TremoloType::C32; break;
963  }
964  tremolo->setTremoloType(st);
965  if (tremolo->twoNotes()) {
966  int track = chord->track();
967  Segment* ss = 0;
968  for (Segment* ps = m->first(SegmentType::ChordRest); ps; ps = ps->next(SegmentType::ChordRest)) {
969  if (ps->tick() >= e.tick())
970  break;
971  if (ps->element(track))
972  ss = ps;
973  }
974  Chord* pch = 0; // previous chord
975  if (ss) {
976  ChordRest* cr = static_cast<ChordRest*>(ss->element(track));
977  if (cr && cr->type() == ElementType::CHORD)
978  pch = static_cast<Chord*>(cr);
979  }
980  if (pch) {
981  tremolo->setParent(pch);
982  pch->setTremolo(tremolo);
983  chord->setTremolo(0);
984  // force duration to half
985  Fraction pts(timeStretch * pch->globalDuration());
986  int pcrticks = pts.ticks();
987  pch->setDuration(Fraction::fromTicks(pcrticks / 2));
988  chord->setDuration(Fraction::fromTicks(crticks / 2));
989  }
990  else {
991  qDebug("tremolo: first note not found");
992  }
993  crticks /= 2;
994  }
995  else {
996  tremolo->setParent(chord);
997  }
998  }
999  lastTick = e.tick();
1000  e.incTick(crticks);
1001  }
1002  }
1003  else if (tag == "Rest") {
1004  Rest* rest = new Rest(m->score());
1005  rest->setDurationType(TDuration::DurationType::V_MEASURE);
1006  rest->setDuration(m->timesig()/timeStretch);
1007  rest->setTrack(e.track());
1008  readRest(rest, e);
1009  segment = m->getSegment(SegmentType::ChordRest, e.tick());
1010  segment->add(rest);
1011 
1012  if (!rest->duration().isValid()) // hack
1013  rest->setDuration(m->timesig()/timeStretch);
1014 
1015  lastTick = e.tick();
1016  e.incTick(rest->actualTicks());
1017  }
1018  else if (tag == "Breath") {
1019  Breath* breath = new Breath(m->score());
1020  breath->setTrack(e.track());
1021  int tick = e.tick();
1022  breath->read(e);
1023  // older scores placed the breath segment right after the chord to which it applies
1024  // rather than before the next chordrest segment with an element for the staff
1025  // result would be layout too far left if there are other segments due to notes in other staves
1026  // we need to find tick of chord to which this applies, and add its duration
1027  int prevTick;
1028  if (e.tick() < tick)
1029  prevTick = e.tick(); // use our own tick if we explicitly reset to earlier position
1030  else
1031  prevTick = lastTick; // otherwise use tick of previous tick/chord/rest tag
1032  // find segment
1033  Segment* prev = m->findSegment(SegmentType::ChordRest, prevTick);
1034  if (prev) {
1035  // find chordrest
1036  ChordRest* lastCR = static_cast<ChordRest*>(prev->element(e.track()));
1037  if (lastCR)
1038  tick = prevTick + lastCR->actualTicks();
1039  }
1040  segment = m->getSegment(SegmentType::Breath, tick);
1041  segment->add(breath);
1042  }
1043  else if (tag == "endSpanner") {
1044  int id = e.attribute("id").toInt();
1045  Spanner* spanner = e.findSpanner(id);
1046  if (spanner) {
1047  spanner->setTicks(e.tick() - spanner->tick());
1048  // if (spanner->track2() == -1)
1049  // the absence of a track tag [?] means the
1050  // track is the same as the beginning of the slur
1051  if (spanner->track2() == -1)
1052  spanner->setTrack2(spanner->track() ? spanner->track() : e.track());
1053  }
1054  else {
1055  // remember "endSpanner" values
1056  SpannerValues sv;
1057  sv.spannerId = id;
1058  sv.track2 = e.track();
1059  sv.tick2 = e.tick();
1060  e.addSpannerValues(sv);
1061  }
1062  e.readNext();
1063  }
1064  else if (tag == "Slur") {
1065  Slur *sl = new Slur(m->score());
1066  sl->setTick(e.tick());
1067  sl->read(e);
1068  //
1069  // check if we already saw "endSpanner"
1070  //
1071  int id = e.spannerId(sl);
1072  const SpannerValues* sv = e.spannerValues(id);
1073  if (sv) {
1074  sl->setTick2(sv->tick2);
1075  sl->setTrack2(sv->track2);
1076  }
1077  m->score()->addSpanner(sl);
1078  }
1079  else if (tag == "HairPin"
1080  || tag == "Pedal"
1081  || tag == "Ottava"
1082  || tag == "Trill"
1083  || tag == "TextLine"
1084  || tag == "Volta") {
1085  Spanner* sp = static_cast<Spanner*>(Element::name2Element(tag, m->score()));
1086  sp->setTrack(e.track());
1087  sp->setTick(e.tick());
1088  // ?? sp->setAnchor(Spanner::Anchor::SEGMENT);
1089  sp->read(e);
1090  m->score()->addSpanner(sp);
1091  //
1092  // check if we already saw "endSpanner"
1093  //
1094  int id = e.spannerId(sp);
1095  const SpannerValues* sv = e.spannerValues(id);
1096  if (sv) {
1097  sp->setTicks(sv->tick2 - sp->tick());
1098  sp->setTrack2(sv->track2);
1099  }
1100  }
1101  else if (tag == "RepeatMeasure") {
1102  RepeatMeasure* rm = new RepeatMeasure(m->score());
1103  rm->setTrack(e.track());
1104  rm->read(e);
1105  segment = m->getSegment(SegmentType::ChordRest, e.tick());
1106  segment->add(rm);
1107  if (rm->actualDuration().isZero()) { // might happen with 1.3 scores
1108  rm->setDuration(m->len());
1109  }
1110  lastTick = e.tick();
1111  e.incTick(m->ticks());
1112  }
1113  else if (tag == "Clef") {
1114  Clef* clef = new Clef(m->score());
1115  clef->setTrack(e.track());
1116  readClef(clef, e);
1117  if (m->score()->mscVersion() < 113)
1118  clef->setUserOff(QPointF());
1119  clef->setGenerated(false);
1120  // MS3 doesn't support wrong clef for staff type: Default to G
1121  bool isDrumStaff = staff->isDrumStaff(e.tick());
1122  if (clef->clefType() == ClefType::TAB
1123  || (clef->clefType() == ClefType::PERC && !isDrumStaff)
1124  || (clef->clefType() != ClefType::PERC && isDrumStaff)) {
1125  clef->setClefType(ClefType::G);
1126  staff->clefList().erase(e.tick());
1127  staff->clefList().insert(std::pair<int,ClefType>(e.tick(), ClefType::G));
1128  }
1129 
1130  // there may be more than one clef segment for same tick position
1131  // the first clef may be missing and is added later in layout
1132 
1133  bool header;
1134  if (e.tick() != m->tick())
1135  header = false;
1136  else if (!segment)
1137  header = true;
1138  else {
1139  header = true;
1140  for (Segment* s = m->segments().first(); s && !s->rtick(); s = s->next()) {
1141  if (s->isKeySigType() || s->isTimeSigType()) {
1142  // hack: there may be other segment types which should
1143  // generate a clef at current position
1144  header = false;
1145  break;
1146  }
1147  }
1148  }
1149  segment = m->getSegment(header ? SegmentType::HeaderClef : SegmentType::Clef, e.tick());
1150  segment->add(clef);
1151  }
1152  else if (tag == "TimeSig") {
1153  TimeSig* ts = new TimeSig(m->score());
1154  ts->setTrack(e.track());
1155  ts->read(e);
1156  // if time sig not at begining of measure => courtesy time sig
1157  int currTick = e.tick();
1158  bool courtesySig = (currTick > m->tick());
1159  if (courtesySig) {
1160  // if courtesy sig., just add it without map processing
1161  segment = m->getSegment(SegmentType::TimeSigAnnounce, currTick);
1162  segment->add(ts);
1163  }
1164  else {
1165  // if 'real' time sig., do full process
1166  segment = m->getSegment(SegmentType::TimeSig, currTick);
1167  segment->add(ts);
1168 
1169  timeStretch = ts->stretch().reduced();
1170  m->setTimesig(ts->sig() / timeStretch);
1171  }
1172  }
1173  else if (tag == "KeySig") {
1174  KeySig* ks = new KeySig(m->score());
1175  ks->setTrack(e.track());
1176  ks->read(e);
1177  int curTick = e.tick();
1178  if (!ks->isCustom() && !ks->isAtonal() && ks->key() == Key::C && curTick == 0) {
1179  // ignore empty key signature
1180  qDebug("remove keysig c at tick 0");
1181  if (ks->links()) {
1182  if (ks->links()->size() == 1)
1183  e.linkIds().remove(ks->links()->lid());
