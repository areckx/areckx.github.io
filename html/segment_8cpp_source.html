336 
337 Segment* Segment::nextCR(int track, bool sameStaff) const
338  {
339  int strack = track;
340  int etrack;
341  if (sameStaff) {
342  strack &= ~(VOICES-1);
343  etrack = strack + VOICES;
344  }
345  else {
346  etrack = strack + 1;
347  }
348  for (Segment* seg = next1(); seg; seg = seg->next1()) {
349  if (seg->isChordRestType()) {
350  if (track == -1)
351  return seg;
352  for (int t = strack; t < etrack; ++t) {
353  if (seg->element(t))
354  return seg;
355  }
356  }
357  }
358  return 0;
359  }

380 void Segment::insertStaff(int staff)
381  {
382  int track = staff * VOICES;
383  for (int voice = 0; voice < VOICES; ++voice)
384  _elist.insert(_elist.begin() + track, 0);
385  _dotPosX.insert(_dotPosX.begin()+staff, 0.0);
386  _shapes.insert(_shapes.begin()+staff, Shape());
387 
388  for (Element* e : _annotations) {
389  int staffIdx = e->staffIdx();

391  e->setTrack(e->track() + VOICES);
392  }

394  }

430 //---------------------------------------------------------
431 // add
432 //---------------------------------------------------------
433 
434 void Segment::add(Element* el)
435  {
436 // qDebug("%p segment %s add(%d, %d, %s)", this, subTypeName(), tick(), el->track(), el->name());
437 
438  el->setParent(this);
439 
440  int track = el->track();
441  Q_ASSERT(track != -1);
442  Q_ASSERT(el->score() == score());
443  Q_ASSERT(score()->nstaves() * VOICES == int(_elist.size()));
444 
445  switch (el->type()) {
446  case ElementType::REPEAT_MEASURE:
447  _elist[track] = el;
448  setEmpty(false);
449  break;
450 
451  case ElementType::DYNAMIC:
452  case ElementType::HARMONY:
453  case ElementType::SYMBOL:
454  case ElementType::FRET_DIAGRAM:
455  case ElementType::TEMPO_TEXT:
456  case ElementType::STAFF_TEXT:
457  case ElementType::SYSTEM_TEXT:
458  case ElementType::REHEARSAL_MARK:
459  case ElementType::MARKER:
460  case ElementType::IMAGE:
461  case ElementType::TEXT:
462  case ElementType::TREMOLOBAR:
463  case ElementType::TAB_DURATION_SYMBOL:
464  case ElementType::FIGURED_BASS:
465  _annotations.push_back(el);
466  break;
467 
468  case ElementType::STAFF_STATE:
469  if (toStaffState(el)->staffStateType() == StaffStateType::INSTRUMENT) {
470  StaffState* ss = toStaffState(el);
471  Part* part = el->part();
472  part->setInstrument(ss->instrument(), tick());
473  }
474  _annotations.push_back(el);
475  break;
476 
477  case ElementType::INSTRUMENT_CHANGE: {
478  InstrumentChange* is = toInstrumentChange(el);
479  Part* part = is->part();
480  part->setInstrument(is->instrument(), tick());
481  _annotations.push_back(el);
482  break;
483  }
484 
485  case ElementType::CLEF:
486  Q_ASSERT(_segmentType == SegmentType::Clef || _segmentType == SegmentType::HeaderClef);
487  checkElement(el, track);
488  _elist[track] = el;
489  if (!el->generated()) {
490  el->staff()->setClef(toClef(el));
491  updateNoteLines(this, el->track());
492  }
493  setEmpty(false);
494  break;
495 
496  case ElementType::TIMESIG:
497  Q_ASSERT(segmentType() == SegmentType::TimeSig || segmentType() == SegmentType::TimeSigAnnounce);
498  checkElement(el, track);
499  _elist[track] = el;
500  el->staff()->addTimeSig(toTimeSig(el));
501  setEmpty(false);
502  break;
503 
504  case ElementType::KEYSIG:
505  Q_ASSERT(_segmentType == SegmentType::KeySig || _segmentType == SegmentType::KeySigAnnounce);
506  checkElement(el, track);
507  _elist[track] = el;
508  if (!el->generated())
509  el->staff()->setKey(tick(), toKeySig(el)->keySigEvent());
510  setEmpty(false);
511  break;
512 
513  case ElementType::CHORD:
514  case ElementType::REST:
515  Q_ASSERT(_segmentType == SegmentType::ChordRest);
516  {
517  if (track % VOICES) {
518  bool v;
519  if (el->isChord()) {
520  v = false;
521  // consider chord visible if any note is visible
522  Chord* c = toChord(el);
523  for (Note* n : c->notes()) {
524  if (n->visible()) {
525  v = true;
526  break;
527  }
528  }
529  }
530  else
531  v = el->visible();
532 
533  if (v && measure()->score()->ntracks() > track)
534  measure()->setHasVoices(track / VOICES, true);
535  }
536  // the tick position of a tuplet is the tick position of its
537  // first element:
538  ChordRest* cr = toChordRest(el);
539  if (cr->tuplet() && !cr->tuplet()->elements().empty() && cr->tuplet()->elements().front() == cr && cr->tuplet()->tick() < 0)
540  cr->tuplet()->setTick(cr->tick());
541  }
542  // fall through
543 
544  case ElementType::BAR_LINE:
545  case ElementType::BREATH:
546  if (track < score()->nstaves() * VOICES) {
547  checkElement(el, track);
548  _elist[track] = el;
549  }
550  setEmpty(false);
551  break;
552 
553  case ElementType::AMBITUS:
554  Q_ASSERT(_segmentType == SegmentType::Ambitus);
555  checkElement(el, track);
556  _elist[track] = el;
557  setEmpty(false);
558  break;
559 
560  default:
561  qFatal("Segment::add() unknown %s", el->name());
562  }
563  }

565 //---------------------------------------------------------
566 // remove
567 //---------------------------------------------------------
568 
569 void Segment::remove(Element* el)
570  {
571 // qDebug("%p Segment::remove %s %p", this, el->name(), el);
572 
573  int track = el->track();
574 
575  switch(el->type()) {
576  case ElementType::CHORD:
577  case ElementType::REST:
578  {
579  _elist[track] = 0;
580  int staffIdx = el->staffIdx();
581  measure()->checkMultiVoices(staffIdx);
582  // spanners with this cr as start or end element will need relayout
583  SpannerMap& smap = score()->spannerMap();
584  auto spanners = smap.findOverlapping(tick(), tick());
585  for (auto interval : spanners) {
586  Spanner* s = interval.value;
587  if (s->startElement() == el)
588  s->setStartElement(nullptr);
589  if (s->endElement() == el)
590  s->setEndElement(nullptr);
591  }
592  }
593  break;
594 
595  case ElementType::REPEAT_MEASURE:
596  _elist[track] = 0;
597  break;
598 
599  case ElementType::DYNAMIC:
600  case ElementType::FIGURED_BASS:
601  case ElementType::FRET_DIAGRAM:
602  case ElementType::HARMONY:
603  case ElementType::IMAGE:
604  case ElementType::MARKER:
605  case ElementType::REHEARSAL_MARK:
606  case ElementType::STAFF_TEXT:
607  case ElementType::SYSTEM_TEXT:
608  case ElementType::SYMBOL:
609  case ElementType::TAB_DURATION_SYMBOL:
610  case ElementType::TEMPO_TEXT:
611  case ElementType::TEXT:
612  case ElementType::TREMOLOBAR:
613  removeAnnotation(el);
614  break;
615 
616  case ElementType::STAFF_STATE:
617  if (toStaffState(el)->staffStateType() == StaffStateType::INSTRUMENT) {
618  Part* part = el->part();
619  part->removeInstrument(tick());
620  }
621  removeAnnotation(el);
622  break;
623 
624  case ElementType::INSTRUMENT_CHANGE:
625  {
626  InstrumentChange* is = toInstrumentChange(el);
627  Part* part = is->part();
628  part->removeInstrument(tick());
629  }
630  removeAnnotation(el);
631  break;
632 
633  case ElementType::TIMESIG:
634  _elist[track] = 0;
635  el->staff()->removeTimeSig(toTimeSig(el));
636  break;
637 
638  case ElementType::KEYSIG:
639  Q_ASSERT(_elist[track] == el);
640 
641  _elist[track] = 0;
642  if (!el->generated())
643  el->staff()->removeKey(tick());
644  break;
645 
646  case ElementType::CLEF:
647  el->staff()->removeClef(toClef(el));
648  // updateNoteLines(this, el->track());
649  // fall through
650 
651  case ElementType::BAR_LINE:
652  case ElementType::AMBITUS:
653  _elist[track] = 0;
654  break;
655 
656  case ElementType::BREATH:
657  _elist[track] = 0;
658  score()->setPause(tick(), 0);
659  break;
660 
661  default:
662  qFatal("Segment::remove() unknown %s", el->name());
663 
664  }
665  score()->setLayout(tick());
666  checkEmpty();
667  }

702 
703 void Segment::sortStaves(QList<int>& dst)
704  {
705  std::vector<Element*> dl;
706  dl.reserve(dst.size());
707 
708  for (int i = 0; i < dst.size(); ++i) {
709  int startTrack = dst[i] * VOICES;
710  int endTrack = startTrack + VOICES;
711  for (int k = startTrack; k < endTrack; ++k)
712  dl.push_back(_elist[k]);
713  }
714  std::swap(_elist, dl);
715  QMap<int, int> map;
716  for (int k = 0; k < dst.size(); ++k) {
717  map.insert(dst[k], k);
718  }
719  for (Element* e : _annotations) {
720  if (!e->systemFlag())
721  e->setTrack(map[e->staffIdx()] * VOICES + e->voice());
722  }
723  fixStaffIdx();
724  }

803 void Segment::read(XmlReader& e)
804  {
805  while (e.readNextStartElement()) {
806  const QStringRef& tag(e.name());
807 
808  if (tag == "subtype")
809  e.skipCurrentElement();
810  else if (tag == "leadingSpace")
811  _extraLeadingSpace = Spatium(e.readDouble());
812  else if (tag == "trailingSpace") // obsolete
813  e.readDouble();
814  else
815  e.unknown();
816  }
