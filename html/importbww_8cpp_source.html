79 static void xmlSetPitch(Ms::Note* n, char step, int alter, int octave)
80  {
81  int istep = step - 'A';
82  // a b c d e f g
83  static int table[7] = { 9, 11, 0, 2, 4, 5, 7 };
84  if (istep < 0 || istep > 6) {
85  qDebug("xmlSetPitch: illegal pitch %d, <%c>", istep, step);
86  return;
87  }
88  int pitch = table[istep] + alter + (octave+1) * 12;
89 
90  if (pitch < 0)
91  pitch = 0;
92  if (pitch > 127)
93  pitch = 127;
94 
95  n->setPitch(pitch);
96 
97  // a b c d e f g
98  static int table1[7] = { 5, 6, 0, 1, 2, 3, 4 };
99  int tpc = step2tpc(table1[istep], Ms::AccidentalVal(alter));
100  n->setTpc(tpc);
101  }

109 static void setTempo(Ms::Score* score, int tempo)
110  {
111  Ms::TempoText* tt = new Ms::TempoText(score);
112  tt->setTempo(double(tempo)/60.0);
113  tt->setTrack(0);
114  QString tempoText = Ms::TempoText::duration2tempoTextString(Ms::TDuration::DurationType::V_QUARTER);
115  tempoText += QString(" = %1").arg(tempo);
116  tt->setPlainText(tempoText);
117  Ms::Measure* measure = score->firstMeasure();
118  Ms::Segment* segment = measure->getSegment(Ms::SegmentType::ChordRest, 0);
119  segment->add(tt);
120  }

209 void MsScWriter::beginMeasure(const Bww::MeasureBeginFlags mbf)
210  {
211  qDebug() << "MsScWriter::beginMeasure()";
212  ++measureNumber;
213 
214  // create a new measure
215  currentMeasure = new Ms::Measure(score);
216  currentMeasure->setTick(tick);
217  currentMeasure->setTimesig(Ms::Fraction(beats, beat));
218  currentMeasure->setNo(measureNumber);
219  score->measures()->add(currentMeasure);
220 
221  if (mbf.repeatBegin)
222  currentMeasure->setRepeatStart(true);
223 
224  if (mbf.irregular)
225  currentMeasure->setIrregular(true);
226 
227  if (mbf.endingFirst || mbf.endingSecond) {
228  Ms::Volta* volta = new Ms::Volta(score);
229  volta->setTrack(0);
230  volta->endings().clear();
231  if (mbf.endingFirst) {
232  volta->setText("1");
233  volta->endings().append(1);
234  ending = 1;
235  }
236  else {
237  volta->setText("2");
238  volta->endings().append(2);
239  ending = 2;
240  }
241  volta->setTick(currentMeasure->tick());
242  score->addElement(volta);
243  lastVolta = volta;
244  }
245 
246  // set clef, key and time signature in the first measure
247  if (measureNumber == 1) {
248  // clef
249  Ms::Clef* clef = new Ms::Clef(score);
250  clef->setClefType(Ms::ClefType::G);
251  clef->setTrack(0);
252  Ms::Segment* s = currentMeasure->getSegment(Ms::SegmentType::Clef, tick);
253  s->add(clef);
254  // keysig
255  Ms::KeySigEvent key;
256  key.setKey(Ms::Key::D);
257  Ms::KeySig* keysig = new Ms::KeySig(score);
258  keysig->setKeySigEvent(key);
259  keysig->setTrack(0);
260  s = currentMeasure->getSegment(Ms::SegmentType::KeySig, tick);
261  s->add(keysig);
262  // timesig
263  Ms::TimeSig* timesig = new Ms::TimeSig(score);
264  timesig->setSig(Ms::Fraction(beats, beat));
265  timesig->setTrack(0);
266  s = currentMeasure->getSegment(Ms::SegmentType::TimeSig, tick);
267  s->add(timesig);
268  qDebug("tempo %d", tempo);
269  }
270  }

321 void MsScWriter::note(const QString pitch, const QVector<Bww::BeamType> beamList,
322  const QString type, const int dots,
323  bool tieStart, bool /*TODO tieStop */,
324  StartStop triplet,
325  bool grace)
326  {
327  qDebug() << "MsScWriter::note()"
328  << "type:" << type
329  << "dots:" << dots
330  << "grace" << grace
331  ;
332 
333  if (!stepAlterOctMap.contains(pitch)
334  || !typeMap.contains(type)) {
335  // TODO: error message
336  return;
337  }
338  StepAlterOct sao = stepAlterOctMap.value(pitch);
339 
340  int ticks = 4 * Ms::MScore::division / type.toInt();
341  if (dots) ticks = 3 * ticks / 2;
342  qDebug() << "ticks:" << ticks;
343  Ms::TDuration durationType(Ms::TDuration::DurationType::V_INVALID);
344  durationType.setVal(ticks);
345  qDebug() << "duration:" << durationType.name();
346  if (triplet != ST_NONE) ticks = 2 * ticks / 3;
347 
348  Ms::Beam::Mode bm = (beamList.at(0) == Bww::BM_BEGIN) ? Ms::Beam::Mode::BEGIN : Ms::Beam::Mode::AUTO;
349  Ms::Direction sd = Ms::Direction::AUTO;
350 
351  // create chord
352  Ms::Chord* cr = new Ms::Chord(score);
353  //ws cr->setTick(tick);
354  cr->setBeamMode(bm);
355  cr->setTrack(0);
356  if (grace) {
357  cr->setNoteType(Ms::NoteType::GRACE32);
358  cr->setDurationType(Ms::TDuration::DurationType::V_32ND);
359  sd = Ms::Direction::UP;
360  }
361  else {
362  if (durationType.type() == Ms::TDuration::DurationType::V_INVALID)
363  durationType.setType(Ms::TDuration::DurationType::V_QUARTER);
364  cr->setDurationType(durationType);
365  sd = Ms::Direction::DOWN;
366  }
367  cr->setDuration(durationType.fraction());
368  cr->setDots(dots);
369  cr->setStemDirection(sd);
370  // add note to chord
371  Ms::Note* note = new Ms::Note(score);
372  note->setTrack(0);
373  xmlSetPitch(note, sao.s.toLatin1(), sao.a, sao.o);
374  if (tieStart) {
375  Ms::Tie* tie = new Ms::Tie(score);
376  note->setTieFor(tie);
377  tie->setStartNote(note);
378  tie->setTrack(0);
379  }
380  cr->add(note);
381  // add chord to measure
382  if (!grace) {
383  Ms::Segment* s = currentMeasure->getSegment(Ms::SegmentType::ChordRest, tick);
384  s->add(cr);
385  if (!currentGraceNotes.isEmpty()) {
386  for (int i = currentGraceNotes.size() - 1; i >=0; i--)
387  cr->add(currentGraceNotes.at(i));
388  currentGraceNotes.clear();
389  }
390  doTriplet(cr, triplet);
391  int tickBefore = tick;
392  tick += ticks;
393  Ms::Fraction nl(Ms::Fraction::fromTicks(tick - currentMeasure->tick()));
394  currentMeasure->setLen(nl);
395  qDebug() << "MsScWriter::note()"
396  << "tickBefore:" << tickBefore
397  << "tick:" << tick
398  << "nl:" << nl.print()
399  ;
400  }
401  else {
402  currentGraceNotes.append(cr);
403  }
404  }

484 void MsScWriter::doTriplet(Ms::Chord* cr, StartStop triplet)
485  {
486  qDebug() << "MsScWriter::doTriplet(" << triplet << ")"
487  ;
488 
489  if (triplet == ST_START) {
490  tuplet = new Ms::Tuplet(score);
491  tuplet->setTrack(0);
492  tuplet->setRatio(Ms::Fraction(3, 2));
493  tuplet->setTick(tick);
494  currentMeasure->add(tuplet);
495  }
496  else if (triplet == ST_STOP) {
497  if (tuplet) {
498  cr->setTuplet(tuplet);
499  tuplet->add(cr);
500  tuplet = 0;
501  }
502  else
503  qDebug("BWW::import: triplet stop without triplet start");
504  }
505  else if (triplet == ST_CONTINUE) {
506  if (!tuplet)
507  qDebug("BWW::import: triplet continue without triplet start");
508  }
509  else if (triplet == ST_NONE) {
510  if (tuplet)
511  qDebug("BWW::import: triplet none inside triplet");
512  }
513  else
514  qDebug("unknown triplet type %d", triplet);
515  if (tuplet) {
516  cr->setTuplet(tuplet);
517  tuplet->add(cr);
518  }
519  }

530 Score::FileError importBww(MasterScore* score, const QString& path)
531  {
532  qDebug("Score::importBww(%s)", qPrintable(path));
533 
534  QFile fp(path);
535  if(!fp.exists())
536  return Score::FileError::FILE_NOT_FOUND;
537  if (!fp.open(QIODevice::ReadOnly))
538  return Score::FileError::FILE_OPEN_ERROR;
539 
540  QString id("importBww");
541  Part* part = new Part(score);
542  part->setId(id);
543  score->appendPart(part);
544  Staff* staff = new Staff(score);
545  staff->setPart(part);
546  part->staves()->push_back(staff);
547  score->staves().push_back(staff);
548 
549  Bww::Lexer lex(&fp);
550  Bww::MsScWriter wrt;
551  wrt.setScore(score);
552  score->style().set(StyleIdx::measureSpacing, 1.0);
553  Bww::Parser p(lex, wrt);
554  p.parse();
555 
556  score->setSaved(false);
557  score->setCreated(true);
558  score->connectTies();
559  qDebug("Score::importBww() done");
560  return Score::FileError::FILE_NO_ERROR; // OK
561  }
