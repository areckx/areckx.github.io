40 TimeSigProperties::TimeSigProperties(TimeSig* t, QWidget* parent)
41  : QDialog(parent)
42  {
43  setObjectName("TimeSigProperties");
44  setupUi(this);
45  setWindowFlags(this->windowFlags() & ~Qt::WindowContextHelpButtonHint);
46  timesig = t;
47 
48  zText->setText(timesig->numeratorString());
49  nText->setText(timesig->denominatorString());
50  // set validators for numerator and denominator strings
51  // which only accept '+', '(', ')', digits and some time symb conventional representations
52  QRegExp rx("[0-9+CO()\\x00A2\\x00D8]*");
53  QValidator *validator = new QRegExpValidator(rx, this);
54  zText->setValidator(validator);
55  nText->setValidator(validator);
56 
57  Fraction nominal = timesig->sig() / timesig->stretch();
58  nominal.reduce();
59  zNominal->setValue(nominal.numerator());
60  nNominal->setValue(nominal.denominator());
61  Fraction sig(timesig->sig());
62  zActual->setValue(sig.numerator());
63  nActual->setValue(sig.denominator());
64  zNominal->setEnabled(false);
65  nNominal->setEnabled(false);
66 
67  // TODO: fix http://musescore.org/en/node/42341
68  // for now, editing of actual (local) time sig is disabled in dialog
69  // but more importantly, the dialog should make it clear that this is "local" change only
70  // and not normally the right way to add 7/4 to a score
71  zActual->setEnabled(false);
72  nActual->setEnabled(false);
73  switch (timesig->timeSigType()) {
74  case TimeSigType::NORMAL:
75  textButton->setChecked(true);
76  break;
77  case TimeSigType::FOUR_FOUR:
78  fourfourButton->setChecked(true);
79  break;
80  case TimeSigType::ALLA_BREVE:
81  allaBreveButton->setChecked(true);
82  break;
83  }
84 
85  // set ID's of other symbols
86  static const SymId prolatioSymbols[] = {
87  SymId::mensuralProlation1, // tempus perfectum, prol. perfecta
88  SymId::mensuralProlation2, // tempus perfectum, prol. imperfecta
89  SymId::mensuralProlation3, // tempus perfectum, prol. imperfecta, dimin.
90  SymId::mensuralProlation4, // tempus perfectum, prol. perfecta, dimin.
91  SymId::mensuralProlation5, // tempus imperf. prol. perfecta
92 // SymId::mensuralProlation6, // same shape as common time
93  SymId::mensuralProlation7, // tempus imperf., prol. imperfecta, reversed
94  SymId::mensuralProlation8, // tempus imperf., prol. perfecta, dimin.
95 // SymId::mensuralProlation9, // same shape as alla breve
96  SymId::mensuralProlation10, // tempus imperf., prol imperfecta, dimin., reversed
97  SymId::mensuralProlation11, // tempus inperf., prol. perfecta, reversed
98  };
99 
100  ScoreFont* scoreFont = gscore->scoreFont();
101  int idx = 0;
102  for (SymId symId : prolatioSymbols) {
103  const QString& str = scoreFont->toString(symId);
104  if (str.size() > 0) {
105  otherCombo->setItemData(idx, (int)symId);
106  // if time sig matches this symbol string, set as selected
107  if (timesig->timeSigType() == TimeSigType::NORMAL && timesig->denominatorString().isEmpty()
108  && timesig->numeratorString() == str) {
109  textButton->setChecked(false);
110  otherButton->setChecked(true);
111  otherCombo->setCurrentIndex(idx);
112  }
113  }
114  idx++;
115  }
116 
117  Groups g = t->groups();
118  if (g.empty())
119  g = Groups::endings(timesig->sig()); // initialize with default
120  groups->setSig(timesig->sig(), g);
121 
122  MuseScore::restoreGeometry(this);
123  }

129 void TimeSigProperties::accept()
130  {
131  if (zText->text() != timesig->numeratorString())
132  timesig->setNumeratorString(zText->text());
133  if (nText->text() != timesig->denominatorString())
134  timesig->setDenominatorString(nText->text());
135 
136  TimeSigType ts = TimeSigType::NORMAL;
137  if (textButton->isChecked())
138  ts = TimeSigType::NORMAL;
139  else if (fourfourButton->isChecked())
140  ts = TimeSigType::FOUR_FOUR;
141  else if (allaBreveButton->isChecked())
142  ts = TimeSigType::ALLA_BREVE;
143  else if (otherButton->isChecked()) {
144  // if other symbol, set as normal text...
145  ts = TimeSigType::NORMAL;
146  ScoreFont* scoreFont = timesig->score()->scoreFont();
147  SymId symId = (SymId)( otherCombo->itemData(otherCombo->currentIndex()).toInt() );
148  // ...and set numerator to font string for symbol and denominator to empty string
149  timesig->setNumeratorString(scoreFont->toString(symId));
150  timesig->setDenominatorString(QString());
151  }
152 
153  Fraction actual(zActual->value(), nActual->value());
154  Fraction nominal(zNominal->value(), nNominal->value());
155  timesig->setSig(actual, ts);
156  timesig->setStretch(nominal / actual);
157 
158  Groups g = groups->groups();
159  timesig->setGroups(g);
160  QDialog::accept();
161  }
