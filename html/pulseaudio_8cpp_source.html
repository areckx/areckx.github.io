56 void PulseAudio::paCallback(pa_stream* s, size_t len, void* data)
57  {
58  PulseAudio* pa = (PulseAudio*)data;
59  constexpr size_t n = FRAMES * 2 * sizeof(float);
60  if (len > n) {
61  qDebug("PulseAudio:: buffer too large!");
62  len = n;
63  }
64  pa->seq->process(len / (2 * sizeof(float)), pa->buffer);
65  pa_stream_write(s, pa->buffer, len, nullptr, int64_t(0), PA_SEEK_RELATIVE);
66  }

93 static void pa_state_cb(pa_context* c, void* data)
94  {
95  int* pa_ready = (int*)data;
96  switch (pa_context_get_state(c)) {
97  // These are just here for reference
98  case PA_CONTEXT_UNCONNECTED:
99  case PA_CONTEXT_CONNECTING:
100  case PA_CONTEXT_AUTHORIZING:
101  case PA_CONTEXT_SETTING_NAME:
102  default:
103  break;
104  case PA_CONTEXT_FAILED:
105  case PA_CONTEXT_TERMINATED:
106  *pa_ready = 2;
107  break;
108  case PA_CONTEXT_READY:
109  *pa_ready = 1;
110  break;
111  }
112  }

119 bool PulseAudio::init(bool)
120  {
121  pa_ml = pa_mainloop_new();
122  pa_mainloop_api* pa_mlapi = pa_mainloop_get_api(pa_ml);
123  pa_context* pa_ctx = pa_context_new(pa_mlapi, "MuseScore");
124  if (pa_context_connect(pa_ctx, NULL, pa_context_flags_t(0), NULL) != 0) {
125  qDebug("PulseAudio Context Connect Failed with Error: %s", pa_strerror(pa_context_errno(pa_ctx)));
126  return false;
127  }
128 
129  int pa_ready = 0;
130  pa_context_set_state_callback(pa_ctx, pa_state_cb, &pa_ready);
131 
132  while (pa_ready == 0)
133  pa_mainloop_iterate(pa_ml, 1, NULL);
134  if (pa_ready == 2)
135  return false;
136 
137  ss.rate = _sampleRate;
138  ss.channels = 2;
139  ss.format = PA_SAMPLE_FLOAT32LE;
140 
141  pa_stream* playstream = pa_stream_new(pa_ctx, "Playback", &ss, NULL);
142  if (!playstream) {
143  qDebug("pa_stream_new failed: %s", pa_strerror(pa_context_errno(pa_ctx)));
144  return false;
145  }
146  pa_stream_set_write_callback(playstream, paCallback, this);
147 
148  bufattr.fragsize = (uint32_t)-1;
149  bufattr.maxlength = FRAMES * 2 * sizeof(float);
150  bufattr.minreq = FRAMES * 1 * sizeof(float); // pa_usec_to_bytes(0, &ss);
151  bufattr.prebuf = (uint32_t)-1;
152  bufattr.tlength = bufattr.maxlength;
153 
154  int r = pa_stream_connect_playback(playstream, nullptr, &bufattr,
155  PA_STREAM_NOFLAGS, nullptr, nullptr);
156 
157  if (r < 0) {
158  qDebug("pa_stream_connect_playback failed");
159  pa_context_disconnect(pa_ctx);
160  pa_context_unref(pa_ctx);
161  pa_mainloop_free(pa_ml);
162  pa_ml = 0;
163  return false;
164  }
165  return true;
166  }

172 void* PulseAudio::paLoop(void* data)
173  {
174  PulseAudio* pa = (PulseAudio*)data;
175  pa->runState = 2;
176  while (pa->runState == 2)
177  pa_mainloop_iterate(pa->pa_ml, 1, NULL);
178  pa->runState = 0;
179  return 0;
180  }

186 bool PulseAudio::start(bool)
187  {
188  pthread_attr_t* attributes = (pthread_attr_t*) malloc(sizeof(pthread_attr_t));
189  pthread_attr_init(attributes);
190  if (pthread_create(&thread, attributes, paLoop, this))
191  perror("creating thread failed:");
192  pthread_attr_destroy(attributes);
193  return true;

200 bool PulseAudio::stop()
201  {
202  if (runState == 2) {
203  runState = 1;
204  int i = 0;
205  for (;i < 4; ++i) {
206  if (runState == 0)
207  break;
208  sleep(1);
209  }
210  pthread_cancel(thread);
211  pthread_join(thread, 0);
212  }
213  return true;
214  }
