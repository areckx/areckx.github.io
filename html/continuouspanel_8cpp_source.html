50 void ContinuousPanel::paint(const QRect&, QPainter& painter)
51  {
52  qreal _offsetPanel = 0;
53  qreal _y = 0;
54  qreal _oldWidth = 0; // The last final panel width
55  qreal _newWidth = 0; // New panel width
56  qreal _height = 0;
57  qreal _leftMarginTotal = 0; // Sum of all elments left margin
58  qreal _panelRightPadding = 5; // Extra space for the panel after last element
59 
60  Measure* measure = _score->firstMeasure();
61 
62  if (!_active || !measure) {
63  _visible = false;
64  return;
65  }
66 
67  if (measure->mmRest()) {
68  measure = measure->mmRest();
69  }
70 
71  System* system = measure->system();
72  if (system == 0) {
73  _visible = false;
74  return;
75  }
76 
77  Segment* s = measure->first();
78  double _spatium = _score->spatium();
79  if (_width <= 0)
80  _width = s->x();
81 
82  //
83  // Set panel height for whole system
84  //
85  _height = 6 * _spatium;
86  _y = system->staffYpage(0) + system->page()->pos().y();
87  double y2 = 0.0;
88  for (int i = 0; i < _score->nstaves(); ++i) {
89  SysStaff* ss = system->staff(i);
90  if (!ss->show() || !_score->staff(i)->show())
91  continue;
92  y2 = ss->y() + ss->bbox().height();
93  }
94  _height += y2 + 6*_spatium;
95  _y -= 6 * _spatium;
96 
97  //
98  // Check elements at current panel position
99  //
100  _offsetPanel = -(_sv->xoffset()) / _sv->mag();
101  _rect = QRect(_offsetPanel + _width, _y, 1, _height);
102  Page* page = _score->pages().front();
103  QList<Element*> el = page->items(_rect);
104  if (el.empty()) {
105  _visible = false;
106  return;
107  }
108  qStableSort(el.begin(), el.end(), elementLessThan);
109 
110  const Measure*_currentMeasure = 0;
111  for (const Element* e : el) {
112  e->itemDiscovered = 0;
113  if (!e->visible() && !_score->showInvisible())
114  continue;
115 
116  if (e->isMeasure()) {
117  _currentMeasure = toMeasure(e);
118  break;
119  }
120  }
121  if (!_currentMeasure)
122  return;
123 
124  // Don't show panel if staff names are visible
125  if (_currentMeasure == _score->firstMeasure() && _sv->toPhysical(_currentMeasure->canvasPos()).x() > 0) {
126  _visible = false;
127  return;
128  }
129 
130  qreal _xPosMeasure = _currentMeasure->canvasX();
131  qreal _measureWidth = _currentMeasure->width();
132  int tick = _currentMeasure->tick();
133  Fraction _currentTimeSig = _currentMeasure->timesig();
134  //qDebug() << "_sv->xoffset()=" <<_sv->xoffset() << " _sv->mag()="<< _sv->mag() <<" s->x=" << s->x() << " width=" << _width << " currentMeasure=" << _currentMeasure->x() << " _xPosMeasure=" << _xPosMeasure;
135 
136  //---------------------------------------------------------
137  // findElementWidths
138  // determines the max width for each element types
139  //---------------------------------------------------------
140 
141  // The first pass serves to get the maximum width for each elements
142 
143  qreal lineWidthName = 0;
144  qreal _widthClef = 0;
145  qreal _widthKeySig = 0;
146  qreal _widthTimeSig = 0;
147  qreal _xPosTimeSig = 0;
148 
149  for (const Element* e : el) {
150  e->itemDiscovered = 0;
151  if (!e->visible() && !_score->showInvisible())
152  continue;
153 
154  if (e->isRest() && toRest(e)->isGap())
155  continue;
156 
157  if (e->isStaffLines()) {
158  Staff* currentStaff = _score->staff(e->staffIdx());
159  Segment* parent = _score->tick2segment(tick);
160 
161  // Find maximum width for the staff name
162  QList<StaffName>& staffNamesLong = currentStaff->part()->instrument()->longNames();
163  QString staffName = staffNamesLong.isEmpty() ? " " : staffNamesLong[0].name();
164  if (staffName == "") {
165  QList<StaffName>& staffNamesShort = currentStaff->part()->instrument()->shortNames();
166  staffName = staffNamesShort.isEmpty() ? "" : staffNamesShort[0].name();
167  }
168  Text* newName = new Text(_score);
169  newName->setXmlText(staffName);
170  newName->setParent(parent);
171  newName->setTrack(e->track());
172  newName->setFamily("FreeSans");
173  newName->setSizeIsSpatiumDependent(true);
174  newName->layout();
175  newName->setPlainText(newName->plainText());
176  newName->layout();
177 
178  // Find maximum width for the current Clef
179  Clef* newClef = new Clef(_score);
180  ClefType currentClef = currentStaff->clef(tick);
181  newClef->setClefType(currentClef);
182  newClef->setParent(parent);
183  newClef->setTrack(e->track());
184  newClef->layout();
185  if (newClef->width() > _widthClef)
186  _widthClef = newClef->width();
187 
188  // Find maximum width for the current KeySignature
189  KeySig* newKs = new KeySig(_score);
190  KeySigEvent currentKeySigEvent = currentStaff->keySigEvent(tick);
191  newKs->setKeySigEvent(currentKeySigEvent);
192  // The Parent and the Track must be set to have the key signature layout adjusted to different clefs
193  // This also adds naturals to the key signature (if set in the score style)
194  newKs->setParent(parent);
195  newKs->setTrack(e->track());
196  newKs->setHideNaturals(true);
197  newKs->layout();
198  if (newKs->width() > _widthKeySig)
199  _widthKeySig = newKs->width();
200 
201  // Find maximum width for the current TimeSignature
202  TimeSig* newTs = new TimeSig(_score);
203 
204  // Try to get local time signature, if not, get the current measure one
205  TimeSig* currentTimeSig = currentStaff->timeSig(tick);
206  if (currentTimeSig)
207  newTs->setFrom(currentTimeSig);
208  else
209  newTs->setSig(Fraction(_currentTimeSig.numerator(), _currentTimeSig.denominator()), TimeSigType::NORMAL);
210  newTs->setParent(parent);
211  newTs->setTrack(e->track());
212  newTs->layout();
213 
214  if ((newName->width() > lineWidthName) && (newName->xmlText() != ""))
215  lineWidthName = newName->width();
216 
217  if (newTs->width() > _widthTimeSig)
218  _widthTimeSig = newTs->width();
219 
220  delete newClef;
221  delete newName;
222  delete newKs;
223  delete newTs;
224  }
225  }
226 
227  _leftMarginTotal = _score->styleP(StyleIdx::clefLeftMargin);
228  _leftMarginTotal += _score->styleP(StyleIdx::keysigLeftMargin);
229  _leftMarginTotal += _score->styleP(StyleIdx::timesigLeftMargin);
230 
231  _newWidth = _widthClef + _widthKeySig + _widthTimeSig + _leftMarginTotal + _panelRightPadding;
232  _xPosMeasure -= _offsetPanel;
233 
234  lineWidthName += _score->spatium() + _score->styleP(StyleIdx::clefLeftMargin) + _widthClef;
235  if (_newWidth < lineWidthName) {
236  _newWidth = lineWidthName;
237  _oldWidth = 0;
238  }
239  if (_oldWidth == 0) {
240  _oldWidth = _newWidth;
241  _width = _newWidth;
242  }
243  else if (_newWidth > 0) {
244  if (_newWidth == _width) {
245  _oldWidth = _width;
246  _width = _newWidth;
247  }
248  else if (((_xPosMeasure <= _newWidth) && (_xPosMeasure >= _oldWidth)) ||
249  ((_xPosMeasure >= _newWidth) && (_xPosMeasure <= _oldWidth)))
250  _width = _xPosMeasure;
251  else if (((_xPosMeasure+_measureWidth <= _newWidth) && (_xPosMeasure+_measureWidth >= _oldWidth)) ||
252  ((_xPosMeasure+_measureWidth >= _newWidth) && (_xPosMeasure+_measureWidth <= _oldWidth)))
253  _width = _xPosMeasure+_measureWidth;
254  else {
255  _oldWidth = _width;
256  _width = _newWidth;
257  }
258  }
259 
260  _rect = QRect(0, _y, _width, _height);
261 
262  //====================
263 
264  painter.save();
265 
266  // Draw colored rectangle
267  painter.setClipping(false);
268  QPointF pos(_offsetPanel, 0);
269 
270  painter.translate(pos);
271  QPen pen;
272  pen.setWidthF(0.0);
273  pen.setStyle(Qt::NoPen);
274  painter.setPen(pen);
275  painter.setBrush(preferences.fgColor);
276  QRectF bg(_rect);
277 
278  bg.setWidth(_widthClef + _widthKeySig + _widthTimeSig + _leftMarginTotal + _panelRightPadding);
279  QPixmap* fgPixmap = _sv->fgPixmap();
280  if (fgPixmap == 0 || fgPixmap->isNull())
281  painter.fillRect(bg, preferences.fgColor);
282  else {
283  painter.setMatrixEnabled(false);
284  painter.drawTiledPixmap(bg, *fgPixmap, bg.topLeft()
285  - QPoint(lrint(_sv->matrix().dx()), lrint(_sv->matrix().dy())));
286  painter.setMatrixEnabled(true);
287  }
288 
289  painter.setClipRect(_rect);
290  painter.setClipping(true);
291 
292  QColor color(MScore::layoutBreakColor);
293 
294  // Draw measure text number
295  // TODO: simplify (no Text element)
296  QString text = QString("#%1").arg(_currentMeasure->no()+1);
297  Text* newElement = new Text(_score);
298  newElement->setFlag(ElementFlag::MOVABLE, false);
299  newElement->setXmlText(text);
300  newElement->setFamily("FreeSans");
301  newElement->setSizeIsSpatiumDependent(true);
302  newElement->setColor(color);
303  newElement->layout1();
304  pos = QPointF(_score->styleP(StyleIdx::clefLeftMargin) + _widthClef, _y + newElement->height());
305  painter.translate(pos);
306  newElement->draw(&painter);
307  pos += QPointF(_offsetPanel, 0);
308  painter.translate(-pos);
309  delete newElement;
310 
311  // This second pass draws the elements spaced evently using the width of the largest element
312  for (const Element* e : el) {
313  if (!e->visible() && !_score->showInvisible())
314  continue;
315 
316  if (e->isRest() && toRest(e)->isGap())
317  continue;
318 
319  if (e->isStaffLines()) {
320  painter.save();
321  Staff* currentStaff = _score->staff(e->staffIdx());
322  Segment* parent = _score->tick2segmentMM(tick);
323 
324  pos = QPointF (_offsetPanel, e->pagePos().y());
325  painter.translate(pos);
326 
327  // Draw staff lines
328  StaffLines newStaffLines(*toStaffLines(e));
329  newStaffLines.setParent(parent);
330  newStaffLines.setTrack(e->track());
331  newStaffLines.layout();
332  newStaffLines.setColor(color);
333  newStaffLines.setWidth(bg.width());
334  newStaffLines.draw(&painter);
335 
336  // Draw barline
337  BarLine barLine(_score);
338  barLine.setBarLineType(BarLineType::NORMAL);
339  barLine.setParent(parent);
340  barLine.setTrack(e->track());
341  barLine.setSpanStaff(currentStaff->barLineSpan());
342  barLine.setSpanFrom(currentStaff->barLineFrom());
343  barLine.setSpanTo(currentStaff->barLineTo());
344  barLine.layout();
345  barLine.setColor(color);
346  barLine.draw(&painter);
347 
348  // Draw the current staff name
349  QList<StaffName>& staffNamesLong = currentStaff->part()->instrument()->longNames();
350  QString staffName = staffNamesLong.isEmpty() ? " " : staffNamesLong[0].name();
351  if (staffName == "") {
352  QList<StaffName>& staffNamesShort = currentStaff->part()->instrument()->shortNames();
353  staffName = staffNamesShort.isEmpty() ? "" : staffNamesShort[0].name();
354  }
355 
356  Text* newName = new Text(_score);
357  newName->setXmlText(staffName);
358  newName->setParent(parent);
359  newName->setTrack(e->track());
360  newName->setColor(color);
361  newName->setFamily("FreeSans");
362  newName->setSizeIsSpatiumDependent(true);
363  newName->layout();
364  newName->setPlainText(newName->plainText());
365  newName->layout();
366  if (currentStaff->part()->staff(0) == currentStaff) {
367  double _spatium = _score->spatium();
368  pos = QPointF (_score->styleP(StyleIdx::clefLeftMargin) + _widthClef, -_spatium * 2);
369  painter.translate(pos);
370  newName->draw(&painter);
371  painter.translate(-pos);
372  }
373  delete newName;
374 
375  qreal posX = 0.0;
376 
377  // Draw the current Clef
378  Clef clef(_score);
379  clef.setClefType(currentStaff->clef(tick));
380  clef.setParent(parent);
381  clef.setTrack(e->track());
382  clef.setColor(color);
383  clef.layout();
384  posX += _score->styleP(StyleIdx::clefLeftMargin);
385  clef.drawAt(&painter, QPointF(posX, clef.pos().y()));
386  posX += _widthClef;
387 
388  // Draw the current KeySignature
389  KeySig newKs(_score);
390  newKs.setKeySigEvent(currentStaff->keySigEvent(tick));
391 
392  // The Parent and the track must be set to have the key signature layout adjusted to different clefs
393  // This also adds naturals to the key signature (if set in the score style)
394  newKs.setParent(parent);
395  newKs.setTrack(e->track());
396  newKs.setColor(color);
397  newKs.setHideNaturals(true);
398  newKs.layout();
399  posX += _score->styleP(StyleIdx::keysigLeftMargin);
400  newKs.drawAt(&painter, QPointF(posX, 0.0));
401 
402  posX += _widthKeySig + _xPosTimeSig;
403 
404  // Draw the current TimeSignature
405  TimeSig newTs(_score);
406 
407  // Try to get local time signature, if not, get the current measure one
408  TimeSig* currentTimeSig = currentStaff->timeSig(tick);
409  if (currentTimeSig) {
410  newTs.setFrom(currentTimeSig);
411  newTs.setParent(parent);
412  newTs.setTrack(e->track());
413  newTs.setColor(color);
414  newTs.layout();
415  posX += _score->styleP(StyleIdx::timesigLeftMargin);
416  newTs.drawAt(&painter, QPointF(posX, 0.0));
417  }
418  painter.restore();
419  }
420  }
421  painter.restore();
422  _visible = true;
423  }
