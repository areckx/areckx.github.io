29 Volta* Score::searchVolta(int tick) const
30  {
31  for (const std::pair<int,Spanner*>& p : _spanner.map()) {
32  Spanner* s = p.second;
33  if (s->type() != ElementType::VOLTA)
34  continue;
35  Volta* volta = static_cast<Volta*>(s);
36  if (tick >= volta->tick() && tick < volta->tick2())
37  return volta;
38  }
39  return 0;
40  }

46 Measure* Score::searchLabel(const QString& s)
47  {
48  if (s == "start")
49  return firstMeasure();
50  else if (s == "end")
51  return lastMeasure();
52  for (Measure* m = firstMeasure(); m; m = m->nextMeasure()) {
53  for (auto e : m->el()) {
54  if (e->type() == ElementType::MARKER) {
55  const Marker* marker = static_cast<const Marker*>(e);
56  if (marker->label() == s)
57  return m;
58  }
59  }
60  }
61  return 0;
62  }

69  {

72  else if (s == "end")
73  return sectionEndMeasure;
74  for (Measure* m = sectionStartMeasure; m && (m != sectionEndMeasure->nextMeasure()); m = m->nextMeasure()) {
75  for (auto e : m->el()) {
76  if (e->type() == ElementType::MARKER) {
77  const Marker* marker = static_cast<const Marker*>(e);
78  if (marker->label() == s)
79  return m;
80  }
81  }
82  }

84  // if did not find label within section, then search for label in entire score
85  return searchLabel(s);

155 void RepeatList::update()
156  {
157  const TempoMap* tl = _score->tempomap();
158 
159  int utick = 0;
160  qreal t = 0;
161 
162  for(RepeatSegment* s : *this) {
163  s->utick = utick;
164  s->utime = t;
165  qreal ct = tl->tick2time(s->tick);
166  s->timeOffset = t - ct;
167  utick += s->len;
168  t += tl->tick2time(s->tick + s->len) - ct;
169  }
170  }

176 int RepeatList::utick2tick(int tick) const
177  {
178  unsigned n = size();
179  if (n == 0)
180  return tick;
181  if (tick < 0)
182  return 0;
183  unsigned ii = (idx1 < n) && (tick >= at(idx1)->utick) ? idx1 : 0;
184  for (unsigned i = ii; i < n; ++i) {
185  if ((tick >= at(i)->utick) && ((i + 1 == n) || (tick < at(i+1)->utick))) {
186  idx1 = i;
187  return tick - (at(i)->utick - at(i)->tick);
188  }
189  }
190  if (MScore::debugMode) {
191  qFatal("tick %d not found in RepeatList", tick);
192  }
193  return 0;
194  }

201  {
202  for (const RepeatSegment* s : *this) {
203  if (tick >= s->tick && tick < (s->tick + s->len))
204  return s->utick + (tick - s->tick);
205  }
206  return last()->utick + (tick - last()->tick);

278 void RepeatList::unwind()
279  {
280  qDeleteAll(*this);
281  clear();
282  Measure* fm = _score->firstMeasure();
283  if (!fm)
284  return;
285 
286  //qDebug("unwind===================");
287 
288  for (Measure* m = fm; m; m = m->nextMeasure())
289  m->setPlaybackCount(0);
290 
291  MeasureBase* sectionStartMeasureBase = NULL; // NULL indicates haven't discovered starting Measure of section
292  MeasureBase* sectionEndMeasureBase = NULL;
293 
294  // partition score by section breaks and unwind individual sections seperately
295  // note: section breaks may occur on non-Measure frames, so must seach list of all MeasureBases
296  for (MeasureBase* mb = _score->first(); mb; mb = mb->next()) {
297 
298  // unwindSection only deals with real Measures, so sectionEndMeasureBase and sectionStartMeasureBase will only point to real Measures
299  if (mb->isMeasure()) {
300  sectionEndMeasureBase = mb; // ending measure of section is the most recently encountered actual Measure
301 
302  // starting measure of section will be the first non-NULL actual Measure encountered
303  if (sectionStartMeasureBase == NULL)
304  sectionStartMeasureBase = mb;
305  }
306 
307  // if found section break or reached final MeasureBase of score, then unwind
308  if (mb->sectionBreak() || !mb->nextMeasure()) {
309 
310  // only unwind if section starts and ends with actual real measure
311  if (sectionStartMeasureBase && sectionEndMeasureBase) {
312  unwindSection(reinterpret_cast<Measure*>(sectionStartMeasureBase), reinterpret_cast<Measure*>(sectionEndMeasureBase));
313  sectionStartMeasureBase = 0; // reset to NULL to indicate that don't know starting Measure of next section after starting new section
314  sectionEndMeasureBase = 0;
315  }
316  else {
317  qDebug( "Will not unroll a section that doesn't start or end with an actual measure. sectionStartMeasureBase = %p, sectionEndMeasureBase = %p",
318  sectionStartMeasureBase, sectionEndMeasureBase);
319  }
320  }
321  }
322 
323  update();
324  dump();
325  }

328 // unwindSection
329 // unwinds from sectionStartMeasure through sectionEndMeasure
330 // appends repeat segments to rs
331 //---------------------------------------------------------
332 
333 void RepeatList::unwindSection(Measure* sectionStartMeasure, Measure* sectionEndMeasure)
334  {
335 // qDebug("unwind %d-measure section starting %p through %p", sectionEndMeasure->no()+1, sectionStartMeasure, sectionEndMeasure);
336 
337  QList<Jump*> jumps; // take the jumps only once so store them
338 
339  rs = new RepeatSegment;
340  rs->tick = sectionStartMeasure->tick(); // prepare initial repeat segment for start of this section
341 
342  Measure* endRepeat = 0; // measure where the current repeat should stop
343  Measure* continueAt = 0; // measure where the playback should continue after the repeat (To coda)
344  Measure* m = 0;
345  int loop = 0; // keeps track of how many times have repeated a :| (Repeat::END)
346  int repeatCount = 0;
347  bool isGoto = false;
348  bool playRepeats = false;
349 
350  for (Measure* nm = sectionStartMeasure; nm; ) {
351  m = nm;
352  m->setPlaybackCount(m->playbackCount() + 1);
353  bool doJump = false; // process jump after endrepeat
354 
355  // during any DC or DS, will take last time through repeat
356  if (isGoto && !playRepeats && m->repeatEnd())
357  loop = m->repeatCount() - 1;
358 
359  if (endRepeat) {
360  Volta* volta = _score->searchVolta(m->tick());
361  if (volta && !volta->hasEnding(m->playbackCount())) {
362  // skip measure
363  if (rs->tick < m->tick()) {
364  rs->len = m->tick() - rs->tick;
365  append(rs);
366  rs = new RepeatSegment;
367  }
368  rs->tick = m->endTick();
369  }
370  else if (m->repeatJump()) {
371  doJump = true;
372  isGoto = false;
373  }
374  }
375  else if (m->repeatJump()) // Jumps are only accepted outside of other repeats
376  doJump = true;
377 
378  if (isGoto && (endRepeat == m)) {
379  if (continueAt == 0)
380  break;
381  rs->len = m->endTick() - rs->tick;
382  append(rs);
383  rs = new RepeatSegment;
384  rs->tick = continueAt->tick();
385  nm = continueAt;
386  isGoto = false;
387  endRepeat = 0;
388  continue;
389  }
390  else if (m->repeatEnd()) {
391  if (endRepeat == m) {
392  ++loop;
393  if (loop >= repeatCount) {
394  endRepeat = 0;
395  loop = 0;
396  }
397  else {
398  nm = jumpToStartRepeat(m);
399  continue;
400  }
401  }
402  else if (endRepeat == 0) {
403  if (m->playbackCount() >= m->repeatCount())
404  break;
405  endRepeat = m;
406  repeatCount = m->repeatCount();
407  loop = 1;
408  nm = jumpToStartRepeat(m);
409  continue;
410  }
411  else {
412  ++loop;
413  if (loop < repeatCount) {
414  nm = jumpToStartRepeat(m);
415  continue;
416  }
417  }
418  }
419  if (doJump && !isGoto) {
420  Jump* jump = 0;
421  foreach (Element* e, m->el()) {
422  if (e->isJump()) {
423  jump = toJump(e);
424  break;
425  }
426  }
427  // jump only once
428  if (jumps.contains(jump)) {
429  if (endRepeat == _score->searchLabelWithinSectionFirst(jump->playUntil(), sectionStartMeasure, sectionEndMeasure))
430  endRepeat = 0;
431 
432  nm = m->nextMeasure();
433  if (nm == sectionEndMeasure->nextMeasure())
434  break;
435  else
436  continue;
437  }
438  jumps.append(jump);
439  if (jump) {
440  nm = _score->searchLabelWithinSectionFirst(jump->jumpTo() , sectionStartMeasure, sectionEndMeasure);
441  endRepeat = _score->searchLabelWithinSectionFirst(jump->playUntil() , sectionStartMeasure, sectionEndMeasure);
442  continueAt = _score->searchLabelWithinSectionFirst(jump->continueAt(), sectionStartMeasure, sectionEndMeasure);
443 
444  if (nm && endRepeat) {
445  isGoto = true;
446  playRepeats = jump->playRepeats();
447  rs->len = m->endTick() - rs->tick;
448  append(rs);
449  rs = new RepeatSegment;
450  rs->tick = nm->tick();
451  continue;
452  }
453  }
454  else
455  qDebug("Jump not found");
456  }
457 
458  // keep looping until reach end of score or end of the section
459  nm = m->nextMeasure();
460  if (nm == sectionEndMeasure->nextMeasure())
461  break;
462  }
463 
464  // append the final repeat segment of that section
465  if (rs) {
466  rs->len = m->endTick() - rs->tick;
467  if (rs->len)
468  append(rs);
469  else
470  delete rs;
471  }
472  }

478 Measure* RepeatList::jumpToStartRepeat(Measure* m)
479  {
480  // finalize the previous repeat segment
481  rs->len = m->tick() + m->ticks() - rs->tick;
482  append(rs);
483 
484  // search backwards until find start of repeat
485  while (true) {
486 
487  if (m->repeatStart())
488  break;
489 
490  if (m == _score->firstMeasure())
491  break;
492 
493  if (m->prevMeasure()->sectionBreak())
494  break;
495 
496  m = m->prevMeasure();
497  }
498 
499  // initialize the next repeat segment
500  rs = new RepeatSegment;
501  rs->tick = m->tick();
502  return m;
503  }
