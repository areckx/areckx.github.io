114 void Image::draw(QPainter* painter) const
115  {
116  bool emptyImage = false;
117  if (imageType == ImageType::SVG) {
118  if (!svgDoc)
119  emptyImage = true;
120  else
121  svgDoc->render(painter, bbox());
122  }
123  else if (imageType == ImageType::RASTER) {
124  if (rasterDoc == nullptr)
125  emptyImage = true;
126  else {
127  painter->save();
128  QSizeF s;
129  if (_sizeIsSpatium)
130  s = _size * spatium();
131  else
132  s = _size * DPMM;
133  if (score()->printing() && !MScore::svgPrinting) {
134  // use original image size for printing, but not for svg for reasonable file size.
135  painter->scale(s.width() / rasterDoc->width(), s.height() / rasterDoc->height());
136  painter->drawPixmap(QPointF(0, 0), QPixmap::fromImage(*rasterDoc));
137  }
138  else {
139  QTransform t = painter->transform();
140  QSize ss = QSizeF(s.width() * t.m11(), s.height() * t.m22()).toSize();
141  t.setMatrix(1.0, t.m12(), t.m13(), t.m21(), 1.0, t.m23(), t.m31(), t.m32(), t.m33());
142  painter->setWorldTransform(t);
143  if ((buffer.size() != ss || _dirty) && rasterDoc && !rasterDoc->isNull()) {
144  buffer = QPixmap::fromImage(rasterDoc->scaled(ss, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
145  _dirty = false;
146  }
147  if (buffer.isNull())
148  emptyImage = true;
149  else
150  painter->drawPixmap(QPointF(0.0, 0.0), buffer);
151  }
152  painter->restore();
153  }
154  }
155  if (emptyImage) {
156  painter->setBrush(Qt::NoBrush);
157  painter->setPen(Qt::black);
158  painter->drawRect(bbox());
159  painter->drawLine(0.0, 0.0, bbox().width(), bbox().height());
160  painter->drawLine(bbox().width(), 0.0, 0.0, bbox().height());
161  }
162  if (selected() && !(score() && score()->printing())) {
163  painter->setBrush(Qt::NoBrush);
164  painter->setPen(MScore::selectColor[0]);
165  painter->drawRect(bbox());
166  }
167  }

173 void Image::write(XmlWriter& xml) const
174  {
175  // attempt to convert the _linkPath to a path relative to the score
176  //
177  // TODO : on Save As, score()->fileInfo() still contains the old path and fname
178  // if the Save As path is different, image relative path will be wrong!
179  //
180  QString relativeFilePath= QString();
181  if (!_linkPath.isEmpty() && _linkIsValid) {
182  QFileInfo fi(_linkPath);
183  // score()->fileInfo()->canonicalPath() would be better
184  // but we are saving under a temp file name and the 'final' file
185  // might not exist yet, so canonicalFilePath() may return only "/"
186  // OTOH, the score 'final' file name is practically always canonical, at this point
187  QString scorePath = score()->masterScore()->fileInfo()->absolutePath();
188  QString imgFPath = fi.canonicalFilePath();
189  // if imgFPath is in (or below) the directory of scorePath
190  if (imgFPath.startsWith(scorePath, Qt::CaseSensitive)) {
191  // relative img path is the part exceeding scorePath
192  imgFPath.remove(0, scorePath.size());
193  if(imgFPath.startsWith('/'))
194  imgFPath.remove(0, 1);
195  relativeFilePath = imgFPath;
196  }
197  // try 1 level up
198  else {
199  // reduce scorePath by one path level
200  fi.setFile(scorePath);
201  scorePath = fi.path();
202  // if imgFPath is in (or below) the directory up the score directory
203  if (imgFPath.startsWith(scorePath, Qt::CaseSensitive)) {
204  // relative img path is the part exceeding new scorePath plus "../"
205  imgFPath.remove(0, scorePath.size());
206  if (!imgFPath.startsWith('/'))
207  imgFPath.prepend('/');
208  imgFPath.prepend("..");
209  relativeFilePath = imgFPath;
210  }
211  }
212  }
213  // if no match, use full _linkPath
214  if (relativeFilePath.isEmpty())
215  relativeFilePath = _linkPath;
216 
217  xml.stag("Image");
218  BSymbol::writeProperties(xml);
219  // keep old "path" tag, for backward compatibility and because it is used elsewhere
220  // (for instance by Box:read(), Measure:read(), Note:read(), ...)
221  xml.tag("path", _storeItem ? _storeItem->hashName() : relativeFilePath);
222  xml.tag("linkPath", relativeFilePath);
223 
224  writeProperty(xml, P_ID::AUTOSCALE);
225  writeProperty(xml, P_ID::SIZE);
226  writeProperty(xml, P_ID::LOCK_ASPECT_RATIO);
227  writeProperty(xml, P_ID::SIZE_IS_SPATIUM);
228 
229  xml.etag();
230  }

236 void Image::read(XmlReader& e)
237  {
238  if (score()->mscVersion() <= 114)
239  _sizeIsSpatium = false;
240 
241  while (e.readNextStartElement()) {
242  const QStringRef& tag(e.name());
243  if (tag == "autoScale")
244  setProperty(P_ID::AUTOSCALE, Ms::getProperty(P_ID::AUTOSCALE, e));
245  else if (tag == "size")
246  setProperty(P_ID::SIZE, Ms::getProperty(P_ID::SIZE, e));
247  else if (tag == "lockAspectRatio")
248  setProperty(P_ID::LOCK_ASPECT_RATIO, Ms::getProperty(P_ID::LOCK_ASPECT_RATIO, e));
249  else if (tag == "sizeIsSpatium")
250  setProperty(P_ID::SIZE_IS_SPATIUM, Ms::getProperty(P_ID::SIZE_IS_SPATIUM, e));
251  else if (tag == "path")
252  _storePath = e.readElementText();
253  else if (tag == "linkPath")
254  _linkPath = e.readElementText();
255  else if (tag == "subtype") // obsolete
256  e.skipCurrentElement();
257  else if (!BSymbol::readProperties(e))
258  e.unknown();
259  }
260 
261  // once all paths are read, load img or retrieve it from store
262  // loading from file is tried first to update the stored image, if necessary
263 
264  qDebug("linkPath <%s>", qPrintable(_linkPath));
265  qDebug("storePath <%s>", qPrintable(_storePath));
266 
267  QString path;
268  bool loaded = false;
269  // if a store path is given, attempt to get the image from the store
270  if (!_storePath.isEmpty()) {
271  _storeItem = imageStore.getImage(_storePath);
272  if (_storeItem) {
273  _storeItem->reference(this);
274  loaded = true;
275  }
276  // if no image in store, attempt to load from path (for backward compatibility)
277  else
278  loaded = load(_storePath);
279  path = _storePath;
280  }
281  // if no succes from store path, attempt loading from link path (for .mscx files)
282  if (!loaded) {
283  _linkIsValid = load(_linkPath);
284  path = _linkPath;
285  }
286 
287  if (path.endsWith(".svg"))
288  setImageType(ImageType::SVG);
289  else
290  setImageType(ImageType::RASTER);
291  }

299 bool Image::load(const QString& ss)
300  {
301  qDebug("Image::load <%s>", qPrintable(ss));
302  QString path(ss);
303  // if file path is relative, prepend score path
304  QFileInfo fi(path);
305  if (fi.isRelative()) {
306  path.prepend(score()->masterScore()->fileInfo()->absolutePath() + "/");
307  fi.setFile(path);
308  }
309 
310  _linkIsValid = false; // assume link fname is invalid
311  QFile f(path);
312  if (!f.open(QIODevice::ReadOnly)) {
313  qDebug("Image::load<%s> failed", qPrintable(path));
314  return false;
315  }
316  QByteArray ba = f.readAll();
317  f.close();
318 
319  _linkIsValid = true;
320  _linkPath = fi.canonicalFilePath();
321  _storeItem = imageStore.add(_linkPath, ba);
322  _storeItem->reference(this);
323  if (path.endsWith(".svg"))
324  setImageType(ImageType::SVG);
325  else
326  setImageType(ImageType::RASTER);
327  return true;
328  }

376 void Image::editDrag(EditData& ed)
377  {
378  qreal ratio = _size.width() / _size.height();
379  qreal dx = ed.delta.x();
380  qreal dy = ed.delta.y();
381  if (_sizeIsSpatium) {
382  qreal _spatium = spatium();
383  dx /= _spatium;
384  dy /= _spatium;
385  }
386  else {
387  dx /= DPMM;
388  dy /= DPMM;
389  }
390  if (ed.curGrip == Grip::START) {
391  _size.setWidth(_size.width() + dx);
392  if (_lockAspectRatio)
393  _size.setHeight(_size.width() / ratio);
394  }
395  else {
396  _size.setHeight(_size.height() + dy);
397  if (_lockAspectRatio)
398  _size.setWidth(_size.height() * ratio);
399  }
400  layout();
401  }

447 void Image::layout()
448  {
449  setPos(0.0, 0.0);
450  if (imageType == ImageType::SVG && !svgDoc) {
451  if (_storeItem)
452  svgDoc = new QSvgRenderer(_storeItem->buffer());
453  }
454  else if (imageType == ImageType::RASTER && !rasterDoc) {
455  if (_storeItem) {
456  rasterDoc = new QImage;
457  rasterDoc->loadFromData(_storeItem->buffer());
458  if (!rasterDoc->isNull())
459  _dirty = true;
460  }
461  }
462  if (_size.isNull())
463  _size = pixel2size(imageSize());
464 
465  // if autoscale && inside a box, scale to box relevant size
466  if (autoScale() && parent() && ((parent()->isHBox() || parent()->isVBox()))) {
467  if (_lockAspectRatio) {
468  qreal f = _sizeIsSpatium ? spatium() : DPMM;
469  QSizeF size(imageSize());
470  qreal ratio = size.width() / size.height();
471  qreal w = parent()->width();
472  qreal h = parent()->height();
473  if ((w / h) < ratio) {
474  _size.setWidth(w / f);
475  _size.setHeight((w / ratio) / f);
476  }
477  else {
478  _size.setHeight(h / f);
479  _size.setWidth(h * ratio / f);
480  }
481  }
482  else
483  _size = pixel2size(parent()->bbox().size());
484  }
485 
486  // in any case, adjust position relative to parent
487  adjustReadPos();
488  setbbox(QRectF(QPointF(), size2pixel(_size)));
489  }
