53 void TestKeySig::keysig()
54  {
55  QString writeFile1("keysig01-test.mscx");
56  QString reference1(DIR + "keysig01-ref.mscx"); // with D maj
57  QString writeFile2("keysig02-test.mscx");
58  QString reference2(DIR + "keysig02-ref.mscx"); // with Eb maj
59  QString writeFile3("keysig03-test.mscx");
60  QString reference3(DIR + "keysig.mscx"); // orig
61  QString writeFile4("keysig04-test.mscx");
62  QString reference4(DIR + "keysig02-ref.mscx"); // with Eb maj
63  QString writeFile5("keysig05-test.mscx");
64  QString reference5(DIR + "keysig01-ref.mscx"); // with D maj
65  QString writeFile6("keysig06-test.mscx");
66  QString reference6(DIR + "keysig.mscx"); // orig
67 
68  // read file
69  MasterScore* score = readScore(DIR + "keysig.mscx");
70  Measure* m2 = score->firstMeasure()->nextMeasure();
71 
72  // add a key signature (D major) in measure 2
73  KeySigEvent ke2;
74  ke2.setKey(Key::D);
75  score->startCmd();
76  score->undoChangeKeySig(score->staff(0), m2->tick(), ke2);
77  score->endCmd();
78  QVERIFY(saveCompareScore(score, writeFile1, reference1));
79 
80  // change key signature in measure 2 to E flat major
81  KeySigEvent ke_3;
82  ke_3.setKey(Key(-3));
83  score->startCmd();
84  score->undoChangeKeySig(score->staff(0), m2->tick(), ke_3);
85  score->endCmd();
86  QVERIFY(saveCompareScore(score, writeFile2, reference2));
87 
88  // remove key signature in measure 2
89  Segment* s = m2->first();
90  while (!(s->isKeySigType()))
91  s = s->next();
92  Element* e = s->element(0);
93  score->startCmd();
94  score->undoRemoveElement(e);
95  score->endCmd();
96  QVERIFY(saveCompareScore(score, writeFile3, reference3));
97 
98  // undo remove
99  score->undoStack()->undo(&ed);
100  QVERIFY(saveCompareScore(score, writeFile4, reference4));
101 
102  // undo change
103  score->undoStack()->undo(&ed);
104  QVERIFY(saveCompareScore(score, writeFile5, reference5));
105 
106  // undo add
107  score->undoStack()->undo(&ed);
108  QVERIFY(saveCompareScore(score, writeFile6, reference6));
109 
110  delete score;
111  }

122 
123  Measure* m1 = score->firstMeasure();
