27 void MuseScore::showOmrPanel(bool visible)
28  {
29  QAction* a = getAction("omr");
30  if (visible) {
31  if (!omrPanel) {
32  omrPanel = new OmrPanel();
33  connect(omrPanel, SIGNAL(omrPanelVisible(bool)), a, SLOT(setChecked(bool)));
34  addDockWidget(Qt::RightDockWidgetArea, omrPanel);
35  }
36  }
37  if (omrPanel)
38  omrPanel->setVisible(visible);
39  if (omrPanel && visible) {
40  if (cv && cv->omrView())
41  omrPanel->setOmrView(cv->omrView());
42  else
43  omrPanel->setOmrView(0);
44  }
45  a->setChecked(visible);
46  }

166 void OmrPanel::setOmrView(OmrView* v)
167  {
168  omrView = v;
169  if (omrView) {
170  enableGui(true);
171  blockSignals(true);
172 
173  op.showBarlines->setChecked(omrView->showBarlines());
174  op.showLines->setChecked(omrView->showLines());
175  op.showSlices->setChecked(omrView->showSlices());
176  op.showStaves->setChecked(omrView->showStaves());
177  op.spatium->setValue(omrView->omr()->spatiumMM());
178 
179  blockSignals(false);
180  }
181  else {
182  enableGui(false);
183  }
184  }
