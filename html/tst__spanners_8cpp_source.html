86  QVERIFY(chord && chord->type() == ElementType::CHORD);

93  note->drop(dropData);
94

100  QVERIFY(seg);
101  chord = static_cast<Ms::Chord*>(seg->element(0)); // voice 0 of staff 0

259 void TestSpanners::spanners04()
260  {
261  MasterScore* score = readScore(DIR + "glissando-cloning01.mscx");
262  QVERIFY(score);
263 
264  // add a linked staff to the existing staff
265  // (copied and adapted from void MuseScore::editInstrList() in mscore/instrdialog.cpp)
266  Staff* oldStaff = score->staff(0);
267  Staff* newStaff = new Staff(score);
268  newStaff->setPart(oldStaff->part());
269  newStaff->initFromStaffType(oldStaff->staffType(0));
270  newStaff->setDefaultClefType(ClefTypeList(ClefType::G));
271 
272  KeySigEvent ke;
273  ke.setKey(Key::C);
274  newStaff->setKey(0, ke);
275 
276  score->undoInsertStaff(newStaff, 1, false);
277  Excerpt::cloneStaff(oldStaff, newStaff);
278 
279  QVERIFY(saveCompareScore(score, "glissando-cloning01.mscx", DIR + "glissando-cloning01-ref.mscx"));
280  delete score;
281  }
