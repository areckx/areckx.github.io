15 #
16 # This library file contains definitions of classes needed to decompose C
17 # source code files into a series of multi-line `blocks'. There are two
18 # kinds of blocks.
19 #
20 # - Normal blocks, which contain source code or ordinary comments.
21 #
22 # - Documentation blocks, which have restricted formatting, and whose text
23 # always start with a documentation markup tag like `<Function>',
24 # `<Type>', etc.
25 #
26 # The routines to process the content of documentation blocks are contained
27 # in file `content.py'; the classes and methods found here only deal with
28 # text parsing and basic documentation block extraction.
29 #

59 #
60 # Format 1 documentation comment blocks.
61 #
62 # /************************************/ (at least 2 asterisks)
63 # /* */
64 # /* */
65 # /* */
66 # /************************************/ (at least 2 asterisks)
67 #
68 start = r'''
69  \s* # any number of whitespace
70  /\*{2,}/ # followed by '/' and at least two asterisks then '/'
71  \s*$ # probably followed by whitespace
72 '''
73 
74 column = r'''
75  \s* # any number of whitespace
76  /\*{1} # followed by '/' and precisely one asterisk
77  ([^*].*) # followed by anything (group 1)
78  \*{1}/ # followed by one asterisk and a '/'
79  \s*$ # probably followed by whitespace
80 '''
81 
82 re_source_block_format1 = SourceBlockFormat( 1, start, column, start )
83 
84 
85 #
86 # Format 2 documentation comment blocks.
87 #
88 # /************************************ (at least 2 asterisks)
89 # *
90 # * (1 asterisk)
91 # *
92 # */ (1 or more asterisks)
93 #
94 start = r'''
95  \s* # any number of whitespace
96  /\*{2,} # followed by '/' and at least two asterisks
97  \s*$ # probably followed by whitespace
98 '''
99 
100 column = r'''
101  \s* # any number of whitespace
102  \*{1}(?![*/]) # followed by precisely one asterisk not followed by `/'
103  (.*) # then anything (group1)
104 '''
105 
106 end = r'''
107  \s* # any number of whitespace
108  \*+/ # followed by at least one asterisk, then '/'
109 '''
110 
111 re_source_block_format2 = SourceBlockFormat( 2, start, column, end )

121 #
122 # The following regular expressions correspond to markup tags within the
123 # documentation comment blocks. They are equivalent despite their different
124 # syntax.
125 #
126 # A markup tag consists of letters or character `-', to be found in group 1.
127 #
128 # Notice that a markup tag _must_ begin a new paragraph.
129 #
130 re_markup_tag1 = re.compile( r'''\s*<((?:\w|-)*)>''' ) # <xxxx> format
131 re_markup_tag2 = re.compile( r'''\s*@((?:\w|-)*):''' ) # @xxxx: format
132 
133 #
134 # The list of supported markup tags. We could add new ones quite easily.
135 #
136 re_markup_tags = [re_markup_tag1, re_markup_tag2]
137 
138 
139 #
140 # A regular expression to detect a cross reference, after markup tags have
141 # been stripped off. Group 1 is the reference, group 2 the rest of the
142 # line.
143 #
144 # A cross reference consists of letters, digits, or characters `-' and `_'.
145 #
146 re_crossref = re.compile( r'@((?:\w|-)*)(.*)' ) # @foo
147 
148 #
149 # Two regular expressions to detect italic and bold markup, respectively.
150 # Group 1 is the markup, group 2 the rest of the line.
151 #
152 # Note that the markup is limited to words consisting of letters, digits,
153 # the characters `_' and `-', or an apostrophe (but not as the first
154 # character).
155 #
156 re_italic = re.compile( r"_((?:\w|-)(?:\w|'|-)*)_(.*)" ) # _italic_
157 re_bold = re.compile( r"\*((?:\w|-)(?:\w|'|-)*)\*(.*)" ) # *bold*
158 
159 #
160 # This regular expression code to identify an URL has been taken from
161 #
162 # http://mail.python.org/pipermail/tutor/2002-September/017228.html
163 #
164 # (with slight modifications).
165 #
166 urls = r'(?:https?|telnet|gopher|file|wais|ftp)'
167 ltrs = r'\w'
168 gunk = r'/#~:.?+=&%@!\-'
169 punc = r'.:?\-'
170 any = "%(ltrs)s%(gunk)s%(punc)s" % { 'ltrs' : ltrs,
171  'gunk' : gunk,
172  'punc' : punc }
173 url = r"""
174  (
175  \b # start at word boundary
176  %(urls)s : # need resource and a colon
177  [%(any)s] +? # followed by one or more of any valid
178  # character, but be conservative and
179  # take only what you need to...
180  (?= # [look-ahead non-consumptive assertion]
181  [%(punc)s]* # either 0 or more punctuation
182  (?: # [non-grouping parentheses]
183  [^%(any)s] | $ # followed by a non-url char
184  # or end of the string
185  )
186  )
187  )
188  """ % {'urls' : urls,
189  'any' : any,
190  'punc' : punc }
191 
192 re_url = re.compile( url, re.VERBOSE | re.MULTILINE )
193 
194 #
195 # A regular expression that stops collection of comments for the current
196 # block.
197 #
198 re_source_sep = re.compile( r'\s*/\*\s*\*/' ) # /* */
199 
200 #
201 # A regular expression to find possible C identifiers while outputting
202 # source code verbatim, covering things like `*foo' or `(bar'. Group 1 is
203 # the prefix, group 2 the identifier -- since we scan lines from left to
204 # right, sequentially splitting the source code into prefix and identifier
205 # is fully sufficient for our purposes.
206 #
207 re_source_crossref = re.compile( r'(\W*)(\w*)' )
208 
209 #
210 # A regular expression that matches a list of reserved C source keywords.
211 #
212 re_source_keywords = re.compile( '''\\b ( typedef |
213  struct |
214  enum |
215  union |
216  const |
217  char |
218  int |
219  short |
220  long |
221  void |
222  signed |
223  unsigned |
224  \#include |
225  \#define |
226  \#undef |
227  \#if |
228  \#ifdef |
229  \#ifndef |
230  \#else |
231  \#endif ) \\b''', re.VERBOSE )

251  def __init__( self, processor, filename, lineno, lines ):
252  self.processor = processor
253  self.filename = filename
254  self.lineno = lineno
255  self.lines = lines[:]
256  self.format = processor.format
257  self.content = []
258 
259  if self.format == None:
260  return
261 
262  words = []
263 
264  # extract comment lines
265  lines = []
266 
267  for line0 in self.lines:
268  m = self.format.column.match( line0 )
269  if m:
270  lines.append( m.group( 1 ) )
271 
272  # now, look for a markup tag
273  for l in lines:
274  l = string.strip( l )
275  if len( l ) > 0:
276  for tag in re_markup_tags:
277  if tag.match( l ):
278  self.content = lines
279  return

317 
318  def __init__( self ):
319  """Initialize a source processor."""
320  self.blocks = []
321  self.filename = None
322  self.format = None
323  self.lines = []

325  def reset( self ):
326  """Reset a block processor and clean up all its blocks."""
327  self.blocks = []
328  self.format = None
329 
330  def parse_file( self, filename ):
331  """Parse a C source file and add its blocks to the processor's
332  list."""
333  self.reset()
334 
335  self.filename = filename
336 
337  fileinput.close()
338  self.format = None
339  self.lineno = 0
340  self.lines = []
341 
342  for line in fileinput.input( filename ):
343  # strip trailing newlines, important on Windows machines!
344  if line[-1] == '\012':
345  line = line[0:-1]
346 
347  if self.format == None:
348  self.process_normal_line( line )
349  else:
350  if self.format.end.match( line ):
351  # A normal block end. Add it to `lines' and create a
352  # new block
353  self.lines.append( line )
354  self.add_block_lines()
355  elif self.format.column.match( line ):
356  # A normal column line. Add it to `lines'.
357  self.lines.append( line )
358  else:
359  # An unexpected block end. Create a new block, but
360  # don't process the line.
361  self.add_block_lines()
362 
363  # we need to process the line again
364  self.process_normal_line( line )
365 
366  # record the last lines
367  self.add_block_lines()
368 
369  def process_normal_line( self, line ):
370  """Process a normal line and check whether it is the start of a new
371  block."""
372  for f in re_source_block_formats:
373  if f.start.match( line ):
374  self.add_block_lines()
375  self.format = f
376  self.lineno = fileinput.filelineno()
377 
378  self.lines.append( line )
379 
380  def add_block_lines( self ):
381  """Add the current accumulated lines and create a new block."""
382  if self.lines != []:
383  block = SourceBlock( self,
384  self.filename,
385  self.lineno,
386  self.lines )
387 
388  self.blocks.append( block )
389  self.format = None
390  self.lines = []
391 
392  # debugging only, not used in normal operations
393  def dump( self ):
394  """Print all blocks in a processor."""
395  for b in self.blocks:
396  b.dump()
