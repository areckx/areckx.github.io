57 // A score with:
58 // 3 staves,
59 // bracket across all 3 staves
60 // bar lines across all 3 staves
61 // systems with each staff hidden in turn because empty
62 // is loaded, laid out and bracket/bar line sizes are checked.

68 // actual 3-staff bracket should be high 28.6 SP ca.: allow for some layout margin
69 static const qreal BRACKET0_HEIGHT_MIN = 27;
70 static const qreal BRACKET0_HEIGHT_MAX = 30;
71 // actual 2-staff bracket should be high 18.1 SP ca.
72 static const qreal BRACKET_HEIGHT_MIN = 17;
73 static const qreal BRACKET_HEIGHT_MAX = 20;
74 // actual 3-staff bar line should be high 25 SP
75 static const qreal BARLINE0_HEIGHT_MIN = 24;
76 static const qreal BARLINE0_HEIGHT_MAX = 26;
77 // actual 2-staff bar line should be high 14.5 SP
78 static const qreal BARLINE_HEIGHT_MIN = 14;
79 static const qreal BARLINE_HEIGHT_MAX = 15;

81 void TestBarline::barline01()
82  {
83  char msg[256];
84  Score* score = readScore(DIR + "barline01.mscx");
85 
86  qreal height, heightMin, heightMax;
87  qreal spatium = score->spatium();
88  int sysNo = 0;
89  for (System* sys : score->systems()) {
90  // check number of the brackets of each system
91  sprintf(msg, "Wrong number of brackets in system %d.", sysNo+1);
92  QVERIFY2(sys->brackets().count() == 1, msg);
93 
94  // check height of the bracket of each system
95  // (bracket height is different between first system (3 staves) and other systems (2 staves) )
96  Bracket* bracket = sys->brackets().at(0);
97  height = bracket->bbox().height() / spatium;
98  heightMin = (sysNo == 0) ? BRACKET0_HEIGHT_MIN : BRACKET_HEIGHT_MIN;
99  heightMax = (sysNo == 0) ? BRACKET0_HEIGHT_MAX : BRACKET_HEIGHT_MAX;
100  sprintf(msg, "Wrong bracket height in system %d.", sysNo+1);
101  QVERIFY2(height > heightMin && height < heightMax, msg);
102 
103  // check presence and height of the bar line of each measure of each system
104  // (2 measure for each system)
105  heightMin = (sysNo == 0) ? BARLINE0_HEIGHT_MIN : BARLINE_HEIGHT_MIN;
106  heightMax = (sysNo == 0) ? BARLINE0_HEIGHT_MAX : BARLINE_HEIGHT_MAX;
107  for (int msrNo=0; msrNo < 2; ++msrNo) {
108  BarLine* bar = nullptr;
109  Measure* msr = toMeasure(sys->measure(msrNo));
110  Segment* seg = msr->findSegment(SegmentType::EndBarLine, msr->tick()+msr->ticks());
111  sprintf(msg, "No SegEndBarLine in measure %d of system %d.", msrNo+1, sysNo+1);
112  QVERIFY2(seg != nullptr, msg);
113 
114  bar = toBarLine(seg->element(0));
115  sprintf(msg, "No barline in measure %d of system %d.", msrNo+1, sysNo+1);
116  QVERIFY2(bar != nullptr, msg);
117 
118 #if 0 // not valid anymore
119  height = bar->bbox().height() / spatium;
120  sprintf(msg, "Wrong barline height %f %f %f in measure %d of system %d.",
121  heightMin, height, heightMax, msrNo+1, sysNo+1);
122  QVERIFY2(height > heightMin && height < heightMax, msg);
123 #endif
124  }
125  sysNo++;
126  }
127 
128 // QVERIFY(saveCompareScore(score, "barline01.mscx", DIR + "barline01-ref.mscx"));
129  delete score;
130  }

150

152  int msrNo = 1;

156  sprintf(msg, "No SegEndBarLine in measure %d.", msrNo);
157  QVERIFY2(seg != nullptr, msg);
158 
159  BarLine* bar = static_cast<BarLine*>(seg->element(0));
160  sprintf(msg, "No barline in measure %d.", msrNo);
161  QVERIFY2(bar != nullptr, msg);
162 
163  // bar line should be generated if NORMAL, except the END one at the end
164  sprintf(msg, "Barline in measure %d changed into 'non-generated'.", msrNo);
165 // ws: end barline is also generated
166 // bool test = (bar->barLineType() == BarLineType::NORMAL) ? bar->generated() : !bar->generated();

168  QVERIFY2(test, msg);

264  {
265  Score* score = readScore(DIR + "barline05.mscx");
266  QVERIFY(score);
267  score->doLayout();
268 
269  // 'go' to 4th measure
270  Measure* msr = score->firstMeasure();
271  for (int i=0; i < 3; i++)
272  msr = msr->nextMeasure();
273  // create and add a LineBreak element
274  LayoutBreak* lb = new LayoutBreak(score);
275  lb->setLayoutBreakType(LayoutBreak::Type::LINE);
276  lb->setTrack(-1); // system-level element
277  lb->setParent(msr);
278  score->undoAddElement(lb);
279  score->doLayout();
280 
281  // check an end-repeat bar line has been created at the end of this measure and it is generated
282  Segment* seg = msr->findSegment(SegmentType::EndBarLine, msr->tick()+msr->ticks());
283  QVERIFY2(seg != nullptr, "No SegEndBarLine segment in measure 4.");
284  BarLine* bar = static_cast<BarLine*>(seg->element(0));
285  QVERIFY2(bar != nullptr, "No end-repeat barline in measure 4.");
286  QVERIFY2(bar->barLineType() == BarLineType::END_REPEAT, "Barline at measure 4 is not END-REPEAT");
287  QVERIFY2(bar->generated(), "End-repeat barline in measure 4 is non-generated.");
288 
289  // // check an end-repeat bar line has been created at the beginning of the next measure and it is not generated
290  // check an end-repeat bar line has been created at the beginning of the next measure and it is generated
291  msr = msr->nextMeasure();
292  seg = msr->findSegment(SegmentType::StartRepeatBarLine, msr->tick());
293  QVERIFY2(seg != nullptr, "No SegStartRepeatBarLine segment in measure 5.");
294  bar = static_cast<BarLine*>(seg->element(0));
295  QVERIFY2(bar != nullptr, "No start-repeat barline in measure 5.");
296  QVERIFY2(bar->generated(), "Start-reapeat barline in measure 5 is not generated.");
297 
298 // QVERIFY(saveCompareScore(score, "barline05.mscx", DIR + "barline05-ref.mscx"));

311 void TestBarline::barline06()
312  {
313  char msg[256];
314  Score* score = readScore(DIR + "barline06.mscx");
315  QVERIFY(score);
316  score->doLayout();
317 
318  // scan each measure
319  Measure* msr = score->firstMeasure();
320  int msrNo = 1;
321  for (int i=0; i < 3; i++) {
322  // check measure endbarline type
323  sprintf(msg, "EndBarLineType not NORMAL in measure %d.", msrNo);
324 //TODO QVERIFY2(msr->endBarLineType() == BarLineType::NORMAL, msg);
325  // locate end-measure bar line segment
326  Segment* seg = msr->findSegment(SegmentType::EndBarLine, msr->tick()+msr->ticks());
327  sprintf(msg, "No SegEndBarLine in measure %d.", msr->no());
328  QVERIFY2(seg != nullptr, msg);
329 
330  // check only i-th staff has custom bar line type
331  for (int j=0; j < 3; j++) {
332  BarLine* bar = static_cast<BarLine*>(seg->element(j*VOICES));
333  // if not the i-th staff, bar should be normal and not custom
334  if (j != i) {
335  sprintf(msg, "barline type NOT NORMAL or CUSTOM TYPE in staff %d of measure %d.", j+1, msrNo);
336  QVERIFY2(bar->barLineType() == BarLineType::NORMAL, msg);
337 // QVERIFY2(bar->customSubtype() == false, msg);
338  }
339  // in the i-th staff, the bar line should be of type DOUBLE and custom type should be true
340  else {
341  sprintf(msg, "No barline for staff %d in measure %d", j+1, msrNo);
342  QVERIFY2(bar != nullptr, msg);
343  sprintf(msg, "barline type NOT DOUBLE or NOT CUSTOM TYPE in staff %d of measure %d.", j+1, msrNo);
344  QVERIFY2(bar->barLineType() == BarLineType::DOUBLE, msg);
345 // QVERIFY2(bar->customSubtype() == true, msg);
346  }
347  }
348 
349  msr = msr->nextMeasure();
350  msrNo++;
351  }
352 // QVERIFY(saveCompareScore(score, "barline06.mscx", DIR + "barline06-ref.mscx"));
353  delete score;
354  }

397
