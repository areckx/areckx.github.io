81 void PartEdit::setPart(Part* p, Channel* a)
82  {
83  Channel dummy;
84  channel = a;
85  part = p;
86  QString s = part->partName();
87  if (!a->name.isEmpty()) {
88  if (a->name != "normal") {
89  s += "-";
90  s += qApp->translate("InstrumentsXML", a->name.toUtf8().data());
91  }
92  }
93  partName->setText(s);
94  _setValue(volume, a->volume);
95  volume->setDclickValue1(dummy.volume);
96  volume->setDclickValue2(dummy.volume);
97  _setValue(reverb, a->reverb);
98  reverb->setDclickValue1(dummy.reverb);
99  reverb->setDclickValue2(dummy.reverb);
100  _setValue(chorus, a->chorus);
101 
102  chorus->setDclickValue1(dummy.chorus);
103  chorus->setDclickValue2(dummy.chorus);
104  _setValue(pan, a->pan);
105 
106  pan->setDclickValue1(0);
107  pan->setDclickValue2(0);
108  for (int i = 0; i < patch->count(); ++i) {
109  MidiPatch* p = (MidiPatch*)patch->itemData(i, Qt::UserRole).value<void*>();
110  if (a->synti == p->synti && a->program == p->prog && a->bank == p->bank) {
111  patch->setCurrentIndex(i);
112  break;
113  }
114  }
115  _setChecked(drumset, p->instrument()->useDrumset());
116  _setValue(portSpinBox, part->masterScore()->midiMapping(a->channel)->port + 1);
117  _setValue(channelSpinBox, part->masterScore()->midiMapping(a->channel)->channel + 1);
118 
119  QHBoxLayout* hb = voiceButtonBox;
120  int idx = 0;
121  for (Staff* staff : *part->staves()) {
122  for (int voice = 0; voice < VOICES; ++voice) {
123  if (!voiceButtons.value(idx)) {
124  QToolButton* tb = new QToolButton;
125  tb->setText(QString("%1").arg(voice+1));
126  tb->setCheckable(true);
127  tb->setChecked(staff->playbackVoice(voice));
128  tb->setFocusPolicy(Qt::ClickFocus);
129  tb->setToolButtonStyle(Qt::ToolButtonTextOnly);
130  QPalette p(tb->palette());
131  p.setColor(QPalette::Base, MScore::selectColor[voice]);
132  tb->setPalette(p);
133  hb->addWidget(tb);
134  voiceButtons.append(tb);
135  connect(tb, SIGNAL(clicked()), SLOT(playbackVoiceChanged()));
136  }
137  ++idx;
138  }
139  hb->addStretch(5);
140  }
141  while (voiceButtons.value(idx)) {
142  QToolButton* tb = voiceButtons.value(idx);
143  if (!tb)
144  break;
145  voiceButtons.takeAt(idx);
146  delete tb;
147  }
148  }

154 void PartEdit::playbackVoiceChanged()
155  {
156  int idx = 0;
157  Score* score = part->score();
158  score->startCmd();
159  for (Staff* staff : *part->staves()) {
160  for (int voice = 0; voice < VOICES; ++voice) {
161  QToolButton* tb = voiceButtons[idx];
162  bool val = tb->isChecked();
163  if (val != staff->playbackVoice(voice)) {
164  switch (voice) {
165  case 0:
166  printf("undo\n");
167  score->undoChangeProperty(staff, P_ID::PLAYBACK_VOICE1, val);
168  break;
169  case 1:
170  score->undoChangeProperty(staff, P_ID::PLAYBACK_VOICE2, val);
171  break;
172  case 2:
173  score->undoChangeProperty(staff, P_ID::PLAYBACK_VOICE3, val);
174  break;
175  case 3:
176  score->undoChangeProperty(staff, P_ID::PLAYBACK_VOICE4, val);
177  break;
178  }
179  }
180  ++idx;
181  }
182  }
183  score->endCmd();
184  }

281 void Mixer::updateAll(MasterScore* score)
282  {
283  cs = score;
284  int n = -vb->count();
285  if (n < 0) {
286  QLayoutItem* wi = vb->itemAt(vb->count()-1);
287  vb->removeItem(wi);
288  delete wi;
289  n += 1;
290  }
291  if (cs) {
292  QList<MidiMapping>* mm = cs->midiMapping();
293  n += mm->size();
294  }
295  while (n < 0) {
296  QLayoutItem* wi = vb->itemAt(0);
297  vb->removeItem(wi);
298  delete wi->widget();
299  delete wi;
300  ++n;
301  }
302  while (n > 0) {
303  PartEdit* pe = new PartEdit;
304  connect(pe, SIGNAL(soloChanged(bool)), SLOT(updateSolo(bool)));
305  vb->addWidget(pe);
306  --n;
307  }
308  vb->addStretch(5);
309  patchListChanged();
310  }

329 void Mixer::patchListChanged()
330  {
331  if (!cs)
332  return;
333 
334  int idx = 0;
335  QList<MidiMapping>* mm = cs->midiMapping();
336  const QList<MidiPatch*> pl = synti->getPatchInfo();
337  for (const MidiMapping& m : *mm) {
338  QLayoutItem* wi = (QWidgetItem*)(vb->itemAt(idx));
339  if (!wi->widget())
340  continue;
341  PartEdit* pe = (PartEdit*)(wi->widget());
342  bool drum = m.part->instrument()->useDrumset();
343  pe->patch->clear();
344  for (const MidiPatch* p : pl) {
345  if (p->drum == drum)
346  pe->patch->addItem(p->name, QVariant::fromValue<void*>((void*)p));
347  }
348  pe->setPart(m.part, m.articulation);
349  idx++;
350  }
351  // Update solo & mute only after creating all controls (we need to sync all controls)
352  idx = 0;
353  for (const MidiMapping& m : *mm) {
354  QLayoutItem* wi = (QWidgetItem*)(vb->itemAt(idx));
355  if (!wi->widget())
356  continue;
357  PartEdit* pe = (PartEdit*)(wi->widget());
358  pe->mute->setChecked(m.articulation->mute);
359  pe->solo->setChecked(m.articulation->solo);
360  idx++;
361  }
362  }

498 
499 void PartEdit::soloToggled(bool val, bool syncControls)
500  {
501  channel->solo = val;
502  channel->soloMute = !val;
503  if (val) {
504  mute->setChecked(false);
505  for (Part* p : part->score()->parts()) {
506  const InstrumentList* il = p->instruments();
507  for (auto i = il->begin(); i != il->end(); ++i) {
508  const Instrument* instr = i->second;
509  for (Channel* a : instr->channel()) {
510  a->soloMute = (channel != a && !a->solo);
511  a->solo = (channel == a || a->solo);
512  if (a->soloMute)
513  seq->stopNotes(a->channel);
514  }
515  }
516  }
517  emit soloChanged(true);
518  }
519  else { //do nothing except if it's the last solo to be switched off
520  bool found = false;
521  for (Part* p : part->score()->parts()) {
522  const InstrumentList* il = p->instruments();
523  for (auto i = il->begin(); i != il->end(); ++i) {
524  const Instrument* instr = i->second;
525  for (Channel* a : instr->channel()) {
526  if (a->solo){
527  found = true;
528  break;
529  }
530  }
531  }
532  }
533  if (!found){
534  foreach(Part* p, part->score()->parts()) {
535  const InstrumentList* il = p->instruments();
536  for (auto i = il->begin(); i != il->end(); ++i) {
537  const Instrument* instr = i->second;
538  for (Channel* a : instr->channel()) {
539  a->soloMute = false;
540  a->solo = false;
541  }
542  }
543  }
544  emit soloChanged(false);
545  }
546  }
547  sync(syncControls);
548  }

615 //---------------------------------------------------------
616 // sync
617 // synchronizes controls with same MIDI port and channel
618 //---------------------------------------------------------
619 
620 void PartEdit::sync(bool syncControls)
621  {
622  if (!syncControls)
623  return;
624  int count = this->parentWidget()->layout()->count();
625  for(int i = 0; i < count; i++) {
626  QWidgetItem* wi = (QWidgetItem*)(this->parentWidget()->layout()->itemAt(i));
627  PartEdit* pe = (PartEdit*)(wi->widget());
628  if (pe != 0 && pe != this
629  && this->channelSpinBox->value() == pe->channelSpinBox->value()
630  && this->portSpinBox->value() == pe->portSpinBox->value()) {
631 
632  if (volume->value() != pe->volume->value()) {
633  _setValue(pe->volume, this->volume->value());
634  emit pe->volChanged(this->volume->value(), false);
635  }
636  if (pan->value() != pe->pan->value()) {
637  _setValue(pe->pan, this->pan->value());
638  emit pe->panChanged(this->pan->value(), false);
639  }
640  if (reverb->value() != pe->reverb->value()) {
641  _setValue(pe->reverb, this->reverb->value());
642  emit pe->reverbChanged(this->reverb->value(), false);
643  }
644  if (chorus->value() != pe->chorus->value()) {
645  _setValue(pe->chorus, this->chorus->value());
646  emit pe->chorusChanged(this->chorus->value(), false);
647  }
648  if (mute->isChecked() != pe->mute->isChecked()) {
649  _setChecked(pe->mute, channel->mute);
650  emit pe->muteChanged(channel->mute, false);
651  }
652  if (solo->isChecked() != pe->solo->isChecked()) {
653  _setChecked(pe->solo, channel->solo);
654  emit pe->soloToggled(channel->solo, false);
655  }
656 
657  if (drumset->isChecked() != pe->drumset->isChecked()) {
658  _setChecked(pe->drumset, drumset->isChecked());
659  emit pe->drumsetToggled(drumset->isChecked(), false);
660  }
661  if (patch->currentIndex() != pe->patch->currentIndex()) {
662  pe->patch->blockSignals(true);
663  pe->patch->setCurrentIndex(this->patch->currentIndex());
664  pe->patch->blockSignals(false);
665  }
666  }
667  }
668  }

670 //---------------------------------------------------------
671 // midiChannelChanged
672 // handles MIDI port & channel change
673 //---------------------------------------------------------
674 
675 void PartEdit::midiChannelChanged(int)
676  {
677  if (part == 0)
678  return;
679  seq->stopNotes(channel->channel);
680  int p = portSpinBox->value() - 1;
681  int c = channelSpinBox->value() - 1;
682 
683  // 1 is for going up, -1 for going down
684  int direction = copysign(1, c - part->masterScore()->midiMapping(channel->channel)->channel);
685 
686  // Channel 9 is special for drums
687  if (part->instrument()->useDrumset() && c != 9) {
688  _setValue(channelSpinBox, 10);
689  return;
690  }
691  else if (!part->instrument()->useDrumset() && c == 9) {
692  c = 9 + direction;
693  }
694 
695  if (c == 16) {
696  c = 0;
697  p++;
698  }
699 
700  int newChannel = p*16+c;
701 
702  // If there is an instrument with the same MIDI port and channel, sync this instrument to a found one
703  bool needSync = true;
704  int elementsInMixer = parentWidget()->layout()->count();
705  for (int i = 0; i < elementsInMixer; i++) {
706  QWidgetItem* wi = (QWidgetItem*)(this->parentWidget()->layout()->itemAt(i));
707  PartEdit* pe = (PartEdit*)(wi->widget());
708  if (pe != 0 && pe != this
709  && pe->channelSpinBox->value() == this->channelSpinBox->value()
710  && pe->portSpinBox->value() == this->portSpinBox->value()) {
711  // Show datails if parameters are different
712  QString detailedText;
713  if (patch->currentIndex() != pe->patch->currentIndex())
714  detailedText += QString(tr("Sound: '%1'' vs '%2'\n")).arg(patch->itemText(patch->currentIndex()), pe->patch->itemText(pe->patch->currentIndex()));
715  if (this->volume->value() != pe->volume->value())
716  detailedText += QString(tr("Volume: %1 vs %2\n")).arg(QString::number(volume->value()), QString::number(pe->volume->value()));
717  if (this->pan->value() != pe->pan->value())
718  detailedText += QString(tr("Pan: %1 vs %2\n")).arg(QString::number(pan->value()), QString::number(pe->pan->value()));
719  if (this->reverb->value() != pe->reverb->value())
