42  e.setTrack(-1);
43  const QStringRef& tag(e.name());
44  if (tag == "Staff")
45  readStaff(e);
46  else if (tag == "Omr") {
47 #ifdef OMR
48  masterScore()->setOmr(new Omr(this));
49  masterScore()->omr()->read(e);
50 #else
51  e.skipCurrentElement();
52 #endif
53  }
54  else if (tag == "Audio") {
55  _audio = new Audio;
56  _audio->read(e);
57  }
58  else if (tag == "showOmr")
59  masterScore()->setShowOmr(e.readInt());
60  else if (tag == "playMode")
61  _playMode = PlayMode(e.readInt());
62  else if (tag == "LayerTag") {
63  int id = e.intAttribute("id");
64  const QString& tag = e.attribute("tag");
65  QString val(e.readElementText());
66  if (id >= 0 && id < 32) {
67  _layerTags[id] = tag;
68  _layerTagComments[id] = val;
69  }
70  }
71  else if (tag == "Layer") {
72  Layer layer;
73  layer.name = e.attribute("name");
74  layer.tags = e.attribute("mask").toUInt();
75  _layer.append(layer);
76  e.readNext();
77  }
78  else if (tag == "currentLayer")
79  _currentLayer = e.readInt();
80  else if (tag == "Synthesizer")
81  _synthesizerState.read(e);
82  else if (tag == "Division")
83  _fileDivision = e.readInt();
84  else if (tag == "showInvisible")
85  _showInvisible = e.readInt();
86  else if (tag == "showUnprintable")
87  _showUnprintable = e.readInt();
88  else if (tag == "showFrames")
89  _showFrames = e.readInt();
90  else if (tag == "showMargins")
91  _showPageborders = e.readInt();
92  else if (tag == "Style") {
93  qreal sp = style().value(StyleIdx::spatium).toDouble();
94  style().load(e);
95  // if (_layoutMode == LayoutMode::FLOAT || _layoutMode == LayoutMode::SYSTEM) {
96  if (_layoutMode == LayoutMode::FLOAT) {
97  // style should not change spatium in
98  // float mode
99  style().set(StyleIdx::spatium, sp);
100  }
101  _scoreFont = ScoreFont::fontFactory(style().value(StyleIdx::MusicalSymbolFont).toString());
102  }
103  else if (tag == "copyright" || tag == "rights") {
104  Text* text = new Text(this);
105  text->read(e);
106  setMetaTag("copyright", text->xmlText());
107  delete text;
108  }
109  else if (tag == "movement-number")
110  setMetaTag("movementNumber", e.readElementText());
111  else if (tag == "movement-title")
112  setMetaTag("movementTitle", e.readElementText());
113  else if (tag == "work-number")
114  setMetaTag("workNumber", e.readElementText());
115  else if (tag == "work-title")
116  setMetaTag("workTitle", e.readElementText());
117  else if (tag == "source")
118  setMetaTag("source", e.readElementText());
119  else if (tag == "metaTag") {
120  QString name = e.attribute("name");
121  setMetaTag(name, e.readElementText());
122  }
123  else if (tag == "Part") {
124  Part* part = new Part(this);
125  part->read(e);
126  _parts.push_back(part);
127  }
128  else if ((tag == "HairPin")
129  || (tag == "Ottava")
130  || (tag == "TextLine")
131  || (tag == "Volta")
132  || (tag == "Trill")
133  || (tag == "Slur")
134  || (tag == "Pedal")) {
135  Spanner* s = static_cast<Spanner*>(Element::name2Element(tag, this));
136  s->read(e);
137  addSpanner(s);
138  }
139  else if (tag == "Excerpt") {
140  if (MScore::noExcerpts)
141  e.skipCurrentElement();
142  else {
143  if (isMaster()) {
144  Excerpt* ex = new Excerpt(static_cast<MasterScore*>(this));
145  ex->read(e);
146  excerpts().append(ex);
147  }
148  else {
149  qDebug("Score::read(): part cannot have parts");
150  e.skipCurrentElement();
151  }
152  }
153  }
154  else if (e.name() == "Tracklist") {
155  int strack = e.intAttribute("sTrack", -1);
156  int dtrack = e.intAttribute("dstTrack", -1);
157  if (strack != -1 && dtrack != -1)
158  e.tracks().insert(strack, dtrack);
159  e.skipCurrentElement();
160  }
161  else if (tag == "Score") { // recursion
162  if (MScore::noExcerpts)
163  e.skipCurrentElement();
164  else {
165  e.tracks().clear(); // ???
166  MasterScore* m = masterScore();
167  Score* s = new Score(m);
168  Excerpt* ex = new Excerpt(m);
169 
170  ex->setPartScore(s);
171  ex->setTracks(e.tracks());
172  e.setLastMeasure(nullptr);
173  s->read(e);
174  m->addExcerpt(ex);
175  }
176  }
177  else if (tag == "name") {
178  QString n = e.readElementText();
179  if (!isMaster()) //ignore the name if it's not a child score
180  excerpt()->setTitle(n);
181  }
182  else if (tag == "layoutMode") {
183  QString s = e.readElementText();
184  if (s == "line")
185  _layoutMode = LayoutMode::LINE;
186  else if (s == "system")
187  _layoutMode = LayoutMode::SYSTEM;
188  else
189  qDebug("layoutMode: %s", qPrintable(s));
190  }
191  else
192  e.unknown();
193  }
194  if (e.error() != QXmlStreamReader::NoError) {
195  qDebug("%s: xml read error at line %lld col %lld: %s",
196  qPrintable(e.getDocName()), e.lineNumber(), e.columnNumber(),
197  e.name().toUtf8().data());
198  MScore::lastError = QObject::tr("XML read error at line %1 column %2: %3").arg(e.lineNumber()).arg(e.columnNumber()).arg(e.name().toString());
199  return false;
200  }
201 
202  connectTies();
203 
204  _fileDivision = MScore::division;
205 
206 #if 0 // TODO:barline
207  //
208  // sanity check for barLineSpan
209  //
210  for (Staff* st : staves()) {
211  int barLineSpan = st->barLineSpan();
212  int idx = st->idx();
213  int n = nstaves();
214  if (idx + barLineSpan > n) {
215  qDebug("bad span: idx %d span %d staves %d", idx, barLineSpan, n);
216  // span until last staff
217  barLineSpan = n - idx;
218  st->setBarLineSpan(barLineSpan);
219  }
220  else if (idx == 0 && barLineSpan == 0) {
221  qDebug("bad span: idx %d span %d staves %d", idx, barLineSpan, n);
222  // span from the first staff until the start of the next span
223  barLineSpan = 1;
224  for (int i = 1; i < n; ++i) {
225  if (staff(i)->barLineSpan() == 0)
226  ++barLineSpan;
227  else
228  break;
229  }
230  st->setBarLineSpan(barLineSpan);
231  }
232  // check spanFrom
233  int minBarLineFrom = st->lines(0) == 1 ? BARLINE_SPAN_1LINESTAFF_FROM : MIN_BARLINE_SPAN_FROMTO;
234  if (st->barLineFrom() < minBarLineFrom)
235  st->setBarLineFrom(minBarLineFrom);
236  if (st->barLineFrom() > st->lines(0) * 2)
237  st->setBarLineFrom(st->lines(0) * 2);
238  // check spanTo
239  Staff* stTo = st->barLineSpan() <= 1 ? st : staff(idx + st->barLineSpan() - 1);
240  // 1-line staves have special bar line spans
241  int maxBarLineTo = stTo->lines(0) == 1 ? BARLINE_SPAN_1LINESTAFF_TO : stTo->lines(0) * 2;
242  if (st->barLineTo() < MIN_BARLINE_SPAN_FROMTO)
243  st->setBarLineTo(MIN_BARLINE_SPAN_FROMTO);
244  if (st->barLineTo() > maxBarLineTo)
245  st->setBarLineTo(maxBarLineTo);
246  // on single staff span, check spanFrom and spanTo are distant enough
247  if (st->barLineSpan() == 1) {
248  if (st->barLineTo() - st->barLineFrom() < MIN_BARLINE_FROMTO_DIST) {
249  st->setBarLineFrom(0);
250  st->setBarLineTo(0);
251  }
252  }
253  }
254 #endif
255 
256  if (!masterScore()->omr())
257  masterScore()->setShowOmr(false);
258 
259  fixTicks();
260  masterScore()->rebuildMidiMapping();
261  masterScore()->updateChannel();
262  createPlayEvents();
263  return true;
264  }

305 Score::FileError MasterScore::read300(XmlReader& e)
306  {
307  bool top = true;
308  while (e.readNextStartElement()) {
309  const QStringRef& tag(e.name());
310  if (tag == "programVersion") {
311  setMscoreVersion(e.readElementText());
312  parseVersion(mscoreVersion());
313  }
314  else if (tag == "programRevision")
315  setMscoreRevision(e.readIntHex());
316  else if (tag == "Score") {
317  MasterScore* score;
318  if (top) {
319  score = this;
320  top = false;
321  }
322  else {
323  score = new MasterScore();
324  addMovement(score);
325  }
326  if (!score->read(e))
327  return FileError::FILE_BAD_FORMAT;
328  }
329  else if (tag == "Revision") {
330  Revision* revision = new Revision;
331  revision->read(e);
332  revisions()->add(revision);
333  }
334  }
335  return FileError::FILE_NO_ERROR;
336  }
