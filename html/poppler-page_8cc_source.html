73 Link* PageData::convertLinkActionToLink(::LinkAction * a, DocumentData *parentDoc, const QRectF &linkArea)
74 {
75  if ( !a )
76  return NULL;
77 
78  Link * popplerLink = NULL;
79  switch ( a->getKind() )
80  {
81  case actionGoTo:
82  {
83  LinkGoTo * g = (LinkGoTo *) a;
84  const LinkDestinationData ldd( g->getDest(), g->getNamedDest(), parentDoc, false );
85  // create link: no ext file, namedDest, object pointer
86  popplerLink = new LinkGoto( linkArea, QString::null, LinkDestination( ldd ) );
87  }
88  break;
89 
90  case actionGoToR:
91  {
92  LinkGoToR * g = (LinkGoToR *) a;
93  // copy link file
94  const QString fileName = UnicodeParsedString( g->getFileName() );
95  const LinkDestinationData ldd( g->getDest(), g->getNamedDest(), parentDoc, !fileName.isEmpty() );
96  // ceate link: fileName, namedDest, object pointer
97  popplerLink = new LinkGoto( linkArea, fileName, LinkDestination( ldd ) );
98  }
99  break;
100 
101  case actionLaunch:
102  {
103  LinkLaunch * e = (LinkLaunch *)a;
104  GooString * p = e->getParams();
105  popplerLink = new LinkExecute( linkArea, e->getFileName()->getCString(), p ? p->getCString() : 0 );
106  }
107  break;
108 
109  case actionNamed:
110  {
111  const char * name = ((LinkNamed *)a)->getName()->getCString();
112  if ( !strcmp( name, "NextPage" ) )
113  popplerLink = new LinkAction( linkArea, LinkAction::PageNext );
114  else if ( !strcmp( name, "PrevPage" ) )
115  popplerLink = new LinkAction( linkArea, LinkAction::PagePrev );
116  else if ( !strcmp( name, "FirstPage" ) )
117  popplerLink = new LinkAction( linkArea, LinkAction::PageFirst );
118  else if ( !strcmp( name, "LastPage" ) )
119  popplerLink = new LinkAction( linkArea, LinkAction::PageLast );
120  else if ( !strcmp( name, "GoBack" ) )
121  popplerLink = new LinkAction( linkArea, LinkAction::HistoryBack );
122  else if ( !strcmp( name, "GoForward" ) )
123  popplerLink = new LinkAction( linkArea, LinkAction::HistoryForward );
124  else if ( !strcmp( name, "Quit" ) )
125  popplerLink = new LinkAction( linkArea, LinkAction::Quit );
126  else if ( !strcmp( name, "GoToPage" ) )
127  popplerLink = new LinkAction( linkArea, LinkAction::GoToPage );
128  else if ( !strcmp( name, "Find" ) )
129  popplerLink = new LinkAction( linkArea, LinkAction::Find );
130  else if ( !strcmp( name, "FullScreen" ) )
131  popplerLink = new LinkAction( linkArea, LinkAction::Presentation );
132  else if ( !strcmp( name, "Print" ) )
133  popplerLink = new LinkAction( linkArea, LinkAction::Print );
134  else if ( !strcmp( name, "Close" ) )
135  {
136  // acroread closes the document always, doesnt care whether
137  // its presentation mode or not
138  // popplerLink = new LinkAction( linkArea, LinkAction::EndPresentation );
139  popplerLink = new LinkAction( linkArea, LinkAction::Close );
140  }
141  else
142  {
143  // TODO
144  }
145  }
146  break;
147 
148  case actionURI:
149  {
150  popplerLink = new LinkBrowse( linkArea, ((LinkURI *)a)->getURI()->getCString() );
151  }
152  break;
153 
154  case actionSound:
155  {
156  ::LinkSound *ls = (::LinkSound *)a;
157  popplerLink = new LinkSound( linkArea, ls->getVolume(), ls->getSynchronous(), ls->getRepeat(), ls->getMix(), new SoundObject( ls->getSound() ) );
158  }
159  break;
160 
161  case actionJavaScript:
162  {
163  ::LinkJavaScript *ljs = (::LinkJavaScript *)a;
164  popplerLink = new LinkJavaScript( linkArea, UnicodeParsedString(ljs->getScript()) );
165  }
166  break;
167 
168  case actionMovie:
169  {
170  ::LinkMovie *lm = (::LinkMovie *)a;
171 
172  const QString title = ( lm->hasAnnotTitle() ? UnicodeParsedString( lm->getAnnotTitle() ) : QString() );
173 
174  Ref reference;
175  reference.num = reference.gen = -1;
176  if ( lm->hasAnnotRef() )
177  reference = *lm->getAnnotRef();
178 
179  LinkMovie::Operation operation = LinkMovie::Play;
180  switch ( lm->getOperation() )
181  {
182  case ::LinkMovie::operationTypePlay:
183  operation = LinkMovie::Play;
184  break;
185  case ::LinkMovie::operationTypePause:
186  operation = LinkMovie::Pause;
187  break;
188  case ::LinkMovie::operationTypeResume:
189  operation = LinkMovie::Resume;
190  break;
191  case ::LinkMovie::operationTypeStop:
192  operation = LinkMovie::Stop;
193  break;
194  };
195 
196  popplerLink = new LinkMovie( linkArea, operation, title, reference );
197  }
198  break;
199 
200  case actionRendition:
201  {
202  ::LinkRendition *lrn = (::LinkRendition *)a;
203 
204  Ref reference;
205  reference.num = reference.gen = -1;
206  if ( lrn->hasScreenAnnot() )
207  reference = lrn->getScreenAnnot();
208 
209  popplerLink = new LinkRendition( linkArea, lrn->getMedia() ? lrn->getMedia()->copy() : NULL, lrn->getOperation(), UnicodeParsedString( lrn->getScript() ), reference );
210  }
211  break;
212 
213  case actionOCGState:
214  break;
215 
216  case actionUnknown:
217  break;
218  }
219 
220  return popplerLink;
221 }

223 inline TextPage *PageData::prepareTextSearch(const QString &text, Page::Rotation rotate, QVector<Unicode> *u)
224 {
225  const QChar * str = text.unicode();
226  const int len = text.length();
227  u->resize(len);
228  for (int i = 0; i < len; ++i) (*u)[i] = str[i].unicode();
229 
230  const int rotation = (int)rotate * 90;
231 
232  // fetch ourselves a textpage
233  TextOutputDev td(NULL, gTrue, 0, gFalse, gFalse);
234  parentDoc->doc->displayPage( &td, index + 1, 72, 72, rotation, false, true, false,
235  NULL, NULL, NULL, NULL, gTrue);
236  TextPage *textPage=td.takeText();
237 
238  return textPage;
239 }

241 inline GBool PageData::performSingleTextSearch(TextPage* textPage, QVector<Unicode> &u, double &sLeft, double &sTop, double &sRight, double &sBottom, Page::SearchDirection direction, GBool sCase, GBool sWords)
242 {
243  if (direction == Page::FromTop)
244  return textPage->findText( u.data(), u.size(),
245  gTrue, gTrue, gFalse, gFalse, sCase, gFalse, sWords, &sLeft, &sTop, &sRight, &sBottom );
246  else if ( direction == Page::NextResult )
247  return textPage->findText( u.data(), u.size(),
248  gFalse, gTrue, gTrue, gFalse, sCase, gFalse, sWords, &sLeft, &sTop, &sRight, &sBottom );
249  else if ( direction == Page::PreviousResult )
250  return textPage->findText( u.data(), u.size(),
251  gFalse, gTrue, gTrue, gFalse, sCase, gTrue, sWords, &sLeft, &sTop, &sRight, &sBottom );
252 
253  return gFalse;
254 }

256 inline QList<QRectF> PageData::performMultipleTextSearch(TextPage* textPage, QVector<Unicode> &u, GBool sCase, GBool sWords)
257 {
258  QList<QRectF> results;
259  double sLeft = 0.0, sTop = 0.0, sRight = 0.0, sBottom = 0.0;
260 
261  while(textPage->findText( u.data(), u.size(),
262  gFalse, gTrue, gTrue, gFalse, sCase, gFalse, sWords, &sLeft, &sTop, &sRight, &sBottom ))
263  {
264  QRectF result;
265 
266  result.setLeft(sLeft);
267  result.setTop(sTop);
268  result.setRight(sRight);
269  result.setBottom(sBottom);
270 
271  results.append(result);
272  }
273 
274  return results;

291 QImage Page::renderToImage(double xres, double yres, int x, int y, int w, int h, Rotation rotate) const
292 {
293  int rotation = (int)rotate * 90;
294  QImage img;
295  switch(m_page->parentDoc->m_backend)
296  {
297  case Poppler::Document::SplashBackend:
298  {
299 #if defined(HAVE_SPLASH)
300  SplashColor bgColor;
301  GBool overprintPreview = gFalse;
302 #if SPLASH_CMYK
303  overprintPreview = m_page->parentDoc->m_hints & Document::OverprintPreview ? gTrue : gFalse;
304  if (overprintPreview)
305  {
306  Guchar c, m, y, k;
307 
308  c = 255 - m_page->parentDoc->paperColor.blue();
309  m = 255 - m_page->parentDoc->paperColor.red();
310  y = 255 - m_page->parentDoc->paperColor.green();
311  k = c;
312  if (m < k) {
313  k = m;
314  }
315  if (y < k) {
316  k = y;
317  }
318  bgColor[0] = c - k;
319  bgColor[1] = m - k;
320  bgColor[2] = y - k;
321  bgColor[3] = k;
322  for (int i = 4; i < SPOT_NCOMPS + 4; i++) {
323  bgColor[i] = 0;
324  }
325  }
326  else
327 #endif
328  {
329  bgColor[0] = m_page->parentDoc->paperColor.blue();
330  bgColor[1] = m_page->parentDoc->paperColor.green();
331  bgColor[2] = m_page->parentDoc->paperColor.red();
332  }
333 
334  SplashColorMode colorMode = splashModeXBGR8;
335 #if SPLASH_CMYK
336  if (overprintPreview) colorMode = splashModeDeviceN8;
337 #endif
338 
339  SplashThinLineMode thinLineMode = splashThinLineDefault;
340  if (m_page->parentDoc->m_hints & Document::ThinLineShape) thinLineMode = splashThinLineShape;
341  if (m_page->parentDoc->m_hints & Document::ThinLineSolid) thinLineMode = splashThinLineSolid;
342 
343  const bool ignorePaperColor = m_page->parentDoc->m_hints & Document::IgnorePaperColor;
344 
345  SplashOutputDev splash_output(
346  colorMode, 4,
347  gFalse,
348  ignorePaperColor ? NULL : bgColor,
349  gTrue,
350  thinLineMode,
351  overprintPreview);
352 
353  splash_output.setFontAntialias(m_page->parentDoc->m_hints & Document::TextAntialiasing ? gTrue : gFalse);
354  splash_output.setVectorAntialias(m_page->parentDoc->m_hints & Document::Antialiasing ? gTrue : gFalse);
355  splash_output.setFreeTypeHinting(m_page->parentDoc->m_hints & Document::TextHinting ? gTrue : gFalse,
356  m_page->parentDoc->m_hints & Document::TextSlightHinting ? gTrue : gFalse);
357 
358  splash_output.startDoc(m_page->parentDoc->doc);
359 
360  m_page->parentDoc->doc->displayPageSlice(&splash_output, m_page->index + 1, xres, yres,
361  rotation, false, true, false, x, y, w, h,
362  NULL, NULL, NULL, NULL, gTrue);
363 
364  SplashBitmap *bitmap = splash_output.getBitmap();
365 
366  const int bw = bitmap->getWidth();
367  const int bh = bitmap->getHeight();
368  const int brs = bitmap->getRowSize();
369 
370  // If we use DeviceN8, convert to XBGR8.
371  // If requested, also transfer Splash's internal alpha channel.
372  const SplashBitmap::ConversionMode mode = ignorePaperColor
373  ? SplashBitmap::conversionAlphaPremultiplied
374  : SplashBitmap::conversionOpaque;
375 
376  const QImage::Format format = ignorePaperColor
377  ? QImage::Format_ARGB32_Premultiplied
378  : QImage::Format_RGB32;
379 
380  if (bitmap->convertToXBGR(mode)) {
381  SplashColorPtr data = bitmap->takeData();
382 
383  if (QSysInfo::ByteOrder == QSysInfo::BigEndian) {
384  // Convert byte order from RGBX to XBGR.
385  for (int i = 0; i < bh; ++i) {
386  for (int j = 0; j < bw; ++j) {
387  SplashColorPtr pixel = &data[i * brs + j];
388 
389  qSwap(pixel[0], pixel[3]);
390  qSwap(pixel[1], pixel[2]);
391  }
392  }
393  }
394 
395  // Construct a Qt image holding (and also owning) the raw bitmap data.
396  img = QImage(data, bw, bh, brs, format, gfree, data);
397  }
398 #endif
399  break;
400  }
401  case Poppler::Document::ArthurBackend:
402  {
403  QSize size = pageSize();
404  QImage tmpimg(w == -1 ? qRound( size.width() * xres / 72.0 ) : w, h == -1 ? qRound( size.height() * yres / 72.0 ) : h, QImage::Format_ARGB32);
405 
406  QPainter painter(&tmpimg);
407  renderToPainter(&painter, xres, yres, x, y, w, h, rotate, DontSaveAndRestore);
408  painter.end();
409  img = tmpimg;
410  break;
411  }
412  }
413 
414  return img;
415 }

417 bool Page::renderToPainter(QPainter* painter, double xres, double yres, int x, int y, int w, int h, Rotation rotate, PainterFlags flags) const
418 {
419  if (!painter)
420  return false;
421 
422  switch(m_page->parentDoc->m_backend)
423  {
424  case Poppler::Document::SplashBackend:
425  return false;
426  case Poppler::Document::ArthurBackend:
427  {
428  const bool savePainter = !(flags & DontSaveAndRestore);
429  if (savePainter)
430  painter->save();
431  if (m_page->parentDoc->m_hints & Document::Antialiasing)
432  painter->setRenderHint(QPainter::Antialiasing);
433  if (m_page->parentDoc->m_hints & Document::TextAntialiasing)
434  painter->setRenderHint(QPainter::TextAntialiasing);
435  painter->translate(x == -1 ? 0 : -x, y == -1 ? 0 : -y);
436  ArthurOutputDev arthur_output(painter);
437  arthur_output.startDoc(m_page->parentDoc->doc->getXRef());
438  m_page->parentDoc->doc->displayPageSlice(&arthur_output,
439  m_page->index + 1,
440  xres,
441  yres,
442  (int)rotate * 90,
443  false,
444  true,
445  false,
446  x,
447  y,
448  w,
449  h);
450  if (savePainter)
451  painter->restore();
452  return true;
453  }
454  }
455  return false;
456 }

568 QList<TextBox*> Page::textList(Rotation rotate) const
569 {
570  TextOutputDev *output_dev;
571 
572  QList<TextBox*> output_list;
573 
574  output_dev = new TextOutputDev(0, gFalse, 0, gFalse, gFalse);
575 
576  int rotation = (int)rotate * 90;
577 
578  m_page->parentDoc->doc->displayPageSlice(output_dev, m_page->index + 1, 72, 72,
579  rotation, false, false, false, -1, -1, -1, -1,
580  NULL, NULL, NULL, NULL, gTrue);
581 
582  TextWordList *word_list = output_dev->makeWordList();
583 
584  if (!word_list) {
585  delete output_dev;
586  return output_list;
587  }
588 
589  QHash<TextWord *, TextBox*> wordBoxMap;
590 
591  output_list.reserve(word_list->getLength());
592  for (int i = 0; i < word_list->getLength(); i++) {
593  TextWord *word = word_list->get(i);
594  GooString *gooWord = word->getText();
595  QString string = QString::fromUtf8(gooWord->getCString());
596  delete gooWord;
597  double xMin, yMin, xMax, yMax;
598  word->getBBox(&xMin, &yMin, &xMax, &yMax);
599 
600  TextBox* text_box = new TextBox(string, QRectF(xMin, yMin, xMax-xMin, yMax-yMin));
601  text_box->m_data->hasSpaceAfter = word->hasSpaceAfter() == gTrue;
602  text_box->m_data->charBBoxes.reserve(word->getLength());
603  for (int j = 0; j < word->getLength(); ++j)
604  {
605  word->getCharBBox(j, &xMin, &yMin, &xMax, &yMax);
606  text_box->m_data->charBBoxes.append(QRectF(xMin, yMin, xMax-xMin, yMax-yMin));
607  }
608 
609  wordBoxMap.insert(word, text_box);
610 
611  output_list.append(text_box);
612  }
613 
614  for (int i = 0; i < word_list->getLength(); i++) {
