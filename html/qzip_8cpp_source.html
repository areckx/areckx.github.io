138 static void writeMSDosDate(uchar *dest, const QDateTime& dt)
139 {
140  if (dt.isValid()) {
141  quint16 time =
142  (dt.time().hour() << 11) // 5 bit hour
143  | (dt.time().minute() << 5) // 6 bit minute
144  | (dt.time().second() >> 1); // 5 bit double seconds
145 
146  dest[0] = time & 0xff;
147  dest[1] = time >> 8;
148 
149  quint16 date =
150  ((dt.date().year() - 1980) << 9) // 7 bit year 1980-based
151  | (dt.date().month() << 5) // 4 bit month
152  | (dt.date().day()); // 5 bit day
153 
154  dest[2] = char(date);
155  dest[3] = char(date >> 8);
156  } else {
157  dest[0] = 0;
158  dest[1] = 0;
159  dest[2] = 0;
160  dest[3] = 0;
161  }
162 }

164 static quint32 permissionsToMode(QFile::Permissions perms)
165 {
166  quint32 mode = 0;
167  if (perms & QFile::ReadOwner)
168  mode |= S_IRUSR;
169  if (perms & QFile::WriteOwner)

194 static int inflate(Bytef *dest, ulong *destLen, const Bytef *source, ulong sourceLen)
195 {
196  z_stream stream;
197  int err;
198 
199  stream.next_in = (Bytef*)source;
200  stream.avail_in = (uInt)sourceLen;
201  if ((uLong)stream.avail_in != sourceLen)
202  return Z_BUF_ERROR;
203 
204  stream.next_out = dest;
205  stream.avail_out = (uInt)*destLen;
206  if ((uLong)stream.avail_out != *destLen)
207  return Z_BUF_ERROR;
208 
209  stream.zalloc = (alloc_func)0;
210  stream.zfree = (free_func)0;
211 
212  err = inflateInit2(&stream, -MAX_WBITS);
213  if (err != Z_OK)
214  return err;
215 
216  err = inflate(&stream, Z_FINISH);
217  if (err != Z_STREAM_END) {
218  inflateEnd(&stream);
219  if (err == Z_NEED_DICT || (err == Z_BUF_ERROR && stream.avail_in == 0))
220  return Z_DATA_ERROR;
221  return err;
222  }
223  *destLen = stream.total_out;
224 
225  err = inflateEnd(&stream);
226  return err;
227 }

229 static int deflate (Bytef *dest, ulong *destLen, const Bytef *source, ulong sourceLen)
230 {
231  z_stream stream;
232  int err;
233 
234  stream.next_in = (Bytef*)source;
235  stream.avail_in = (uInt)sourceLen;
236  stream.next_out = dest;
237  stream.avail_out = (uInt)*destLen;
238  if ((uLong)stream.avail_out != *destLen) return Z_BUF_ERROR;
239 
240  stream.zalloc = (alloc_func)0;
241  stream.zfree = (free_func)0;
242  stream.opaque = (voidpf)0;
243 
244  err = deflateInit2(&stream, Z_DEFAULT_COMPRESSION, Z_DEFLATED, -MAX_WBITS, 8, Z_DEFAULT_STRATEGY);
245  if (err != Z_OK) return err;
246 
247  err = deflate(&stream, Z_FINISH);
248  if (err != Z_STREAM_END) {
249  deflateEnd(&stream);
250  return err == Z_OK ? Z_BUF_ERROR : err;
251  }
252  *destLen = stream.total_out;
253 
254  err = deflateEnd(&stream);
255  return err;
256 }

288 static QDateTime readMSDosDate(const uchar *src)
289 {
290  uint dosDate = readUInt(src);
291  quint64 uDate;
292  uDate = (quint64)(dosDate >> 16);
293  uint tm_mday = (uDate & 0x1f);
294  uint tm_mon = ((uDate & 0x1E0) >> 5);
295  uint tm_year = (((uDate & 0x0FE00) >> 9) + 1980);
296  uint tm_hour = ((dosDate & 0xF800) >> 11);
297  uint tm_min = ((dosDate & 0x7E0) >> 5);
298  uint tm_sec = ((dosDate & 0x1f) << 1);
299 
300  return QDateTime(QDate(tm_year, tm_mon, tm_mday), QTime(tm_hour, tm_min, tm_sec));
301 }

484 void MQZipReaderPrivate::scanFiles()
485 {
486  if (!dirtyFileTree)
487  return;
488 
489  if (! (device->isOpen() || device->open(QIODevice::ReadOnly))) {
490  status = MQZipReader::FileOpenError;
491  return;
492  }
493 
494  if ((device->openMode() & QIODevice::ReadOnly) == 0) { // only read the index from readable files.
495  status = MQZipReader::FileReadError;
496  return;
497  }
498 
499  dirtyFileTree = false;
500  uchar tmp[4];
501  device->read((char *)tmp, 4);
502  if (readUInt(tmp) != 0x04034b50) {
503  qWarning() << "QZip: not a zip file!";
504  return;
505  }
506 
507  // find EndOfDirectory header
508  int i = 0;
509  int start_of_directory = -1;
510  int num_dir_entries = 0;
511  EndOfDirectory eod;
512  while (start_of_directory == -1) {
513  int pos = device->size() - sizeof(EndOfDirectory) - i;
514  if (pos < 0 || i > 65535) {
515  qWarning() << "QZip: EndOfDirectory not found";
516  return;
517  }
518 
519  device->seek(pos);
520  device->read((char *)&eod, sizeof(EndOfDirectory));
521  if (readUInt(eod.signature) == 0x06054b50)
522  break;
523  ++i;
524  }
525 
526  // have the eod
527  start_of_directory = readUInt(eod.dir_start_offset);
528  num_dir_entries = readUShort(eod.num_dir_entries);
529  ZDEBUG("start_of_directory at %d, num_dir_entries=%d", start_of_directory, num_dir_entries);
530  int comment_length = readUShort(eod.comment_length);
531  if (comment_length != i)
532  qWarning() << "QZip: failed to parse zip file.";
533  comment = device->read(qMin(comment_length, i));
534 
535 
536  device->seek(start_of_directory);
537  for (i = 0; i < num_dir_entries; ++i) {
538  FileHeader header;
539  int read = device->read((char *) &header.h, sizeof(MCentralFileHeader));
540  if (read < (int)sizeof(MCentralFileHeader)) {
541  qWarning() << "QZip: Failed to read complete header, index may be incomplete";
542  break;
543  }
544  if (readUInt(header.h.signature) != 0x02014b50) {
545  qWarning() << "QZip: invalid header signature, index may be incomplete";
546  break;
547  }
548 
549  int l = readUShort(header.h.file_name_length);
550  header.file_name = device->read(l);
551  if (header.file_name.length() != l) {
552  qWarning() << "QZip: Failed to read filename from zip index, index may be incomplete";
553  break;
554  }
555  l = readUShort(header.h.extra_field_length);
556  header.extra_field = device->read(l);
557  if (header.extra_field.length() != l) {
558  qWarning() << "QZip: Failed to read extra field in zip file, skipping file, index may be incomplete";
559  break;
560  }
561  l = readUShort(header.h.file_comment_length);
562  header.file_comment = device->read(l);
563  if (header.file_comment.length() != l) {
564  qWarning() << "QZip: Failed to read read file comment, index may be incomplete";
565  break;
566  }
567 
568  ZDEBUG("found file '%s'", header.file_name.data());
569  fileHeaders.append(header);
570  }
571 }

573 void MQZipWriterPrivate::addEntry(EntryType type, const QString &fileName, const QByteArray &contents/*, QFile::Permissions permissions, QZip::Method m*/)
574 {
575 #ifndef NDEBUG
576  static const char *entryTypes[] = {
577  "directory",
578  "file ",
579  "symlink " };
580  ZDEBUG() << "adding" << entryTypes[type] <<":" << fileName.toUtf8().data() << (type == 2 ? QByteArray(" -> " + contents).constData() : "");
581 #endif
582 
583  if (! (device->isOpen() || device->open(QIODevice::WriteOnly))) {
584  status = MQZipWriter::FileOpenError;
585  return;
586  }
587  device->seek(start_of_directory);
588 
589  // don't compress small files
590  MQZipWriter::CompressionPolicy compression = compressionPolicy;
591  if (compressionPolicy == MQZipWriter::AutoCompress) {
592  if (contents.length() < 64)
593  compression = MQZipWriter::NeverCompress;
594  else
595  compression = MQZipWriter::AlwaysCompress;
596  }
597 
598  FileHeader header;
599  memset(&header.h, 0, sizeof(MCentralFileHeader));
600  writeUInt(header.h.signature, 0x02014b50);
601 
602  writeUShort(header.h.version_needed, 0x14);
603  writeUInt(header.h.uncompressed_size, contents.length());
604  writeMSDosDate(header.h.last_mod_file, QDateTime::currentDateTime());
605  QByteArray data = contents;
606  if (compression == MQZipWriter::AlwaysCompress) {
607  writeUShort(header.h.compression_method, 8);
608 
609  ulong len = contents.length();
610  // shamelessly copied form zlib
611  len += (len >> 12) + (len >> 14) + 11;
612  int res;
613  do {
614  data.resize(len);
615  res = deflate((uchar*)data.data(), &len, (const uchar*)contents.constData(), contents.length());
616 
617  switch (res) {
618  case Z_OK:
619  data.resize(len);
620  break;
621  case Z_MEM_ERROR:
622  qWarning("QZip: Z_MEM_ERROR: Not enough memory to compress file, skipping");
623  data.resize(0);
624  break;
625  case Z_BUF_ERROR:
626  len *= 2;
627  break;
628  }
629  } while (res == Z_BUF_ERROR);
630  }
631 // TODO add a check if data.length() > contents.length(). Then try to store the original and revert the compression method to be uncompressed
632  writeUInt(header.h.compressed_size, data.length());
633  uint crc_32 = ::crc32(0, 0, 0);
634  crc_32 = ::crc32(crc_32, (const uchar *)contents.constData(), contents.length());
635  writeUInt(header.h.crc_32, crc_32);
636 
637  header.file_name = fileName.toUtf8();
638  if (header.file_name.size() > 0xffff) {
639  qWarning("QZip: Filename too long, chopping it to 65535 characters");
640  header.file_name = header.file_name.left(0xffff);
641  }
642  writeUShort(header.h.file_name_length, header.file_name.length());
643  //h.extra_field_length[2];
644 
645  writeUShort(header.h.version_made, 3 << 8);
646  //uchar internal_file_attributes[2];
647  //uchar external_file_attributes[4];
648  quint32 mode = permissionsToMode(permissions);
649  switch (type) {
650  case File: mode |= S_IFREG; break;
651  case Directory: mode |= S_IFDIR; break;
652  case Symlink: mode |= S_IFLNK; break;
653  }
654  writeUInt(header.h.external_file_attributes, mode << 16);
655  writeUInt(header.h.offset_local_header, start_of_directory);
656 
657 
658  fileHeaders.append(header);
659 
660  LocalFileHeader h = header.h.toLocalHeader();
661  device->write((const char *)&h, sizeof(LocalFileHeader));
662  device->write(header.file_name);
663  device->write(data);
664  start_of_directory = device->pos();
665  dirtyFileTree = true;
666 }

850 QByteArray MQZipReader::fileData(const QString &fileName) const
851 {
852  d->scanFiles();
853  int i;
854  for (i = 0; i < d->fileHeaders.size(); ++i) {
855  if (QString::fromUtf8(d->fileHeaders.at(i).file_name) == fileName)
856  break;
857  }
858  if (i == d->fileHeaders.size())
859  return QByteArray();
860 
861  FileHeader header = d->fileHeaders.at(i);
862 
863  int compressed_size = readUInt(header.h.compressed_size);
864  int uncompressed_size = readUInt(header.h.uncompressed_size);
865  int start = readUInt(header.h.offset_local_header);
866  //qDebug("uncompressing file %d: local header at %d", i, start);
867 
868  d->device->seek(start);
869  LocalFileHeader lh;
870  d->device->read((char *)&lh, sizeof(LocalFileHeader));
871  uint skip = readUShort(lh.file_name_length) + readUShort(lh.extra_field_length);
872  d->device->seek(d->device->pos() + skip);
873 
874  int compression_method = readUShort(lh.compression_method);
875  //qDebug("file=%s: compressed_size=%d, uncompressed_size=%d", fileName.toLocal8Bit().data(), compressed_size, uncompressed_size);
876 
877  //qDebug("file at %lld", d->device->pos());
878  QByteArray compressed = d->device->read(compressed_size);
879  if (compression_method == 0) {
880  // no compression
881  compressed.truncate(uncompressed_size);
882  return compressed;
883  } else if (compression_method == 8) {
884  // Deflate
885  //qDebug("compressed=%d", compressed.size());
886  compressed.truncate(compressed_size);
887  QByteArray baunzip;
888  ulong len = qMax(uncompressed_size, 1);
889  int res;
890  do {
891  baunzip.resize(len);
892  res = inflate((uchar*)baunzip.data(), &len,
893  (uchar*)compressed.constData(), compressed_size);
894 
895  switch (res) {
896  case Z_OK:
897  if ((int)len != baunzip.size())
898  baunzip.resize(len);
899  break;
900  case Z_MEM_ERROR:
901  qWarning("QZip: Z_MEM_ERROR: Not enough memory");
902  break;
903  case Z_BUF_ERROR:
904  len *= 2;
905  break;
906  case Z_DATA_ERROR:
907  qWarning("QZip: Z_DATA_ERROR: Input data is corrupted");
908  break;
909  }
910  } while (res == Z_BUF_ERROR);
911  return baunzip;
912  }
913  qWarning() << "QZip: Unknown compression method";
914  return QByteArray();
915 }

923 {

925

928  foreach (FileInfo fi, allFiles) {

930  if (fi.isDir) {

935  }
936  }
937

940  const QString absPath = destinationDir + QDir::separator() + fi.filePath;

943  if (destination.isEmpty())
944  return false;
945  QFileInfo linkFi(absPath);
946  if (!QFile::exists(linkFi.absolutePath()))
947  QDir::root().mkpath(linkFi.absolutePath());
948  if (!QFile::link(destination, absPath))
949  return false;
950  /* cannot change permission of links
951  if (!QFile::setPermissions(absPath, fi.permissions))
952  return false;
953  */
954  }
955  }
956 
957  foreach (FileInfo fi, allFiles) {
958  const QString absPath = destinationDir + QDir::separator() + fi.filePath;
959  if (fi.isFile) {
960  QFile f(absPath);
961  if (!f.open(QIODevice::WriteOnly))
962  return false;
963  f.write(fileData(fi.filePath));
