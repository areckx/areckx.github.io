28 static inline qreal wrap(qreal a, qreal d = 1.0) {
29  qreal r = fmod(a, d);
30  return (r < 0.0 ? d + r : (r > 0.0 ? r : 0.0));
31  }

44 KHCY::KHCY(const QColor& color) {
45  qreal r = gamma(color.redF());
46  qreal g = gamma(color.greenF());
47  qreal b = gamma(color.blueF());
48  a = color.alphaF();
49 
50  // luma component
51  y = lumag(r, g, b);
52 
53  // hue component
54  qreal p = qMax(qMax(r, g), b);
55  qreal n = qMin(qMin(r, g), b);
56  qreal d = 6.0 * (p - n);
57  if (n == p)
58  h = 0.0;
59  else if (r == p)
60  h = ((g - b) / d);
61  else if (g == p)
62  h = ((b - r) / d) + (1.0 / 3.0);
63  else
64  h = ((r - g) / d) + (2.0 / 3.0);
65 
66  // chroma component
67  if (r == g && g == b)
68  c = 0.0;
69  else
70  c = qMax( (y - n) / y, (p - y) / (1 - y) );
71  }

77 QColor KHCY::qColor() const {
78  // start with sane component values
79  qreal _h = wrap(h);
80  qreal _c = normalize(c);
81  qreal _y = normalize(y);
82 
83  // calculate some needed variables
84  qreal _hs = _h * 6.0, th, tm;
85  if (_hs < 1.0) {
86  th = _hs;
87  tm = yc[0] + yc[1] * th;
88  }
89  else if (_hs < 2.0) {
90  th = 2.0 - _hs;
91  tm = yc[1] + yc[0] * th;
92  }
93  else if (_hs < 3.0) {
94  th = _hs - 2.0;
95  tm = yc[1] + yc[2] * th;
96  }
97  else if (_hs < 4.0) {
98  th = 4.0 - _hs;
99  tm = yc[2] + yc[1] * th;
100  }
101  else if (_hs < 5.0) {
102  th = _hs - 4.0;
103  tm = yc[2] + yc[0] * th;
104  }
105  else {
106  th = 6.0 - _hs;
107  tm = yc[0] + yc[2] * th;
108  }
109 
110  // calculate RGB channels in sorted order
111  qreal tn, to, tp;
112  if (tm >= _y) {
113  tp = _y + _y * _c * (1.0 - tm) / tm;
114  to = _y + _y * _c * (th - tm) / tm;
115  tn = _y - (_y * _c);
116  }
117  else {
118  tp = _y + (1.0 - _y) * _c;
119  to = _y + (1.0 - _y) * _c * (th - tm) / (1.0 - tm);
120  tn = _y - (1.0 - _y) * _c * tm / (1.0 - tm);
121  }
122 
123  // return RGB channels in appropriate order
124  if (_hs < 1.0)
125  return QColor::fromRgbF(igamma(tp), igamma(to), igamma(tn), a);
126  else if (_hs < 2.0)
127  return QColor::fromRgbF(igamma(to), igamma(tp), igamma(tn), a);
128  else if (_hs < 3.0)
129  return QColor::fromRgbF(igamma(tn), igamma(tp), igamma(to), a);
130  else if (_hs < 4.0)
131  return QColor::fromRgbF(igamma(tn), igamma(to), igamma(tp), a);
132  else if (_hs < 5.0)
133  return QColor::fromRgbF(igamma(to), igamma(tn), igamma(tp), a);
134  else
135  return QColor::fromRgbF(igamma(tp), igamma(tn), igamma(to), a);
136  }

168 qreal ColorUtils::contrastRatio(const QColor& c1, const QColor& c2) {
169  qreal y1 = luma(c1), y2 = luma(c2);
170  if (y1 > y2)
171  return (y1 + 0.05) / (y2 + 0.05);
172  else
173  return (y2 + 0.05) / (y1 + 0.05);
174  }

180 QColor ColorUtils::tint(const QColor& base, const QColor& color, qreal amount) {
181  if (amount <= 0.0)
182  return base;
183  if (amount >= 1.0)
184  return color;
185 #ifdef Q_OS_MAC
186  if (isnan(amount))
187 #else
188  if (std::isnan(amount))
189 #endif
190  return base;
191 
192  double ri = contrastRatio(base, color);
193  double rg = 1.0 + ((ri + 1.0) * amount * amount * amount);
194  double u = 1.0, l = 0.0;
195  QColor result;
196  for (int i = 12 ; i ; --i) {
197  double a = 0.5 * (l + u);
198  result = tintHelper(base, color, a);
199  double ra = contrastRatio(base, result);
200  if (ra > rg)
201  u = a;
202  else
203  l = a;
204  }
205  return result;
206  }

212 QColor ColorUtils::mix(const QColor& c1, const QColor& c2, qreal bias) {
213  if (bias <= 0.0)
214  return c1;
215  if (bias >= 1.0)
216  return c2;
217 #ifdef Q_OS_MAC
218  if (isnan(bias))
219 #else
220  if (std::isnan(bias))
221 #endif
222  return c1;
223 
224  qreal r = mixQreal(c1.redF(), c2.redF(), bias);
225  qreal g = mixQreal(c1.greenF(), c2.greenF(), bias);
226  qreal b = mixQreal(c1.blueF(), c2.blueF(), bias);
227  qreal a = mixQreal(c1.alphaF(), c2.alphaF(), bias);
228 
229  return QColor::fromRgbF(r, g, b, a);
230  }

236 QColor ColorUtils::lighten(const QColor& color, qreal ky, qreal kc) {
237  KHCY c(color);
238  c.y = 1.0 - normalize((1.0 - c.y) * (1.0 - ky));
239  c.c = 1.0 - normalize((1.0 - c.c) * kc);
240  return c.qColor();
241  }
