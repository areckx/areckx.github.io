198 static void collectNote(EventMap* events, int channel, const Note* note, int velo, int tickOffset)
199  {
200  if (!note->play() || note->hidden()) // do not play overlapping notes
201  return;
202  int pitch = note->ppitch();
203  Chord* chord = note->chord();
204 
205  int ticks;
206  int tieLen = 0;
207  if (chord->isGrace()) {
208  Q_ASSERT( !graceNotesMerged(chord)); // this function should not be called on a grace note if grace notes are merged
209  chord = static_cast<Chord*>(chord->parent());
210  ticks = chord->actualTicks(); // ticks of the parent note
211  tieLen = 0;
212  }
213  else {
214  ticks = chord->actualTicks(); // ticks of the actual note
215  // calculate additional length due to ties forward
216  // taking NoteEvent length adjustments into account
217  // but stopping at any note with multiple NoteEvents
218  // and processing those notes recursively
219  if (note->tieFor()) {
220  Note* n = note->tieFor()->endNote();
221  while (n) {
222  NoteEventList nel = n->playEvents();
223  if (nel.size() == 1) {
224  // add value of this note to main note
225  // if we wish to suppress first note of ornament,
226  // then do this regardless of number of NoteEvents
227  tieLen += (n->chord()->actualTicks() * (nel[0].len())) / 1000;
228  }
229  else {
230  // recurse
231  collectNote(events, channel, n, velo, tickOffset);
232  break;
233  }
234  if (n->tieFor() && n != n->tieFor()->endNote())
235  n = n->tieFor()->endNote();
236  else
237  break;
238  }
239  }
240  }
241 
242  int tick1 = chord->tick() + tickOffset;
243  bool tieFor = note->tieFor();
244  bool tieBack = note->tieBack();
245 
246  NoteEventList nel = note->playEvents();
247  int nels = nel.size();
248  for (int i = 0; i < nels; ++i) {
249  const NoteEvent& e = nel[i]; // we make an explict const ref, not a const copy. no need to copy as we won't change the original object.
250 
251  // skip if note has a tie into it and only one NoteEvent
252  // its length was already added to previous note
253  // if we wish to suppress first note of ornament
254  // then change "nels == 1" to "i == 0", and change "break" to "continue"
255  if (tieBack && nels == 1)
256  break;
257  int p = pitch + e.pitch();
258  if (p < 0)
259  p = 0;
260  else if (p > 127)
261  p = 127;
262  int on = tick1 + (ticks * e.ontime())/1000;
263  int off = on + (ticks * e.len())/1000 - 1;
264  if (tieFor && i == nels - 1)
265  off += tieLen;
266  playNote(events, note, channel, p, velo, on, off);
267  }
268 
269  // Bends
270  for (Element* e : note->el()) {
271  if (e == 0 || e->type() != ElementType::BEND)
272  continue;
273  Bend* bend = static_cast<Bend*>(e);
274  if (!bend->playBend())
275  break;
276  const QList<PitchValue>& points = bend->points();
277  int pitchSize = points.size();
278 
279  double noteLen = note->playTicks();
280  int lastPointTick = tick1;
281  for (int pitchIndex = 0; pitchIndex < pitchSize-1; pitchIndex++) {
282  PitchValue pitchValue = points[pitchIndex];
283  PitchValue nextPitch = points[pitchIndex+1];
284  int nextPointTick = tick1 + nextPitch.time / 60.0 * noteLen;
285  int pitch = pitchValue.pitch;
286 
287  if (pitchIndex == 0 && (pitch == nextPitch.pitch)) {
288  int midiPitch = (pitch * 16384) / 1200 + 8192;
289  int msb = midiPitch / 128;
290  int lsb = midiPitch % 128;
291  NPlayEvent ev(ME_PITCHBEND, channel, lsb, msb);
292  events->insert(std::pair<int, NPlayEvent>(lastPointTick, ev));
293  lastPointTick = nextPointTick;
294  continue;
295  }
296  if (pitch == nextPitch.pitch && !(pitchIndex == 0 && pitch != 0)) {
297  lastPointTick = nextPointTick;
298  continue;
299  }
300 
301  double pitchDelta = nextPitch.pitch - pitch;
302  double tickDelta = nextPitch.time - pitchValue.time;
303  /* B
304  /. pitch is 1/100 semitones
305  bend / . pitchDelta time is in noteDuration/60
306  / . midi pitch is 12/16384 semitones
307  A....
308  tickDelta */
309  for (int i = lastPointTick; i <= nextPointTick; i += 16) {
310  double dx = ((i-lastPointTick) * 60) / noteLen;
311  int p = pitch + dx * pitchDelta / tickDelta;
312 
313  // We don't support negative pitch, but Midi does. Let's center by adding 8192.
314  int midiPitch = (p * 16384) / 1200 + 8192;
315  // Representing pitch as two bytes
316  int msb = midiPitch / 128;
317  int lsb = midiPitch % 128;
318  NPlayEvent ev(ME_PITCHBEND, channel, lsb, msb);
319  events->insert(std::pair<int, NPlayEvent>(i, ev));
320  }
321  lastPointTick = nextPointTick;
322  }
323  NPlayEvent ev(ME_PITCHBEND, channel, 0, 64); // 0:64 is 8192 - no pitch bend
324  events->insert(std::pair<int, NPlayEvent>(tick1+noteLen, ev));
325  }
326  }

355 static void collectMeasureEvents(EventMap* events, Measure* m, Staff* staff, int tickOffset)
356  {
357  int firstStaffIdx = staff->idx();
358  int nextStaffIdx = firstStaffIdx + 1;
359 
360  SegmentType st = SegmentType::ChordRest;
361  int strack = firstStaffIdx * VOICES;
362  int etrack = nextStaffIdx * VOICES;
363 
364  for (Segment* seg = m->first(st); seg; seg = seg->next(st)) {
365  int tick = seg->tick();
366  for (int track = strack; track < etrack; ++track) {
367  // skip linked staves, except primary
368  if (!m->score()->staff(track / VOICES)->primaryStaff()) {
369  track += VOICES-1;
370  continue;
371  }
372  Element* cr = seg->element(track);
373  if (cr == 0 || cr->type() != ElementType::CHORD)
374  continue;
375 
376  Chord* chord = static_cast<Chord*>(cr);
377  Staff* staff = chord->staff();
378  int velocity = staff->velocities().velo(seg->tick());
379  Instrument* instr = chord->part()->instrument(tick);
380  int channel = instr->channel(chord->upNote()->subchannel())->channel;
381 
382  for (Articulation* a : chord->articulations())
383  instr->updateVelocity(&velocity,channel, a->articulationName());
384 
385  if ( !graceNotesMerged(chord))
386  for (Chord* c : chord->graceNotesBefore())
387  for (const Note* note : c->notes())
388  collectNote(events, channel, note, velocity, tickOffset);
389 
390  for (const Note* note : chord->notes())
391  collectNote(events, channel, note, velocity, tickOffset);
392 
393  if ( !graceNotesMerged(chord))
394  for (Chord* c : chord->graceNotesAfter())
395  for (const Note* note : c->notes())
396  collectNote(events, channel, note, velocity, tickOffset);
397  }
398  }
399 
400  //
401  // collect program changes and controller
402  //
403  for (Segment* s = m->first(SegmentType::ChordRest); s; s = s->next(SegmentType::ChordRest)) {
404  // int tick = s->tick();
405  for (Element* e : s->annotations()) {
406  if (e->type() != ElementType::STAFF_TEXT
407  || e->staffIdx() < firstStaffIdx
408  || e->staffIdx() >= nextStaffIdx)
409  continue;
410  const StaffText* st = static_cast<const StaffText*>(e);
411  int tick = s->tick() + tickOffset;
412 
413  Instrument* instr = e->part()->instrument(tick);
414  for (const ChannelActions& ca : *st->channelActions()) {
415  int channel = instr->channel().at(ca.channel)->channel;
416  for (const QString& ma : ca.midiActionNames) {
417  NamedEventList* nel = instr->midiAction(ma, ca.channel);
418  if (!nel)
419  continue;
420  for (MidiCoreEvent event : nel->events) {
421  event.setChannel(channel);
422  NPlayEvent e(event);
423  if (e.dataA() == CTRL_PROGRAM)
424  events->insert(std::pair<int, NPlayEvent>(tick-1, e));
425  else
426  events->insert(std::pair<int, NPlayEvent>(tick, e));
427  }
428  }
429  }
430  if (st->setAeolusStops()) {
431  Staff* staff = st->staff();
432  int voice = 0;
433  int channel = staff->channel(tick, voice);
434 
435  for (int i = 0; i < 4; ++i) {
436  static int num[4] = { 12, 13, 16, 16 };
437  for (int k = 0; k < num[i]; ++k)
438  aeolusSetStop(tick, channel, i, k, st->getAeolusStop(i, k), events);
439  }
440  }
441  }
442  }
443  }

477 void Score::updateHairpin(Hairpin* h)
478  {
479  Staff* st = h->staff();
480  int tick = h->tick();
481  int velo = st->velocities().velo(tick);
482  int incr = h->veloChange();
483  int tick2 = h->tick2();
484 
485  //
486  // If velocity increase/decrease is zero, then assume
487  // the end velocity is taken from the next velocity
488  // event (the next dynamics symbol after the hairpin).
489  //
490 
491  int endVelo = velo;
492  if (h->hairpinType() == HairpinType::CRESC_HAIRPIN || h->hairpinType() == HairpinType::CRESC_LINE) {
493  if (incr == 0 && velo < st->velocities().nextVelo(tick2-1))
494  endVelo = st->velocities().nextVelo(tick2-1);
495  else
496  endVelo += incr;
497  }
498  else {
499  if (incr == 0 && velo > st->velocities().nextVelo(tick2-1))
500  endVelo = st->velocities().nextVelo(tick2-1);
501  else
502  endVelo -= incr;
503  }
504 
505  if (endVelo > 127)
506  endVelo = 127;
507  else if (endVelo < 1)
508  endVelo = 1;
509 
510  switch (h->dynRange()) {
511  case Dynamic::Range::STAFF:
512  st->velocities().setVelo(tick, VeloEvent(VeloType::RAMP, velo));
513  st->velocities().setVelo(tick2-1, VeloEvent(VeloType::FIX, endVelo));
514  break;
515  case Dynamic::Range::PART:
516  for (Staff* s : *st->part()->staves()) {
517  s->velocities().setVelo(tick, VeloEvent(VeloType::RAMP, velo));
518  s->velocities().setVelo(tick2-1, VeloEvent(VeloType::FIX, endVelo));
519  }
520  break;
521  case Dynamic::Range::SYSTEM:
522  for (Staff* s : _staves) {
523  s->velocities().setVelo(tick, VeloEvent(VeloType::RAMP, velo));
524  s->velocities().setVelo(tick2-1, VeloEvent(VeloType::FIX, endVelo));
525  }
526  break;
527  }
528  }

565 void Score::updateVelo()
566  {

568  // collect Dynamics
569  //

571  return;
572 
573  for (Staff* st : _staves) {
574  VeloList& velo = st->velocities();
575  velo.clear();
576  velo.setVelo(0, 80);
577  }
578  for (int staffIdx = 0; staffIdx < nstaves(); ++staffIdx) {
579  Staff* st = staff(staffIdx);
580  VeloList& velo = st->velocities();
581  Part* prt = st->part();
582  int partStaves = prt->nstaves();
583  int partStaff = Score::staffIdx(prt);
584 
585  for (Segment* s = firstMeasure()->first(); s; s = s->next1()) {
586  int tick = s->tick();
587  for (const Element* e : s->annotations()) {
588  if (e->staffIdx() != staffIdx)
589  continue;
590  if (e->type() != ElementType::DYNAMIC)
591  continue;
592  const Dynamic* d = static_cast<const Dynamic*>(e);
593  int v = d->velocity();
594  if (v < 1) // illegal value
595  continue;
596  int dStaffIdx = d->staffIdx();
597  switch(d->dynRange()) {
598  case Dynamic::Range::STAFF:
599  if (dStaffIdx == staffIdx)
600  velo.setVelo(tick, v);
601  break;
602  case Dynamic::Range::PART:
603  if (dStaffIdx >= partStaff && dStaffIdx < partStaff+partStaves) {
604  for (int i = partStaff; i < partStaff+partStaves; ++i)
605  staff(i)->velocities().setVelo(tick, v);
606  }
607  break;
608  case Dynamic::Range::SYSTEM:
609  for (int i = 0; i < nstaves(); ++i)
610  staff(i)->velocities().setVelo(tick, v);
611  break;
612  }
613  }
614  }
615  for (const auto& sp : _spanner.map()) {
616  Spanner* s = sp.second;
617  if (s->type() != ElementType::HAIRPIN || sp.second->staffIdx() != staffIdx)
618  continue;
619  Hairpin* h = static_cast<Hairpin*>(s);
620  updateHairpin(h);
621  }
622  }

646  break;
647  }

655 void Score::renderSpanners(EventMap* events, int staffIdx)
656  {
657  for (const RepeatSegment* rs : *repeatList()) {
658  int tickOffset = rs->utick - rs->tick;
659  int utick1 = rs->utick;
660  int tick1 = repeatList()->utick2tick(utick1);
661  int tick2 = tick1 + rs->len;
662  std::map<int, std::vector<std::pair<int, bool>>> channelPedalEvents = std::map<int, std::vector<std::pair<int, bool>>>();
663  for (const auto& sp : _spanner.map()) {
664  Spanner* s = sp.second;
665  if (s->type() != ElementType::PEDAL || (staffIdx != -1 && s->staffIdx() != staffIdx))
666  continue;
667 
668  int idx = s->staff()->channel(s->tick(), 0);
669  int channel = s->part()->instrument(s->tick())->channel(idx)->channel;
670  channelPedalEvents.insert({channel, std::vector<std::pair<int, bool>>()});
671  std::vector<std::pair<int, bool>> pedalEventList = channelPedalEvents.at(channel);
672  std::pair<int, bool> lastEvent;
673 
674  if (!pedalEventList.empty())
675  lastEvent = pedalEventList.back();
676  else

910 //---------------------------------------------------------
911 // articulationExcursion -- an articulation such as a trill, or modant consists of several notes
912 // played in succession. The pitch offsets of each such note in the sequence can be represented either
913 // as a number of steps in the diatonic scale, or in half steps as on a piano keyboard.
914 // this function, articulationExcursion, takes deltastep indicating the number of steps in the
915 // diatonic scale, and calculates (and returns) the number of half steps, taking several things into account.
916 // E.g., the key signature, a trill from e to f, is to be understood as a trill between E and F# if we are
917 // in the key of G.
918 // E.g., if previously (looking backward in time) in the same measure there is another note on the same
919 // staff line/space, and that note has an accidental (sharp,flat,natural,etc), then we want to match that
920 // tone exactly.
921 // E.g., If there are multiple notes on the same line/space, then we only consider the most
922 // recent one, but avoid looking forward in time after the current note.
923 // E.g., Also if there is an accidental // on a note one (or more) octaves above or below we
924 // observe its accidental as well.
925 // E.g., Still another case is that if two staffs are involved (such as a glissando between two
926 // notes on different staffs) then we have to search both staffs for the most recent accidental.
927 //
928 // noteL is the note to measure the deltastep from, i.e., ornaments are w.r.t. this note
929 // noteR is the note to search backward from to find accidentals.
930 // for ornament calculation noteL and noteR are the same, but for glissando they are
931 // the start end end note of glissando.
932 // deltastep is the desired number of diatonic steps between the base note and this articulation step.
933 //---------------------------------------------------------
934 
935 int articulationExcursion(Note *noteL, Note *noteR, int deltastep)
936  {
937  if (0 == deltastep)
938  return 0;
939  Chord *chordL = noteL->chord();
940  Chord *chordR = noteR->chord();
941  int pitchL = noteL->pitch();
942  int tickL = chordL->tick();
943  // we canot use staffL = chord->staff() because that won't correspond to the noteL->line()
944  // in the case the user has pressed Shift-Cmd->Up or Shift-Cmd-Down.
945  // Therefore we have to take staffMove() into account using vStaffIdx().
946  Staff * staffL = noteL->score()->staff(chordL->vStaffIdx());
947  ClefType clefL = staffL->clef(tickL);
948  // line represents the ledger line of the staff. 0 is the top line, 1, is the space between the top 2 lines,
949  // ... 8 is the bottom line.
950  int lineL = noteL->line();
951  // we use line - deltastep, because lines are oriented from top to bottom, while step is oriented from bottom to top.
952  int lineL2 = lineL - deltastep;
953  Measure* measureR = chordR->segment()->measure();
954 
955  Segment* segment = noteL->chord()->segment();
956  int lineR2 = convertLine(lineL2, noteL, noteR);
957  // is there another note in this segment on the same line?
958  // if so, use its pitch exactly.
959  int halfsteps = 0;
960  int staffIdx = noteL->chord()->staff()->idx(); // cannot use staffL->idx() because of staffMove()
961  int startTrack = staffIdx * VOICES;
962  int endTrack = startTrack + VOICES;
963  bool done = false;
964  for (int track = startTrack; track < endTrack; ++track) {
965  Element *e = segment->element(track);
966  if (!e || e->type() != ElementType::CHORD)
967  continue;
968  Chord* chord = static_cast<Chord*>(e);
969  for (Note* note : chord->notes()) {
970  if (note->tieBack())
971  continue;
972  int pc = (note->line() + 700) % 7;
973  int pc2 = (lineL2 + 700) % 7;
974  if (pc2 == pc) {
975  // e.g., if there is an F# note at this staff/tick, then force every F to be F#.
976  int octaves = (note->line() - lineL2) / 7;
977  halfsteps = note->pitch() + 12 * octaves - pitchL;
978  done = true;
979  break;
980  }
981  }
982  if (!done) {
983  if (staffL->isPitchedStaff(segment->tick())) {
984  bool error = false;
985  AccidentalVal acciv2 = measureR->findAccidental(chordR->segment(), chordR->staff()->idx(), lineR2, error);
986  int acci2 = int(acciv2);
987  // we have to add ( noteL->ppitch() - noteL->epitch() ) which is the delta for transposing instruments.
988  halfsteps = line2pitch(lineL-deltastep, clefL, Key::C) + noteL->ppitch() - noteL->epitch() + acci2 - pitchL;
989  }
990  else {
991  // cannot rely on accidentals or key signatures
992  halfsteps = deltastep;
993  }
994  }
995  }
996  return halfsteps;
997  }

1015 //---------------------------------------------------------
1016 // renderNoteArticulation
1017 // prefix, vector of int, normally something like {0,-1,0,1} modeling the prefix of tremblement relative to the base note
1018 // body, vector of int, normally something like {0,-1,0,1} modeling the possibly repeated tremblement relative to the base note
1019 // tickspernote, number of ticks, either _16h or _32nd, i.e., MScore::division/4 or MScore::division/8
1020 // repeatp, true means repeat the body as many times as possible to fill the time slice.
1021 // sustainp, true means the last note of the body is sustained to fill remaining time slice
1022 //---------------------------------------------------------
1023 
1024 bool renderNoteArticulation(NoteEventList* events, Note* note, bool chromatic, int requestedTicksPerNote,
1025  const vector<int>& prefix, const vector<int>& body,
1026  bool repeatp, bool sustainp, const vector<int>& suffix,
1027  int fastestFreq=16, int slowestFreq=8 // 16 Hz and 8 Hz
1028  )
1029  {
1030  events->clear();
1031  Chord *chord = note->chord();
1032  int maxticks = totalTiedNoteTicks(note);
1033  int space = 1000 * maxticks;
1034  int numrepeat = 1;
1035  int sustain = 0;
1036  int ontime = 0;
1037 
1038  int gnb = note->chord()->graceNotesBefore().size();
1039  int p = prefix.size();
1040  int b = body.size();
1041  int s = suffix.size();
1042  int gna = note->chord()->graceNotesAfter().size();
1043 
1044  int ticksPerNote = 0;
1045 
1046  if (gnb + p + b + s + gna <= 0 )
1047  return false;
1048 
1049  int tick = chord->tick();
1050  qreal tempo = chord->score()->tempo(tick);
1051  int ticksPerSecond = tempo * MScore::division;
1052 
1053  int minTicksPerNote = int(ticksPerSecond / fastestFreq);
1054  int maxTicksPerNote = (0 == slowestFreq) ? 0 : int(ticksPerSecond / slowestFreq);
1055 
1056  // for fast tempos, we have to slow down the tremblement frequency, i.e., increase the ticks per note
1057  if (requestedTicksPerNote >= minTicksPerNote)
1058  ;
1059  else { // try to divide the requested frequency by a power of 2 if possible, if not, use the maximum frequency, ie., minTicksPerNote
1060  ticksPerNote = requestedTicksPerNote;
1061  while (ticksPerNote < minTicksPerNote) {
1062  ticksPerNote *= 2; // decrease the tremblement frequency
1063  }
1064  if (ticksPerNote > maxTicksPerNote)
1065  ticksPerNote = minTicksPerNote;
1066  }
1067 
1068  ticksPerNote = max(requestedTicksPerNote, minTicksPerNote);
1069 
1070  if (slowestFreq <= 0) // no slowest freq given such as something silly like glissando with 4 notes over 8 counts.
1071  ;
1072  else if (ticksPerNote <= maxTicksPerNote) // in a good range, so we don't need to adjust ticksPerNote
1073  ;
1074  else {
1075  // for slow tempos, such as adagio, we may need to speed up the tremblement freqency, i.e., decrease the ticks per note, to make it sound reasonable.
1076  ticksPerNote = requestedTicksPerNote ;
1077  while (ticksPerNote > maxTicksPerNote) {
1078  ticksPerNote /= 2;
1079  }
1080  if (ticksPerNote < minTicksPerNote)
1081  ticksPerNote = minTicksPerNote;
1082  }
1083  // calculate whether to shorten the duration value.
1084  if ( ticksPerNote*(gnb + p + b + s + gna) <= maxticks )
1085  ; // plenty of space to play the notes without changing the requested trill note duration
1086  else if ( ticksPerNote == minTicksPerNote )
1087  return false; // the ornament is impossible to implement respecting the minimum duration and all the notes it contains
1088  else {
1089  ticksPerNote = maxticks / (gnb + p + b + s + gna); // integer division ignoring remainder
1090  if ( slowestFreq <= 0 )
1091  ;
1092  else if ( ticksPerNote < minTicksPerNote )
1093  return false;
1094  }
1095 
1096  int millespernote = space * ticksPerNote / maxticks; // rescale duration into per mille
1097 
1098  // local function:
1099  // look ahead in the given vector to see if the current note is the same pitch as the next note or next several notes.
1100  // If so, increment the duration by the appropriate note duration, and increment the index, j, to the next note index
1101  // of a different pitch.
1102  // The total duration of the tied note is returned, and the index is modified.
1103  auto tieForward = [millespernote] (int & j, const vector<int> & vec) {
1104  int size = vec.size();
1105  int duration = millespernote;
1106  while ( j < size-1 && vec[j] == vec[j+1] ) {
1107  duration += millespernote;
1108  j++;
1109  }
1110  return duration;
1111  };
1112 
1113  // local function:
1114  // append a NoteEvent either by calculating an articulationExcursion or by
1115  // the given chromatic relative pitch.
1116  // RETURNS the new ontime value. The caller is expected to assign this value.
1117  auto makeEvent = [note,chord,chromatic,events] (int pitch, int ontime, int duration) {
1118  events->append( NoteEvent(chromatic ? pitch : articulationExcursion(note,note,pitch),
1119  ontime/chord->actualTicks(),
1120  duration/chord->actualTicks()));
1121  return ontime + duration;
1122  };
1123 
1124  // local function:
1125  // Given a chord from a grace note, (normally the chord contains a single note) and create
1126  // a NoteEvent as if the grace note were part of the articulation (such as trill). This
1127  // local function works for the graceNotesBefore() and also graceNotesAfter().
1128  // If the grace note has play=false, then it will sound as a rest, but the other grace
1129  // notes will still play. This means graceExtend simply omits the call to append( NoteEvent(...))
1130  // but still updates ontime +=millespernote.
1131  // RETURNS the new value of ontime, so caller must make an assignment to the return value.
1132  auto graceExtend = [millespernote,chord,events] (int notePitch, QVector<Chord*> graceNotes, int ontime) {
1133  for (Chord* c : graceNotes) {
1134  for (Note* n : c->notes()) {
1135  // NoteEvent takes relative pitch as first argument.
1136  // The pitch is relative to the pitch of the note, the event is rendering
1137  if (n->play())
1138  events->append( NoteEvent(n->pitch() - notePitch,
1139  ontime/chord->actualTicks(),
1140  millespernote/chord->actualTicks()));
1141  }
1142  ontime += millespernote;
1143  }
1144  return ontime;
1145  };
1146 
1147  // calculate the number of times to repeat the body, and sustain the last note of the body
1148  // 1000 = P + numrepeat*B+sustain + S
1149  if (repeatp)
1150  numrepeat = (space - millespernote*(gnb + p + s + gna)) / (millespernote * b);
1151  if (sustainp)
1152  sustain = space - millespernote*(gnb + p + numrepeat * b + s + gna);
1153  // render the graceNotesBefore
1154  ontime = graceExtend(note->pitch(),note->chord()->graceNotesBefore(), ontime);
1155 
1156  // render the prefix
1157  for (int j=0; j < p; j++)
1158  ontime = makeEvent(prefix[j], ontime, tieForward(j,prefix));
1159 
1160  if (b > 0) {
1161  // render the body, but not the final repetion
1162  for (int r = 0; r < numrepeat-1; r++) {
1163  for (int j=0; j < b; j++)
1164  ontime = makeEvent(body[j], ontime, millespernote);
1165  }
1166  // render the final repetion of body, but not the final note of the repition
1167  for (int j = 0; j < b - 1; j++)
1168  ontime = makeEvent(body[j], ontime, millespernote);
1169  // render the final note of the final repeat of body
1170  ontime = makeEvent(body[b-1], ontime, millespernote+sustain);
1171  }
1172  // render the suffix
1173  for (int j = 0; j < s; j++)
1174  ontime = makeEvent(suffix[j], ontime, tieForward(j,suffix));
1175  // render graceNotesAfter
1176  ontime = graceExtend(note->pitch(), note->chord()->graceNotesAfter(), ontime);
1177  return true;
1178  }

1180 // This struct specifies how to render an articulation.
1181 // atype - the articulation type to implement, such as SymId::ornamentTurn
1182 // ostyles - the actual ornament has a property called ornamentStyle whose value is
1183 // a value of type MScore::OrnamentStyle. This ostyles field indicates the
1184 // the set of ornamentStyles which apply to this rendition.
1185 // duration - the default duration for each note in the rendition, the final duration
1186 // rendered might be less than this if an articulation is attached to a note of
1187 // short duration.
1188 // prefix - vector of integers. indicating which notes to play at the beginning of rendering the
1189 // articulation. 0 represents the principle note, 1==> the note diatonically 1 above
1190 // -1 ==> the note diatonically 1 below. E.g., in the key of G, if a turn articulation
1191 // occures above the note F#, then 0==>F#, 1==>G, -1==>E.
1192 // These integers indicate which notes actual notes to play when rendering the ornamented
1193 // note. However, if the same integer appears several times adjacently such as {0,0,0,1}
1194 // That means play the notes tied. e.g., F# followed by G, but the duration of F# is 3x the
1195 // duration of the G.
1196 // body - notes to play comprising the body of the rendered ornament.
1197 // The body differs from the prefix and suffix in several ways.
1198 // * body does not support tied notes: {0,0,0,1} means play 4 distinct notes (not tied).
1199 // * if there is sufficient duration in the principle note, AND repeatep is true, then body
1200 // will be rendered multiple times, as the duration allows.
1201 // * to avoid a time gap (or rest) in rendering the articulation, if sustainp is true,
1202 // then the final note of the body will be sustained to fill the left-over time.
1203 // suffix - similar to prefix but played once at the end of the rendered ornament.
1204 // repeatp - whether the body is repeatable in its entirety.

1495 void Score::createGraceNotesPlayEvents(int tick, Chord* chord, int &ontime, int &trailtime)
1496  {
1497  QVector<Chord*> gnb= chord->graceNotesBefore();
1498  QVector<Chord*> gna= chord->graceNotesAfter();
1499  int nb = gnb.size();
1500  int na = gna.size();
1501  if (0 == nb+na){
1502  return; // return immediately if no grace notes to deal with
1503  }
1504  // return immediately if the chord has a trill or articulation which effectively plays the graces notes.
1505  if (graceNotesMerged(chord)) {
1506  return;
1507  }
1508  // if there are graceNotesBefore and also graceNotesAfter, and the before grace notes are
1509  // not ACCIACCATURA, then the total time of all of them will be 50% of the time of the main note.
1510  // if the before grace notes are ACCIACCATURA then the grace notes after (if there are any).
1511  // get 50% of the time of the main note.
1512  // this is achieved by the two floating point weights: weighta and weightb whose total is 1.0
1513  // assuring that all the grace notes get the same duration, and their total is 50%.
1514  // exception is if the note is dotted or double-dotted; see below.
1515  float weighta = float(na) / (nb+na);
1516  float weightb = float(nb) / (nb+na);
1517  if (nb) {

1519  // render grace notes:
1520  // simplified implementation:
1521  // - grace notes start on the beat of the main note
1522  // - duration: appoggiatura: 0.5 * duration of main note (2/3 for dotted notes, 4/7 for double-dotted)
1523  // acciacatura: min of 0.5 * duration or 65ms fixed (independent of duration or tempo)

1525  // - the grace note duration as notated does not matter

1535  weightb=0.0;
1536  weighta=1.0;
1537  }
1538  else if (chord->dots() == 1)
1539  ontime = floor(667*weightb);
1540  else if (chord->dots() == 2)
1541  ontime = floor(571*weightb);
1542  else
1543  ontime = floor(500*weightb);
1544 
1545  int graceDuration = ontime / nb;
1546 
1547  int on = 0;
1548  for (int i = 0; i < nb; ++i) {
1549  QList<NoteEventList> el;
1550  Chord* gc = gnb.at(i);
1551  int nn = gc->notes().size();
