367 QVariant getProperty(P_ID id, XmlReader& e)
368  {
369  switch (propertyType(id)) {
370  case P_TYPE::BOOL:
371  return QVariant(bool(e.readInt()));
372  case P_TYPE::SUBTYPE:
373  case P_TYPE::ZERO_INT:
374  case P_TYPE::INT:
375  return QVariant(e.readInt());
376  case P_TYPE::REAL:
377  case P_TYPE::SPATIUM:
378  case P_TYPE::SP_REAL:
379  case P_TYPE::TEMPO:
380  return QVariant(e.readDouble());
381  case P_TYPE::FRACTION:
382  return QVariant::fromValue(e.readFraction());
383  case P_TYPE::COLOR:
384  return QVariant(e.readColor());
385  case P_TYPE::POINT:
386  return QVariant(e.readPoint());
387  case P_TYPE::SCALE:
388  case P_TYPE::SIZE:
389  return QVariant(e.readSize());
390  case P_TYPE::FONT:
391  case P_TYPE::STRING:
392  return QVariant(e.readElementText());
393  case P_TYPE::GLISSANDO_STYLE: {
394  QString value(e.readElementText());
395  if ( value == "whitekeys")
396  return QVariant(int(MScore::GlissandoStyle::WHITE_KEYS));
397  else if ( value == "blackkeys")
398  return QVariant(int(MScore::GlissandoStyle::BLACK_KEYS));
399  else if ( value == "diatonic")
400  return QVariant(int(MScore::GlissandoStyle::DIATONIC));
401  else // e.g., normally "Chromatic"
402  return QVariant(int(MScore::GlissandoStyle::CHROMATIC));
403  }
404  break;
405  case P_TYPE::ORNAMENT_STYLE: {
406  QString value(e.readElementText());
407  if ( value == "baroque")
408  return QVariant(int(MScore::OrnamentStyle::BAROQUE));
409  return QVariant(int(MScore::OrnamentStyle::DEFAULT));
410  }
411 
412  case P_TYPE::DIRECTION:
413  return QVariant::fromValue(Direction(e.readElementText()));
414 
415  case P_TYPE::DIRECTION_H:
416  {
417  QString value(e.readElementText());
418  if (value == "left" || value == "1")
419  return QVariant(int(MScore::DirectionH::LEFT));
420  else if (value == "right" || value == "2")
421  return QVariant(int(MScore::DirectionH::RIGHT));
422  else if (value == "auto")
423  return QVariant(int(MScore::DirectionH::AUTO));
424  }
425  break;
426  case P_TYPE::LAYOUT_BREAK: {
427  QString value(e.readElementText());
428  if (value == "line")
429  return QVariant(int(LayoutBreak::LINE));
430  if (value == "page")
431  return QVariant(int(LayoutBreak::PAGE));
432  if (value == "section")
433  return QVariant(int(LayoutBreak::SECTION));
434  if (value == "nobreak")
435  return QVariant(int(LayoutBreak::NOBREAK));
436  qDebug("getProperty: invalid P_TYPE::LAYOUT_BREAK: <%s>", qPrintable(value));
437  }
438  break;
439  case P_TYPE::VALUE_TYPE: {
440  QString value(e.readElementText());
441  if (value == "offset")
442  return QVariant(int(Note::ValueType::OFFSET_VAL));
443  else if (value == "user")
444  return QVariant(int(Note::ValueType::USER_VAL));
445  }
446  break;
447  case P_TYPE::PLACEMENT: {
448  QString value(e.readElementText());
449  if (value == "above")
450  return QVariant(int(Element::Placement::ABOVE));
451  else if (value == "below")
452  return QVariant(int(Element::Placement::BELOW));
453  }
454  break;
455  case P_TYPE::BARLINE_TYPE: {
456  bool ok;
457  const QString& val(e.readElementText());
458  int ct = val.toInt(&ok);
459  if (ok)
460  return QVariant(ct);
461  else {
462  BarLineType t = BarLine::barLineType(val);
463  return QVariant::fromValue(t);
464  }
465  }
466  break;
467  case P_TYPE::BEAM_MODE: // TODO
468  return QVariant(int(0));
469 
470  case P_TYPE::GROUPS:
471  {
472  Groups g;
473  g.read(e);
474  return QVariant::fromValue(g);
475  }
476  case P_TYPE::SYMID:
477  return QVariant::fromValue(Sym::name2id(e.readElementText()));
478  break;
479  case P_TYPE::HEAD_GROUP:
480  return QVariant::fromValue(NoteHead::name2group(e.readElementText()));;
481  case P_TYPE::HEAD_TYPE:
482  return QVariant::fromValue(NoteHead::name2type(e.readElementText()));
483  case P_TYPE::POINT_MM: // not supported
484  case P_TYPE::TDURATION:
485  case P_TYPE::SIZE_MM:
486  case P_TYPE::TEXT_STYLE:
487  case P_TYPE::INT_LIST:
488  return QVariant();
489  case P_TYPE::SUB_STYLE:
490  return int(subStyleFromName(e.readElementText()));
491  case P_TYPE::ALIGN: {
492  QString s = e.readElementText();
493  QStringList sl = s.split(',');
494  if (sl.size() != 2) {
495  qDebug("bad align text <%s>", qPrintable(s));
496  return QVariant();
497  }
498  Align align = Align::LEFT;
499  if (sl[0] == "center")
500  align = align | Align::HCENTER;
501  else if (sl[0] == "right")
502  align = align | Align::RIGHT;
503  else if (sl[0] == "left")
504  ;
505  else {
506  qDebug("bad align text <%s>", qPrintable(sl[0]));
507  return QVariant();
508  }
509  if (sl[1] == "center")
510  align = align | Align::VCENTER;
511  else if (sl[1] == "bottom")
512  align = align | Align::BOTTOM;
513  else if (sl[1] == "baseline")
514  align = align | Align::BASELINE;
515  else if (sl[1] == "top")
516  ;
517  else {
518  qDebug("bad align text <%s>", qPrintable(sl[1]));
519  return QVariant();
520  }
521  return int(align);
522  }
523  default:
524  qFatal("unhandled PID type");
525  break;
526  }
527  return QVariant();
528  }
