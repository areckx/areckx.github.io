125 Element* TBox::drop(EditData& data)
126  {
127  Element* e = data.element;
128  switch (e->type()) {
129  case ElementType::TEXT:
130  {
131  Text* t = toText(e);
132  _text->undoSetText(t->xmlText());
133  _text->undoChangeProperty(P_ID::SUB_STYLE, int(t->subStyle()));
134  delete e;
135  return _text;
136  }
137  default:
138  return VBox::drop(data);
139  }
140  }
