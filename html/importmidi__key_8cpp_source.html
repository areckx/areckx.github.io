44 void assignKeyListToStaff(const KeyList &kl, Staff *staff)
45  {
46  Score* score = staff->score();
47  const int track = staff->idx() * VOICES;
48  Key pkey = Key::C;
49 
50  for (auto it = kl.begin(); it != kl.end(); ++it) {
51  const int tick = it->first;
52  Key key = it->second.key();
53  if ((key == Key::C) && (key == pkey)) // dont insert uneccessary C key
54  continue;
55  pkey = key;
56  KeySig* ks = new KeySig(score);
57  ks->setTrack(track);
58  ks->setGenerated(false);
59  ks->setKey(key);
60  ks->setMag(staff->mag(tick));
61  Measure* m = score->tick2measure(tick);
62  if (!m)
63  continue;
64  Segment* seg = m->getSegment(SegmentType::KeySig, tick);
65  seg->add(ks);
66  }
67  }

69 Key findKey(const QList<MTrack> &tracks)
70  {
71  const int octave = 12;
72  std::vector<int> counts(octave);
73 
74  for (const auto &track: tracks) {
75  if (track.mtrack->drumTrack())
76  continue;
77  for (auto it = track.chords.begin(); it != track.chords.end(); ++it) {
78  const auto next = std::next(it);
79  if (next == track.chords.end())
80  continue;
81 
82  for (const auto &note1: it->second.notes) {
83  for (const auto &note2: next->second.notes) {
84  if (qAbs(note1.pitch - note2.pitch) == 1)
85  ++counts[qMin(note1.pitch, note2.pitch) % octave];
86  }
87  }
88  }
89  }

92  {Key::C_B, counts[3] + counts[10]}
93  , {Key::G_B, counts[10] + counts[5]}
94  , {Key::D_B, counts[5] + counts[0]}
95  , {Key::A_B, counts[0] + counts[7]}
96  , {Key::E_B, counts[7] + counts[2]}
97  , {Key::B_B, counts[2] + counts[9]}
98  , {Key::F, counts[9] + counts[4]}
99  , {Key::C, counts[4] + counts[11]}
100  , {Key::G, counts[11] + counts[6]}
101  , {Key::D, counts[6] + counts[1]}
102  , {Key::A, counts[1] + counts[8]}
103  , {Key::E, counts[8] + counts[3]}
104  , {Key::B, counts[3] + counts[10]}
105  , {Key::F_S, counts[10] + counts[5]}
