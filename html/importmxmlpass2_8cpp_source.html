121 // find the duration of the chord starting at or after s in track and ending at tick
122 
123 static int lastChordTicks(const Segment* s, const int track, const int tick)
124  {
125  while (s && s->tick() < tick) {
126  Element* el = s->element(track);
127  if (el && el->isChordRest()) {
128  ChordRest* cr = static_cast<ChordRest*>(el);
129  if (cr->tick() + cr->actualTicks() == tick)
130  return cr->actualTicks();
131  }
132  s = s->nextCR(track, true);
133  }
134  return 0;
135  }

141 // set extend for lyric no in track to end at tick
142 // called when lyric (with or without "extend") or note with "extend type=stop" is found
143 // note that no == -1 means all lyrics in this track
144 
145 void MusicXmlLyricsExtend::setExtend(const int no, const int track, const int tick)
146  {
147  QList<Lyrics*> list;
148  foreach(Lyrics* l, _lyrics) {
149  Element* const el = l->parent();
150  if (el->type() == ElementType::CHORD) { // TODO: rest also possible ?
151  ChordRest* const par = static_cast<ChordRest*>(el);
152  if (par->track() == track && (no == -1 || l->no() == no)) {
153  int lct = lastChordTicks(l->segment(), track, tick);
154  if (lct > 0) {
155  // set lyric tick to the total length fron the lyric note
156  // plus all notes covered by the melisma minus the last note length
157  l->setTicks(tick - par->tick() - lct);
158  }
159  list.append(l);
160  }
161  }
162  }
163  // cleanup
164  foreach(Lyrics* l, list) {
165  _lyrics.remove(l);
166  }
167  }

177 static Fraction noteTypeToFraction(const QString& type)
178  {
179  if (type == "1024th")
180  return Fraction(1, 1024);
181  else if (type == "512th")
182  return Fraction(1, 512);
183  else if (type == "256th")
184  return Fraction(1, 256);
185  else if (type == "128th")
186  return Fraction(1, 128);
187  else if (type == "64th")
188  return Fraction(1, 64);
189  else if (type == "32nd")
190  return Fraction(1, 32);
191  else if (type == "16th")
192  return Fraction(1, 16);
193  else if (type == "eighth")
194  return Fraction(1, 8);
195  else if (type == "quarter")
196  return Fraction(1, 4);
197  else if (type == "half")
198  return Fraction(1, 2);
199  else if (type == "whole")
200  return Fraction(1, 1);
201  else if (type == "breve")
202  return Fraction(2, 1);
203  else if (type == "long")
204  return Fraction(4, 1);

214 
219 static Fraction calculateFraction(const QString& type, const int dots, const Fraction timeMod)
220  {
221  // type
222  Fraction f = noteTypeToFraction(type);
223  if (f.isValid()) {
224  // dot(s)
225  Fraction f_no_dots = f;
226  for (int i = 0; i < dots; ++i)
227  f += (f_no_dots / (2 << i));
228  // tuplet
229  if (timeMod.isValid())
230  f *= timeMod;
231  // clean up (just in case)
232  f.reduce();
233  }
234  return f;
235  }

247 static int MusicXMLStepAltOct2Pitch(int step, int alter, int octave)
248  {
249  // c d e f g a b
250  static int table[7] = { 0, 2, 4, 5, 7, 9, 11 };
251  if (step < 0 || step > 6) {
252  qDebug("MusicXMLStepAltOct2Pitch: illegal step %d", step);
253  return -1;
254  }
255  int pitch = table[step] + alter + (octave+1) * 12;
256 
257  if (pitch < 0)
258  pitch = -1;
259  if (pitch > 127)
260  pitch = -1;
261 
262  return pitch;
263  }

308 static void fillGap(Measure* measure, int track, int tstart, int tend)
309  {
310  int ctick = tstart;
311  int restLen = tend - tstart;
312  // qDebug("\nfillGIFV fillGap(measure %p track %d tstart %d tend %d) restLen %d len",
313  // measure, track, tstart, tend, restLen);
314  // note: as MScore::division (#ticks in a quarter note) equals 480
315  // MScore::division / 64 (#ticks in a 256th note) uequals 7.5 but is rounded down to 7
316  while (restLen > MScore::division / 64) {
317  int len = restLen;
318  TDuration d(TDuration::DurationType::V_INVALID);
319  if (measure->ticks() == restLen)
320  d.setType(TDuration::DurationType::V_MEASURE);
321  else
322  d.setVal(len);
323  Rest* rest = new Rest(measure->score(), d);
324  rest->setDuration(Fraction::fromTicks(len));
325  rest->setTrack(track);
326  rest->setVisible(false);
327  Segment* s = measure->getSegment(SegmentType::ChordRest, tstart);
328  s->add(rest);
329  len = rest->globalDuration().ticks();
330  // qDebug(" %d", len);
331  ctick += len;
332  restLen -= len;
333  }
334  }

337 // fillGapsInFirstVoices
338 //---------------------------------------------------------
339 
344 static void fillGapsInFirstVoices(Measure* measure, Part* part)
345  {
346  Q_ASSERT(measure);
347  Q_ASSERT(part);
348 
349  int measTick = measure->tick();
350  int measLen = measure->ticks();
351  int nextMeasTick = measTick + measLen;
352  int staffIdx = part->score()->staffIdx(part);
353  /*
354  qDebug("fillGIFV measure %p part %p idx %d nstaves %d tick %d - %d (len %d)",
355  measure, part, staffIdx, part->nstaves(),
356  measTick, nextMeasTick, measLen);
357  */
358  for (int st = 0; st < part->nstaves(); ++st) {
359  int track = (staffIdx + st) * VOICES;
360  int endOfLastCR = measTick;
361  for (Segment* s = measure->first(); s; s = s->next()) {
362  // qDebug("fillGIFV segment %p tp %s", s, s->subTypeName());
363  Element* el = s->element(track);
364  if (el) {
365  // qDebug(" el[%d] %p", track, el);
366  if (s->isChordRestType()) {
367  ChordRest* cr = static_cast<ChordRest*>(el);
368  int crTick = cr->tick();
369  int crLen = cr->globalDuration().ticks();
370  int nextCrTick = crTick + crLen;
371  /*
372  qDebug(" chord/rest tick %d - %d (len %d)",
373  crTick, nextCrTick, crLen);
374  */
375  if (crTick > endOfLastCR) {
376  /*
377  qDebug(" GAP: track %d tick %d - %d",
378  track, endOfLastCR, crTick);
379  */
380  fillGap(measure, track, endOfLastCR, crTick);
381  }
382  endOfLastCR = nextCrTick;
383  }
384  }
385  }
386  if (nextMeasTick > endOfLastCR) {
387  /*
388  qDebug("fillGIFV measure end GAP: track %d tick %d - %d",
389  track, endOfLastCR, nextMeasTick);
390  */
391  fillGap(measure, track, endOfLastCR, nextMeasTick);
392  }
393  }

592 //---------------------------------------------------------
593 
598 static QString text2syms(const QString& t)
599  {
600  //QTime time;
601  //time.start();
602 
603  // first create a map from symbol (Unicode) text to symId
604  // note that this takes about 1 msec on a Core i5,
605  // caching does not gain much
606 
607  ScoreFont* sf = ScoreFont::fallbackFont();
608  QMap<QString, SymId> map;
609  int maxStringSize = 0; // maximum string size found
610 
611  for (int i = int(SymId::noSym); i < int(SymId::lastSym); ++i) {
612  SymId id((SymId(i)));
613  QString string(sf->toString(id));
614  // insert all syms except space to prevent matching all regular spaces
615  if (id != SymId::space)
616  map.insert(string, id);
617  if (string.size() > maxStringSize)
618  maxStringSize = string.size();
619  }
620  //qDebug("text2syms map count %d maxsz %d filling time elapsed: %d ms",
621  // map.size(), maxStringSize, time.elapsed());
622 
623  // then look for matches
624  QString in = t;
625  QString res;
626 
627  while (in != "") {
628  // try to find the largest match possible
629  int maxMatch = qMin(in.size(), maxStringSize);
630  QString sym;
631  while (maxMatch > 0) {
632  QString toBeMatched = in.left(maxMatch);
633  if (map.contains(toBeMatched)) {
634  sym = Sym::id2name(map.value(toBeMatched));
635  break;
636  }
637  maxMatch--;
638  }
639  if (maxMatch > 0) {
640  // found a match, add sym to res and remove match from string in
641  res += "<sym>";
642  res += sym;
643  res += "</sym>";
644  in.remove(0, maxMatch);
645  }
646  else {
647  // not found, move one char from res to in
648  res += in.left(1);
649  in.remove(0, 1);
650  }
651  }
652 
653  //qDebug("text2syms total time elapsed: %d ms, res '%s'", time.elapsed(), qPrintable(res));
654  return res;
655  }

1962  || (alter == 0

1965  && !accTmp.value(ln, false))
