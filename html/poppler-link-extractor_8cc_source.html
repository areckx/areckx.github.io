52 void LinkExtractorOutputDev::processLink(::AnnotLink *link)
53 {
54  if (!link->isOk())
55  return;
56 
57  double left, top, right, bottom;
58  int leftAux, topAux, rightAux, bottomAux;
59  link->getRect(&left, &top, &right, &bottom);
60  QRectF linkArea;
61 
62  cvtUserToDev(left, top, &leftAux, &topAux);
63  cvtUserToDev(right, bottom, &rightAux, &bottomAux);
64  linkArea.setLeft((double)leftAux / m_pageCropWidth);
65  linkArea.setTop((double)topAux / m_pageCropHeight);
66  linkArea.setRight((double)rightAux / m_pageCropWidth);
67  linkArea.setBottom((double)bottomAux / m_pageCropHeight);
68 
69  Link *popplerLink = m_data->convertLinkActionToLink(link->getAction(), linkArea);
70  if (popplerLink)
71  {
72  m_links.append(popplerLink);
73  }
74  OutputDev::processLink(link);
75 }
