103 static void SetCapGraceDuration(Chord* chord,ChordObj* o)
104  {

106  if (o->nTremoloBars > 0)
107  nt = NoteType::ACCIACCATURA;
108  ((Chord*)chord)->setNoteType(nt);
109  if (o->t == TIMESTEP::D4) {
110  ((Chord*)chord)->setNoteType(NoteType::GRACE4);
111  chord->setDurationType(TDuration::DurationType::V_QUARTER);
112  }
113  else if (o->t == TIMESTEP::D_BREVE)
114  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_BREVE);
115  else if (o->t == TIMESTEP::D1)
116  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_WHOLE);
117  else if (o->t == TIMESTEP::D2)
118  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_HALF);
119  else if (o->t == TIMESTEP::D16) {
120  ((Chord*)chord)->setNoteType(NoteType::GRACE16);
121  chord->setDurationType(TDuration::DurationType::V_16TH);
122  }
123  else if (o->t == TIMESTEP::D32) {
124  ((Chord*)chord)->setNoteType(NoteType::GRACE32);
125  chord->setDurationType(TDuration::DurationType::V_32ND);
126  }
127  else if (o->t == TIMESTEP::D64)
128  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_64TH);
129  else if (o->t == TIMESTEP::D128)
130  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_128TH);
131  else if (o->t == TIMESTEP::D256)
132  ((Chord*)chord)->setDurationType(TDuration::DurationType::V_256TH);

375 // TupletFractionCap

377 
378 Fraction TupletFractionCap(int tupletCount, bool tuplettrp, bool tupletprol)
379  {
380  int dd = 0;
381  int nn = 0;
382  qreal exponent = 0;
383  qreal count = tupletCount;
384  Fraction f(3,2);
385  if ((count > 0) && (count <= 15)) {
386  if (tuplettrp)
387  exponent = qFloor(qLn(count/3.0)/qLn(2.0));
388  else
389  exponent = qFloor(qLn(count)/qLn(2.0));
390  }
391  else {
392  qDebug("Unknown tuplet, count = %d",tupletCount);
393  return f;
394  }
395  if (tupletprol)
396  exponent += 1.0;
397  if (exponent < 0.0)
398  exponent = 0.0;
399  nn = tupletCount;
400  dd = static_cast<int>(qPow(2.0, exponent));
401  if (tuplettrp)
402  dd = dd * 3;
403  qDebug("Tuplet Fraction: %d / %d",nn,dd);
404  f.setNumerator(nn);
405  f.setDenominator(dd);
406  return f;
407  }

411 // return true on success (both begin and end found)

413 
414 static bool findChordRests(BasicDrawObj const* const o, Score* score, const int track, const int tick,
415  ChordRest*& cr1, ChordRest*& cr2, NoteObj* no, QList<NoteObj*> objects)
416  {
417  cr1 = 0; // ChordRest where BasicDrawObj o begins
418  cr2 = 0; // ChordRest where BasicDrawObj o ends
419 
420  // find the ChordRests where o begins and ends
421  int n = o->nNotes + 1; // # notes in BasicDrawObj (nNotes is # notes following the first note)
422  int graceNumber = 0;
423  int graceNumber1 = 0;
424  bool foundcr1 = false;
425  int tick2 = tick;
426  foreach(NoteObj* nobj, objects) {
427  BasicDurationalObj* d = 0;
428  if (nobj->type() == CapellaNoteObjectType::REST) {
429  d = static_cast<BasicDurationalObj*>(static_cast<RestObj*>(nobj));
430  graceNumber = 0;
431  }
432  else if (nobj->type() == CapellaNoteObjectType::CHORD) {
433  ChordObj* cho = static_cast<ChordObj*>(nobj);
434  d = static_cast<BasicDurationalObj*>(cho);
435  if (!(cho->invisible) && (cho->ticks() == 0)) // grace note
436  ++graceNumber;
437  else
438  graceNumber = 0;
439  }
440  if (!d)
441  continue;
442  if (nobj == no) {
443  foundcr1 = true;
444  graceNumber1 = graceNumber;
445  }
446  int ticks = 0;
447  if (foundcr1) {
448  --n; // found the object corresponding to cr1, count down to find the second one
449  ticks = d->ticks();
450  if (d->count) {
451  Fraction f = TupletFractionCap(d->count, d->tripartite, d->isProlonging);
452  ticks = ticks*f.denominator()/f.numerator();
453  }
454  if (nobj->type() == CapellaNoteObjectType::REST) {
455  RestObj* ro = static_cast<RestObj*>(nobj);
456  if (ro->fullMeasures) {
457  Measure* m = score->getCreateMeasure(tick2);
458  int ft = m->ticks();
459  ticks = ft * ro->fullMeasures;
460  }
461  }
462  if (n == 0)
463  break;
464  tick2 += ticks;
465  }
466  }
467  // Now we have the tick (tick) and the level of grace note (graceNumber1, if "no" is a grace note) for the first ChordRest
468  // and the tick (tick2) and the level of grace note (graceNumber, if the target is a grace note) for the 2nd ChordRest
469  for (Segment* seg = score->tick2segment(tick); seg; seg = seg->next1()) {
470  if (seg->segmentType() != SegmentType::ChordRest)
471  continue;
472  ChordRest* cr = static_cast<ChordRest*>(seg->element(track));
473  if (cr) {
474  if (graceNumber1 > 0) { // the spanner is starting from a grace note
475  Chord* chord = static_cast<Chord*>(cr);
476  foreach(Chord* cc, chord->graceNotes()) {
477  --graceNumber1;
478  if ((graceNumber1 == 0) && (!cr1))
479  cr1 = static_cast<ChordRest*>(cc); // found first ChordRest
480  }
481  }
482  if (!cr1) cr1 = cr; // found first ChordRest
483  break;
484  }
485  }
486  for (Segment* seg = score->tick2segment(tick2); seg; seg = seg->next1()) {
487  if (seg->segmentType() != SegmentType::ChordRest)
488  continue;
489  ChordRest* cr = static_cast<ChordRest*>(seg->element(track));
490  if (cr) {
491  if ((graceNumber > 0) && (cr->type() == ElementType::CHORD)) { // the spanner is ending on a grace note
492  Chord* chord = static_cast<Chord*>(cr);
493  foreach(Chord* cc, chord->graceNotes()) {
494  --graceNumber;
495  if ((graceNumber == 0) && (!cr2))
496  cr2 = static_cast<ChordRest*>(cc); // found 2nd ChordRest
497  }
498  }
499  if (!cr2) cr2 = cr; // found 2nd ChordRest
500  break;
501  }
502  }
503  qDebug("findChordRests o %p nNotes %d score %p track %d tick %d cr1 %p cr2 %p", o, o->nNotes, score, track, tick, cr1, cr2);
504 
505  if (!(cr1 && cr2)) {
506  qDebug("first or second anchor for BasicDrawObj not found (tick %d type %d track %d first %p second %p)",
507  tick, int(o->type), track, cr1, cr2);
508  return false;
509  }
510  return true;
511  }

518  {

521 
522  qDebug("readCapVoice 1");
523  //
524  // pass I
525  //
526  int startTick = tick;
527 
528  Tuplet* tuplet = 0;
529  int tupletCount = 0;
530  bool tuplettrp = false;
531  bool tupletprol = false;
532  int nTuplet = 0;
533  int tupletTick = 0;
534 
535  qDebug(" read voice: tick %d track: %d)", tick, track);
536  QList<Chord*> graceNotes;
537  foreach(NoteObj* no, cvoice->objects) {
538  switch (no->type()) {
539  case CapellaNoteObjectType::REST:
540  {
541  qDebug(" <Rest>");
542  Measure* m = score->getCreateMeasure(tick);
543  RestObj* o = static_cast<RestObj*>(no);
544  int ticks = o->ticks();
545  if (o->invisible && ticks == 0) { // get rid of placeholders
546  break;
547  }
548  TDuration d;
549  d.setVal(ticks);
550  if (o->count) {
551  if (tuplet == 0) {
552  tupletCount = o->count;
553  tuplettrp = o->tripartite;
554  tupletprol = o->isProlonging;
555  nTuplet = 0;
556  tupletTick = tick;
557  tuplet = new Tuplet(score);
558  Fraction f = TupletFractionCap(tupletCount,tuplettrp,tupletprol);
559  tuplet->setRatio(f);
560  tuplet->setBaseLen(d);
561  tuplet->setTrack(track);
562  tuplet->setTick(tick);
563  tuplet->setParent(m);
564  int nn = ((tupletCount * ticks) * f.denominator()) / f.numerator();
565  tuplet->setDuration(Fraction::fromTicks(nn));
566  }
567  }
568 
569  int ft = m->ticks();
570  if (o->fullMeasures) {
571  ticks = ft * o->fullMeasures;
572  if (!o->invisible) {
573  for (unsigned i = 0; i < o->fullMeasures; ++i) {
574  Measure* m = score->getCreateMeasure(tick + i * ft);
575  Segment* s = m->getSegment(SegmentType::ChordRest, tick + i * ft);
576  Rest* rest = new Rest(score);
577  rest->setDurationType(TDuration(TDuration::DurationType::V_MEASURE));
578  rest->setDuration(m->len());
579  rest->setTrack(staffIdx * VOICES + voice);
580  s->add(rest);
581  }
582  }
583  }
584  if (!o->invisible || voice == 0) {
585  Segment* s = m->getSegment(SegmentType::ChordRest, tick);
586  Rest* rest = new Rest(score);
587  if (tuplet) {
588  rest->setTuplet(tuplet);
589  tuplet->add(rest);
590  }
591  TDuration d;
592  if (o->fullMeasures) {
593  d.setType(TDuration::DurationType::V_MEASURE);
594  rest->setDuration(m->len());
595  }
596  else {
597  d.setVal(ticks);
598  rest->setDuration(d.fraction());
599  }
600  rest->setDurationType(d);
601  rest->setTrack(track);
602  rest->setVisible(!o->invisible);
603  s->add(rest);
604  processBasicDrawObj(o->objects, s, track, rest);
605  }
606 
607  if (tuplet) {
608  if (++nTuplet >= tupletCount) {
609  tick = tupletTick + tuplet->actualTicks();
610  tuplet = 0;
611  }
612  else {
613  tick += (ticks * tuplet->ratio().denominator()) / tuplet->ratio().numerator();
614  }
615  }
616  else
617  tick += ticks;
618  }
619  break;
620  case CapellaNoteObjectType::CHORD:
621  {
622  qDebug(" <Chord>");
623  ChordObj* o = static_cast<ChordObj*>(no);
624  int ticks = o->ticks();
625  if (o->invisible && ticks == 0) { // get rid of placeholders
626  break;
627  }
628  TDuration d;
629  d.setVal(ticks);
630  Measure* m = score->getCreateMeasure(tick);
631 
632  bool isgracenote = (!(o->invisible) && (ticks==0));
633  if (o->count) {
634  if (tuplet == 0) {
635  tupletCount = o->count;
636  tuplettrp = o->tripartite;
637  tupletprol = o->isProlonging;
638  nTuplet = 0;
639  tupletTick = tick;
640  tuplet = new Tuplet(score);
641  Fraction f = TupletFractionCap(tupletCount,tuplettrp,tupletprol);
642  tuplet->setRatio(f);
643  tuplet->setBaseLen(d);
644  tuplet->setTrack(track);
645  tuplet->setTick(tick);
646  tuplet->setParent(m);
647  int nn = ((tupletCount * ticks) * f.denominator()) / f.numerator();
