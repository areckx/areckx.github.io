33 bool ScoreView::event(QEvent* event)
34  {
35  if (event->type() == QEvent::KeyPress && editData.element) {
36  QKeyEvent* ke = static_cast<QKeyEvent*>(event);
37  if (ke->key() == Qt::Key_Tab || ke->key() == Qt::Key_Backtab) {
38  if (editData.element->isText())
39  return true;
40  bool rv = true;
41  if (ke->key() == Qt::Key_Tab) {
42  rv = editData.element->nextGrip(editData);
43  updateGrips();
44  _score->update();
45  if (rv)
46  return true;
47  }
48  else if (ke->key() == Qt::Key_Backtab)
49  rv = editData.element->prevGrip(editData);
50  updateGrips();
51  _score->update();
52  if (rv)
53  return true;
54  }
55  }
56 // else if (event->type() == CloneDrag) {
57 //TODO:drag Element* e = static_cast<CloneEvent*>(event)->element();
58 // cloneElement(e);
59 // }
60  else if (event->type() == QEvent::Gesture) {
61  return gestureEvent(static_cast<QGestureEvent*>(event));
62  }
63  else if (event->type() == QEvent::MouseButtonPress && qApp->focusWidget() != this) {
64  QMouseEvent* me = static_cast<QMouseEvent*>(event);
65  if (me->button() == Qt::LeftButton)
66  this->setFocus();
67  }
68  return QWidget::event(event);
69  }

76 bool ScoreView::gestureEvent(QGestureEvent *event)
77  {
78  if (QGesture *gesture = event->gesture(Qt::PinchGesture)) {
79  // Zoom in/out when receiving a pinch gesture
80  QPinchGesture *pinch = static_cast<QPinchGesture *>(gesture);
81 
82  static qreal magStart = 1.0;
83  if (pinch->state() == Qt::GestureStarted) {
84  magStart = lmag();
85  }
86  if (pinch->changeFlags() & QPinchGesture::ScaleFactorChanged) {
87  // On Windows, totalScaleFactor() contains the net magnification.
88  // On OS X, totalScaleFactor() is 1, and scaleFactor() contains the net magnification.
89  qreal value = pinch->totalScaleFactor();
90  if (value == 1) {
91  value = pinch->scaleFactor();
92  }
93  zoom(magStart*value, pinch->centerPoint());
94  }
95  }
96  return true;
97  }

103 void ScoreView::wheelEvent(QWheelEvent* event)
104  {
105 #define PIXELSSTEPSFACTOR 5
106 
107  QPoint pixelsScrolled = event->pixelDelta();
108  QPoint stepsScrolled = event->angleDelta();
109 
110  int dx = 0, dy = 0, n = 0;
111  qreal nReal = 0.0;
112 
113  if (!pixelsScrolled.isNull()) {
114  dx = pixelsScrolled.x();
115  dy = pixelsScrolled.y();
116  nReal = static_cast<qreal>(dy) / PIXELSSTEPSFACTOR;
117  }
118  else if (!stepsScrolled.isNull()) {
119  dx = static_cast<qreal>(stepsScrolled.x()) * qMax(2, width() / 10) / 120;
120  dy = static_cast<qreal>(stepsScrolled.y()) * qMax(2, height() / 10) / 120;
121  nReal = static_cast<qreal>(stepsScrolled.y()) / 120;
122  }
123 
124  n = (int) nReal;
125 
126  //this functionality seems currently blocked by the context menu
127  if (event->buttons() & Qt::RightButton) {
128  bool up = n > 0;
129  if (!up)
130  n = -n;
131  score()->startCmd();
132  for (int i = 0; i < n; ++i)
133  score()->upDown(up, UpDownMode::CHROMATIC);
134  score()->endCmd();
135  return;
136  }
137 
138  if (event->modifiers() & Qt::ControlModifier) { // Windows touch pad pinches also execute this
139  QApplication::sendPostedEvents(this, 0);
140  zoomStep(nReal, event->pos());
141  return;
142  }
143 
144  //make shift+scroll go horizontally
145  if (event->modifiers() & Qt::ShiftModifier && dx == 0) {
146  dx = dy;
147  dy = 0;
148  }
149 
150  if (dx == 0 && dy == 0)
151  return;
152 
153  constraintCanvas(&dx, &dy);
154 
155  _matrix.setMatrix(_matrix.m11(), _matrix.m12(), _matrix.m13(), _matrix.m21(),
156  _matrix.m22(), _matrix.m23(), _matrix.dx()+dx, _matrix.dy()+dy, _matrix.m33());
157  imatrix = _matrix.inverted();
158 
159  scroll(dx, dy, QRect(0, 0, width(), height()));
160  emit viewRectChanged();
161  emit offsetChanged(_matrix.dx(), _matrix.dy());
162  }

288 void ScoreView::mousePressEventNormal(QMouseEvent* ev)
289  {
290  _score->masterScore()->cmdState().reset(); // DEBUG: should not be necessary
291 
292  Qt::KeyboardModifiers keyState = ev->modifiers();
293  SelectType st = SelectType::SINGLE;
294  if (keyState == Qt::NoModifier)
295  st = SelectType::SINGLE;
296  else if (keyState & Qt::ShiftModifier)
297  st = SelectType::RANGE;
298  else if (keyState & Qt::ControlModifier)
299  st = SelectType::ADD;
300  Element* e = editData.element;
301  if (e) {
302  if (keyState == (Qt::ShiftModifier | Qt::ControlModifier)) {
303  cloneElement(e);
304  return;
305  }
306  if (e->isKeySig() && (keyState != Qt::ControlModifier) && st == SelectType::SINGLE) {
307  // special case: select for all staves
308  Segment* s = toKeySig(e)->segment();
309  bool first = true;
310  for (int staffIdx = 0; staffIdx < _score->nstaves(); ++staffIdx) {
311  Element* e = s->element(staffIdx * VOICES);
312  if (e) {
313  e->score()->select(e, first ? SelectType::SINGLE : SelectType::ADD);
314  first = false;
315  }
316  }
317  }
318  else {
319  if (st == SelectType::ADD && e->selected())
320  e->score()->deselect(e);
321  else
322  e->score()->select(e, st, -1);
323  }
324  if (e->isText())
325  changeState(ViewState::EDIT);
326  else if (e->isNote())
327  mscore->play(e);
328  if (e) {
329  _score = e->score();
330  _score->setUpdateAll();
331  }
332  }
333  else {
334  // special case: chacke if measure is selected
335  int staffIdx;
336  Measure* m = _score->pos2measure(editData.startMove, &staffIdx, 0, 0, 0);
337  if (m && m->staffLines(staffIdx)->abbox().contains(editData.startMove)) {
338  _score->select(m, st, staffIdx);
339  _score->setUpdateAll();
340  }
341  else
342  _score->deselectAll();
343  }
344  _score->update();
345  mscore->endCmd();
346  }

352 void ScoreView::mousePressEvent(QMouseEvent* ev)
353  {
354  editData.startMove = toLogical(ev->pos());
355  editData.buttons = ev->buttons();
356  Element* e = elementNear(editData.startMove);
357  qDebug("%s", e ? e->name() : "--");
358 
359  switch (state) {
360  case ViewState::NORMAL:
361  if (ev->button() == Qt::RightButton) // context menu?
362  break;
363  editData.element = e;
364  mousePressEventNormal(ev);
365  break;
366 
367  case ViewState::FOTO:
368  editData.element = _foto;
369  if (editElementDragTransition(ev))
370  changeState(ViewState::FOTO_DRAG_EDIT);
371  else if (_foto->canvasBoundingRect().contains(editData.startMove))
372  changeState(ViewState::FOTO_DRAG_OBJECT);
373  else
374  changeState(ViewState::FOTO_DRAG);
375  break;
376 
377  case ViewState::NOTE_ENTRY:
378  noteEntryButton(ev);
379  break;
380 
381  case ViewState::EDIT:
382  if (editElementDragTransition(ev))
383  changeState(ViewState::DRAG_EDIT);
384  else {
385  endEdit();
386  editData.element = e;
387  if (e && e->isText()) {
388  changeState(ViewState::EDIT);
389  _score->update();
390  mscore->endCmd();
391  }
392  else {
393  changeState(ViewState::NORMAL);
394  mousePressEventNormal(ev);
395  }
396  }
397  break;
398 
399  default:
400  qDebug("mousePressEvent in state %d", int(state));
401  break;
402  }
403  }

442 void ScoreView::_keyPressEvent(QKeyEvent* ev)
443  {
444  if (editData.element->isLyrics()) {
445  if (editKeyLyrics(ev))
446  return;
447  }
448  else if (editData.element->isHarmony()) {
449  if (editData.key == Qt::Key_Space && !(editData.modifiers & CONTROL_MODIFIER)) {
450  harmonyBeatsTab(true, editData.modifiers & Qt::ShiftModifier);
451  return;
452  }
453  }
454  else if (editData.element->isFiguredBass()) {
455  if (editData.key == Qt::Key_Space && !(editData.modifiers & CONTROL_MODIFIER)) {
456  figuredBassTab(false, editData.modifiers & Qt::ShiftModifier);
457  return;
458  }
459  }
460 
461 #ifdef Q_OS_WIN // Japenese IME on Windows needs to know when Contrl/Alt/Shift/CapsLock is pressed while in predit
462  if (editData.element->isText()) {
463  Text* text = toText(editData.element);
464  if (text->cursor(editData)->format()->preedit() && QGuiApplication::inputMethod()->locale().script() == QLocale::JapaneseScript &&
465  ((editData.key == Qt::Key_Control || (editData.modifiers & Qt::ControlModifier)) ||
466  (editData.key == Qt::Key_Alt || (editData.modifiers & Qt::AltModifier)) ||
467  (editData.key == Qt::Key_Shift || (editData.modifiers & Qt::ShiftModifier)) ||
468  (editData.key == Qt::Key_CapsLock))) {
469  return; // musescore will ignore this key event so that the IME can handle it
470  }
471  }
472 #endif
473 
474  if (!((editData.modifiers & Qt::ShiftModifier) && (editData.key == Qt::Key_Backtab))) {
475  if (editData.element->edit(editData)) {
476  if (editData.element->isText())
477  mscore->textTools()->updateTools(editData);
478  updateGrips();
479  return;
480  }
481  }
482  QPointF delta;
483  qreal _spatium = editData.element->spatium();
484 
485  qreal xval, yval;
486  if (editData.element->isBeam()) {
487  xval = 0.25 * _spatium;
488  if (editData.modifiers & Qt::ControlModifier)
489  xval = _spatium;
490  else if (editData.modifiers & Qt::AltModifier)
491  xval = 4 * _spatium;
492  }
493  else {
494  xval = MScore::nudgeStep * _spatium;
495  if (editData.modifiers & Qt::ControlModifier)
496  xval = MScore::nudgeStep10 * _spatium;
497  else if (editData.modifiers & Qt::AltModifier)
498  xval = MScore::nudgeStep50 * _spatium;
499  }
500  yval = xval;
501 
502  if (mscore->vRaster()) {
503  qreal vRaster = _spatium / MScore::vRaster();
504  if (yval < vRaster)
505  yval = vRaster;
506  }
507  if (mscore->hRaster()) {
508  qreal hRaster = _spatium / MScore::hRaster();
509  if (xval < hRaster)
510  xval = hRaster;
511  }
512  // TODO: if raster, then xval/yval should be multiple of raster
513 
514  switch (editData.key) {
515  case Qt::Key_Left:
516  delta = QPointF(-xval, 0);
517  break;
518  case Qt::Key_Right:
519  delta = QPointF(xval, 0);
520  break;
521  case Qt::Key_Up:
522  delta = QPointF(0, -yval);
523  break;
524  case Qt::Key_Down:
525  delta = QPointF(0, yval);
526  break;
527  default:
528  ev->ignore();
529  return;
530  }
531  editData.init();
532  editData.delta = delta;
533  editData.view = this;
534  editData.hRaster = mscore->hRaster();
535  editData.vRaster = mscore->vRaster();
536  if (editData.curGrip != Grip::NO_GRIP && int(editData.curGrip) < editData.grips)
537  editData.pos = editData.grip[int(editData.curGrip)].center() + delta;
538  editData.element->startEditDrag(editData);
539  editData.element->editDrag(editData);
540  editData.element->endEditDrag(editData);
541  updateGrips();
542  }
