160  tmpVarList << QVariant::fromValue(QString() + QChar((ushort)1) + QChar((ushort)2) + QChar((ushort)3) + QChar((ushort)3)); // (("\u0001\u0002\u0003\u0003"));
161  tmpVarList << QVariant::fromValue(tmpVector);

177  tmpVarList.clear();

186  assertEquals("diff_linesToChars: More than 256 (setup).", n, chars.length());
