250 const QColor& StyleHelper::calcMidColor(const QColor& color) const {
251  const quint64 key(color.rgba());
252  QColor* out = m_midColorCache.object(key);
253  if (!out) {
254  out = new QColor(ColorScheme::shade(color, ColorScheme::MidShade, _contrast - 1.0));
255  m_midColorCache.insert(key, out);
256  }
257  return *out;
258  }
