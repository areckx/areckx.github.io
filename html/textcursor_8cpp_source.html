54 void PositionCursor::paint(QPainter* p)
55  {
56  if (!visible())
57  return;
58  QPointF points[3];
59  qreal h = _sv->score()->spatium() * 2;
60 
61  qreal x = _rect.left();
62  qreal y = _rect.top();
63 
64  switch(_type) {
65  case CursorType::LOOP_IN: // draw a right-pointing triangle
66  {
67  qreal tx = x - 1.0;
68  p->setPen(QPen(_color, 2.0, Qt::SolidLine, Qt::FlatCap, Qt::MiterJoin));
69  p->drawLine(x, y, x, _rect.bottom());
70  points[0] = QPointF(tx, y);
71  points[1] = QPointF(tx, y + h);
72  points[2] = QPointF(tx + h, y + h * .5);
73  p->setBrush(_color);
74  p->drawConvexPolygon(points, 3);
75  }
76  break;
77  case CursorType::LOOP_OUT: // draw a left-pointing triangle
78  p->setPen(QPen(_color, 2.0, Qt::SolidLine, Qt::FlatCap, Qt::MiterJoin));
79  p->drawLine(x, y, x, _rect.bottom());
80  points[0] = QPointF(x, y);
81  points[1] = QPointF(x, y + h);
82  points[2] = QPointF(x - h, y + h * .5);
83  p->setBrush(_color);
84  p->drawConvexPolygon(points, 3);
85  break;
86  default: // fill the rectangle and add TAB string marks, if required
87  p->fillRect(_rect, color());
88  if (_sv->score()->noteEntryMode()) {
89  int track = _sv->score()->inputTrack();
90  if (track >= 0) {
91  Staff* staff = _sv->score()->staff(track2staff(track));
92  StaffType* staffType = staff->staffType(0);
93  if (staffType && staffType->group() == StaffGroup::TAB)
94  staffType->drawInputStringMarks(p, _sv->score()->inputState().string(),
95  track2voice(track), _rect);
96  }
97  }
98  break;
99  }
100  }

106 QRectF PositionCursor::bbox() const
107  {
108  QRectF r;
109  qreal h = _sv->score()->spatium() * 2;
110 
111  switch(_type) {
112  case CursorType::LOOP_IN:
113  r.setRect(_rect.x(), _rect.y(), h, _rect.height());
114  break;
115  case CursorType::LOOP_OUT:
116  r.setRect(_rect.x() - h, _rect.y(), h, _rect.height());
117  break;
118  default:
119  r = _rect;
120  break;
121  }
122  return r.adjusted(-2, -2, 2, 2);
123  }

130 void PositionCursor::move(int tick)
131  {
132  QRectF r(bbox());
133  //
134  // set mark height for whole system
135  //
136  if (_type == CursorType::LOOP_OUT)
137  tick --;
138  Score* score = _sv->score();
139  Measure* measure = score->tick2measureMM(tick);
140  if (measure == 0)
141  return;
142  qreal x;
143  int offset = 0;
144 
145  Segment* s;
146  for (s = measure->first(SegmentType::ChordRest); s;) {
147  int t1 = s->tick();
148  int x1 = s->canvasPos().x();
149  qreal x2;
150  int t2;
151  Segment* ns = s->next(SegmentType::ChordRest);
152  if (ns) {
153  t2 = ns->tick();
154  x2 = ns->canvasPos().x();
155  }
156  else {
157  t2 = measure->endTick();
158  x2 = measure->canvasPos().x() + measure->width();
159  }
160  t1 += offset;
161  t2 += offset;
162  if (tick >= t1 && tick < t2) {
163  int dt = t2 - t1;
164  qreal dx = x2 - x1;
165  x = x1 + dx * (tick-t1) / dt;
166  break;
167  }
168  s = ns;
169  }
170  if (s == 0)
171  return;
172 
173  System* system = measure->system();
174  if (system == 0)
175  return;
176  double y = system->staffYpage(0) + system->page()->pos().y();
177  double _spatium = score->spatium();
178 
179  qreal mag = _spatium / SPATIUM20;
180  double w = (_spatium * 2.0 + score->scoreFont()->width(SymId::noteheadBlack, mag))/3;
181  double h = 6 * _spatium;
182  //
183  // set cursor height for whole system
184  //
185  double y2 = 0.0;
186 
187  for (int i = 0; i < score->nstaves(); ++i) {
188  SysStaff* ss = system->staff(i);
189  if (!ss->show() || !score->staff(i)->show())
190  continue;
191  y2 = ss->y() + ss->bbox().height();
192  }
193  h += y2;
194  y -= 3 * _spatium;
195 
196  if (_type == CursorType::LOOP_IN) {
197  x = x - _spatium + w/1.5;
198  }
199  else {
200  x = x - _spatium * .5;
201  }
202  _tick = tick;
203  _rect = QRectF(x, y, w, h);
204  _sv->update(_sv->matrix().mapRect(r | bbox()).toRect().adjusted(-1,-1,1,1));
205  }
206 }
