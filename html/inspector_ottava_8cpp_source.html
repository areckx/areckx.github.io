100 void InspectorOttava::updateLineType()
101  {
102  TextLineBaseSegment* ts = static_cast<TextLineBaseSegment*>(inspector->element());
103  TextLineBase* t = ts->textLineBase();
104  bool userDash = t->lineStyle() == Qt::CustomDashLine;
105 
106  l.dashLineLength->setVisible(userDash);
107  l.dashGapLength->setVisible(userDash);
108  l.resetDashLineLength->setVisible(userDash);
109  l.resetDashGapLength->setVisible(userDash);
110  l.dashLineLengthLabel->setVisible(userDash);
111  l.dashGapLengthLabel->setVisible(userDash);
112  }

148 void InspectorOttava::setElement()
149  {
150  TextLineBaseSegment* ts = static_cast<TextLineBaseSegment*>(inspector->element());
151  TextLineBase* t = ts->textLineBase();
152 
153  bool bt = !t->beginText().isEmpty();
154  bool ct = !t->continueText().isEmpty();
155  bool et = !t->endText().isEmpty();
156 
157  tl.hasBeginText->setChecked(bt);
158  tl.hasContinueText->setChecked(ct);
159  tl.hasEndText->setChecked(et);
160 
161  tl.beginWidget->setVisible(bt);
162  tl.continueWidget->setVisible(ct);
163  tl.endWidget->setVisible(et);
164 
165  updateLineType();
166  updateBeginHookType();
167  updateEndHookType();
168 
169  InspectorElementBase::setElement();
170  }

176 void InspectorOttava::valueChanged(int idx)
177  {
178  InspectorBase::valueChanged(idx);
179  if (iList[idx].t == P_ID::LINE_STYLE)
180  updateLineType();
181  else if (iList[idx].t == P_ID::BEGIN_HOOK_TYPE)
182  updateBeginHookType();
183  else if (iList[idx].t == P_ID::END_HOOK_TYPE)
184  updateEndHookType();
185  }
