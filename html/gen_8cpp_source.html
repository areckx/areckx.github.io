97 void fluid_gen_set_default_values(Generator* gen)
98  {
99  for (int i = 0; i < GEN_LAST; i++) {
100  gen[i].flags = GEN_UNUSED;
101  gen[i].mod = 0.0;
102  gen[i].nrpn = 0.0;
103  gen[i].val = fluid_gen_info[i].def;
104  }
105  }

108 /* fluid_gen_init
109  *
110  * Set an array of generators to their initial value
111  */
112 void fluid_gen_init(Generator* gen, Channel* channel)
113  {
114  fluid_gen_set_default_values(gen);
115 
116  for (int i = 0; i < GEN_LAST; i++) {
117  gen[i].nrpn = channel->getGen(i);
118 
119  /* This is an extension to the SoundFont standard. More
120  * documentation is available at the fluid_synth_set_gen2()
121  * function. */
122  if (channel->getGenAbs(i))
123  gen[i].flags = GEN_ABS_NRPN;
124  }
125  }

127 float fluid_gen_scale(int gen, float value)
128  {
129  return (fluid_gen_info[gen].min
130  + value * (fluid_gen_info[gen].max - fluid_gen_info[gen].min));
131  }

133 float fluid_gen_scale_nrpn(int gen, int data)
134  {
135  float value = qBound(-8192.0, (float) data - 8192.0, 8192.0);
136  return value * (float) fluid_gen_info[gen].nrpn_scale;
