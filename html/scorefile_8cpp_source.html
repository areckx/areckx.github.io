62 static void writeMeasure(XmlWriter& xml, MeasureBase* m, int staffIdx, bool writeSystemElements, bool forceTimeSig)
63  {
64  //
65  // special case multi measure rest
66  //
67  if (m->isMeasure() || staffIdx == 0)
68  m->write(xml, staffIdx, writeSystemElements, forceTimeSig);
69 
70  if (m->score()->styleB(StyleIdx::createMultiMeasureRests) && m->isMeasure() && toMeasure(m)->mmRest())
71  toMeasure(m)->mmRest()->write(xml, staffIdx, writeSystemElements, forceTimeSig);
72 
73  xml.setCurTick(m->endTick());
74  }

77 // writeMovement
78 //---------------------------------------------------------
79 
80 void Score::writeMovement(XmlWriter& xml, bool selectionOnly)
81  {
82  // if we have multi measure rests and some parts are hidden,
83  // then some layout information is missing:
84  // relayout with all parts set visible
85 
86  QList<Part*> hiddenParts;
87  bool unhide = false;
88  if (styleB(StyleIdx::createMultiMeasureRests)) {
89  for (Part* part : _parts) {
90  if (!part->show()) {
91  if (!unhide) {
92  startCmd();
93  unhide = true;
94  }
95  part->undoChangeProperty(P_ID::VISIBLE, true);
96  hiddenParts.append(part);
97  }
98  }
99  }
100  if (unhide) {
101  doLayout();
102  for (Part* p : hiddenParts)
103  p->setShow(false);
104  }
105 
106  xml.stag("Score");
107  if (excerpt()) {
108  Excerpt* e = excerpt();
109  QMultiMap<int, int> trackList = e->tracks();
110  QMapIterator<int, int> i(trackList);
111  if (!(trackList.size() == e->parts().size() * VOICES) && !trackList.isEmpty()) {
112  while (i.hasNext()) {
113  i.next();
114  xml.tagE(QString("Tracklist sTrack=\"%1\" dstTrack=\"%2\"").arg(i.key()).arg(i.value()));
115  }
116  }
117  }
118 
119  if (_layoutMode == LayoutMode::LINE)
120  xml.tag("layoutMode", "line");
121 
122 #ifdef OMR
123  if (masterScore()->omr() && xml.writeOmr())
124  masterScore()->omr()->write(xml);
125 #endif
126  if (isMaster() && masterScore()->showOmr() && xml.writeOmr())
127  xml.tag("showOmr", masterScore()->showOmr());
128  if (_audio && xml.writeOmr()) {
129  xml.tag("playMode", int(_playMode));
130  _audio->write(xml);
131  }
132 
133  for (int i = 0; i < 32; ++i) {
134  if (!_layerTags[i].isEmpty()) {
135  xml.tag(QString("LayerTag id=\"%1\" tag=\"%2\"").arg(i).arg(_layerTags[i]),
136  _layerTagComments[i]);
137  }
138  }
139  int n = _layer.size();
140  for (int i = 1; i < n; ++i) { // dont save default variant
141  const Layer& l = _layer[i];
142  xml.tagE(QString("Layer name=\"%1\" mask=\"%2\"").arg(l.name).arg(l.tags));
143  }
144  xml.tag("currentLayer", _currentLayer);
145 
146  if (isTopScore() && !MScore::testMode)
147  _synthesizerState.write(xml);
148 
149  if (pageNumberOffset())
150  xml.tag("page-offset", pageNumberOffset());
151  xml.tag("Division", MScore::division);
152  xml.setCurTrack(-1);
153 
154  if (isTopScore()) // only top score
155  style().save(xml, true); // save only differences to buildin style
156 
157  xml.tag("showInvisible", _showInvisible);
158  xml.tag("showUnprintable", _showUnprintable);
159  xml.tag("showFrames", _showFrames);
160  xml.tag("showMargins", _showPageborders);
161 
162  QMapIterator<QString, QString> i(_metaTags);
163  while (i.hasNext()) {
164  i.next();
165  // do not output "platform" and "creationDate" in test mode
166  if ((!MScore::testMode && !MScore::saveTemplateMode) || (i.key() != "platform" && i.key() != "creationDate"))
167  xml.tag(QString("metaTag name=\"%1\"").arg(i.key().toHtmlEscaped()), i.value());
168  }
169 
170  xml.setCurTrack(0);
171  int staffStart;
172  int staffEnd;
173  MeasureBase* measureStart;
174  MeasureBase* measureEnd;
175 
176  if (selectionOnly) {
177  staffStart = _selection.staffStart();
178  staffEnd = _selection.staffEnd();
179  // make sure we select full parts
180  Staff* sStaff = staff(staffStart);
181  Part* sPart = sStaff->part();
182  Staff* eStaff = staff(staffEnd - 1);
183  Part* ePart = eStaff->part();
184  staffStart = staffIdx(sPart);
185  staffEnd = staffIdx(ePart) + ePart->nstaves();
186  measureStart = _selection.startSegment()->measure();
187  if (_selection.endSegment())
188  measureEnd = _selection.endSegment()->measure()->next();
189  else
190  measureEnd = 0;
191  }
192  else {
193  staffStart = 0;
194  staffEnd = nstaves();
195  measureStart = first();
196  measureEnd = 0;
197  }
198 
199  // Let's decide: write midi mapping to a file or not
200  masterScore()->checkMidiMapping();
201  for (const Part* part : _parts) {
202  if (!selectionOnly || ((staffIdx(part) >= staffStart) && (staffEnd >= staffIdx(part) + part->nstaves())))
203  part->write(xml);
204  }
205 
206  xml.setCurTrack(0);
207  xml.setTrackDiff(-staffStart * VOICES);
208  if (measureStart) {
209  for (int staffIdx = staffStart; staffIdx < staffEnd; ++staffIdx) {
210  xml.stag(QString("Staff id=\"%1\"").arg(staffIdx + 1 - staffStart));
211  xml.setCurTick(measureStart->tick());
212  xml.setTickDiff(xml.curTick());
213  xml.setCurTrack(staffIdx * VOICES);
214  bool writeSystemElements = (staffIdx == staffStart);
215  bool firstMeasureWritten = false;
216  bool forceTimeSig = false;
217  for (MeasureBase* m = measureStart; m != measureEnd; m = m->next()) {
218  // force timesig if first measure and selectionOnly
219  if (selectionOnly && m->isMeasure()) {
220  if (!firstMeasureWritten) {
221  forceTimeSig = true;
222  firstMeasureWritten = true;
223  }
224  else
225  forceTimeSig = false;
226  }
227  writeMeasure(xml, m, staffIdx, writeSystemElements, forceTimeSig);
228  }
229  xml.etag();
230  }
231  }
232  xml.setCurTrack(-1);
233  if (isMaster()) {
234  if (!selectionOnly) {
235  for (const Excerpt* excerpt : excerpts()) {
236  if (excerpt->partScore() != this)
237  excerpt->partScore()->write(xml, false); // recursion
238  }
239  }
240  }
241  else
242  xml.tag("name", excerpt()->title());
243  xml.etag();
244 
245  if (unhide) {
246  endCmd();
247  EditData ed;
248  ed.init();
249  undoRedo(true, ed); // undo
250  }
251  }

257 void Score::write(XmlWriter& xml, bool selectionOnly)
258  {
259  if (isMaster()) {
260  MasterScore* score = static_cast<MasterScore*>(this);
261  while (score->prev())
262  score = score->prev();
263  while (score) {
264  score->writeMovement(xml, selectionOnly);
265  score = score->next();
266  }
267  }
268  else
269  writeMovement(xml, selectionOnly);
270  }

276 void Score::readStaff(XmlReader& e)
277  {
278  int staff = e.intAttribute("id", 1) - 1;
279  e.initTick(0);
280  e.setTrack(staff * VOICES);
281 
282  if (staff == 0) {
283  while (e.readNextStartElement()) {
284  const QStringRef& tag(e.name());
285 
286  if (tag == "Measure") {
287  Measure* measure = 0;
288  measure = new Measure(this);
289  measure->setTick(e.tick());
290  //
291  // inherit timesig from previous measure
292  //
293  Measure* m = e.lastMeasure(); // measure->prevMeasure();
294  Fraction f(m ? m->timesig() : Fraction(4,4));
295  measure->setLen(f);
296  measure->setTimesig(f);
297 
298  measure->read(e, staff);
299  measure->checkMeasure(staff);
300  if (!measure->isMMRest()) {
301  measures()->add(measure);
302  e.setLastMeasure(measure);
303  e.initTick(measure->tick() + measure->ticks());
304  }
305  else {
306  // this is a multi measure rest
307  // always preceded by the first measure it replaces
308  Measure* m = e.lastMeasure();
309 
310  if (m) {
311  m->setMMRest(measure);
312  measure->setTick(m->tick());
313  }
314  }
315  }
316  else if (tag == "HBox" || tag == "VBox" || tag == "TBox" || tag == "FBox") {
317  MeasureBase* mb = static_cast<MeasureBase*>(Element::name2Element(tag, this));
318  mb->read(e);
319  mb->setTick(e.tick());
320  measures()->add(mb);
321  }
322  else if (tag == "tick")
323  e.initTick(fileDivision(e.readInt()));
324  else
325  e.unknown();
326  }
327  }
328  else {
329  Measure* measure = firstMeasure();
330  while (e.readNextStartElement()) {
331  const QStringRef& tag(e.name());
332 
333  if (tag == "Measure") {
334  if (measure == 0) {
335  qDebug("Score::readStaff(): missing measure!");
336  measure = new Measure(this);
337  measure->setTick(e.tick());
338  measures()->add(measure);
339  }
340  e.initTick(measure->tick());
341  measure->read(e, staff);
342  measure->checkMeasure(staff);
343  if (measure->isMMRest())
344  measure = e.lastMeasure()->nextMeasure();
345  else {
346  e.setLastMeasure(measure);
347  if (measure->mmRest())
348  measure = measure->mmRest();
349  else
350  measure = measure->nextMeasure();
351  }
352  }
353  else if (tag == "tick")
354  e.initTick(fileDivision(e.readInt()));
355  else
356  e.unknown();
357  }
358  }
359  }

370 bool MasterScore::saveFile()
371  {
372  QString suffix = info.suffix();
373  if (info.exists() && !info.isWritable()) {
374  MScore::lastError = QObject::tr("The following file is locked: \n%1 \n\nTry saving to a different location.").arg(info.filePath());
375  return false;
376  }
377  //
378  // step 1
379  // save into temporary file to prevent partially overwriting
380  // the original file in case of "disc full"
381  //
382 
383  QString tempName = info.filePath() + QString(".temp");
384  QFile temp(tempName);
385  if (!temp.open(QIODevice::WriteOnly)) {
386  MScore::lastError = QObject::tr("Open Temp File\n%1\nfailed: %2").arg(tempName).arg(QString(strerror(errno)));
387  return false;
388  }
389  bool rv = suffix == "mscx" ? Score::saveFile(&temp, false) : Score::saveCompressedFile(&temp, info, false);
390  if (!rv) {
391  return false;
392  }
393 
394  if (temp.error() != QFile::NoError) {
395  MScore::lastError = QObject::tr("Save File failed: %1").arg(temp.errorString());
396  return false;
397  }
398  temp.close();
399 
400  QString name(info.filePath());
401  QDir dir(info.path());
402  if (!saved()) {
403  // if file was already saved in this session
404  // save but don't overwrite backup again
405 
406  //
407  // step 2
408  // remove old backup file if exists
409  //
410  QString backupName = QString(".") + info.fileName() + QString(",");
411  if (dir.exists(backupName)) {
412  if (!dir.remove(backupName)) {
413 // if (!MScore::noGui)
414 // QMessageBox::critical(0, QObject::tr("Save File"),
415 // QObject::tr("Removing old backup file ") + backupName + tr(" failed"));
416  }
417  }
418 
419  //
420  // step 3
421  // rename old file into backup
422  //
423  if (dir.exists(name)) {
424  if (!dir.rename(name, backupName)) {
425 // if (!MScore::noGui)
426 // QMessageBox::critical(0, tr("Save File"),
427 // tr("Renaming old file <")
428 // + name + tr("> to backup <") + backupName + tr("> failed"));
429  }
430  }
431 #ifdef Q_OS_WIN
432  QFileInfo fileBackup(dir, backupName);
433  QString backupNativePath = QDir::toNativeSeparators(fileBackup.absoluteFilePath());
434  SetFileAttributes((LPCTSTR)backupNativePath.toLocal8Bit(), FILE_ATTRIBUTE_HIDDEN);
435 #endif
436  }
437  else {
438  // file has previously been saved - remove the old file
439  if (dir.exists(name)) {
440  if (!dir.remove(name)) {
441 // if (!MScore::noGui)
442 // QMessageBox::critical(0, tr("Save File"),
443 // tr("Removing old file") + name + tr(" failed"));
444  }
445  }
446  }
447 
448  //
449  // step 4
450  // rename temp name into file name
451  //
452  if (!QFile::rename(tempName, name)) {
453  MScore::lastError = QObject::tr("Renaming temp. file <%1> to <%2> failed:\n%3").arg(tempName).arg(name).arg(QString(strerror(errno)));
454  return false;
455  }
456  // make file readable by all
457  QFile::setPermissions(name, QFile::ReadOwner | QFile::WriteOwner | QFile::ReadUser
458  | QFile::ReadGroup | QFile::ReadOther);
459 
460  undoStack()->setClean();
461  setSaved(true);
462  info.refresh();
463  update();
464  return true;
465  }

485 
486 QImage Score::createThumbnail()
487  {
488  LayoutMode mode = layoutMode();
489  setLayoutMode(LayoutMode::PAGE);
490  doLayout();
491 
492  Page* page = pages().at(0);
493  QRectF fr = page->abbox();
494  qreal mag = 256.0 / qMax(fr.width(), fr.height());
495  int w = int(fr.width() * mag);
496  int h = int(fr.height() * mag);
497 
498  QImage pm(w, h, QImage::Format_ARGB32_Premultiplied);
499 
500  int dpm = lrint(DPMM * 1000.0);
501  pm.setDotsPerMeterX(dpm);
502  pm.setDotsPerMeterY(dpm);
503  pm.fill(0xffffffff);
504 
505  double pr = MScore::pixelRatio;
506  MScore::pixelRatio = 1.0;
507 
508  QPainter p(&pm);
509  p.setRenderHint(QPainter::Antialiasing, true);
510  p.setRenderHint(QPainter::TextAntialiasing, true);
511  p.scale(mag, mag);
512  print(&p, 0);
513  p.end();
514 
515  MScore::pixelRatio = pr;
516 
517  if (layoutMode() != mode) {
518  setLayoutMode(mode);
519  doLayout();
520  }
521  return pm;
522  }

717 
718 QString readRootFile(MQZipReader* uz, QList<QString>& images)
719  {
720  QString rootfile;
721 
722  QByteArray cbuf = uz->fileData("META-INF/container.xml");
723  if (cbuf.isEmpty()) {
724  qDebug("can't find container.xml");
725  return rootfile;
726  }
727 
728  XmlReader e(0, cbuf);
729 
730  while (e.readNextStartElement()) {
731  if (e.name() != "container") {
732  e.unknown();
733  continue;
734  }
735  while (e.readNextStartElement()) {
736  if (e.name() != "rootfiles") {
737  e.unknown();
738  continue;
739  }
740  while (e.readNextStartElement()) {
