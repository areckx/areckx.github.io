618 //---------------------------------------------------------
619 // Phase
620 /* Purpose:
621 * Playing pointer for voice playback
622 *
623 * When a sample is played back at a different pitch, the playing pointer in the
624 * source sample will not advance exactly one sample per output sample.
625 * This playing pointer is implemented using Phase.
626 * It is a 64 bit number. The higher 32 bits contain the 'index' (number of
627 * the current sample), the lower 32 bits the fractional part.
628 * Access is possible in two ways:
629 * -through the 64 bit part 'b64', if the architecture supports 64 bit integers
630 * -through 'index' and 'fract'
631 * Note: b64 and index / fract share the same memory location!
632 */
633 
634 struct Phase {
635  qint64 data;
636 
637  void operator+=(const Phase& p) { data += p.data; }
638  void setInt(qint32 b) { data = qint64(b) << 32; }
639  void setFloat(double b) {
640  data = (((qint64)(b)) << 32) | (quint32) (((double)(b) - (int)(b)) * (double)FLUID_FRACT_MAX);
641  }
642 
643  void operator-=(const Phase& b) { data -= b.data; }
644  void operator-=(int b) { data -= (qint64(b) << 32); }
645  int index() const { return data >> 32; }
646  quint32 fract() const { return quint32(data & 0xffffffff); }
647  quint32 index_round() const { return quint32((data+0x80000000) >> 32); }
648 
649  Phase() {}
650  Phase(qint64 v) : data(v) {}
651  };
652 
653 /* Purpose:
654  * Takes the fractional part of the argument phase and
655  * calculates the corresponding position in the interpolation table.
656  * The fractional position of the playing pointer is calculated with a quite high
657  * resolution (32 bits). It would be unpractical to keep a set of interpolation
658  * coefficients for each possible fractional part...
659  */
660 #define fluid_phase_fract_to_tablerow(_x) \
661  ((int)(((_x).fract() & FLUID_INTERP_BITS_MASK) >> FLUID_INTERP_BITS_SHIFT))
662 
663 #define fluid_phase_double(_x) \
664  ((double)((_x).index()) + ((double)((_x).fract()) / FLUID_FRACT_MAX))
665 
666 /* Purpose:
667  * The playing pointer is _phase. How many output samples are produced, until the point _p1 in the sample is reached,
668  * if _phase advances in steps of _incr?
669  */
670 #define fluid_phase_steps(_phase,_idx,_incr) \
671  (int)(((double)(_idx) - fluid_phase_double(_phase)) / (double)_incr)
672 
673 /* Purpose:
