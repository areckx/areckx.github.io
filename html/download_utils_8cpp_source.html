48 void DownloadUtils::download()
49  {
50  QUrl url = QUrl::fromEncoded(_target.toLocal8Bit());
51  QNetworkRequest request(url);
52  QEventLoop loop;
53  QNetworkReply* reply = manager.get(request);
54  QObject::connect(reply, SIGNAL(downloadProgress(qint64,qint64)), this, SLOT(downloadProgress(qint64,qint64)));
55  QObject::connect(&manager, SIGNAL(finished(QNetworkReply*)), this, SLOT(downloadFinished(QNetworkReply*)));
56  QObject::connect(reply, SIGNAL(finished()), &loop, SLOT(quit()));
57  loop.exec();
58  QObject::disconnect(reply, SIGNAL(downloadProgress(qint64,qint64)), this, SLOT(downloadProgress(qint64,qint64)));
59  QObject::disconnect(&manager, SIGNAL(finished(QNetworkReply*)), this, SLOT(downloadFinished(QNetworkReply*)));
60  QObject::disconnect(reply, SIGNAL(finished()), &loop, SLOT(quit()));
61  }
