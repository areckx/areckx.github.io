192 template< typename T > void MenuBarDataV2::mouseMoveEvent( const QObject* object ) {
193  const T* local = qobject_cast<const T*>( object );
194  if ( !local ) return;
195  if ( local->activeAction() == currentAction().data() ) return;
196 
197  // check if current position match another action
198  if ( local->activeAction() && local->activeAction()->isEnabled() && !local->activeAction()->isSeparator()) {
199 
200  if ( timer_.isActive() ) timer_.stop();
201 
202  QAction* activeAction( local->activeAction() );
203 
204  // update previous rect if the current action is valid
205  QRect activeRect( local->actionGeometry( activeAction ) );
206  if ( currentAction() ) {
207  if ( !progressAnimation().data()->isRunning() ) {
208 
209  setPreviousRect( currentRect() );
210 
211  }
212  else if ( progress() < 1 && currentRect().isValid() && previousRect().isValid() ) {
213 
214  // re-calculate previous rect so that animatedRect
215  // is unchanged after currentRect is updated
216  // this prevents from having jumps in the animation
217  qreal ratio = progress() / (1.0 - progress());
218  previousRect_.adjust(
219  ratio * ( currentRect().left() - activeRect.left() ),
220  ratio * ( currentRect().top() - activeRect.top() ),
221  ratio * ( currentRect().right() - activeRect.right() ),
222  ratio * ( currentRect().bottom() - activeRect.bottom() ) );
223 
224  }
225 
226  // update current action
227  setCurrentAction( activeAction );
228  setCurrentRect( activeRect );
229  if ( animation().data()->isRunning() ) animation().data()->stop();
230  if ( !progressAnimation().data()->isRunning() ) progressAnimation().data()->start();
231 
232  }
233  else {
234 
235  // update current action
236  setCurrentAction( activeAction );
237  setCurrentRect( activeRect );
238  setPreviousRect( activeRect );
239 
240  clearAnimatedRect();
241  if ( progressAnimation().data()->isRunning() ) progressAnimation().data()->stop();
242  animation().data()->setDirection( Animation::Forward );
243 
244  if ( !animation().data()->isRunning() ) animation().data()->start();
245 
246  }
247 
248  }
249  else if ( currentAction() ) {
250 
251  timer_.start( 150, this );
252 
253  }
254 
255  return;
256 
257  }
