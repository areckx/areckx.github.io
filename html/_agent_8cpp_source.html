58 void Agent::accept(const Event &e, double err, int beats)
59  {
60  beatTime = e.time;
61  events.push_back(e);
62  if (std::fabs(initialBeatInterval - beatInterval - err / correctionFactor)
63  < maxChange * initialBeatInterval) {
64  beatInterval += err / correctionFactor; // Adjust tempo
65  }
66  beatCount += beats;
67  double conFactor = 1.0 - CONF_FACTOR * err
68  / (err > 0 ? postMargin: -preMargin);
69  phaseScore += conFactor * e.salience;
70  }

72 bool Agent::considerAsBeat(const Event &e, AgentList &a)
73  {
74  if (beatTime < 0) { // first event
75  accept(e, 0, 1);
76  return true;
77  }
78  else { // subsequent events
79  EventList::iterator last = events.end();
80  --last;
81  if (e.time - last->time > expiryTime) {
82  phaseScore = -1.0; // flag agent to be deleted
83  return false;
84  }
85  double beats = nearbyint((e.time - beatTime) / beatInterval);
86  double err = e.time - beatTime - beats * beatInterval;
87  if ((beats > 0) && (-preMargin <= err) && (err <= postMargin)) {
88  if (std::fabs(err) > innerMargin) {
89  // Create new agent that skips this event (avoids
90  // large phase jump)
91  a.add(clone());
92  }
93  accept(e, err, (int)beats);
94  return true;
95  }
96  }
97  return false;
98  }

100 void Agent::fillBeats(double start)
101  {
102  EventList::iterator it = events.begin();
103  if (it == events.end())
104  return;
105  double prevBeat = it->time;
106  for (++it; it != events.end(); ++it) {
107  double nextBeat = it->time;
108  double beats = nearbyint((nextBeat - prevBeat) / beatInterval - 0.01); // prefer slow
109  double currentInterval = (nextBeat - prevBeat) / beats;
110  for ( ; (nextBeat > start) && (beats > 1.5); --beats) {
111  prevBeat += currentInterval;
112  events.insert(it, Event(prevBeat, 0));
113  }
114  prevBeat = nextBeat;
115  }
