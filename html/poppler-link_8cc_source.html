240  static void cvtUserToDev(::Page *page, double xu, double yu, int *xd, int *yd) {
241  double ctm[6];
242 
243  page->getDefaultCTM(ctm, 72.0, 72.0, 0, false, true);
244  *xd = (int)(ctm[0] * xu + ctm[2] * yu + ctm[4] + 0.5);
245  *yd = (int)(ctm[1] * xu + ctm[3] * yu + ctm[5] + 0.5);
246  }

248  LinkDestination::LinkDestination(const LinkDestinationData &data)
249  : d( new LinkDestinationPrivate )
250  {
251  bool deleteDest = false;
252  LinkDest *ld = data.ld;
253 
254  if ( data.namedDest && !ld && !data.externalDest )
255  {
256  deleteDest = true;
257  ld = data.doc->doc->findDest( data.namedDest );
258  }
259  // in case this destination was named one, and it was not resolved
260  if ( data.namedDest && !ld )
261  {
262  d->name = QString::fromLatin1( data.namedDest->getCString() );
263  }
264 
265  if (!ld) return;
266 
267  if (ld->getKind() == ::destXYZ) d->kind = destXYZ;
268  else if (ld->getKind() == ::destFit) d->kind = destFit;
269  else if (ld->getKind() == ::destFitH) d->kind = destFitH;
270  else if (ld->getKind() == ::destFitV) d->kind = destFitV;
271  else if (ld->getKind() == ::destFitR) d->kind = destFitR;
272  else if (ld->getKind() == ::destFitB) d->kind = destFitB;
273  else if (ld->getKind() == ::destFitBH) d->kind = destFitBH;
274  else if (ld->getKind() == ::destFitBV) d->kind = destFitBV;
275 
276  if ( !ld->isPageRef() ) d->pageNum = ld->getPageNum();
277  else
278  {
279  Ref ref = ld->getPageRef();
280  d->pageNum = data.doc->doc->findPage( ref.num, ref.gen );
281  }
282  double left = ld->getLeft();
283  double bottom = ld->getBottom();
284  double right = ld->getRight();
285  double top = ld->getTop();
286  d->zoom = ld->getZoom();
287  d->changeLeft = ld->getChangeLeft();
288  d->changeTop = ld->getChangeTop();
289  d->changeZoom = ld->getChangeZoom();
290 
291  int leftAux = 0, topAux = 0, rightAux = 0, bottomAux = 0;
292 
293  if (!data.externalDest) {
294  ::Page *page;
295  if (d->pageNum > 0 &&
296  d->pageNum <= data.doc->doc->getNumPages() &&
297  (page = data.doc->doc->getPage( d->pageNum )))
298  {
299  cvtUserToDev( page, left, top, &leftAux, &topAux );
300  cvtUserToDev( page, right, bottom, &rightAux, &bottomAux );
301 
302  d->left = leftAux / (double)page->getCropWidth();
303  d->top = topAux / (double)page->getCropHeight();
304  d->right = rightAux/ (double)page->getCropWidth();
305  d->bottom = bottomAux / (double)page->getCropHeight();
306  }
307  else d->pageNum = 0;
308  }
309 
310  if (deleteDest) delete ld;
311  }

313  LinkDestination::LinkDestination(const QString &description)
314  : d( new LinkDestinationPrivate )
315  {
316  QStringList tokens = description.split( ';' );
317  d->kind = static_cast<Kind>(tokens.at(0).toInt());
318  d->pageNum = tokens.at(1).toInt();
319  d->left = tokens.at(2).toDouble();
320  d->bottom = tokens.at(3).toDouble();
321  d->right = tokens.at(4).toDouble();
322  d->top = tokens.at(5).toDouble();
323  d->zoom = tokens.at(6).toDouble();
324  d->changeLeft = static_cast<bool>(tokens.at(7).toInt());
325  d->changeTop = static_cast<bool>(tokens.at(8).toInt());
326  d->changeZoom = static_cast<bool>(tokens.at(9).toInt());
327  }
