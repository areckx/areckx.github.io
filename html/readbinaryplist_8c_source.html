45 /* A note about memory management:
46 Strings and possibly other values are unique and because the values
47 associated with IDs are cached, you end up with a directed graph rather
48 than a tree. It is tricky to free the data because if you do a simple
49 depth-first search to free nodes, you will free nodes twice. I decided
50 to allocate memory from blocks of 1024 bytes and keep the blocks in a
51 list associated with but private to this module. So the user should
52 access this module by calling:
53  bplist_read_file() or bplist_read_user_pref() or
54  bplist_read_system_pref()
55 which returns a value. When you are done with the value, call
56  bplist_free_data()
57 This will of course free the value_ptr returned by bplist_read_*()
58 
59 To deal with memory exhaustion (what happens when malloc returns
60 NULL?), use setjmp/longjmp -- a single setjmp protects the whole
61 parser, and allocate uses longjmp to abort. After abort, memory
62 is freed and NULL is returned to caller. There is not much here
63 in the way of error reporting.
64 
65 Memory is obtained by calling allocate which either returns the
66 memory requested or calls longjmp, so callers don't have to check.
67 
68 */

105 /********* MEMORY MANAGEMENT ********/
106 #define BLOCK_SIZE 1024
107 // memory is aligned to multiples of this; assume malloc automatically
108 // aligns to this number and assume this number is > sizeof(void *)
109 #define ALIGNMENT 8
110 static void *block_list = NULL;
111 static char *free_ptr = NULL;
112 static char *end_ptr = NULL;
113 static jmp_buf abort_parsing;
114 
115 static void *allocate(size_t size)
116 {
117  void *result;
118  if (free_ptr + size > end_ptr) {
119  size_t how_much = BLOCK_SIZE;
120  // align everything to 8 bytes
121  if (size > BLOCK_SIZE - ALIGNMENT) {
122  how_much = size + ALIGNMENT;
123  }
124  result = malloc(how_much);
125  if (result == NULL) {
126  /* serious problem */
127  longjmp(abort_parsing, 1);
128  }
129  *((void **)result) = block_list;
130  block_list = result;
131  free_ptr = ((char *) result) + ALIGNMENT;
132  end_ptr = ((char *) result) + how_much;
133  }
134  // now, there is enough rooom at free_ptr
135  result = free_ptr;
136  free_ptr += size;
137  return result;
138 }

140 void bplist_free_data()
141 {
142  while (block_list) {
143  void *next = *(void **)block_list;
144  free(block_list);
145  block_list = next;
146  }
147  free_ptr = NULL;
148  end_ptr = NULL;
149 }

318 value_ptr bplist_read_file(char *filename)
319 {
320  struct stat stbuf;
321  pldata_node pldata;
322  FILE *file;
323  size_t n;
324  value_ptr value;
325  int rslt = stat(filename, &stbuf);
326  if (rslt) {
327  perror("in stat: ");
328  bplist_log("Could not stat %s, error %d\n", filename, rslt);
329  return NULL;
330  }
331  pldata.len = stbuf.st_size;
332  // note: this is supposed to be malloc, not allocate. It is separate
333  // from the graph structure, large, and easy to free right after
334  // parsing.
335  pldata.data = (uint8_t *) malloc(pldata.len);
336  if (!pldata.data) {
337  bplist_log("Could not allocate %d bytes for %s\n",
338  (long) pldata.len, filename);
339  return NULL;
340  }
341  file = fopen(filename, "rb");
342  if (!file) {
343  bplist_log("Could not open %s\n", filename);
344  free(pldata.data);
345  return NULL;
346  }
347  n = fread(pldata.data, 1, pldata.len, file);
348  if (n != pldata.len) {
349  bplist_log("Error reading from %s\n", filename);
350  free(pldata.data);
351  return NULL;
352  }
353  value = bplist_read_pldata(&pldata);
354  free(pldata.data);
355  return value;
356 }

391 // data is stored with high-order bytes first.
392 // read from plist data in a machine-independent fashion
393 //
394 uint64_t convert_uint64(uint8_t *ptr)
395 {
396  uint64_t rslt = 0;
397  int i;
398  // shift in bytes, high-order first
399  for (i = 0; i < sizeof(uint64_t); i++) {
400  rslt <<= 8;
401  rslt += ptr[i];
402  }
403  return rslt;
404 }

407 value_ptr bplist_read_pldata(pldata_ptr data)
408 {
409  value_ptr result = NULL;
410  bplist_info_node bplist;
411  uint8_t *ptr;
412  uint64_t top_level_object;
413  int i;
414 
415  if (data == NULL) return NULL;
416  if (!is_binary_plist(data)) {
417  bplist_log("Bad binary plist: too short or invalid header.\n");
418  return NULL;
419  }
420 
421  // read trailer
422  ptr = (uint8_t *) (data->data + data->len - kTRAILER_SIZE);
423  bplist.offset_int_size = ptr[6];
424  bplist.object_ref_size = ptr[7];
425  bplist.object_count = convert_uint64(ptr + 8);
426  top_level_object = convert_uint64(ptr + 16);
427  bplist.offset_table_offset = convert_uint64(ptr + 24);
428 
429  // Basic sanity checks
430  if (bplist.offset_int_size < 1 || bplist.offset_int_size > 8 ||
431  bplist.object_ref_size < 1 || bplist.object_ref_size > 8 ||
432  bplist.offset_table_offset < kHEADER_SIZE) {
433  bplist_log("Bad binary plist: trailer declared insane.\n");
434  return NULL;
435  }
436 
437  // Ensure offset table is inside file
438  uint64_t offsetTableSize = bplist.offset_int_size * bplist.object_count;
439  if (offsetTableSize + bplist.offset_table_offset + kTRAILER_SIZE >
440  data->len) {
441  bplist_log("Bad binary plist: offset table overlaps end of container.\n");
442  return NULL;
443  }
444 
445  bplist.data_bytes = data->data;
446  bplist.length = data->len;
447  bplist.cache = NULL; /* dictionary is empty */
448 
449  bplist_log_verbose("Got a sane bplist with %llu items, offset_int_size: %u, object_ref_size: %u\n",
450  bplist.object_count, bplist.offset_int_size,
451  bplist.object_ref_size);
452  /* at this point, we are ready to do some parsing which allocates
453  memory for the result data structure. If memory allocation (using
454  allocate fails, a longjmp will return to here and we simply give up
455  */
456  i = setjmp(abort_parsing);
457  if (i == 0) {
458  result = extract_object(&bplist, top_level_object);
459  } else {
460  bplist_log("allocate() failed to allocate memory. Giving up.\n");
461  result = NULL;
462  }
463  if (!result) {
464  bplist_free_data();
465  }
466  return result;
467 }

470 static value_ptr extract_object(bplist_info_ptr bplist, uint64_t objectRef)
471 {
472  uint64_t offset;
473  value_ptr result = NULL;
474  uint8_t objectTag;
475 
476  if (objectRef >= bplist->object_count) {
477  // Out-of-range object reference.
478  bplist_log("Bad binary plist: object index is out of range.\n");
479  return NULL;
480  }
481 
482  // Use cached object if it exists
483  result = cache_lookup(bplist->cache, objectRef);
484  if (result != NULL) return result;
485 
486  // Otherwise, find object in file.
487  offset = read_offset(bplist, objectRef);
488  if (offset > bplist->length) {
489  // Out-of-range offset.
490  bplist_log("Bad binary plist: object outside container.\n");
491  return NULL;
492  }
493  objectTag = *(bplist->data_bytes + offset);
494  switch (objectTag & 0xF0) {
495  case kTAG_SIMPLE:
496  result = extract_simple(bplist, offset);
497  break;
498 
499  case kTAG_INT:
500  result = extract_int(bplist, offset);
501  break;
502 
503  case kTAG_REAL:
504  result = extract_real(bplist, offset);
505  break;
506 
507  case kTAG_DATE:
508  result = extract_date(bplist, offset);
509  break;
510 
511  case kTAG_DATA:
512  result = extract_data(bplist, offset);
513  break;
514 
515  case kTAG_ASCIISTRING:
516  result = extract_ascii_string(bplist, offset);
517  break;
518 
519  case kTAG_UNICODESTRING:
520  result = extract_unicode_string(bplist, offset);
521  break;
522 
523  case kTAG_UID:
524  result = extract_uid(bplist, offset);
525  break;
526 
527  case kTAG_ARRAY:
528  result = extract_array(bplist, offset);
529  break;
530 
531  case kTAG_DICTIONARY:
532  result = extract_dictionary(bplist, offset);
533  break;
534 
535  default:
536  // Unknown tag.
537  bplist_log("Bad binary plist: unknown tag 0x%X.\n",
538  (objectTag & 0x0F) >> 4);
539  result = NULL;
540  }
541 
542  // Cache and return result.
543  if (result != NULL)
544  cache_insert(&bplist->cache, objectRef, result);
545  return result;
546 }

549 static uint64_t read_sized_int(bplist_info_ptr bplist, uint64_t offset,
550  uint8_t size)
551 {
552  assert(bplist->data_bytes != NULL && size >= 1 && size <= 8 &&
553  offset + size <= bplist->length);
554 
555  uint64_t result = 0;
556  const uint8_t *byte = bplist->data_bytes + offset;
557 
558  do {
559  // note that ints seem to be high-order first
560  result = (result << 8) | *byte++;
561  } while (--size);
562 
563  return result;
564 }

577 static BOOL read_self_sized_int(bplist_info_ptr bplist, uint64_t offset,
578  uint64_t *outValue, size_t *outSize)
579 {
580  uint32_t size;
581  int64_t value;
582 
583  assert(bplist->data_bytes != NULL && offset < bplist->length);
584 
585  size = 1 << (bplist->data_bytes[offset] & 0x0F);
586  if (size > 8) {
587  // Maximum allowable size in this implementation is 1<<3 = 8 bytes.
588  // This also happens to be the biggest we can handle.
589  return NO;
590  }
591 
592  if (offset + 1 + size > bplist->length) {
593  // Out of range.
594  return NO;
595  }
596 
597  value = read_sized_int(bplist, offset + 1, size);
598 
599  if (outValue != NULL) *outValue = value;
600  if (outSize != NULL) *outSize = size + 1; // +1 for tag byte.
601  return YES;
602 }

608  value_ptr value = value_create();
609

612  value->tag = kVALUE_NULL;
613  return value;
614 
615  case kVALUE_TRUE:
616  value->tag = kVALUE_TRUE;
617  return value;
618 
619  case kVALUE_FALSE:
620  value->tag = kVALUE_FALSE;
621  return value;
622  }
623

625  bplist_log("Bad binary plist: invalid atom.\n");
626  free(value);
627  return NULL;

648 static value_ptr extract_real(bplist_info_ptr bplist, uint64_t offset)
649 {
650  value_ptr value = value_create();
651  uint32_t size;
652 
653  assert(bplist->data_bytes != NULL && offset < bplist->length);
654 
655  size = 1 << (bplist->data_bytes[offset] & 0x0F);
656 
657  // FIXME: what to do if faced with other sizes for float/double?
658  assert (sizeof (float) == sizeof (uint32_t) &&
659  sizeof (double) == sizeof (uint64_t));
660 
661  if (offset + 1 + size > bplist->length) {
662  bplist_log("Bad binary plist: %s object overlaps end of container.\n",
663  "floating-point number");
664  free(value);
665  return NULL;
666  }
667 
668  if (size == sizeof (float)) {
669  uint32_t i = read_sized_int(bplist, offset + 1, size);
670  // Note that this handles byte swapping.
671  value_set_real(value, *(float *)&i);
672  return value;
673  } else if (size == sizeof (double)) {
674  uint64_t i = read_sized_int(bplist, offset + 1, size);
675  // Note that this handles byte swapping.
676  value_set_real(value, *(double *)&i);
677  return value;
678  } else {
679  // Can't handle floats of other sizes.
680  bplist_log("Bad binary plist: can't handle %u-byte float.\n", size);
681  free(value);
682  return NULL;
683  }
684 }

715 uint64_t bplist_get_a_size(bplist_info_ptr bplist,
716  uint64_t *offset_ptr, char *msg)
717 {
718  uint64_t size = bplist->data_bytes[*offset_ptr] & 0x0F;
719  (*offset_ptr)++;
720  if (size == 0x0F) {
721  // 0x0F means separate int size follows.
722  // Smaller values are used for short data.
723  size_t extra; // the length of the data size we are about to read
724  if ((bplist->data_bytes[*offset_ptr] & 0xF0) != kTAG_INT) {
725  // Bad data, mistagged size int
726  bplist_log("Bad binary plist: %s object size is not tagged as int.\n",
727  msg);
728  return UINT64_MAX; // error
729  }
730 
731  // read integer data as size, extra tells how many bytes to skip
732  if (!read_self_sized_int(bplist, *offset_ptr, &size, &extra)) {
733  bplist_log("Bad binary plist: invalid %s object size tag.\n",
734  "data");
735  return UINT64_MAX; // error
736  }
737  (*offset_ptr) += extra;
738  }
739 
740  if (*offset_ptr + size > bplist->length) {
741  bplist_log("Bad binary plist: %s object overlaps end of container.\n",
742  "data");
743  return UINT64_MAX; // error
744  }
745  return size;
746 }

835 static value_ptr extract_array(bplist_info_ptr bplist, uint64_t offset)
836 {
837  uint64_t i, count;
838  uint64_t size;
839  uint64_t elementID;
840  value_ptr element = NULL;
841  value_ptr *array = NULL;
842  value_ptr value = NULL;
843  BOOL ok = YES;
844 
845  assert(bplist->data_bytes != NULL && offset < bplist->length);
846 
847  if ((count = bplist_get_a_size(bplist, &offset, "array")) == UINT64_MAX)
848  return NULL;
849 
850  if (count > UINT64_MAX / bplist->object_ref_size - offset) {
851  // Offset overflow.
852  bplist_log("Bad binary plist: %s object overlaps end of container.\n",
853  "array");
854  return NULL;
855  }
856 
857  size = bplist->object_ref_size * count;
858  if (size + offset > bplist->length) {
859  bplist_log("Bad binary plist: %s object overlaps end of container.\n",
860  "array");
861  return NULL;
862  }
863 
864  // got count, the number of array elements
865 
866  value = value_create();
867  assert(value);
868 
869  if (count == 0) {
870  value_set_array(value, array, count);
871  return value;
872  }
873 
874  array = allocate(sizeof(value_ptr) * count);
875 
876  for (i = 0; i != count; ++i) {
877  bplist_log_verbose("[%u]\n", i);
878  elementID = read_sized_int(bplist, offset + i * bplist->object_ref_size,
879  bplist->object_ref_size);
880  element = extract_object(bplist, elementID);
881  if (element != NULL) {
882  array[i] = element;
883  } else {
884  ok = NO;
885  break;
886  }
887  }
888  if (ok) {
889  value_set_array(value, array, count);
890  }
891 
892  return value;

1016 void plist_print(value_ptr v)
1017 {
1018  size_t i;
1019  int comma_needed;
1020  dict_ptr dict;
1021  if (!v) {
1022  printf("NULL");
1023  return;
1024  }
1025  switch (v->tag & 0xF0) {

1027  switch (v->tag) {
1028  case kVALUE_NULL:
1029  printf("NULL@%p", v); break;
1030  case kVALUE_FALSE:
1031  printf("FALSE@%p", v); break;
1032  case kVALUE_TRUE:
1033  printf("TRUE@%p", v); break;
1034  default:
1035  printf("UNKNOWN tag=%x@%p", v->tag, v); break;
1036  }
1037  break;
1038  case kTAG_INT:
1039  printf("%lld@%p", v->integer, v); break;
1040  case kTAG_REAL:
1041  printf("%g@%p", v->real, v); break;
1042  case kTAG_DATE:
1043  printf("date:%g@%p", v->real, v); break;
1044  case kTAG_DATA:
1045  printf("data@%p->%p:[%p:", v, v->data, v->data->data);
1046  for (i = 0; i < v->data->len; i++) {
1047  printf(" %2x", v->data->data[i]);
1048  }
1049  printf("]"); break;
1050  case kTAG_ASCIISTRING:
1051  printf("%p:\"%s\"@%p", v->string, v->string, v); break;
1052  case kTAG_UNICODESTRING:
1053  printf("unicode:%p:\"%s\"@%p", v->string, v->string, v); break;
1054  case kTAG_UID:
1055  printf("UID:%llu@%p", v->uinteger, v); break;
1056  case kTAG_ARRAY:
1057  comma_needed = FALSE;
1058  printf("%p->%p:[%p:", v, v->array, v->array->array);
1059  for (i = 0; i < v->array->length; i++) {
1060  if (comma_needed) printf(", ");
1061  plist_print(v->array->array[i]);
1062  comma_needed = TRUE;
1063  }
1064  printf("]"); break;
1065  case kTAG_DICTIONARY:
1066  comma_needed = FALSE;
1067  printf("%p:[", v);
1068  dict = v->dict;
1069  while (dict) {
1070  if (comma_needed) printf(", ");
1071  printf("%p:", dict);
1072  plist_print(dict->key);
1073  printf("->");
1074  plist_print(dict->value);
1075  comma_needed = TRUE;
1076  dict = dict->next;
1077  }
1078  printf("]"); break;
1079  default:
1080  printf("UNKNOWN tag=%x", v->tag);
1081  break;
1082  }
