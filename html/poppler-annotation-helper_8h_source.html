116 int XPDFReader::lookupNumArray( Dict * dict, char * type, double * dest, int len )
117 {
118  Object arrObj;
119  dict->lookup( type, &arrObj );
120  if ( arrObj.isNull() )
121  return 0;
122  Object numObj;
123  if ( arrObj.isArray() )
124  {
125  len = qMin( len, arrObj.arrayGetLength() );
126  for ( int i = 0; i < len; i++ )
127  {
128  dest[i] = arrObj.arrayGet( i, &numObj )->getNum();
129  numObj.free();
130  }
131  }
132  else
133  {
134  len = 0;
135  qDebug() << type << "is not Array." << endl;
136  }
137  arrObj.free();
138  return len;
139 }

182 void XPDFReader::invTransform( double * M, const QPointF &p, double &x, double &y )
183 {
184  const double det = M[0]*M[3] - M[1]*M[2];
185  Q_ASSERT(det != 0);
186 
187  const double invM[4] = { M[3]/det, -M[1]/det, -M[2]/det, M[0]/det };
188  const double xt = p.x() - M[4];
189  const double yt = p.y() - M[5];
190 
191  x = invM[0] * xt + invM[2] * yt;
192  y = invM[1] * xt + invM[3] * yt;
193 }
