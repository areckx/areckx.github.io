67 bool JpegWriter::init(FILE *f, int width, int height, int hDPI, int vDPI)
68 {
69  // Setup error handler
70  priv->cinfo.err = jpeg_std_error(&priv->jerr);
71  priv->jerr.output_message = &outputMessage;
72 
73  // Initialize libjpeg
74  jpeg_create_compress(&priv->cinfo);
75 
76  // First set colorspace and call jpeg_set_defaults() since
77  // jpeg_set_defaults() sets default values for all fields in
78  // cinfo based on the colorspace.
79  switch (priv->format) {
80  case RGB:
81  priv->cinfo.in_color_space = JCS_RGB;
82  break;
83  case GRAY:
84  priv->cinfo.in_color_space = JCS_GRAYSCALE;
85  break;
86  case CMYK:
87  priv->cinfo.in_color_space = JCS_CMYK;
88  break;
89  default:
90  return false;
91  }
92  jpeg_set_defaults(&priv->cinfo);
93 
94  // Set destination file
95  jpeg_stdio_dest(&priv->cinfo, f);
96 
97  // Set libjpeg configuration
98  priv->cinfo.image_width = width;
99  priv->cinfo.image_height = height;
100  priv->cinfo.density_unit = 1; // dots per inch
101  priv->cinfo.X_density = hDPI;
102  priv->cinfo.Y_density = vDPI;
103  switch (priv->format) {
104  case GRAY:
105  priv->cinfo.input_components = 1;
106  break;
107  case RGB:
108  priv->cinfo.input_components = 3;
109  break;
110  case CMYK:
111  priv->cinfo.input_components = 4;
112  jpeg_set_colorspace(&priv->cinfo, JCS_YCCK);
113  priv->cinfo.write_JFIF_header = TRUE;
114  break;
115  default:
116  return false;
117  }
118 
119  // Set quality
120  if (priv->quality >= 0 && priv->quality <= 100) {
121  jpeg_set_quality(&priv->cinfo, priv->quality, TRUE);
122  }
123 
124  // Use progressive mode
125  if (priv->progressive) {
126  jpeg_simple_progression(&priv->cinfo);
127  }
128 
129  // Get ready for data
130  jpeg_start_compress(&priv->cinfo, TRUE);
131 
132  return true;
133 }

135 bool JpegWriter::writePointers(unsigned char **rowPointers, int rowCount)
136 {
137  if (priv->format == CMYK) {
138  for (int y = 0; y < rowCount; y++) {
139  unsigned char *row = rowPointers[y];
140  for (unsigned int x = 0; x < priv->cinfo.image_width; x++) {
141  for (int n = 0; n < 4; n++) {
142  *row = 0xff - *row;
143  row++;
144  }
145  }
146  }
147  }
148  // Write all rows to the file
149  jpeg_write_scanlines(&priv->cinfo, rowPointers, rowCount);
150 
151  return true;
152 }
