221 void Animations::registerWidget( QWidget* widget ) const {
222  if (!widget)
223  return;
224 
225  // these are needed to not register animations for kwin widgets
226  if (widget->objectName() == "decoration widget")
227  return;
228  if (widget->inherits("KCommonDecorationButton"))
229  return;
230  if (widget->inherits("QShapedPixmapWidget"))
231  return;
232 
233  // all widgets are registered to the enability engine.
234  widgetEnabilityEngine().registerWidget( widget, AnimationEnable );
235 
236  // install animation timers
237  // for optimization, one should put with most used widgets here first
238  if ( qobject_cast<QToolButton*>(widget) ) {
239 
240  toolButtonEngine().registerWidget( widget, AnimationHover );
241  bool isInToolBar( qobject_cast<QToolBar*>(widget->parent()) );
242  if ( isInToolBar ) {
243 
244  if ( MgStyleConfigData::toolBarAnimationType == MgStyleConfigData::TB_FADE ) {
245  widgetStateEngine().registerWidget( widget, AnimationHover );
246  }
247 
248  }
249  else widgetStateEngine().registerWidget( widget, AnimationHover | AnimationFocus );
250 
251  }
252  else if ( qobject_cast<QAbstractButton*>(widget) ) {
253 
254  if ( qobject_cast<QToolBox*>( widget->parent() ) ) {
255  toolBoxEngine().registerWidget( widget );
256  }
257 
258  widgetStateEngine().registerWidget( widget, AnimationHover | AnimationFocus );
259 
260  }
261  else if ( qobject_cast<QDial*>(widget) ) {
262 
263  widgetStateEngine().registerWidget( widget, AnimationHover | AnimationFocus );
264 
265  }
266 
267  // groupboxes
268  else if ( QGroupBox* groupBox = qobject_cast<QGroupBox*>( widget ) ) {
269  if ( groupBox->isCheckable() ) {
270  widgetStateEngine().registerWidget( widget, AnimationHover | AnimationFocus );
271  }
272  }
273 
274  // scrollbar
275  else if ( qobject_cast<QScrollBar*>( widget ) ) {
276  scrollBarEngine().registerWidget( widget );
277  }
278  else if ( qobject_cast<QSlider*>( widget ) ) {
279  sliderEngine().registerWidget( widget );
280  }
281  else if ( qobject_cast<QProgressBar*>( widget ) ) {
282  progressBarEngine().registerWidget( widget );
283  }
284  else if ( qobject_cast<QSplitterHandle*>( widget ) ) {
285  splitterEngine().registerWidget( widget );
286  }
287  else if ( qobject_cast<QMainWindow*>( widget ) ) {
288  dockSeparatorEngine().registerWidget( widget );
289  }
290  else if ( qobject_cast<QHeaderView*>( widget ) ) {
291  headerViewEngine().registerWidget( widget );
292  }
293  // menu
294  else if ( qobject_cast<QMenu*>( widget ) ) {
295  menuEngine().registerWidget( widget );
296  }
297  else if ( qobject_cast<QMenuBar*>( widget ) ) {
298  menuBarEngine().registerWidget( widget );
299  }
300  else if ( qobject_cast<QTabBar*>( widget ) ) {
301  tabBarEngine().registerWidget( widget );
302  }
303  else if ( qobject_cast<QToolBar*>( widget ) ) {
304  toolBarEngine().registerWidget( widget );
305  }
306 
307  // editors
308  else if ( qobject_cast<QComboBox*>( widget ) ) {
309  comboBoxEngine().registerWidget( widget, AnimationHover );
310  lineEditEngine().registerWidget( widget, AnimationHover | AnimationFocus );
311  }
312  else if ( qobject_cast<QSpinBox*>( widget ) ) {
313  spinBoxEngine().registerWidget( widget );
314  lineEditEngine().registerWidget( widget, AnimationHover | AnimationFocus );
315  }
316  else if ( qobject_cast<QLineEdit*>( widget ) ) {
317  lineEditEngine().registerWidget( widget, AnimationHover | AnimationFocus );
318  }
319  else if ( qobject_cast<QTextEdit*>( widget ) ) {
320  lineEditEngine().registerWidget( widget, AnimationHover | AnimationFocus );
321  }
322 
323  // lists
324  else if ( qobject_cast<QAbstractItemView*>( widget ) || widget->inherits("Q3ListView") ) {
325  lineEditEngine().registerWidget( widget, AnimationHover | AnimationFocus );
326  }
327 
328  // mdi subwindows
329  else if ( qobject_cast<QMdiSubWindow*>( widget ) ) {
330  mdiWindowEngine().registerWidget( widget );
331  }
332 
333  return;
334 
335  }

337 void Animations::unregisterWidget( QWidget* widget ) const {
338  if ( !widget ) return;
339 
340  /*
341  these are the engines that have not been stored
342  inside the list, because they can be register widgets in combination
343  with other engines
344  */
345  widgetEnabilityEngine().unregisterWidget( widget );
346  spinBoxEngine().unregisterWidget( widget );
347  comboBoxEngine().unregisterWidget( widget );
348  toolButtonEngine().unregisterWidget( widget );
349  toolBoxEngine().unregisterWidget( widget );
350 
351  // the following allows some optimization of widget unregistration
352  // it assumes that a widget can be registered atmost in one of the
353  // engines stored in the list.
354  foreach( const BaseEngine::Pointer & engine, engines_ ) {
355  if ( engine && engine.data()->unregisterWidget( widget ) ) break;
356  }
357 
358  }
