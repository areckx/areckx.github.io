65 void Transitions::registerWidget( QWidget* widget ) const {
66 
67  if ( !widget ) return;
68 
69  if ( QLabel* label = qobject_cast<QLabel*>( widget ) ) {
70 
71  // do not animate labels from tooltips
72  if ( widget->window() && widget->window()->windowFlags().testFlag( Qt::ToolTip ) ) return;
73  else if ( widget->window() && widget->window()->inherits( "KWin::GeometryTip" ) ) return;
74  else labelEngine().registerWidget( label );
75 
76  }
77  else if ( QComboBox* comboBox = qobject_cast<QComboBox*>( widget ) ) {
78 
79  comboBoxEngine().registerWidget( comboBox );
80 
81  }
82  else if ( QLineEdit* lineEdit = qobject_cast<QLineEdit*>( widget ) ) {
83 
84  lineEditEngine().registerWidget( lineEdit );
85 
86  }
87  else if ( QStackedWidget* stack = qobject_cast<QStackedWidget*>( widget ) ) {
88 
89  stackedWidgetEngine().registerWidget( stack );
90 
91  }
92 
93  }

96 void Transitions::unregisterWidget( QWidget* widget ) const {
97 
98  if ( !widget ) return;
99 
100  // the following allows some optimization of widget unregistration
101  // it assumes that a widget can be registered atmost in one of the
102  // engines stored in the list.
103  foreach( const BaseEngine::Pointer & engine, engines_ ) {
104  if ( engine && engine.data()->unregisterWidget( widget ) ) break;
105  }
