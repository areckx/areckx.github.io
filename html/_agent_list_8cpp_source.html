54 void AgentList::removeDuplicates()
55  {
56  sort();
57  for (iterator itr = begin(); itr != end(); ++itr) {
58  if ((*itr)->phaseScore < 0.0) // already flagged for deletion
59  continue;
60  iterator itr2 = itr;
61  for (++itr2; itr2 != end(); ++itr2) {
62  if ((*itr2)->beatInterval - (*itr)->beatInterval > DEFAULT_BI)
63  break;
64  if (std::fabs((*itr)->beatTime - (*itr2)->beatTime) > DEFAULT_BT)
65  continue;
66  if ((*itr)->phaseScore < (*itr2)->phaseScore) {
67  (*itr)->phaseScore = -1.0; // flag for deletion
68  break;
69  }
70  else {
71  (*itr2)->phaseScore = -1.0; // flag for deletion
72  }
73  }
74  }
75  int removed = 0;
76  for (iterator itr = begin(); itr != end(); ) {
77  if ((*itr)->phaseScore < 0.0) {
78  ++removed;
79  delete *itr;
80  list.erase(itr);
81  }
82  else {

84  }

89 void AgentList::beatTrack(const EventList &el,
90  const AgentParameters &params,
91  double stop)
92  {
93  EventList::const_iterator ei = el.begin();
94  bool phaseGiven = !empty() && ((*begin())->beatTime >= 0); // if given for one, assume given for others
95  while (ei != el.end()) {
96  Event ev = *ei;
97  ++ei;
98  if ((stop > 0) && (ev.time > stop))
99  break;
100  bool created = phaseGiven;
101  double prevBeatInterval = -1.0;
102  // cc: Duplicate our list of agents, and scan through the
103  // copy. This means we can safely add agents to our own
104  // list while scanning without disrupting our scan. Each
105  // agent needs to be re-added to our own list explicitly
106  // (since it is modified by e.g. considerAsBeat)
107  Container currentAgents = list;
108  list.clear();
109  for (Container::iterator ai = currentAgents.begin();
110  ai != currentAgents.end(); ++ai) {
111  Agent *currentAgent = *ai;
112  if (currentAgent->beatInterval != prevBeatInterval) {
113  if ((prevBeatInterval >= 0) && !created && (ev.time < 5.0)) {
114  // Create new agent with different phase
115  Agent *newAgent = new Agent(params, prevBeatInterval);
116  // This may add another agent to our list as well
117  newAgent->considerAsBeat(ev, *this);
118  add(newAgent);
119  }
120  prevBeatInterval = currentAgent->beatInterval;
121  created = phaseGiven;
122  }
123  if (currentAgent->considerAsBeat(ev, *this))
124  created = true;
125  add(currentAgent);
126  } // loop for each agent
127  removeDuplicates();
128  } // loop for each event
129  }

131 Agent *AgentList::bestAgent()
132  {

136  if ((*itr)->events.empty())
