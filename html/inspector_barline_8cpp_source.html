66 void InspectorBarLine::setAsStaffDefault()
67  {
68  BarLine* bl = toBarLine(inspector->element());
69  Staff* staff = bl->staff();
70  Score* score = bl->score();
71  score->startCmd();
72  staff->undoChangeProperty(P_ID::STAFF_BARLINE_SPAN, bl->getProperty(P_ID::BARLINE_SPAN));
73  staff->undoChangeProperty(P_ID::STAFF_BARLINE_SPAN_FROM, bl->getProperty(P_ID::BARLINE_SPAN_FROM));
74  staff->undoChangeProperty(P_ID::STAFF_BARLINE_SPAN_TO, bl->getProperty(P_ID::BARLINE_SPAN_TO));
75  if (bl->barLineType() == BarLineType::NORMAL)
76  bl->setGenerated(true);
77  score->endCmd();
78  }

86 void InspectorBarLine::setElement()
87  {
88  InspectorElementBase::setElement();
89  BarLine* bl = toBarLine(inspector->element());
90 
91  // enable / disable individual type combo items according to score and selected bar line status
92  bool bMultiStaff = bl->score()->nstaves() > 1;
93  BarLineType blt = bl->barLineType();
94 // bool isRepeat = blt & (BarLineType::START_REPEAT | BarLineType::END_REPEAT | BarLineType::END_START_REPEAT);
95  bool isRepeat = blt & (BarLineType::START_REPEAT | BarLineType::END_REPEAT);
96 
97  const QStandardItemModel* model = qobject_cast<const QStandardItemModel*>(b.type->model());
98  int i = 0;
99  for (auto& k : BarLine::barLineTable) {
100  QStandardItem* item = model->item(i);
101  // if combo item is repeat type, should be disabled for multi-staff scores
102 // if (k.type & (BarLineType::START_REPEAT | BarLineType::END_REPEAT | BarLineType::END_START_REPEAT)) {
103  if (k.type & (BarLineType::START_REPEAT | BarLineType::END_REPEAT)) {
104  // disable / enable
105  item->setFlags(bMultiStaff ?
106  item->flags() & ~(Qt::ItemIsSelectable|Qt::ItemIsEnabled) :
107  item->flags() | (Qt::ItemFlags)(Qt::ItemIsSelectable|Qt::ItemIsEnabled) );
108  }
109  // if combo item is NOT repeat type, should be disabled if selected bar line is a repeat
110  else {
111  item->setFlags(isRepeat ?
112  item->flags() & ~(Qt::ItemIsSelectable|Qt::ItemIsEnabled) :
113  item->flags() | (Qt::ItemFlags)(Qt::ItemIsSelectable|Qt::ItemIsEnabled) );
114  }
115  ++i;
116  }
117 #if 0
118  blockSpanDataSignals(true);
119  manageSpanData();
120  blockSpanDataSignals(false);
121 #endif
122  }
