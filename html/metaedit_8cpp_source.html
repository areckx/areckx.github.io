93 void MetaEditDialog::accept()
94  {
95  if (dirty) {
96  QGridLayout* grid = static_cast<QGridLayout*>(scrollWidget->layout());
97  int idx = grid->rowCount();
98  QMap<QString, QString> m;
99  for (int i = 0; i < idx; ++i) {
100  QLayoutItem* labelItem = grid->itemAtPosition(i, 0);
101  QLayoutItem* dataItem = grid->itemAtPosition(i, 1);
102  if (labelItem && dataItem) {
103  QLabel* label = static_cast<QLabel*>(labelItem->widget());
104  QLineEdit* le = static_cast<QLineEdit*>(dataItem->widget());
105  m.insert(label->text(), le->text());
106  }
107  }
108  score->undo(new ChangeMetaTags(score, m));
109  }
110  QDialog::accept();
111  }
