77 bool NoteList::stavesOverlap(const int staff1, const int staff2) const
78  {
79  for (int i = 0; i < _staffNoteLists.at(staff1).size(); ++i)

82  // printf(" %d-%d", staff1, staff2);
83  return true;
84  }
85  return false;

153 void ValidatorMessageHandler::handleMessage(QtMsgType type, const QString& description,
154  const QUrl& /* identifier */, const QSourceLocation& sourceLocation)
155  {
156  // convert description from html to text
157  QDomDocument desc;
158  QString contentError;
159  int contentLine;
160  int contentColumn;
161  if (!desc.setContent(description, false, &contentError, &contentLine,
162  &contentColumn)) {
163  qDebug("ValidatorMessageHandler: could not parse validation error line %d column %d: %s",
164  contentLine, contentColumn, qPrintable(contentError));
165  return;
166  }
167  QDomElement e = desc.documentElement();
168  if (e.tagName() != "html") {
169  qDebug("ValidatorMessageHandler: description is not html");
170  return;
171  }
172  QString descText = e.text();
173 
174  QString strType;
175  switch (type) {
176  case 0: strType = "Debug"; break;
177  case 1: strType = "Warning"; break;
178  case 2: strType = "Critical"; break;
179  case 3: strType = "Fatal"; break;
180  default: strType = "Unknown"; break;
181  }
182 
183  QString errorStr = QString("%1 error: line %2 column %3 %4")
184  .arg(strType)
185  .arg(sourceLocation.line())
186  .arg(sourceLocation.column())
187  .arg(descText);
188 
189  // append error, separated by newline if necessary
190  if (errors != "")
191  errors += "\n";
192  errors += errorStr;
193  }

199 static QString domElementPath(const QDomElement& e)
200  {
201  QString s;
202  QDomNode dn(e);
203  while (!dn.parentNode().isNull()) {
204  dn = dn.parentNode();
205  const QDomElement& e = dn.toElement();
206  const QString k(e.tagName());
207  if (!s.isEmpty())
208  s += ":";
209  s += k;
210  }
211  return s;
212  }

351 Fraction MxmlSupport::calculateFraction(QString type, int dots, int normalNotes, int actualNotes)
352  {
353  // type
354  Fraction f = MxmlSupport::noteTypeToFraction(type);
355  if (f.isValid()) {
356  // dot(s)
357  Fraction f_no_dots = f;
358  for (int i = 0; i < dots; ++i)
359  f += (f_no_dots / (2 << i));
360  // tuplet
361  if (actualNotes > 0 && normalNotes > 0) {
362  f *= normalNotes;
363  f /= actualNotes;
364  }
365  // clean up (just in case)
366  f.reduce();
367  }
368  return f;
369  }
