957 void Note::add(Element* e)
958  {
959  e->setParent(this);
960  e->setTrack(track());

962  switch(e->type()) {
963  case ElementType::NOTEDOT:
964  _dots.append(toNoteDot(e));
965  break;
966  case ElementType::SYMBOL:
967  case ElementType::IMAGE:
968  case ElementType::FINGERING:
969  case ElementType::TEXT:
970  case ElementType::BEND:

972  break;

977  tie->setTrack(track());
978  setTieFor(tie);
979  if (tie->endNote())
980  tie->endNote()->setTieBack(tie);
981  int n = tie->spannerSegments().size();
982  for (int i = 0; i < n; ++i) {
983  SpannerSegment* ss = tie->spannerSegments().at(i);
984  if (ss->system())
985  ss->system()->add(ss);
986  }
987  }
988  break;
989  case ElementType::ACCIDENTAL:
990  _accidental = toAccidental(e);
991  break;
992  case ElementType::TEXTLINE:
993  case ElementType::GLISSANDO:
994  addSpanner(static_cast<Spanner*>(e));
995  break;
996  default:
997  qDebug("Note::add() not impl. %s", e->name());
998  break;
999  }
1000  score()->setLayout(tick());
1001  }
