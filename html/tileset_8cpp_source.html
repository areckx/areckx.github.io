27 void TileSet::initPixmap( PixmapList& pixmaps, const QPixmap& pix, int w, int h, const QRect& rect) {
28  QSize size( w, h );
29  if (!size.isValid()) {
30  pixmaps.push_back(QPixmap());
31  }
32  else if (size != rect.size()) {
33  const QPixmap tile(pix.copy(rect));
34  QPixmap pixmap(w, h);
35 
36  pixmap.fill(Qt::transparent);
37  QPainter p(&pixmap);
38  p.drawTiledPixmap(QRect(0, 0, w, h), tile);
39 
40  pixmaps.push_back( pixmap );
41  }
42  else
43  pixmaps.push_back(pix.copy(rect));
44  }

55 TileSet::TileSet(const QPixmap& pix, int w1, int h1, int w2, int h2)
56  : _w1(w1), _h1(h1), _w3(0), _h3(0) {
57  _pixmaps.reserve(9);
58  if (pix.isNull())
59  return;
60 
61  _w3 = pix.width() - (w1 + w2);
62  _h3 = pix.height() - (h1 + h2);
63  int w = w2;
64  while (w < 32 && w2 > 0)
65  w += w2;
66  int h = h2;
67  while (h < 32 && h2 > 0)
68  h += h2;
69 
70  // initialise pixmap array
71  initPixmap(_pixmaps, pix, _w1, _h1, QRect(0, 0, _w1, _h1));
72  initPixmap(_pixmaps, pix, w, _h1, QRect(_w1, 0, w2, _h1));
73  initPixmap(_pixmaps, pix, _w3, _h1, QRect(_w1 + w2, 0, _w3, _h1));
74  initPixmap(_pixmaps, pix, _w1, h, QRect(0, _h1, _w1, h2));
75  initPixmap(_pixmaps, pix, w, h, QRect(_w1, _h1, w2, h2));
76  initPixmap(_pixmaps, pix, _w3, h, QRect(_w1 + w2, _h1, _w3, h2));
77  initPixmap(_pixmaps, pix, _w1, _h3, QRect(0, _h1 + h2, _w1, _h3));
78  initPixmap(_pixmaps, pix, w, _h3, QRect(_w1, _h1 + h2, w2, _h3));
79  initPixmap(_pixmaps, pix, _w3, _h3, QRect(_w1 + w2, _h1 + h2, _w3, _h3));
80  }

119 void TileSet::render(const QRect& r, QPainter* p, Tiles t) const
120  {
121  // check initialization
122  if (_pixmaps.size() < 9)
123  return;
124 
125  int x0, y0, w, h;
126  r.getRect(&x0, &y0, &w, &h);
127 
128  // calculate pixmaps widths
129  int wLeft(0);
130  int wRight(0);
131  if (_w1 + _w3 > 0) {
132  qreal wRatio(qreal(_w1) / qreal(_w1 + _w3));
133  wLeft = (t & Right) ? qMin(_w1, int(w * wRatio)) : _w1;
134  wRight = (t & Left) ? qMin(_w3, int(w * (1.0 - wRatio))) : _w3;
135  }
136 
137  // calculate pixmap heights
138  int hTop(0);
139  int hBottom(0);
140  if (_h1 + _h3 > 0) {
141  qreal hRatio( qreal( _h1 ) / qreal( _h1 + _h3 ) );
142  hTop = (t & Bottom) ? qMin( _h1, int(h * hRatio) ) : _h1;
143  hBottom = (t & Top) ? qMin( _h3, int(h * (1.0 - hRatio)) ) : _h3;
144  }
145 
146  // calculate corner locations
147  w -= wLeft + wRight;
148  h -= hTop + hBottom;
149  int x1 = x0 + wLeft;
150  int x2 = x1 + w;
151  int y1 = y0 + hTop;
152  int y2 = y1 + h;
153 
154  // corner
155  if (bits(t, Top | Left))
156  p->drawPixmap(x0, y0, _pixmaps.at(0), 0, 0, wLeft, hTop);
157  if (bits(t, Top | Right))
158  p->drawPixmap(x2, y0, _pixmaps.at(2), _w3 - wRight, 0, wRight, hTop);
159  if (bits(t, Bottom | Left))
160  p->drawPixmap(x0, y2, _pixmaps.at(6), 0, _h3 - hBottom, wLeft, hBottom);
161  if (bits(t, Bottom | Right))
162  p->drawPixmap(x2, y2, _pixmaps.at(8), _w3 - wRight, _h3 - hBottom, wRight, hBottom);
163 
164  // top and bottom
165  if (w > 0) {
166  if (t & Top)
167  p->drawTiledPixmap(x1, y0, w, hTop, _pixmaps.at(1));
168  if (t & Bottom)
169  p->drawTiledPixmap(x1, y2, w, hBottom, _pixmaps.at(7), 0, _h3 - hBottom);
170  }
171 
172  // left and right
173  if (h > 0) {
174  if (t & Left)
175  p->drawTiledPixmap(x0, y1, wLeft, h, _pixmaps.at(3));
176  if (t & Right)
177  p->drawTiledPixmap(x2, y1, wRight, h, _pixmaps.at(5), _w3 - wRight, 0);
178  }
179 
180  // center
181  if ((t & Center) && h > 0 && w > 0)
182  p->drawTiledPixmap(x1, y1, w, h, _pixmaps.at(4));
183  }
