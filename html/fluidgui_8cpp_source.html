177 void FluidGui::soundFontAddClicked()
178  {
179  QFileInfoList l = FluidS::Fluid::sfFiles();
180 
181  SfListDialog ld(this);
182  foreach (const QFileInfo& fi, l)
183  ld.add(fi.fileName(), fi.absoluteFilePath());
184  if (!ld.exec())
185  return;
186 
187  QString sfName = ld.name();
188  QString sfPath = ld.path();
189 
190  int n = soundFonts->count();
191  QStringList sl;
192  for (int i = 0; i < n; ++i) {
193  QListWidgetItem* item = soundFonts->item(i);
194  sl.append(item->text());
195  }
196 
197  if (sl.contains(sfPath)) {
198  QMessageBox::warning(this,
199  tr("MuseScore"),
200  tr("SoundFont %1 already loaded").arg(sfPath));
201  }
202  else {
203  bool loaded = fluid()->addSoundFont(sfPath);
204  if (!loaded) {
205  QMessageBox::warning(this,
206  tr("MuseScore"),
207  tr("Cannot load SoundFont %1").arg(sfPath));
208  }
209  else {
210  soundFonts->insertItem(0, sfName);
211  emit sfChanged();
212  emit valueChanged();
213  }
214  }
215  updateUpDownButtons();
216  }
