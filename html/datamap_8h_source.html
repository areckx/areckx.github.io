57  Value find( Key key ) {
58  if ( !( enabled() && key ) ) return Value();
59  if ( key == lastKey_ ) return lastValue_;
60  else {
61  Value out;
62  typename QMap<Key, Value>::iterator iter( QMap<Key, Value>::find( key ) );
63  if ( iter != QMap<Key, Value>::end() ) out = iter.value();
64  lastKey_ = key;
65  lastValue_ = out;

67  }

71  bool unregisterWidget( Key key ) {
72 
73  // check key
74  if ( !key ) return false;
75 
76  // clear last value if needed
77  if ( key == lastKey_ ) {
78 
79  if ( lastValue_ ) lastValue_.clear();
80  lastKey_ = NULL;
81 
82  }
83 
84  // find key in map
85  typename QMap<Key, Value>::iterator iter( QMap<Key, Value>::find( key ) );
86  if ( iter == QMap<Key, Value>::end() ) return false;
87 
88  // delete value from map if found
89  if ( iter.value() ) iter.value().data()->deleteLater();
90  QMap<Key, Value>::erase( iter );
91 
92  return true;
93 
94  }
