47 void Lasso::editDrag(EditData& ed)
48  {
49  Qt::CursorShape cursorShape = ed.view->cursor().shape();
50  switch (int(ed.curGrip)) {
51  case 0:
52  cursorShape = Qt::SizeFDiagCursor;
53  bbox().setTopLeft(bbox().topLeft() + ed.delta);
54  break;
55  case 1:
56  cursorShape = Qt::SizeBDiagCursor;
57  bbox().setTopRight(bbox().topRight() + ed.delta);
58  break;
59  case 2:
60  cursorShape = Qt::SizeFDiagCursor;
61  bbox().setBottomRight(bbox().bottomRight() + ed.delta);
62  break;
63  case 3:
64  cursorShape = Qt::SizeBDiagCursor;
65  bbox().setBottomLeft(bbox().bottomLeft() + ed.delta);
66  break;
67  case 4:
68  cursorShape = Qt::SizeVerCursor;
69  bbox().setTop(bbox().top() + ed.delta.y());
70  break;
71  case 5:
72  cursorShape = Qt::SizeHorCursor;
73  bbox().setRight(bbox().right() + ed.delta.x());
74  break;
75  case 6:
76  cursorShape = Qt::SizeVerCursor;
77  bbox().setBottom(bbox().bottom() + ed.delta.y());
78  break;
79  case 7:
80  cursorShape = Qt::SizeHorCursor;
81  bbox().setLeft(bbox().left() + ed.delta.x());
82  break;
83  }
84  ed.view->setCursor(QCursor(cursorShape));
85  }

91 void Lasso::updateGrips(EditData& ed) const
92  {
93  ed.grip[0].translate(bbox().topLeft());
94  ed.grip[1].translate(bbox().topRight());
95  ed.grip[2].translate(bbox().bottomRight());
96  ed.grip[3].translate(bbox().bottomLeft());
97  ed.grip[4].translate(bbox().x() + bbox().width() * .5, bbox().top());
98  ed.grip[5].translate(bbox().right(), bbox().y() + bbox().height() * .5);
99  ed.grip[6].translate(bbox().x() + bbox().width()*.5, bbox().bottom());
100  ed.grip[7].translate(bbox().left(), bbox().y() + bbox().height() * .5);
101  }

117 bool Lasso::setProperty(P_ID propertyId, const QVariant& v)
118  {
119  switch (propertyId) {
120  case P_ID::LASSO_POS:
121  bbox().moveTo(v.toPointF());
122  break;
123  case P_ID::LASSO_SIZE:
124  bbox().setSize(v.toSizeF());
125  break;
126  default:
127  if (!Element::setProperty(propertyId, v))
128  return false;
129  break;
130  }
131  score()->setUpdateAll();
132  return true;
133  }
