101 void TestRepeat::repeat(const char* f1, const QString & ref)
102  {
103  Score* score = readScore(DIR + f1);
104  QVERIFY(score);
105  score->updateRepeatList(true);
106  QStringList sl;
107  for (const RepeatSegment* rs : *score->repeatList()) {
108  int startTick = rs->tick;
109  int endTick = startTick + rs->len;
110  for (Measure* m = score->tick2measure(startTick); m; m = m->nextMeasure()) {
111  sl.append(QString::number(m->no()+1));
112  if (m->tick() + m->ticks() >= endTick)
113  break;
114  }
115  }
116  QString s = sl.join(";");
117  QString ref1 = ref;
118  ref1.replace(" ","");
119  qDebug("File <%s> sequence %s", f1, qPrintable(s));
120  QCOMPARE(s, ref1);
121  delete score;
122  }
