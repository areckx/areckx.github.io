110 void ScrollBarData::hoverMoveEvent( QObject* object, QEvent* event ) {
111 
112  // try cast object to scrollbar
113  QScrollBar* scrollBar( qobject_cast<QScrollBar*>( object ) );
114  if ( !scrollBar || scrollBar->isSliderDown() ) return;
115 
116  // retrieve scrollbar option
117  QStyleOptionSlider opt( qt_qscrollbarStyleOption( qobject_cast<QScrollBar*>( object ) ) );
118 
119  // cast event
120  QHoverEvent* he = static_cast<QHoverEvent*>(event);
121  QStyle::SubControl hoverControl = scrollBar->style()->hitTestComplexControl(QStyle::CC_ScrollBar, &opt, he->pos(), scrollBar);
122  updateAddLineArrow( hoverControl );
123  updateSubLineArrow( hoverControl );
124 
125  }
