33 LineEditData::LineEditData( QObject* parent, QLineEdit* target, int duration ):
34  TransitionData( parent, target, duration ),

36  hasClearButton_( false ),
37  edited_( false ) {
38  target_.data()->installEventFilter( this );
39 
40  checkClearButton();
41 
42  connect( target_.data(), SIGNAL( destroyed() ), SLOT( targetDestroyed() ) );
43  connect( target_.data(), SIGNAL( textEdited( const QString& ) ), SLOT( textEdited( void ) ) );

68 bool LineEditData::eventFilter( QObject* object, QEvent* event ) {
69 
70  if ( !( enabled() && object && object == target_.data() ) ) {
71  return TransitionData::eventFilter( object, event );
72  }
73 
74  switch ( event->type() ) {
75  case QEvent::Show:
76  case QEvent::Resize:
77  case QEvent::Move:
78  if ( !recursiveCheck() ) {
79  timer_.start( 0, this );
80  break;
81  }
82 
83  default:
84  break;
85  }
86 
87  return TransitionData::eventFilter( object, event );
88 
89  }

93  if ( event->timerId() == timer_.timerId() ) {
94 
95  timer_.stop();
96  checkClearButton();
97  if ( enabled() && transition() && target_ && target_.data()->isVisible() ) {
98  setRecursiveCheck( true );
99  transition().data()->setEndPixmap( transition().data()->grab( target_.data(), targetRect() ) );
100  setRecursiveCheck( false );
101  }
102 
103  }
104  else if ( event->timerId() == animationLockTimer_.timerId() ) {
105 
106  unlockAnimations();
107 
108  }
109  else return TransitionData::timerEvent( event );

146 void LineEditData::textChanged( void ) {
147 
148  // check whether text change was triggered manually
149  // in which case do not start transition
150  if ( edited_ ) {
151  edited_ = false;
152  return;
153  }
154 
155  if ( transition().data()->isAnimated() ) {
156  transition().data()->endAnimation();
157  }
158 
159  if ( isLocked() ) {
160  // if locked one do not start the new animation, to prevent flicker
161  // instead, one hides the transition pixmap, trigger an update, and return.
162  // animations are re-locked.
163  transition().data()->hide();
164  lockAnimations();
165  timer_.start( 0, this );
166  return;
167  }
168 
169  if ( initializeAnimation() ) {
170 
171  lockAnimations();
172  animate();
173 
174  }
175  else {
176 
177  transition().data()->hide();
178 
179  }
180  }

183 bool LineEditData::initializeAnimation( void ) {
184  if ( !( enabled() && target_ && target_.data()->isVisible() ) ) return false;
185 
186  if ( recursiveCheck() ) return false;
187 
188  QRect current( targetRect() );
189 
190  transition().data()->setOpacity(0);
191  transition().data()->setGeometry( current );
192 
193  if ( widgetRect_.isValid() &&
194  !transition().data()->currentPixmap().isNull() &&
195  widgetRect_ != current ) {
196 
197  // if label geometry has changed since last animation
198  // one must clone the pixmap to make it match the right
199  // geometry before starting the animation.
200  QPixmap pixmap( current.size() );
201  pixmap.fill( Qt::transparent );
202  QPainter p( &pixmap );
203  p.drawPixmap( widgetRect_.topLeft() - current.topLeft(), transition().data()->currentPixmap() );
204  p.end();
205  transition().data()->setStartPixmap( pixmap );
206 
207  }
208  else {
209 
210  transition().data()->setStartPixmap( transition().data()->currentPixmap() );
211 
212  }
213 
214  bool valid( !transition().data()->startPixmap().isNull() );
215  if ( valid ) {
216  transition().data()->show();
217  transition().data()->raise();
218  }
219 
220  setRecursiveCheck( true );
221  transition().data()->setEndPixmap( transition().data()->grab( target_.data(), targetRect() ) );
222  setRecursiveCheck( false );
223 
224  return valid;
225 
226  }
