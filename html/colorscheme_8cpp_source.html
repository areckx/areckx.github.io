63 StateEffects::StateEffects(QPalette::ColorGroup state)
64  : _color(0, 0, 0, 0) {
65  QString group;
66  if (state == QPalette::Disabled)
67  group = "ColorEffects:Disabled";
68  else if (state == QPalette::Inactive)
69  group = "ColorEffects:Inactive";
70  else {
71  _effects[0] = 0;
72  _effects[1] = 0;
73  _effects[2] = 0;
74  }
75 
76  // NOTE: keep this in sync with kdebase/workspace/kcontrol/colors/colorscm.cpp
77  if (!group.isEmpty()) {
78  bool dp = state == QPalette::Disabled;
79  _effects[Intensity] = (int)(dp ? IntensityDarken : IntensityNoEffect);
80  _effects[Color] = (int)(dp ? ColorNoEffect : ColorFade);
81  _effects[Contrast] = (int)(dp ? ContrastFade : ContrastTint);
82  _amount[Intensity] = dp ? 0.10 : 0.0;
83  _amount[Color] = dp ? 0.0 : 0.025;
84  _amount[Contrast] = dp ? 0.65 : 0.10;
85  if (_effects[Color] > ColorNoEffect)
86  _color = dp ? QColor(56, 56, 56) : QColor(112, 111, 110);
87  }
88  }

94 QBrush StateEffects::brush(const QBrush& background) const {
95  QColor color = background.color(); // TODO - actually work on brushes
96  switch (_effects[Intensity]) {
97  case IntensityShade:
98  color = ColorUtils::shade(color, _amount[Intensity]);
99  break;
100  case IntensityDarken:
101  color = ColorUtils::darken(color, _amount[Intensity]);
102  break;
103  case IntensityLighten:
104  color = ColorUtils::lighten(color, _amount[Intensity]);
105  break;
106  }
107  switch (_effects[Color]) {
108  case ColorDesaturate:
109  color = ColorUtils::darken(color, 0.0, 1.0 - _amount[Color]);
110  break;
111  case ColorFade:
112  color = ColorUtils::mix(color, _color, _amount[Color]);
113  break;
114  case ColorTint:
115  color = ColorUtils::tint(color, _color, _amount[Color]);
116  break;
117  }
118  return QBrush(color);
119  }

121 QBrush StateEffects::brush(const QBrush& foreground, const QBrush& background) const {
122  QColor color = foreground.color(); // TODO - actually work on brushes
123  QColor bg = background.color();
124  // Apply the foreground effects
125  switch (_effects[Contrast]) {
126  case ContrastFade:
127  color = ColorUtils::mix(color, bg, _amount[Contrast]);
128  break;
129  case ContrastTint:
130  color = ColorUtils::tint(color, bg, _amount[Contrast]);
131  break;
132  }
133  // Now apply global effects
134  return brush(color);
135  }

294 void ColorSchemePrivate::init(QPalette::ColorGroup state, const char* /*group*/, SetDefaultColors defaults) {
295  // loaded-from-config colors
296  _brushes.fg[0] = SET_DEFAULT(NormalText);
297  _brushes.fg[1] = SET_DEFAULT(InactiveText);
298  _brushes.fg[2] = SET_DEFAULT(ActiveText);
299  _brushes.fg[3] = SET_DEFAULT(LinkText);
300  _brushes.fg[4] = SET_DEFAULT(VisitedText);
301  _brushes.fg[5] = SET_DEFAULT(NegativeText);
302  _brushes.fg[6] = SET_DEFAULT(NeutralText);
303  _brushes.fg[7] = SET_DEFAULT(PositiveText);
304 
305  _brushes.deco[0] = DECO_DEFAULT(Hover);
306  _brushes.deco[1] = DECO_DEFAULT(Focus);
307 
308  // apply state adjustments
309  if (state != QPalette::Active) {
310  StateEffects effects(state);
311  for (int i = 0; i < 8; i++) {
312  _brushes.fg[i] = effects.brush(_brushes.fg[i], _brushes.bg[0]);
313  }
314  _brushes.deco[0] = effects.brush(_brushes.deco[0], _brushes.bg[0]);
315  _brushes.deco[1] = effects.brush(_brushes.deco[1], _brushes.bg[0]);
316  _brushes.bg[0] = effects.brush(_brushes.bg[0]);
317  _brushes.bg[1] = effects.brush(_brushes.bg[1]);
318  }
319 
320  // calculated backgrounds
321  _brushes.bg[2] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[2].color());
322  _brushes.bg[3] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[3].color());
323  _brushes.bg[4] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[4].color());
324  _brushes.bg[5] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[5].color());
325  _brushes.bg[6] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[6].color());
326  _brushes.bg[7] = ColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[7].color());
327  }

416 ColorScheme::ColorScheme(QPalette::ColorGroup state, ColorSet set) {
417  switch (set) {
418  case Window:
419  d = new ColorSchemePrivate(state, "Colors:Window", defaultWindowColors);
420  break;
421  case Button:
422  d = new ColorSchemePrivate(state, "Colors:Button", defaultButtonColors);
423  break;
424  case Selection: {
425  // KConfigGroup group(config,"ColorEffects:Inactive");
426  // NOTE: keep this in sync with kdebase/workspace/kcontrol/colors/colorscm.cpp
427  bool inactiveSelectionEffect = true; // group.readEntry("ChangeSelectionColor", group.readEntry("Enable", true));
428  // if enabled, inactiver/disabled uses Window colors instead, ala gtk
429  // ...except tinted with the Selection:NormalBackground color so it looks more like selection
430  if (state == QPalette::Active || (state == QPalette::Inactive && !inactiveSelectionEffect))
431  d = new ColorSchemePrivate(state, "Colors:Selection", defaultSelectionColors);
432  else if (state == QPalette::Inactive)
433  d = new ColorSchemePrivate(state, "Colors:Window", defaultWindowColors,
434  ColorScheme(QPalette::Active, Selection).background());
435  else // disabled (...and still want this branch when inactive+disabled exists)
436  d = new ColorSchemePrivate(state, "Colors:Window", defaultWindowColors);
437  }
438  break;
439  case Tooltip:
440  d = new ColorSchemePrivate(state, "Colors:Tooltip", defaultTooltipColors);
441  break;
442  default:
443  d = new ColorSchemePrivate(state, "Colors:View", defaultViewColors);
444  }
445  }

483 QColor ColorScheme::shade(const QColor& color, ShadeRole role, qreal contrast, qreal chromaAdjust) {
484  // nan -> 1.0
485  contrast = (1.0 > contrast ? (-1.0 < contrast ? contrast : -1.0) : 1.0);
486  qreal y = ColorUtils::luma(color), yi = 1.0 - y;
487 
488  // handle very dark colors (base, mid, dark, shadow == midlight, light)
489  if (y < 0.006) {
490  switch (role) {
491  case ColorScheme::LightShade:
492  return ColorUtils::shade(color, 0.05 + 0.95 * contrast, chromaAdjust);
493  case ColorScheme::MidShade:
494  return ColorUtils::shade(color, 0.01 + 0.20 * contrast, chromaAdjust);
495  case ColorScheme::DarkShade:
496  return ColorUtils::shade(color, 0.02 + 0.40 * contrast, chromaAdjust);
497  default:
498  return ColorUtils::shade(color, 0.03 + 0.60 * contrast, chromaAdjust);
499  }
500  }
501 
502  // handle very light colors (base, midlight, light == mid, dark, shadow)
503  if (y > 0.93) {
504  switch (role) {
505  case ColorScheme::MidlightShade:
506  return ColorUtils::shade(color, -0.02 - 0.20 * contrast, chromaAdjust);
507  case ColorScheme::DarkShade:
508  return ColorUtils::shade(color, -0.06 - 0.60 * contrast, chromaAdjust);
509  case ColorScheme::ShadowShade:
510  return ColorUtils::shade(color, -0.10 - 0.90 * contrast, chromaAdjust);
511  default:
512  return ColorUtils::shade(color, -0.04 - 0.40 * contrast, chromaAdjust);
513  }
514  }
515 
516  // handle everything else
517  qreal lightAmount = (0.05 + y * 0.55) * (0.25 + contrast * 0.75);
518  qreal darkAmount = ( - y ) * (0.55 + contrast * 0.35);
519  switch (role) {
520  case ColorScheme::LightShade:
521  return ColorUtils::shade(color, lightAmount, chromaAdjust);
522  case ColorScheme::MidlightShade:
523  return ColorUtils::shade(color, (0.15 + 0.35 * yi) * lightAmount, chromaAdjust);
524  case ColorScheme::MidShade:
525  return ColorUtils::shade(color, (0.35 + 0.15 * y) * darkAmount, chromaAdjust);
526  case ColorScheme::DarkShade:
527  return ColorUtils::shade(color, darkAmount, chromaAdjust);
528  default:
529  return ColorUtils::darken(ColorUtils::shade(color, darkAmount, chromaAdjust), 0.5 + 0.3 * y);
530  }
531  }
