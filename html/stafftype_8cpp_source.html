130 bool StaffType::operator==(const StaffType& st) const
131  {
132  if (!isSameStructure(st) || st._xmlName != _xmlName) { // common to all type groups
133  return false;
134  }
135  if (_group == StaffGroup::TAB) { // TAB-specific
136  bool v = st._durationFontIdx == _durationFontIdx
137  && st._durationFontSize == _durationFontSize
138  && st._durationFontUserY == _durationFontUserY
139  && st._fretFontIdx == _fretFontIdx
140  && st._fretFontSize == _fretFontSize
141  && st._fretFontUserY == _fretFontUserY
142  ;
143  return v;
144  }
145  return true;
146  }

154 bool StaffType::isSameStructure(const StaffType& st) const
155  {
156  if (st.group() != group() // common to all type groups
157  || st._lines != _lines
158  || st._stepOffset != _stepOffset
159  || st._lineDistance != _lineDistance
160  || st._genClef != _genClef
161  || st._showBarlines != _showBarlines
162  || st._slashStyle != _slashStyle
163  || st._genTimesig != _genTimesig)
164  return false;
165  if (_group == StaffGroup::STANDARD) // standard specific
166  if (st._noteHeadScheme != _noteHeadScheme)
167  return false;
168  if (_group != StaffGroup::TAB) { // common to pitched and percussion
169  return st._genKeysig == _genKeysig
170  && st._showLedgerLines == _showLedgerLines
171  ;
172  }
173  else { // TAB-specific
174  return st._genDurations == _genDurations
175  && st._symRepeat == _symRepeat
176  && st._linesThrough == _linesThrough
177  && st._minimStyle == _minimStyle
178  && st._onLines == _onLines
179  && st._showBackTied == _showBackTied
180  && st._showRests == _showRests
181  && st._stemsDown == _stemsDown
182  && st._stemsThrough == _stemsThrough
183  && st._upsideDown == _upsideDown
184  && st._showTabFingering == _showTabFingering
185  && st._useNumbers == _useNumbers
186  ;
187  }
188  }

194 void StaffType::write(XmlWriter& xml) const
195  {
196  xml.stag(QString("StaffType group=\"%1\"").arg(fileGroupNames[(int)_group]));
197  if (!_xmlName.isEmpty())
198  xml.tag("name", _xmlName);
199  if (_lines != 5)
200  xml.tag("lines", _lines);
201  if (_lineDistance.val() != 1.0)
202  xml.tag("lineDistance", _lineDistance.val());
203  if (_yoffset.val() != 0.0)
204  xml.tag("yoffset", _yoffset.val());
205  if (_userMag != 1.0)
206  xml.tag("mag", _userMag);
207  if (_small)
208  xml.tag("small", _small);
209  if (_stepOffset)
210  xml.tag("stepOffset", _stepOffset);
211  if (!_genClef)
212  xml.tag("clef", _genClef);
213  if (_slashStyle)
214  xml.tag("slashStyle", _slashStyle);
215  if (!_showBarlines)
216  xml.tag("barlines", _showBarlines);
217  if (!_genTimesig)
218  xml.tag("timesig", _genTimesig);
219  if (_group == StaffGroup::STANDARD) {
220  xml.tag("noteheadScheme", StaffType::scheme2name(_noteHeadScheme), StaffType::scheme2name(NoteHeadScheme::HEAD_NORMAL));
221  }
222  if (_group == StaffGroup::STANDARD || _group == StaffGroup::PERCUSSION) {
223  if (!_genKeysig)
224  xml.tag("keysig", _genKeysig);
225  if (!_showLedgerLines)
226  xml.tag("ledgerlines", _showLedgerLines);
227  }
228  else {
229  xml.tag("durations", _genDurations);
230  xml.tag("durationFontName", _durationFonts[_durationFontIdx].displayName); // write font names anyway for backward compatibility
231  xml.tag("durationFontSize", _durationFontSize);
232  xml.tag("durationFontY", _durationFontUserY);
233  xml.tag("fretFontName", _fretFonts[_fretFontIdx].displayName);
234  xml.tag("fretFontSize", _fretFontSize);
235  xml.tag("fretFontY", _fretFontUserY);
236  if (_symRepeat != TablatureSymbolRepeat::NEVER)
237  xml.tag("symbolRepeat", int(_symRepeat));
238  xml.tag("linesThrough", _linesThrough);
239  xml.tag("minimStyle", int(_minimStyle));
240  xml.tag("onLines", _onLines);
241  xml.tag("showRests", _showRests);
242  xml.tag("stemsDown", _stemsDown);
243  xml.tag("stemsThrough", _stemsThrough);
244  xml.tag("upsideDown", _upsideDown);
245  xml.tag("showTabFingering", _showTabFingering, false);
246  xml.tag("useNumbers", _useNumbers);
247  // only output "showBackTied" if different from !"slashStyle"
248  // to match the behaviour in 2.0.2 scores (or older)
249  if (_showBackTied != !_slashStyle)
250  xml.tag("showBackTied", _showBackTied);
251  }
252  xml.etag();
253  }

259 void StaffType::read(XmlReader& e)
260  {
261  QString group = e.attribute("group", fileGroupNames[(int)StaffGroup::STANDARD]);
262  if (group == fileGroupNames[(int)StaffGroup::TAB])
263  _group = StaffGroup::TAB;
264  else if (group == fileGroupNames[(int)StaffGroup::PERCUSSION])
265  _group = StaffGroup::PERCUSSION;
266  else if (group == fileGroupNames[(int)StaffGroup::STANDARD])
267  _group = StaffGroup::STANDARD;
268  else {
269  qDebug("StaffType::read: unknown group: %s", qPrintable(group));
270  _group = StaffGroup::STANDARD;
271  }
272 
273  while (e.readNextStartElement()) {
274  const QStringRef& tag(e.name());
275  if (tag == "name")
276  setXmlName(e.readElementText());
277  else if (tag == "lines")
278  setLines(e.readInt());
279  else if (tag == "lineDistance")
280  setLineDistance(Spatium(e.readDouble()));
281  else if (tag == "clef")
282  setGenClef(e.readInt());
283  else if (tag == "slashStyle") {
284  bool val = e.readInt() != 0;
285  setSlashStyle(val);
286  setShowBackTied(!val); // for compatibility with 2.0.2 scores where this prop
287  } // was lacking and controlled by "slashStyle" instead
288  else if (tag == "yoffset")
289  _yoffset = Spatium(e.readDouble());
290  else if (tag == "mag")
291  _userMag = e.readDouble();
292  else if (tag == "small")
293  _small = e.readBool();
294  else if (tag == "barlines")
295  setShowBarlines(e.readInt());
296  else if (tag == "timesig")
297  setGenTimesig(e.readInt());
298  else if (tag == "noteheadScheme")
299  setNoteHeadScheme(StaffType::name2scheme(e.readElementText()));
300  else if (tag == "keysig")
301  _genKeysig = e.readInt();
302  else if (tag == "ledgerlines")
303  _showLedgerLines = e.readInt();
304  else if (tag == "durations")
305  setGenDurations(e.readBool());
306  else if (tag == "durationFontName")
307  setDurationFontName(e.readElementText());
308  else if (tag == "durationFontSize")
309  setDurationFontSize(e.readDouble());
310  else if (tag == "durationFontY")
311  setDurationFontUserY(e.readDouble());
312  else if (tag == "fretFontName")
313  setFretFontName(e.readElementText());
314  else if (tag == "fretFontSize")
315  setFretFontSize(e.readDouble());
316  else if (tag == "fretFontY")
317  setFretFontUserY(e.readDouble());
318  else if (tag == "symbolRepeat")
319  setSymbolRepeat( (TablatureSymbolRepeat) e.readInt() );
320  else if (tag == "linesThrough")
321  setLinesThrough(e.readBool());
322  else if (tag == "minimStyle")
323  setMinimStyle( (TablatureMinimStyle) e.readInt() );
324  else if (tag == "onLines")
325  setOnLines(e.readBool());
326  else if (tag == "showRests")
327  setShowRests(e.readBool());
328  else if (tag == "stemsDown")
329  setStemsDown(e.readBool());
330  else if (tag == "stemsThrough")
331  setStemsThrough(e.readBool());
332  else if (tag == "upsideDown")
333  setUpsideDown(e.readBool());
334  else if (tag == "showTabFingering")
335  setShowTabFingering(e.readBool());
336  else if (tag == "useNumbers")
337  setUseNumbers(e.readBool());
338  else if (tag == "showBackTied") // must be after reading "slashStyle" prop, as in older
339  setShowBackTied(e.readBool()); // scores, this prop was lacking and controlled by "slashStyle"
340  else
341  e.unknown();
342  }
343  }

409 //---------------------------------------------------------
410 // setDurationMetrics
411 // checks whether the internally computed metrics are is still valid and re-computes them, if not
412 //---------------------------------------------------------
413 
414 void StaffType::setDurationMetrics()
415  {
416  if (_durationMetricsValid && _refDPI == DPI) // metrics are still valid
417  return;
418 
419 // QFontMetrics[F]() returns results unreliably rounded to integral pixels;
420 // use a scaled up font and then scale computed values down
421 // QFontMetricsF fm(durationFont());
422  QFont font(durationFont());
423  font.setPointSizeF(_durationFontSize);
424  QFontMetricsF fm(font, MScore::paintDevice());
425  QString txt(_durationFonts[_durationFontIdx].displayValue, int(TabVal::NUM_OF));
426  QRectF bb( fm.tightBoundingRect(txt) );
427  // raise symbols by a default margin and, if marks are above lines, by half the line distance
428  // (converted from spatium units to raster units)
429  _durationGridYOffset = ( TAB_DEFAULT_DUR_YOFFS - (_onLines ? 0.0 : lineDistance().val()*0.5) ) * SPATIUM20;
430  // this is the bottomest point of any duration sign
431  _durationYOffset = _durationGridYOffset;
432  // move symbols so that the lowest margin 'sits' on the base line:
433  // move down by the whole part above (negative) the base line
434  // ( -bb.y() ) then up by the whole height ( -bb.height() )
435  _durationYOffset -= (bb.height() + bb.y()) / 100.0;
436  _durationBoxH = bb.height() / 100.0;
437  _durationBoxY = _durationGridYOffset - bb.height() / 100.0;
438  // keep track of the conditions under which metrics have been computed
439  _refDPI = DPI;
440  _durationMetricsValid = true;
441  }
442 
443 void StaffType::setFretMetrics()
444  {
445  if (_fretMetricsValid && _refDPI == DPI)
446  return;
447 
448  QFontMetricsF fm(fretFont(), MScore::paintDevice());
449  QRectF bb;
450  // compute vertical displacement
451  if (_useNumbers) {
452  // compute total height of used characters
453  QString txt = QString();
454  for (int idx = 0; idx < 10; idx++) // use only first 10 digits
455  txt.append(_fretFonts[_fretFontIdx].displayDigit[idx]);
456  bb = fm.tightBoundingRect(txt);
457  // for numbers: centre on '0': move down by the whole part above (negative)
458  // the base line ( -bb.y() ) then up by half the whole height ( -bb.height()/2 )
459  QRectF bx( fm.tightBoundingRect(_fretFonts[_fretFontIdx].displayDigit[0]) );
460  _fretYOffset = -(bx.y() + bx.height()/2.0);
461  // _fretYOffset = -(bb.y() + bb.height()/2.0); // <- using bbox of all chars
462  }
463  else {
464  // compute total height of used characters
465  QString txt(_fretFonts[_fretFontIdx].displayLetter, NUM_OF_LETTERFRETS);
466  bb = fm.tightBoundingRect(txt);
467  // for letters: centre on the 'a' ascender, by moving down half of the part above the base line in bx
468  QRectF bx( fm.tightBoundingRect(_fretFonts[_fretFontIdx].displayLetter[0]) );
469  _fretYOffset = -bx.y() / 2.0;
470  }
471  // if on string, we are done; if between strings, raise by half line distance
472  if (!_onLines)
473  _fretYOffset -= lineDistance().val() * SPATIUM20 * 0.5;
474 
475  // from _fretYOffset, compute _fretBoxH and _fretBoxY
476  _fretBoxH = bb.height();
477  _fretBoxY = bb.y() + _fretYOffset;
478 
479  // keep track of the conditions under which metrics have been computed
480  _refDPI = DPI;
481  _fretMetricsValid = true;
482  }

556 //---------------------------------------------------------
557 // chordRestStemPosY / chordStemPos / chordStemPosBeam / chordStemLength
558 //
559 // computes the stem data for the given chord, according to TAB settings
560 // NOTE: unit: spatium, position: relative to chord (DIFFERENT from Chord own functions)
561 //
562 // chordRestStemPosY
563 // returns the vertical position of stem start point
564 //---------------------------------------------------------
565 
566 qreal StaffType::chordRestStemPosY(const ChordRest *chordRest) const
567  {
568  if (stemThrough()) // does not make sense for "stems through staves" setting; just return top line vert. position
569  return 0.0;
570 
571  // if stems beside staff, position are fixed, but take into account delta for half notes
572  qreal delta = // displacement for half note stems (if used)
573  // if half notes have not a short stem OR not a half note => 0
574  (minimStyle() != TablatureMinimStyle::SHORTER || chordRest->durationType().type() != TDuration::DurationType::V_HALF) ?
575  0.0 :
576  // if stem is up, displace of half stem length down (positive)
577  // if stem is down, displace of half stem length up (negative)
578  (chordRest->up() ?
579  -STAFFTYPE_TAB_DEFAULTSTEMLEN_UP : STAFFTYPE_TAB_DEFAULTSTEMLEN_DN) * 0.5;
580  // if fret marks above lines and chordRest is up, move half a line distance up
581  if (!onLines() && chordRest->up())
582  delta -= _lineDistance.val() *0.5;
583  qreal y = (chordRest->up() ? STAFFTYPE_TAB_DEFAULTSTEMPOSY_UP : (_lines-1)*_lineDistance.val() + STAFFTYPE_TAB_DEFAULTSTEMPOSY_DN)
584  + delta;
585  return y;
586  }

588 //---------------------------------------------------------
589 // chordStemPos
590 // return position of note at other side of beam
591 //---------------------------------------------------------
592 
593 QPointF StaffType::chordStemPos(const Chord* chord) const
594  {
595  qreal y;
596  if (stemThrough())
597  // if stems are through staff, stem goes from fartest note string
598  y = (chord->up() ? chord->downString() : chord->upString()) * _lineDistance.val();
599  else
600  // if stems are beside staff, stem start point has a fixed vertical position,
601  // according to TAB parameters and stem up/down
602  y = chordRestStemPosY(chord);
603  return QPointF(chordStemPosX(chord), y);
604  }

618 //---------------------------------------------------------
619 // chordStemLength
620 // return length of stem
621 //---------------------------------------------------------
622 
623 qreal StaffType::chordStemLength(const Chord* chord) const
624  {
625  qreal stemLen;
626  // if stems are through staff, length should be computed by relevant chord algorithm;
627  // here, just return default length (= 1 'octave' = 3.5 line spaces)
628  if (stemThrough())
629  return STAFFTYPE_TAB_DEFAULTSTEMLEN_THRU * _lineDistance.val();
630  // if stems beside staff, length is fixed, but take into account shorter half note stems
631  else {
632  bool shrt = (minimStyle() == TablatureMinimStyle::SHORTER) && (chord->durationType().type() == TDuration::DurationType::V_HALF);
633  stemLen = (stemsDown() ? STAFFTYPE_TAB_DEFAULTSTEMLEN_DN : STAFFTYPE_TAB_DEFAULTSTEMLEN_UP)
634  * (shrt ? STAFFTYPE_TAB_SHORTSTEMRATIO : 1.0);
635  }
636  // scale length by scale of parent chord, but relative to scale of context staff
637  return stemLen * chord->mag() / chord->staff()->mag(chord->tick());
638  }

675 //---------------------------------------------------------
676 // tabBassStringPrefix
677 //
678 // returns a QString (possibly empty) with the prefix identifying a bass string in TAB's;
679 // can deal with non-bass strings (i.e. regular TAB lines).
680 //
681 // Implements the specifics of historic notations for bass lines (i.e. strings outside
682 // the lines of the tab), both Italian and French.
683 //
684 // strg the instrument physical string ordinal (0 = topmost string, may exceed the number
685 // of lines actually present in the TAB to reference a bass string)
686 // bool pntr to a bool receiving the info if notation allows to express a fret number or not
687 // (this is potentially different from the fact that the instrument string itself can be fretted or not)
688 //---------------------------------------------------------
689 
690 QString StaffType::tabBassStringPrefix(int strg, bool* hasFret) const
691  {
692  *hasFret = true; // assume notation allows to fret this string
693  int bassStrgIdx = (strg >= _lines ? strg - _lines + 1 : 0);
694  if (_useNumbers) {
695  // if above the max bass string which can be fretted with number notation
696  // return a number with the string index
697  if (bassStrgIdx > NUM_OF_BASSSTRINGS_WITH_NUMBER) {
698  *hasFret = false;
699  return _fretFonts[_fretFontIdx].displayDigit[strg+1];
700  }
701  // if a frettable bass string, return an empty string
702  return QString();
703  }
704  else {
705  // bass string notation
706  // if above the max bass string which can be fretted with letter notation
707  // return a number with the bass string index itself
708  if (bassStrgIdx > NUM_OF_BASSSTRINGS_WITH_LETTER) {
709  *hasFret = false;
710  return _fretFonts[_fretFontIdx].displayDigit[bassStrgIdx-1];
711  }
712  // if a frettable bass string, return a character with the relevant num. of slashes;
713  // note that the number of slashes is bassStrgIdx-1 (1st bass has no slash)
714  // and slashChar[] is 0-based (slashChar[0] => 1 slash, ...), whence the -2
715  QString prefix = bassStrgIdx > 1 ?
716  QString(_fretFonts[_fretFontIdx].slashChar[bassStrgIdx - 2]) : QString();
717  return prefix;
718  }
719  }

721 //---------------------------------------------------------
722 // drawInputStringMarks
723 //
724 // in TAB's, draws the marks within the input 'blue cursor' required to identify the current target input string.
725 //
726 // Implements the specific of historic TAB styles for instruments with more strings than TAB lines.
727 // For strings normally represented by TAB lines, no mark is required.
728 // For strings not represented by TAB lines (e.g. bass strings in lutes and similar),
729 // either a sequence of slashes OR some ledger line-like lines OR the ordinal of the string
730 // are used, according to the TAB style (French or Italian) and the string position.
731 //
732 // Note: assumes the string parameter is within legal bounds, i.e.:
733 // 0 <= string <= [instrument strings] - 1
734 //
735 // p the QPainter to draw into
736 // string the instrument physical string for which to draw the mark (0 = top string)
737 // voice the current input voice (affects mark colour)
738 // rect the rect of the 'blue rectangle' showing the input position
739 //---------------------------------------------------------
740 
741 static const qreal LEDGER_LINE_THICKNESS = 0.15; // in sp
742 static const qreal LEDGER_LINE_LEFTX = 0.25; // in % of cursor rectangle width
743 static const qreal LEDGER_LINE_RIGHTX = 0.75; // in % of cursor rectangle width
744 
745 void StaffType::drawInputStringMarks(QPainter *p, int string, int voice, QRectF rect) const
746  {
747  if (_group != StaffGroup::TAB)
748  return;
749  qreal spatium = SPATIUM20;
750  qreal lineDist = _lineDistance.val() * spatium;
751  bool hasFret;
752  QString text = tabBassStringPrefix(string, &hasFret);
753 // qreal lw = point(score()->styleS(StyleIdx::ledgerLineWidth)); // no access to score form here
754  qreal lw = LEDGER_LINE_THICKNESS * spatium; // use a fixed width
755  QPen pen(MScore::selectColor[voice].lighter(SHADOW_NOTE_LIGHT), lw);
756  p->setPen(pen);
757  // draw conventional 'ledger lines', if required
758  int numOfLedgerLines = numOfTabLedgerLines(string);
759  qreal x1 = rect.x() + rect.width() * LEDGER_LINE_LEFTX;
760  qreal x2 = rect.x() + rect.width() * LEDGER_LINE_RIGHTX;
761  // cursor rect is 1 line dist. high, and it is:
762  // centred on the line for "frets on strings" => lower top ledger line 1/2 line dist.
763  // sitting on the line for "frets above strings" => lower top ledger line 1 full line dist
764  qreal y = rect.top() + lineDist * (_onLines ? 0.5 : 1.0);
765  for (int i = 0; i < numOfLedgerLines; i++) {
766  p->drawLine(QLineF(x1, y, x2, y));
767  y += lineDist / numOfLedgerLines; // insert other lines between top line and tab body
768  }
769  // draw the text, if any
770  if (!text.isEmpty()) {
771  p->setFont(fretFont());
772  p->drawText(QPointF(rect.left(), rect.top() + lineDist), text);
773  }
774  }

827 //
828 // returns the string Y offset from a string ordinal in physical order:
829 // manages upsideDown and extra bass strings.
830 //
831 // The returned values is in sp. and is relative to the staff top line.
832 //
833 // Note: the difference with physStringToVisual() is that this function takes into account
834 // peculiarities of bass string notations.
835 //---------------------------------------------------------

884 //---------------------------------------------------------
885 // layout
886 //---------------------------------------------------------
887 
888 void TabDurationSymbol::layout()
889  {
890  if(!_tab) {
891  setbbox(QRectF());
892  return;
893  }
894  qreal _spatium = spatium();
895  qreal hbb, wbb, xbb, ybb; // bbox sizes
896  qreal xpos, ypos; // position coords
897 
898  _beamGrid = TabBeamGrid::NONE;
899  Chord* chord = static_cast<Chord*>(parent());
900  // if no chord (shouldn't happens...) or not a special beam mode, layout regular symbol
901  if (!chord || !chord->isChord() ||
902  (chord->beamMode() != Beam::Mode::BEGIN && chord->beamMode() != Beam::Mode::MID &&
903  chord->beamMode() != Beam::Mode::END) ) {
904  QFontMetricsF fm(_tab->durationFont(), MScore::paintDevice());
905  hbb = _tab->durationBoxH();
906  wbb = fm.width(_text);
907  xbb = 0.0;
908  xpos = 0.0;
909  ypos = _tab->durationFontYOffset();
910  ybb = _tab->durationBoxY() - ypos;
911  // with rests, move symbol down by half its displacement from staff
912  if (parent() && parent()->isRest()) {
913  ybb += TAB_RESTSYMBDISPL * _spatium;
914  ypos += TAB_RESTSYMBDISPL * _spatium;
915  }
916  }
917  // if on a chord with special beam mode, layout an 'English'-style duration grid
918  else {
919  TablatureDurationFont font = _tab->_durationFonts[_tab->_durationFontIdx];
920  hbb = font.gridStemHeight * _spatium; // bbox height is stem height
921  wbb = font.gridStemWidth * _spatium; // bbox width is stem width
922  xbb = -wbb * 0.5; // bbox is half at left and half at right of stem centre
923  ybb = -hbb; // bbox top is at top of stem height
924  xpos = 0.75 * _spatium; // conventional centring of stem on fret marks
925  ypos = _tab->durationGridYOffset(); // stem start is at bottom
926  if (chord->beamMode() == Beam::Mode::BEGIN) {
927  _beamGrid = TabBeamGrid::INITIAL;
928  _beamLength = 0.0;
929  }
930  else if (chord->beamMode() == Beam::Mode::MID || chord->beamMode() == Beam::Mode::END) {
931  _beamLevel = (int)(chord->durationType().type()) - (int)(font.zeroBeamLevel);
932  _beamGrid = (_beamLevel < 1 ? TabBeamGrid::INITIAL : TabBeamGrid::MEDIALFINAL);
933  // _beamLength and bbox x and width will be set in layout2(),
934  // once horiz. positions of chords are known
935  }
936  }
937  // set this' mag from parent chord mag (include staff mag)
938  qreal mag = chord != nullptr ? chord->mag() : 1.0;
939  setMag(mag);
940  mag = magS(); // local mag * score mag
941  // set magnified bbox and position
942  bbox().setRect(xbb * mag, ybb * mag, wbb * mag, hbb * mag);
943  setPos(xpos*mag, ypos*mag);
944  }

946 //---------------------------------------------------------
947 // layout2
948 //
949 // Second step: after horizontal positions of elements involved are defined,
950 // compute width of 'grid beams'
951 //---------------------------------------------------------
952 
953 void TabDurationSymbol::layout2()
954  {
955  // if not within a TAB or not a MEDIALFINAL grid element, do nothing
956  if(!_tab || _beamGrid != TabBeamGrid::MEDIALFINAL)
957  return;
958 
959  // get 'grid' beam length from page positions of this' chord and previous chord
960  Chord* chord = static_cast<Chord*>(parent());
961  ChordRest* prevChord = prevChordRest(chord, true);
962  if (chord == nullptr || prevChord == nullptr)
963  return;
964  qreal mags = magS();
965  qreal beamLen = prevChord->pagePos().x() - chord->pagePos().x(); // negative
966  // page pos. difference already includes any magnification in effect:
967  // scale it down, as it will be magnified again during drawing
968  _beamLength = beamLen / mags;
969  // update bbox x and w, but keep current y and h
970  bbox().setX(beamLen);
971  // set bbox width to half a stem width (magnified) plus beam length (already magnified)
972  bbox().setWidth(_tab->_durationFonts[_tab->_durationFontIdx].gridStemWidth * spatium() * 0.5 * mags - beamLen);
973  }

975 //---------------------------------------------------------
976 // draw
977 //---------------------------------------------------------
978 
979 void TabDurationSymbol::draw(QPainter* painter) const
980  {
981  if (!_tab)
982  return;
983  qreal mag = magS();
984  qreal imag = 1.0 / mag;
985 
986  QPen pen(curColor());
987  painter->setPen(pen);
988  painter->scale(mag, mag);
989  if (_beamGrid == TabBeamGrid::NONE) {
990  // if no beam grid, draw symbol
991  QFont f(_tab->durationFont());
992  f.setPointSizeF(f.pointSizeF() * MScore::pixelRatio);
993  painter->setFont(f);
994  painter->drawText(QPointF(0.0, 0.0), _text);
995  }
996  else {
997  // if beam grid, draw stem line
998  TablatureDurationFont& font = _tab->_durationFonts[_tab->_durationFontIdx];
999  qreal _spatium = spatium();
1000  pen.setCapStyle(Qt::FlatCap);
1001  pen.setWidthF(font.gridStemWidth * _spatium);
1002  painter->setPen(pen);
1003  // take stem height from bbox, but de-magnify it, as drawing is already magnified
1004  qreal h = bbox().y() / mag;
1005  painter->drawLine(QPointF(0.0, h), QPointF(0.0, 0.0) );
1006  // if beam grid is medial/final, draw beam lines too: lines go from mid of
1007  // previous stem (delta x stored in _beamLength) to mid of this' stem (0.0)
1008  if (_beamGrid == TabBeamGrid::MEDIALFINAL) {
1009  pen.setWidthF(font.gridBeamWidth * _spatium);
1010  painter->setPen(pen);
1011  // lower heigth available to beams by half a beam width,
1012  // so that top beam upper border aligns with stem top
1013  h += (font.gridBeamWidth * _spatium) * 0.5;
1014  // draw beams equally spaced within the stem height (this is
1015  // different from modern engraving, but common in historic prints)
1016  qreal step = -h / _beamLevel;
1017  qreal y = h;
1018  for (int i = 0; i < _beamLevel; i++, y += step)
1019  painter->drawLine(QPointF(_beamLength, y), QPointF(0.0, y) );
1020  }
1021  }
1022  painter->scale(imag, imag);
1023  }
