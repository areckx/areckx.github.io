95 void TransitionWidget::paintEvent( QPaintEvent* event ) {
96 
97  // fully transparent case
98  if ( opacity() >= 1.0 && endPixmap().isNull() ) return;
99  if ( !paintEnabled_ ) return;
100 
101  // get rect
102  QRect rect = event->rect();
103  if ( !rect.isValid() ) rect = this->rect();
104 
105  // local pixmap
106  const bool paintOnWidget( testFlag( PaintOnWidget ) && !testFlag( Transparent ) );
107  if ( !paintOnWidget ) {
108 
109  if ( currentPixmap_.isNull() || currentPixmap_.size() != size() ) {
110  currentPixmap_ = QPixmap( size() );
111  }
112 
113  }
114 
115  // fill
116  currentPixmap_.fill( Qt::transparent );
117 
118  // copy local pixmap to current
119  {
120 
121  QPainter p;
122 
123  // draw end pixmap first, provided that opacity is small enough
124  if ( opacity() >= 0.004 && !endPixmap_.isNull() ) {
125 
126  // faded endPixmap if parent target is transparent and opacity is
127  if ( opacity() <= 0.996 && testFlag( Transparent ) ) {
128 
129  fade( endPixmap_, currentPixmap_, opacity(), rect );
130  p.begin( &currentPixmap_ );
131  p.setClipRect( event->rect() );
132 
133  }
134  else {
135 
136  if ( paintOnWidget ) p.begin( this );
137  else p.begin( &currentPixmap_ );
138  p.setClipRect( event->rect() );
139  p.drawPixmap( QPoint(), endPixmap_ );
140 
141  }
142 
143  }
144  else {
145 
146  if ( paintOnWidget ) p.begin( this );
147  else p.begin( &currentPixmap_ );
148  p.setClipRect( event->rect() );
149 
150  }
151 
152  // draw fading start pixmap
153  if ( opacity() <= 0.996 && !startPixmap_.isNull() ) {
154  if ( opacity() >= 0.004 ) {
155 
156  fade( startPixmap_, localStartPixmap_, 1.0 - opacity(), rect );
157  p.drawPixmap( QPoint(), localStartPixmap_ );
158 
159  }
160  else p.drawPixmap( QPoint(), startPixmap_ );
161 
162  }
163 
164  p.end();
165  }
166 
167  // copy current pixmap on widget
168  if ( !paintOnWidget ) {
169  QPainter p( this );
170  p.setClipRect( event->rect() );
171  p.drawPixmap( QPoint(0, 0), currentPixmap_ );
172  p.end();
173  }
174  }

180 void TransitionWidget::grabBackground(QPixmap& pixmap, QWidget* widget, QRect& rect) const
181  {
182  if (!widget)
183  return;
184 
185  QWidgetList widgets;
186  if( widget->autoFillBackground() )
187  { widgets.push_back( widget ); }
188 
189  QWidget* parent(0);
190 
191  // get highest level parent
192  for ( parent = widget->parentWidget(); parent; parent = parent->parentWidget() ) {
193 
194  if ( !( parent->isVisible() && parent->rect().isValid() ) ) continue;
195 
196  // store in list
197  widgets.push_back( parent );
198 
199  // stop at topLevel
200  if ( parent->isTopLevel() || parent->autoFillBackground() ) break;
201 
202  }
203 
204  if ( !parent ) parent = widget;
205 
206  // painting
207  QPainter p(&pixmap);
208  p.setClipRect( rect );
209  const QBrush backgroundBrush = parent->palette().brush( parent->backgroundRole());
210  if ( backgroundBrush.style() == Qt::TexturePattern) {
211 
212  p.drawTiledPixmap( rect, backgroundBrush.texture(), widget->mapTo( parent, rect.topLeft() ) );
213 
214  }
215  else {
216 
217  p.fillRect( pixmap.rect(), backgroundBrush );
218 
219  }
220 
221  if ( parent->isTopLevel() && parent->testAttribute(Qt::WA_StyledBackground)) {
222  QStyleOption option;
223  option.initFrom(parent);
224  option.rect = rect;
225  option.rect.translate( widget->mapTo( parent, rect.topLeft() ) );
226  p.translate(-option.rect.topLeft());
227  parent->style()->drawPrimitive ( QStyle::PE_Widget, &option, &p, parent );
228  }
229 
230  p.end();
231 
232  // draw all widgets in parent list
233  // backward
234  QPaintEvent event(rect);
235  for (int i = widgets.size() - 1; i>=0; i--) {
236  QWidget* w = widgets.at(i);
237  w->render(&pixmap, rect.topLeft());
238 // QPainter::setRedirected(w, &pixmap, widget->mapTo(w, rect.topLeft()));
239 // event = QPaintEvent(QRect(QPoint(), rect.size()));
240 // QCoreApplication::sendEvent(w, &event);
241 // QPainter::restoreRedirected(w);
242  }
243  }

255 void TransitionWidget::fade( const QPixmap& source, QPixmap& target, qreal opacity, const QRect& rect ) const
256  {
257  if ( target.isNull() || target.size() != size() ) {
258  target = QPixmap( size() );
259  }
260 
261  // erase target
262  target.fill( Qt::transparent );
263 
264  // check opacity
265  if ( opacity * 255 < 1 ) return;
266 
267  QPainter p( &target );
268  p.setClipRect( rect );
269 
270  // draw pixmap
271  p.drawPixmap( QPoint(0, 0), source );
272 
273  // opacity mask (0.996 corresponds to 254/255)
274  if ( opacity <= 0.996 ) {
275  p.setCompositionMode(QPainter::CompositionMode_DestinationIn);
276  QColor color( Qt::black );
277  color.setAlphaF( opacity );
278  p.fillRect(rect, color );
279  }
280 
281  p.end();
282  return;
283  }
