143 ObjectStream::ObjectStream(XRef *xref, int objStrNumA, int recursion) {
144  Stream *str;
145  Parser *parser;
146  Goffset *offsets;
147  Object objStr, obj1, obj2;
148  Goffset first;
149  int i;
150 
151  objStrNum = objStrNumA;
152  nObjects = 0;
153  objs = NULL;
154  objNums = NULL;
155  ok = gFalse;
156 
157  if (!xref->fetch(objStrNum, 0, &objStr, recursion)->isStream()) {
158  goto err1;
159  }
160 
161  if (!objStr.streamGetDict()->lookup("N", &obj1, recursion)->isInt()) {
162  obj1.free();
163  goto err1;
164  }
165  nObjects = obj1.getInt();
166  obj1.free();
167  if (nObjects <= 0) {
168  goto err1;
169  }
170 
171  objStr.streamGetDict()->lookup("First", &obj1, recursion);
172  if (!obj1.isInt() && !obj1.isInt64()) {
173  obj1.free();
174  goto err1;
175  }
176  if (obj1.isInt())
177  first = obj1.getInt();
178  else
179  first = obj1.getInt64();
180  obj1.free();
181  if (first < 0) {
182  goto err1;
183  }
184 
185  // this is an arbitrary limit to avoid integer overflow problems
186  // in the 'new Object[nObjects]' call (Acrobat apparently limits
187  // object streams to 100-200 objects)
188  if (nObjects > 1000000) {
189  error(errSyntaxError, -1, "Too many objects in an object stream");
190  goto err1;
191  }
192  objs = new Object[nObjects];
193  objNums = (int *)gmallocn(nObjects, sizeof(int));
194  offsets = (Goffset *)gmallocn(nObjects, sizeof(Goffset));
195 
196  // parse the header: object numbers and offsets
197  objStr.streamReset();
198  obj1.initNull();
199  str = new EmbedStream(objStr.getStream(), &obj1, gTrue, first);
200  parser = new Parser(xref, new Lexer(xref, str), gFalse);
201  for (i = 0; i < nObjects; ++i) {
202  parser->getObj(&obj1);
203  parser->getObj(&obj2);
204  if (!obj1.isInt() || !(obj2.isInt() || obj2.isInt64())) {
205  obj1.free();
206  obj2.free();
207  delete parser;
208  gfree(offsets);
209  goto err1;
210  }
211  objNums[i] = obj1.getInt();
212  if (obj2.isInt())
213  offsets[i] = obj2.getInt();
214  else
215  offsets[i] = obj2.getInt64();
216  obj1.free();
217  obj2.free();
218  if (objNums[i] < 0 || offsets[i] < 0 ||
219  (i > 0 && offsets[i] < offsets[i-1])) {
220  delete parser;
221  gfree(offsets);
222  goto err1;
223  }
224  }
225  while (str->getChar() != EOF) ;
226  delete parser;
227 
228  // skip to the first object - this shouldn't be necessary because
229  // the First key is supposed to be equal to offsets[0], but just in
230  // case...
231  for (Goffset pos = first; pos < offsets[0]; ++pos) {
232  objStr.getStream()->getChar();
233  }
234 
235  // parse the objects
236  for (i = 0; i < nObjects; ++i) {
237  obj1.initNull();
238  if (i == nObjects - 1) {
239  str = new EmbedStream(objStr.getStream(), &obj1, gFalse, 0);
240  } else {
241  str = new EmbedStream(objStr.getStream(), &obj1, gTrue,
242  offsets[i+1] - offsets[i]);
243  }
244  parser = new Parser(xref, new Lexer(xref, str), gFalse);
245  parser->getObj(&objs[i]);
246  while (str->getChar() != EOF) ;
247  delete parser;
248  }
249 
250  gfree(offsets);
251  ok = gTrue;
252 
253  err1:
254  objStr.free();
255 }

257 ObjectStream::~ObjectStream() {
258  int i;
259 
260  if (objs) {
261  for (i = 0; i < nObjects; ++i) {
262  objs[i].free();
263  }
264  delete[] objs;
265  }
266  gfree(objNums);
267 }

273  return objs[objIdx].copy(obj);

314 XRef::XRef(BaseStream *strA, Goffset pos, Goffset mainXRefEntriesOffsetA, GBool *wasReconstructed, GBool reconstruct) {
315  Object obj;
316 
317  init();
318  mainXRefEntriesOffset = mainXRefEntriesOffsetA;
319 
320  // read the trailer
321  str = strA;
322  start = str->getStart();
323  prevXRefOffset = mainXRefOffset = pos;
324 
325  if (reconstruct && !(ok = constructXRef(wasReconstructed)))
326  {
327  errCode = errDamaged;
328  return;
329  }
330  else
331  {
332  // if there was a problem with the 'startxref' position, try to
333  // reconstruct the xref table
334  if (prevXRefOffset == 0) {
335  if (!(ok = constructXRef(wasReconstructed))) {
336  errCode = errDamaged;
337  return;
338  }
339 
340  // read the xref table
341  } else {
342  std::vector<Goffset> followedXRefStm;
343  readXRef(&prevXRefOffset, &followedXRefStm, NULL);
344 
345  // if there was a problem with the xref table,
346  // try to reconstruct it
347  if (!ok) {
348  if (!(ok = constructXRef(wasReconstructed))) {
349  errCode = errDamaged;
350  return;
351  }
352  }
353  }
354 
355  // set size to (at least) the size specified in trailer dict
356  trailerDict.dictLookupNF("Size", &obj);
357  if (!obj.isInt()) {
358  error(errSyntaxWarning, -1, "No valid XRef size in trailer");
359  } else {
360  if (obj.getInt() > size) {
361  if (resize(obj.getInt()) != obj.getInt()) {
362  if (!(ok = constructXRef(wasReconstructed))) {
363  obj.free();
364  errCode = errDamaged;
365  return;
366  }
367  }
368  }
369  }
370  obj.free();
371 
372  // get the root dictionary (catalog) object
373  trailerDict.dictLookupNF("Root", &obj);
374  if (obj.isRef()) {
375  rootNum = obj.getRefNum();
376  rootGen = obj.getRefGen();
377  obj.free();
378  } else {
379  obj.free();
380  if (!(ok = constructXRef(wasReconstructed))) {
381  errCode = errDamaged;
382  return;
383  }
384  }
385  }
386  // now set the trailer dictionary's xref pointer so we can fetch
387  // indirect objects from it
388  trailerDict.getDict()->setXRef(this);
389 }

460 int XRef::reserve(int newSize)
461 {
462  if (newSize > capacity) {
463 
464  int realNewSize;
465  for (realNewSize = capacity ? 2 * capacity : 1024;
466  newSize > realNewSize && realNewSize > 0;
467  realNewSize <<= 1) ;
468  if ((realNewSize < 0) ||
469  (realNewSize >= INT_MAX / (int)sizeof(XRefEntry))) {
470  return 0;
471  }
472 
473  void *p = greallocn_checkoverflow(entries, realNewSize, sizeof(XRefEntry));
474  if (p == NULL) {
475  return 0;
476  }
477 
478  entries = (XRefEntry *) p;
479  capacity = realNewSize;
480 
481  }
482 
483  return capacity;
484 }

510 /* Read one xref table section. Also reads the associated trailer
511  * dictionary, and returns the prev pointer (if any).
512  * Arguments:
513  * pos Points to a Goffset containing the offset of the XRef
514  * section to be read. If a prev pointer is found, *pos is
515  * updated with its value
516  * followedXRefStm Used in case of nested readXRef calls to spot circular
517  * references in XRefStm pointers
518  * xrefStreamObjsNum If not NULL, every time a XRef stream is encountered,
519  * its object number is appended
520  * Return value:
521  * gTrue if a prev pointer is found, otherwise gFalse
522  */
523 GBool XRef::readXRef(Goffset *pos, std::vector<Goffset> *followedXRefStm, std::vector<int> *xrefStreamObjsNum) {
524  Parser *parser;
525  Object obj;
526  GBool more;
527 
528  // start up a parser, parse one token
529  obj.initNull();
530  parser = new Parser(NULL,
531  new Lexer(NULL,
532  str->makeSubStream(start + *pos, gFalse, 0, &obj)),
533  gTrue);
534  parser->getObj(&obj, gTrue);
535 
536  // parse an old-style xref table
537  if (obj.isCmd("xref")) {
538  obj.free();
539  more = readXRefTable(parser, pos, followedXRefStm, xrefStreamObjsNum);
540 
541  // parse an xref stream
542  } else if (obj.isInt()) {
543  const int objNum = obj.getInt();
544  obj.free();
545  if (!parser->getObj(&obj, gTrue)->isInt()) {
546  goto err1;
547  }
548  obj.free();
549  if (!parser->getObj(&obj, gTrue)->isCmd("obj")) {
550  goto err1;
551  }
552  obj.free();
553  if (!parser->getObj(&obj)->isStream()) {
554  goto err1;
555  }
556  if (trailerDict.isNone()) {
557  xRefStream = gTrue;
558  }
559  if (xrefStreamObjsNum) {
560  xrefStreamObjsNum->push_back(objNum);
561  }
562  more = readXRefStream(obj.getStream(), pos);
563  obj.free();
564 
565  } else {
566  goto err1;
567  }
568 
569  delete parser;
570  return more;
571 
572  err1:
573  obj.free();
574  delete parser;
575  ok = gFalse;
576  return gFalse;
577 }

579 GBool XRef::readXRefTable(Parser *parser, Goffset *pos, std::vector<Goffset> *followedXRefStm, std::vector<int> *xrefStreamObjsNum) {
580  XRefEntry entry;
581  GBool more;
582  Object obj, obj2;
583  Goffset pos2;
584  int first, n, i;
585 
586  while (1) {
587  parser->getObj(&obj, gTrue);
588  if (obj.isCmd("trailer")) {
589  obj.free();
590  break;
591  }
592  if (!obj.isInt()) {
593  goto err1;
594  }
595  first = obj.getInt();
596  obj.free();
597  if (!parser->getObj(&obj, gTrue)->isInt()) {
598  goto err1;
599  }
600  n = obj.getInt();
601  obj.free();
602  if (first < 0 || n < 0 || first + n < 0) {
603  goto err0;
604  }
605  if (first + n > size) {
606  if (resize(first + n) != first + n) {
607  error(errSyntaxError, -1, "Invalid 'obj' parameters'");
608  goto err0;
609  }
610  }
611  for (i = first; i < first + n; ++i) {
612  parser->getObj(&obj, gTrue);
613  if (obj.isInt()) {
614  entry.offset = obj.getInt();
615  } else if (obj.isInt64()) {
616  entry.offset = obj.getInt64();
617  } else {
618  goto err1;
619  }
620  obj.free();
621  if (!parser->getObj(&obj, gTrue)->isInt()) {
622  goto err1;
623  }
624  entry.gen = obj.getInt();
625  entry.obj.initNull ();
626  entry.flags = 0;
627  obj.free();
628  parser->getObj(&obj, gTrue);
629  if (obj.isCmd("n")) {
630  entry.type = xrefEntryUncompressed;
631  } else if (obj.isCmd("f")) {
632  entry.type = xrefEntryFree;
633  } else {
634  goto err1;
635  }
636  obj.free();
637  if (entries[i].offset == -1) {
638  entries[i] = entry;
639  // PDF files of patents from the IBM Intellectual Property
640  // Network have a bug: the xref table claims to start at 1
641  // instead of 0.
642  if (i == 1 && first == 1 &&
643  entries[1].offset == 0 && entries[1].gen == 65535 &&
644  entries[1].type == xrefEntryFree) {
645  i = first = 0;
646  entries[0] = entries[1];
647  entries[1].offset = -1;
648  }
649  }
650  }
651  }
652 
653  // read the trailer dictionary
654  if (!parser->getObj(&obj)->isDict()) {
655  goto err1;
656  }
657 
658  // get the 'Prev' pointer
659  obj.getDict()->lookupNF("Prev", &obj2);
660  if (obj2.isInt() || obj2.isInt64()) {
661  if (obj2.isInt())
662  pos2 = obj2.getInt();
663  else
664  pos2 = obj2.getInt64();
665  if (pos2 != *pos) {
666  *pos = pos2;
667  more = gTrue;
668  } else {
669  error(errSyntaxWarning, -1, "Infinite loop in xref table");
670  more = gFalse;
671  }
672  } else if (obj2.isRef()) {
673  // certain buggy PDF generators generate "/Prev NNN 0 R" instead
674  // of "/Prev NNN"
675  pos2 = (Guint)obj2.getRefNum();
676  if (pos2 != *pos) {
677  *pos = pos2;
678  more = gTrue;
679  } else {
680  error(errSyntaxWarning, -1, "Infinite loop in xref table");
681  more = gFalse;
682  }
683  } else {
684  more = gFalse;
685  }
686  obj2.free();
687 
688  // save the first trailer dictionary
689  if (trailerDict.isNone()) {
690  obj.copy(&trailerDict);
691  }
692 
693  // check for an 'XRefStm' key
694  obj.getDict()->lookup("XRefStm", &obj2);
695  if (obj2.isInt() || obj2.isInt64()) {
696  if (obj2.isInt())
697  pos2 = obj2.getInt();
698  else
699  pos2 = obj2.getInt64();
700  for (size_t i = 0; ok == gTrue && i < followedXRefStm->size(); ++i) {
701  if (followedXRefStm->at(i) == pos2) {
702  ok = gFalse;
703  }
704  }
705  if (ok) {
706  followedXRefStm->push_back(pos2);
707  readXRef(&pos2, followedXRefStm, xrefStreamObjsNum);
708  }
709  if (!ok) {
710  obj2.free();
711  goto err1;
712  }
713  }
714  obj2.free();
715 
716  obj.free();
717  return more;
718 
719  err1:
720  obj.free();
721  err0:
722  ok = gFalse;
723  return gFalse;
724 }

726 GBool XRef::readXRefStream(Stream *xrefStr, Goffset *pos) {
727  Dict *dict;
728  int w[3];
729  GBool more;
730  Object obj, obj2, idx;
731  int newSize, first, n, i;
732 
733  dict = xrefStr->getDict();
734 
735  if (!dict->lookupNF("Size", &obj)->isInt()) {
736  goto err1;
737  }
738  newSize = obj.getInt();
739  obj.free();
740  if (newSize < 0) {
741  goto err1;
742  }
743  if (newSize > size) {
744  if (resize(newSize) != newSize) {
745  error(errSyntaxError, -1, "Invalid 'size' parameter");
746  goto err0;
747  }
748  }
749 
750  if (!dict->lookupNF("W", &obj)->isArray() ||
751  obj.arrayGetLength() < 3) {
752  goto err1;
753  }
754  for (i = 0; i < 3; ++i) {
755  if (!obj.arrayGet(i, &obj2)->isInt()) {

758  }
759  w[i] = obj2.getInt();
760  obj2.free();

762  goto err1;
763  }
764  }

767  goto err1;
768  }

773  for (i = 0; i+1 < idx.arrayGetLength(); i += 2) {

775  idx.free();
776  goto err1;
777  }
778  first = obj.getInt();
779  obj.free();
780  if (!idx.arrayGet(i+1, &obj)->isInt()) {
781  idx.free();
782  goto err1;
783  }

787  !readXRefStreamSection(xrefStr, w, first, n)) {
788  idx.free();

792  } else {
793  if (!readXRefStreamSection(xrefStr, w, 0, newSize)) {
794  idx.free();
795  goto err0;
796  }
797  }
798  idx.free();

801  if (obj.isInt()) {
802  *pos = obj.getInt();
803  more = gTrue;
804  } else if (obj.isInt64()) {
805  *pos = obj.getInt64();
806  more = gTrue;
807  } else {
808  more = gFalse;
809  }
810  obj.free();
811  if (trailerDict.isNone()) {
812  trailerDict.initDict(dict);
813  }
814

816 
817  err1:
818  obj.free();
819  err0:
820  ok = gFalse;
821  return gFalse;

824 GBool XRef::readXRefStreamSection(Stream *xrefStr, int *w, int first, int n) {
825  unsigned long long offset;
826  int type, gen, c, i, j;
827 
828  if (first + n < 0) {
829  return gFalse;
830  }
831  if (first + n > size) {
832  if (resize(first + n) != size) {
833  error(errSyntaxError, -1, "Invalid 'size' inside xref table");
834  return gFalse;
835  }
836  if (first + n > size) {
837  error(errSyntaxError, -1, "Invalid 'first' or 'n' inside xref table");
838  return gFalse;
839  }
840  }
841  for (i = first; i < first + n; ++i) {
842  if (w[0] == 0) {
843  type = 1;
844  } else {
845  for (type = 0, j = 0; j < w[0]; ++j) {
846  if ((c = xrefStr->getChar()) == EOF) {
847  return gFalse;
848  }
849  type = (type << 8) + c;
850  }
851  }
852  for (offset = 0, j = 0; j < w[1]; ++j) {
853  if ((c = xrefStr->getChar()) == EOF) {
854  return gFalse;
855  }
856  offset = (offset << 8) + c;
857  }
858  if (offset > (unsigned long long)GoffsetMax()) {
859  error(errSyntaxError, -1, "Offset inside xref table too large for fseek");
860  return gFalse;
861  }
862  for (gen = 0, j = 0; j < w[2]; ++j) {
863  if ((c = xrefStr->getChar()) == EOF) {
864  return gFalse;
865  }
866  gen = (gen << 8) + c;
867  }
868  if (entries[i].offset == -1) {
869  switch (type) {
870  case 0:
871  entries[i].offset = offset;
872  entries[i].gen = gen;
873  entries[i].type = xrefEntryFree;
874  break;
875  case 1:
876  entries[i].offset = offset;
877  entries[i].gen = gen;
878  entries[i].type = xrefEntryUncompressed;
879  break;
880  case 2:
881  entries[i].offset = offset;
882  entries[i].gen = gen;
883  entries[i].type = xrefEntryCompressed;
884  break;
885  default:
886  return gFalse;
887  }
888  }
889  }
890 
891  return gTrue;
892 }

895 GBool XRef::constructXRef(GBool *wasReconstructed, GBool needCatalogDict) {
896  Parser *parser;
897  Object newTrailerDict, obj;
898  char buf[256];
899  Goffset pos;
900  int num, gen;
901  int newSize;
902  int streamEndsSize;
903  char *p;
904  GBool gotRoot;
905  char* token = NULL;
906  bool oneCycle = true;
907  int offset = 0;
908 
909  gfree(entries);
910  capacity = 0;
911  size = 0;
912  entries = NULL;
913 
914  gotRoot = gFalse;
915  streamEndsLen = streamEndsSize = 0;
916 
917  if (wasReconstructed)
918  {
919  *wasReconstructed = true;
920  }
921 
922  str->reset();
923  while (1) {
924  pos = str->getPos();
925  if (!str->getLine(buf, 256)) {
926  break;
927  }
928  p = buf;
929 
930  // skip whitespace
931  while (*p && Lexer::isSpace(*p & 0xff)) ++p;
932 
933  oneCycle = true;
934  offset = 0;
935 
936  while( ( token = strstr( p, "endobj" ) ) || oneCycle ) {
937  oneCycle = false;
938 
939  if( token ) {
940  oneCycle = true;
941  token[0] = '\0';
942  offset = token - p;
943  }
944 
945  // got trailer dictionary
946  if (!strncmp(p, "trailer", 7)) {
947  obj.initNull();
948  parser = new Parser(NULL,
949  new Lexer(NULL,
950  str->makeSubStream(pos + 7, gFalse, 0, &obj)),
951  gFalse);

953  if (newTrailerDict.isDict()) {
954  newTrailerDict.dictLookupNF("Root", &obj);
955  if (obj.isRef() && (!gotRoot || !needCatalogDict) && rootNum != obj.getRefNum()) {
956  rootNum = obj.getRefNum();
957  rootGen = obj.getRefGen();
958  if (!trailerDict.isNone()) {
959  trailerDict.free();
960  }
961  newTrailerDict.copy(&trailerDict);
962  gotRoot = gTrue;
963  }
964  obj.free();
965  }
966  newTrailerDict.free();
967  delete parser;
968 
969  // look for object
970  } else if (isdigit(*p & 0xff)) {
971  num = atoi(p);
972  if (num > 0) {
973  do {
974  ++p;
975  } while (*p && isdigit(*p & 0xff));
976  if ((*p & 0xff) == 0 || isspace(*p & 0xff)) {
977  if ((*p & 0xff) == 0) {
978  //new line, continue with next line!
979  str->getLine(buf, 256);
980  p = buf - 1;
981  }
982  do {
983  ++p;
984  } while (*p && isspace(*p & 0xff));
985  if (isdigit(*p & 0xff)) {
986  gen = atoi(p);
987  do {
988  ++p;
989  } while (*p && isdigit(*p & 0xff));
990  if ((*p & 0xff) == 0 || isspace(*p & 0xff)) {
991  if ((*p & 0xff) == 0) {
992  //new line, continue with next line!
993  str->getLine(buf, 256);
994  p = buf - 1;
995  }
996  do {
997  ++p;
998  } while (*p && isspace(*p & 0xff));
999  if (!strncmp(p, "obj", 3)) {
1000  if (num >= size) {
1001  newSize = (num + 1 + 255) & ~255;
1002  if (newSize < 0) {
1003  error(errSyntaxError, -1, "Bad object number");
1004  return gFalse;
1005  }
1006  if (resize(newSize) != newSize) {
1007  error(errSyntaxError, -1, "Invalid 'obj' parameters");
1008  return gFalse;
1009  }
1010  }
1011  if (entries[num].type == xrefEntryFree ||
1012  gen >= entries[num].gen) {
1013  entries[num].offset = pos - start;
1014  entries[num].gen = gen;
1015  entries[num].type = xrefEntryUncompressed;
1016  }
1017  }
1018  }
1019  }
1020  }
1021  }
1022 
1023  } else if (!strncmp(p, "endstream", 9)) {
1024  if (streamEndsLen == streamEndsSize) {
1025  streamEndsSize += 64;
1026  if (streamEndsSize >= INT_MAX / (int)sizeof(int)) {
1027  error(errSyntaxError, -1, "Invalid 'endstream' parameter.");
1028  return gFalse;
1029  }
1030  streamEnds = (Goffset *)greallocn(streamEnds,
1031  streamEndsSize, sizeof(Goffset));
1032  }
1033  streamEnds[streamEndsLen++] = pos;
1034  }
1035  if( token ) {
1036  p = token + 6;// strlen( "endobj" ) = 6
1037  pos += offset + 6;// strlen( "endobj" ) = 6
1038  while (*p && Lexer::isSpace(*p & 0xff)) {
1039  ++p;
1040  ++pos;
1041  }
1042  }
1043  }
1044  }
1045 
1046  if (gotRoot)
1047  return gTrue;
1048 
1049  error(errSyntaxError, -1, "Couldn't find trailer dictionary");
1050  return gFalse;
1051 }

1145 Object *XRef::fetch(int num, int gen, Object *obj, int recursion) {
1146  XRefEntry *e;
1147  Parser *parser;
1148  Object obj1, obj2, obj3;
1149 
1150  xrefLocker();
1151  // check for bogus ref - this can happen in corrupted PDF files
1152  if (num < 0 || num >= size) {
1153  goto err;
1154  }
1155 
1156  e = getEntry(num);
1157  if(!e->obj.isNull ()) { //check for updated object
1158  obj = e->obj.copy(obj);
1159  return obj;
1160  }
1161 
1162  switch (e->type) {
1163 
1164  case xrefEntryUncompressed:
1165  if (e->gen != gen) {
1166  goto err;
1167  }
1168  obj1.initNull();
1169  parser = new Parser(this,
1170  new Lexer(this,
1171  str->makeSubStream(start + e->offset, gFalse, 0, &obj1)),
1172  gTrue);
1173  parser->getObj(&obj1, recursion);
1174  parser->getObj(&obj2, recursion);
1175  parser->getObj(&obj3, recursion);
1176  if (!obj1.isInt() || obj1.getInt() != num ||
1177  !obj2.isInt() || obj2.getInt() != gen ||
1178  !obj3.isCmd("obj")) {
1179  // some buggy pdf have obj1234 for ints that represent 1234
1180  // try to recover here
1181  if (obj1.isInt() && obj1.getInt() == num &&
1182  obj2.isInt() && obj2.getInt() == gen &&
1183  obj3.isCmd()) {
1184  char *cmd = obj3.getCmd();
1185  if (strlen(cmd) > 3 &&
1186  cmd[0] == 'o' &&
1187  cmd[1] == 'b' &&
1188  cmd[2] == 'j') {
1189  char *end_ptr;
1190  long longNumber = strtol(cmd + 3, &end_ptr, 0);
1191  if (longNumber <= INT_MAX && longNumber >= INT_MIN && *end_ptr == '\0') {
1192  int number = longNumber;
1193  error(errSyntaxWarning, -1, "Cmd was not obj but {0:s}, assuming the creator meant obj {1:d}", cmd, number);
1194  obj->initInt(number);
1195  obj1.free();
1196  obj2.free();
1197  obj3.free();
1198  delete parser;
1199  break;
1200  }
1201  }
1202  }
1203  obj1.free();
1204  obj2.free();
1205  obj3.free();
1206  delete parser;
1207  goto err;
1208  }
1209  parser->getObj(obj, gFalse, (encrypted && !e->getFlag(XRefEntry::Unencrypted)) ? fileKey : NULL,
1210  encAlgorithm, keyLength, num, gen, recursion);
1211  obj1.free();
1212  obj2.free();
1213  obj3.free();
1214  delete parser;
1215  break;
1216 
1217  case xrefEntryCompressed:
1218  {
1219 #if 0 // Adobe apparently ignores the generation number on compressed objects
1220  if (gen != 0) {
1221  goto err;
1222  }
1223 #endif
1224  if (e->offset >= (Guint)size ||
1225  entries[e->offset].type != xrefEntryUncompressed) {
1226  error(errSyntaxError, -1, "Invalid object stream");
1227  goto err;
1228  }
1229 
1230  ObjectStream *objStr = NULL;
1231  ObjectStreamKey key(e->offset);
1232  PopplerCacheItem *item = objStrs->lookup(key);
1233  if (item) {
1234  ObjectStreamItem *it = static_cast<ObjectStreamItem *>(item);
1235  objStr = it->objStream;
1236  }
1237 
1238  if (!objStr) {
1239  objStr = new ObjectStream(this, e->offset, recursion + 1);
1240  if (!objStr->isOk()) {
1241  delete objStr;
1242  objStr = NULL;
1243  goto err;
1244  } else {
1245  // XRef could be reconstructed in constructor of ObjectStream:
1246  e = getEntry(num);
1247  ObjectStreamKey *newkey = new ObjectStreamKey(e->offset);
1248  ObjectStreamItem *newitem = new ObjectStreamItem(objStr);
1249  objStrs->put(newkey, newitem);
1250  }
1251  }
1252  objStr->getObject(e->gen, num, obj);
1253  }
1254  break;
1255 
1256  default:
1257  goto err;
1258  }
1259 
1260  return obj;
1261 
1262  err:
1263  if (!xRefStream && !xrefReconstructed) {
1264  error(errInternal, -1, "xref num {0:d} not found but needed, try to reconstruct\n", num);
1265  rootNum = -1;
1266  constructXRef(&xrefReconstructed);
1267  return fetch(num, gen, obj, ++recursion);
1268  }
1269  return obj->initNull();
1270 }
