70 void Voice::init()
71  {
72  // Initialize the coefficients for the interpolation. The math comes
73  // from a mail, posted by Olli Niemitalo to the music-dsp mailing
74  // list (I found it in the music-dsp archives
75  // http://www.smartelectronix.com/musicdsp/).
76 
77  double ff = 1.0 / 32768.0;
78  for (int i = 0; i < INTERP_MAX; i++) {
79  double x = (double) i / (double) INTERP_MAX;
80  interpCoeff[i][0] = (x * (-0.5 + x * (1 - 0.5 * x))) * ff;
81  interpCoeff[i][1] = (1.0 + x * x * (1.5 * x - 2.5)) * ff;
82  interpCoeff[i][2] = (x * (0.5 + x * (2.0 - 1.5 * x))) * ff;
83  interpCoeff[i][3] = (0.5 * x * x * (x - 1.0)) * ff;
84  }
85  static const float MIN_GAIN = -80.0;
86  static const float dbStep = MIN_GAIN / float(EG_SIZE);
87 
88  for (int i = 0; i < EG_SIZE; ++i) {
89  Envelope::egPow[EG_SIZE-i-1] = pow(10.0, (dbStep * i)/20.0);
90  Envelope::egLin[i] = 1.0 - (double(i) / double(EG_SIZE));
91  }
92 
93  }

99 void Voice::start(Channel* c, int key, int v, const Zone* zone, double durSinceNoteOn)
100  {
101  z = zone;
102  _state = VoiceState::ATTACK;
103  //_state = VoiceState::PLAYING;
104  _channel = c;
105  _key = key;
106  _velocity = v;
107  Sample* s = z->sample;
108  audioChan = s->channel();
109  data = s->data() + z->offset * audioChan;
110  eidx = s->frames() * audioChan;
111  _loopMode = z->loopMode;
112  _loopStart = z->loopStart;
113  _loopEnd = z->loopEnd;
114  _samplesSinceStart = 0;
115 
116  _offMode = z->offMode;
117  _offBy = z->offBy;
118 
119  trigger = z->trigger;
120 
121  float offset = -z->ampVeltrack;
122  if (offset <= 0)
123  offset += 100;
124  if (trigger == Trigger::CC)
125  _velocity = 127;
126  float curve = _velocity * _velocity / (127.0 * 127.0);
127 
128  double rt_decay_value = 1.0;
129  if (trigger == Trigger::RELEASE)
130  rt_decay_value = pow(10, (-z->rtDecay * durSinceNoteOn)/20);
131  // the .005 in this calculation is made up like this:
132  // -> (offset + z->ampVeltrack*curve) being a percent value so
133  // this should be divided by 100 or multiplied by 0.01
134  // -> afterwards 0.5 (-6dB) is applied to compensate possible coherent
135  // signals in a stereo output see http://www.sengpielaudio.com/calculator-coherentsources.htm
136  // -> 0.005 = 0.01 * 0.5
137  gain = z->volume * (offset + z->ampVeltrack * curve)
138  * .005 * c->gain() * rt_decay_value;
139 
140  phase.set(0);
141  float sr = float(s->sampleRate()) / _zerberus->sampleRate();
142  double targetcents = ((((key - z->keyBase) * z->pitchKeytrack) + z->keyBase) * 100.0) + z->tune;
143  if (trigger == Trigger::CC)
144  targetcents = z->keyBase * 100;
145  phaseIncr.set(_zerberus->ct2hz(targetcents) * sr/_zerberus->ct2hz(z->keyBase * 100.0));
146 
147  fres = 13500.0;
148  last_fres = -1.0;
149  qreal GEN_FILTERQ = 100.0; // 0 - 960
150  qreal q_db = GEN_FILTERQ / 10.0f - 3.01f;
151  q_lin = pow(10.0f, q_db / 20.0f);
152  filter_gain = 1.0 / sqrt(q_lin);
153 
154  hist1r = 0;
155  hist2r = 0;
156  hist1l = 0;
157  hist2l = 0;
158 
159  filter_startup = true;
160 
161  modenv_val = 0.0;
162  modlfo_val = 0.0;
163 
164  currentEnvelope = V1Envelopes::DELAY;
165 
166  float velPercent = _velocity / 127.0;
167 
168  envelopes[V1Envelopes::DELAY].setTable(Envelope::egLin);
169  envelopes[V1Envelopes::DELAY].setTime(z->ampegDelay + (z->ampegVel2Delay * velPercent), _zerberus->sampleRate());
170  envelopes[V1Envelopes::DELAY].setConstant(0.0);
171 
172  envelopes[V1Envelopes::ATTACK].setTable(Envelope::egLin);
173  envelopes[V1Envelopes::ATTACK].setVariable();
174  envelopes[V1Envelopes::ATTACK].setTime(z->ampegAttack + (z->ampegVel2Attack * velPercent), _zerberus->sampleRate());
175  envelopes[V1Envelopes::ATTACK].offset = z->ampegStart;
176 
177  envelopes[V1Envelopes::HOLD].setTable(Envelope::egLin);
178  envelopes[V1Envelopes::HOLD].setTime(z->ampegHold + (z->ampegVel2Hold * velPercent), _zerberus->sampleRate());
179  envelopes[V1Envelopes::HOLD].setConstant(1.0);
180 
181  envelopes[V1Envelopes::DECAY].setTable(Envelope::egPow);
182  envelopes[V1Envelopes::DECAY].setVariable();
183  envelopes[V1Envelopes::DECAY].setTime(z->ampegDecay + (z->ampegVel2Decay * velPercent), _zerberus->sampleRate());
184  envelopes[V1Envelopes::DECAY].offset = z->ampegSustain;
185 
186  envelopes[V1Envelopes::SUSTAIN].setTable(Envelope::egLin);
187  if (trigger == Trigger::RELEASE || trigger == Trigger::CC) {
188  // Sample is played on noteoff. We need to stop the voice when it's done. Set the sustain duration accordingly.
189  double sampleDur = ((z->sample->frames()/z->sample->channel()) / z->sample->sampleRate()) * 1000; // in ms
190  double scaledSampleDur = sampleDur / (phaseIncr.data / 256.0);
191  double sustainDur = scaledSampleDur - (z->ampegDelay + z->ampegAttack + z->ampegHold + z->ampegDecay + z->ampegRelease);
192  envelopes[V1Envelopes::SUSTAIN].setTime(sustainDur, _zerberus->sampleRate());
193  }
194  else
195  envelopes[V1Envelopes::SUSTAIN].setTime(std::numeric_limits<float>::infinity(), _zerberus->sampleRate());
196  envelopes[V1Envelopes::SUSTAIN].setConstant(qBound(0.0f, z->ampegSustain + (z->ampegVel2Sustain * velPercent), 1.0f));
197 
198  envelopes[V1Envelopes::RELEASE].setTable(Envelope::egPow);
199  envelopes[V1Envelopes::RELEASE].setVariable();
200  envelopes[V1Envelopes::RELEASE].setTime(z->ampegRelease + (z->ampegVel2Release * velPercent), _zerberus->sampleRate());
201  envelopes[V1Envelopes::RELEASE].max = envelopes[V1Envelopes::SUSTAIN].val;
202 
203  _looping = false;
204  }

210 void Voice::updateFilter(float _fres)
211  {
212  // The filter coefficients have to be recalculated (filter
213  // parameters have changed). Recalculation for various reasons is
214  // forced by setting last_fres to -1. The flag filter_startup
215  // indicates, that the DSP loop runs for the first time, in this
216  // case, the filter is set directly, instead of smoothly fading
217  // between old and new settings.
218  //
219  // Those equations from Robert Bristow-Johnson's `Cookbook
220  // formulae for audio EQ biquad filter coefficients', obtained
221  // from Harmony-central.com / Computer / Programming. They are
222  // the result of the bilinear transform on an analogue filter
223  // prototype. To quote, `BLT frequency warping has been taken
224  // into account for both significant frequency relocation and for
225  // bandwidth readjustment'.
226 
227  float sr = _zerberus->sampleRate();
228  float omega = (float) 2.0f * M_PI * (_fres / sr);
229  float sin_coeff = sin(omega);
230  float cos_coeff = cos(omega);
231  float alpha_coeff = sin_coeff / (2.0f * q_lin);
232  float a0_inv = 1.0f / (1.0f + alpha_coeff);
233 
234  /* Calculate the filter coefficients. All coefficients are
235  * normalized by a0. Think of `a1' as `a1/a0'.
236  *
237  * Here a couple of multiplications are saved by reusing common expressions.
238  * The original equations should be:
239  * b0=(1.-cos_coeff)*a0_inv*0.5*voice->filter_gain;
240  * b1=(1.-cos_coeff)*a0_inv*voice->filter_gain;
241  * b2=(1.-cos_coeff)*a0_inv*0.5*voice->filter_gain; */
242 
243  float a1_temp = -2.0f * cos_coeff * a0_inv;
244  float a2_temp = (1.0f - alpha_coeff) * a0_inv;
245  float b1_temp = (1.0f - cos_coeff) * a0_inv * filter_gain;
246  // both b0 -and- b2
247  float b02_temp = b1_temp * 0.5f;
248 
249  if (filter_startup) {
250  /* The filter is calculated, because the voice was started up.
251  * In this case set the filter coefficients without delay.
252  */
253  a1 = a1_temp;
254  a2 = a2_temp;
255  b02 = b02_temp;
256  b1 = b1_temp;
257  filter_coeff_incr_count = 0;
258  filter_startup = false;
259  }
260  else {
261  /* The filter frequency is changed. Calculate an increment
262  * factor, so that the new setting is reached after some time.
263  */
264 
265  static const int FILTER_TRANSITION_SAMPLES = 64;
266 
267  a1_incr = (a1_temp - a1) / FILTER_TRANSITION_SAMPLES;
268  a2_incr = (a2_temp - a2) / FILTER_TRANSITION_SAMPLES;
269  b02_incr = (b02_temp - b02) / FILTER_TRANSITION_SAMPLES;
270  b1_incr = (b1_temp - b1) / FILTER_TRANSITION_SAMPLES;
271  /* Have to add the increments filter_coeff_incr_count times. */
272  filter_coeff_incr_count = FILTER_TRANSITION_SAMPLES;
273  }
274  }

280 void Voice::updateEnvelopes() {
281  if (_state == VoiceState::ATTACK && trigger != Trigger::RELEASE) {
282  while (envelopes[currentEnvelope].step() && currentEnvelope != V1Envelopes::SUSTAIN)
283  currentEnvelope++;
284 
285  // triggered by noteon enter virtually infinite sustain (play state)
286  if (currentEnvelope == V1Envelopes::SUSTAIN)
287  _state = VoiceState::PLAYING;
288  }
289  else if (_state == VoiceState::ATTACK && trigger == Trigger::RELEASE) {
290  while (envelopes[currentEnvelope].step() && currentEnvelope != V1Envelopes::RELEASE)
291  currentEnvelope++;
292 
293  // triggered by noteoff stop sample when entering release
294  if (currentEnvelope == V1Envelopes::RELEASE)
295  _state = VoiceState::STOP;
296  }
297  else if (_state == VoiceState::STOP) {
298  if (envelopes[V1Envelopes::RELEASE].step()) {
299  off();
300  }
301  }
302  }

308 void Voice::process(int frames, float* p)
309  {
310  float modlfo_to_fc = 0.0;
311  float modenv_to_fc = 0.0;
312 
313  float _fres = _zerberus->ct2hz(fres
314  + modlfo_val * modlfo_to_fc
315  + modenv_val * modenv_to_fc);
316 
317  int sr = _zerberus->sampleRate();
318  if (_fres > 0.45f * sr)
319  _fres = 0.45f * sr;
320  else if (_fres < 5.f)
321  _fres = 5.f;
322 
323  if ((fabs(_fres - last_fres) > 0.01f)) {
324  updateFilter(_fres);
325  last_fres = _fres;
326  }
327 
328  if (audioChan == 1) {
329  while (frames--) {
330 
331  updateLoop();
332 
333  int idx = phase.index();
334 
335  if (idx >= eidx) {
336  off();
337  break;
338  }
339  const float* coeffs = interpCoeff[phase.fract()];
340  float f;
341  f = (coeffs[0] * getData(idx-1)
342  + coeffs[1] * getData(idx+0)
343  + coeffs[2] * getData(idx+1)
344  + coeffs[3] * getData(idx+2)) * gain
345  - a1 * hist1l
346  - a2 * hist2l;
347  float v = b02 * (f + hist2l) + b1 * hist1l;
348  hist2l = hist1l;
349  hist1l = f;
350 
351  if (filter_coeff_incr_count) {
352  --filter_coeff_incr_count;
353  a1 += a1_incr;
354  a2 += a2_incr;
355  b02 += b02_incr;
356  b1 += b1_incr;
357  }
358 
359  updateEnvelopes();
360  if (_state == VoiceState::OFF)
361  break;
362  v *= envelopes[currentEnvelope].val * z->ccGain;
363 
364  *p++ += v * _channel->panLeftGain();
365  *p++ += v * _channel->panRightGain();
366  phase += phaseIncr;
367  _samplesSinceStart++;
368  }
369  }
370  else {
371  //
372  // handle interleaved stereo samples
373  //
374  while (frames--) {
375 
376  updateLoop();
377 
378  int idx = phase.index() * 2;
379  if (idx >= eidx) {
380  off();
381  // printf("end of sample\n");
382  break;
383  }
384 
385  const float* coeffs = interpCoeff[phase.fract()];
386  float f1, f2;
387 
388  f1 = (coeffs[0] * getData(idx-2)
389  + coeffs[1] * getData(idx)
390  + coeffs[2] * getData(idx+2)
391  + coeffs[3] * getData(idx+4))
392  * gain * _channel->panLeftGain() * z->ccGain;
393 
394  f2 = (coeffs[0] * getData(idx-1)
395  + coeffs[1] * getData(idx+1)
396  + coeffs[2] * getData(idx+3)
397  + coeffs[3] * getData(idx+5))
398  * gain * _channel->panRightGain() * z->ccGain;
399 
400  updateEnvelopes();
401  if (_state == VoiceState::OFF)
402  break;
403 
404  f1 *= envelopes[currentEnvelope].val;
405  f2 *= envelopes[currentEnvelope].val;
406 
407  f1 += -a1 * hist1l - a2 * hist2l;
408  float vl = b02 * (f1 + hist2l) + b1 * hist1l;
409  hist2l = hist1l;
410  hist1l = f1;
411 
412  f2 += -a1 * hist1r - a2 * hist2r;
413  float vr = b02 * (f2 + hist2r) + b1 * hist1r;
414  hist2r = hist1r;
415  hist1r = f2;
416 
417  if (filter_coeff_incr_count) {
418  --filter_coeff_incr_count;
419  a1 += a1_incr;
420  a2 += a2_incr;
421  b02 += b02_incr;
422  b1 += b1_incr;
423  }
424 
425  *p++ += vl;
426  *p++ += vr;
427  phase += phaseIncr;
428  _samplesSinceStart++;
429  }
430  }
431  }

437 void Voice::updateLoop()
438  {
439  int idx = phase.index();
440  int loopOffset = (audioChan * 3) - 1; // offset due to interpolation
441  bool validLoop = _loopEnd > 0 && _loopStart >= 0 && (_loopEnd <= (eidx/audioChan));
442  bool shallLoop = loopMode() == LoopMode::CONTINUOUS || (loopMode() == LoopMode::SUSTAIN && (_state < VoiceState::STOP));
443 
444  if (!(validLoop && shallLoop)) {
445  _looping = false;
446  return;
447  }
448 
449  if (idx + loopOffset > _loopEnd)
450  _looping = true;
451  if (idx > _loopEnd)
452  phase.setIndex(_loopStart+(idx-_loopEnd-1));
453  }

455 short Voice::getData(int pos) {
456  if (pos < 0 && !_looping)
457  return 0;
458 
459  if (!_looping)
460  return data[pos];
461 
462  int loopEnd = _loopEnd * audioChan;
463  int loopStart = _loopStart * audioChan;
464 
465  if (pos < loopStart)
466  return data[loopEnd + (pos - loopStart) + audioChan];
467  else if (pos > (loopEnd + audioChan - 1))
468  return data[loopStart + (pos - loopEnd) - audioChan];
469  else
470  return data[pos];
471  }
