100 bool DurationElement::readProperties(XmlReader& e)
101  {
102  if (e.name() == "Tuplet") {
103  int i = e.readInt();
104  Tuplet* t = e.findTuplet(i);
105  if (!t) {
106  qDebug("DurationElement:read(): Tuplet id %d not found", i);
107  t = score()->searchTuplet(e, i);
108  if (t) {
109  qDebug(" ...found outside measure, input file corrupted?");
110  e.addTuplet(t);
111  }
112  }
113  if (t) {
114  setTuplet(t);
115  if (!score()->undoStack()->active()) // HACK, also added in Undo::AddElement()
116  t->add(this);
117  }
118  return true;
119  }
120  if (Element::readProperties(e))
121  return true;
122  return false;
123  }
