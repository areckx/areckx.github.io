55 bool ComboBoxData::eventFilter( QObject* object, QEvent* event ) {
56 
57  // make sure engine is enabled
58  if ( !( enabled() && object == target_.data() ) ) {
59  return TransitionData::eventFilter( object, event );
60  }
61 
62  // make sure that target is not editable
63  if ( target_.data()->isEditable() ) {
64  return TransitionData::eventFilter( object, event );
65  }
66 
67  switch ( event->type() ) {
68 
69  case QEvent::Show:
70  case QEvent::Resize:
71  case QEvent::Move:
72  if ( !recursiveCheck() && target_.data()->isVisible() ) {
73  timer_.start( 0, this );
74  }
75  break;
76 
77  default:
78  break;
79  }
80 
81  return TransitionData::eventFilter( object, event );
82 
83  }

102 bool ComboBoxData::initializeAnimation( void ) {
103  if ( !( enabled() && target_ && target_.data()->isVisible() ) ) return false;
104  if ( target_.data()->isEditable() ) {
105  /*
106  do nothing for editable comboboxes because
107  lineEditor animations are handled directly
108  */
109  return false;
110  }
111 
112  transition().data()->setOpacity(0);
113  transition().data()->setGeometry( targetRect() );
114  transition().data()->setStartPixmap( transition().data()->currentPixmap() );
115  transition().data()->show();
116  transition().data()->raise();
117  return true;
118  }
