95 QString readChordName(const MidiEvent &e)
96  {
97  if (e.type() != ME_META || e.metaType() != META_SPECIFIC)
98  return "";
99  if (e.len() < 4)
100  return "";
101 
102  const uchar *data = e.edata();
103  if (data[0] != 0x43 || data[1] != 0x7B || data[2] != 0x01)
104  return "";
105 
106  QString chordName;
107 
108  if (e.len() >= 4)
109  chordName += readChordRoot(data[3]);
110  if (e.len() >= 5)
111  chordName += readChordType(data[4]);
112 
113  if (e.len() >= 6) {
114  const QString chordRoot = readChordRoot(data[5]);
115  if (!chordRoot.isEmpty()) {
116  QString chordType;
117  if (e.len() >= 7)
118  chordType = readChordType(data[6]);
119  if (chordRoot + chordType == chordName)
120  chordName += "/" + chordRoot;
121  else
122  chordName += "/" + chordRoot + chordType;
123  }
124  }
125 
126  return chordName;
127  }

129 QString findChordName(
130  const QList<MidiNote> &notes,
131  const std::multimap<ReducedFraction, QString> &chordNames)
132  {
133  for (const MidiNote &note: notes) {
134  const auto range = chordNames.equal_range(note.origOnTime);
135  if (range.second == range.first)
136  continue;
137  // found chord names (usually only one)
138  QString chordName;
139  for (auto it = range.first; it != range.second; ++it) {
140  if (it != range.first)
141  chordName += "\n" + it->second;
142  else
143  chordName += it->second;
144  }
145  return chordName;
146  }
147  return "";

175  // before quantization, so we look for original on times

178 
179  for (MTrack &track: tracks) {
180  for (const auto &chord: track.chords) {
181  if (usedTimes.find(chord.first) != usedTimes.end())
182  continue;
183

186 
187  if (chordName.isEmpty())
188  continue;
189

191  Score *score = staff->score();
192  const ReducedFraction &onTime = chord.first;
193  usedTimes.insert(onTime);
194 
195  Measure* measure = score->tick2measure(onTime.ticks());
196  Segment* seg = measure->getSegment(SegmentType::ChordRest,
197  onTime.ticks());

199 
200  Harmony* h = new Harmony(score);
201  h->setHarmony(chordName);
