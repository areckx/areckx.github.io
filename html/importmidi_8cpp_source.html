91 bool doNotesOverlap(const MTrack &track)
92  {
93  const auto &chords = track.chords;
94  for (auto i1 = chords.begin(); i1 != chords.end(); ++i1) {
95  const auto &chord1 = i1->second;
96  for (auto noteIt1 = chord1.notes.begin();
97  noteIt1 != chord1.notes.end(); ++noteIt1) {
98  for (auto noteIt2 = std::next(noteIt1);
99  noteIt2 != chord1.notes.end(); ++noteIt2) {
100  if (noteIt2->pitch == noteIt1->pitch)
101  return true;
102  }
103  for (auto i2 = std::next(i1); i2 != chords.end(); ++i2) {
104  if (i2->first >= noteIt1->offTime)
105  break;
106  const auto &chord2 = i2->second;
107  if (chord1.voice != chord2.voice)
108  continue;
109  for (const auto &note2: chord2.notes) {
110  if (note2.pitch == noteIt1->pitch)
111  return true;
112  }
113  }
114  }
115  }
116  return false;
117  }

122  for (const auto &track: tracks)

124  return result;

168 void findAllTupletsForDrums(
169  MTrack &mtrack,
170  TimeSigMap *sigmap,
171  const ReducedFraction &basicQuant)
172  {
173  const size_t drumVoiceCount = 2;
174  // drum track has 2 voices (stem up and stem down),
175  // and tuplet detection is applicable for single voice drum tracks,
176  // so split track chords into 2 voice groups,
177  // detect tuplets and merge chords (and found tuplets) back;
178  // it's a small hack due to the fact that tuplet detection
179  // is designed to work before voice setting
180 
181  std::vector<std::multimap<ReducedFraction, MidiChord> > chords(drumVoiceCount);
182  for (const auto &chord: mtrack.chords) {
183  const MidiChord &c = chord.second;
184 
185  Q_ASSERT(c.voice == 0 || c.voice == 1);
186 
187  chords[c.voice].insert({chord.first, c});
188  }
189 
190  std::vector<std::multimap<ReducedFraction,
191  MidiTuplet::TupletData> > tuplets(drumVoiceCount);
192  for (size_t voice = 0; voice < drumVoiceCount; ++voice) {
193  if (!chords[voice].empty())
194  MidiTuplet::findAllTuplets(tuplets[voice], chords[voice], sigmap, basicQuant);
195  }
196  mtrack.chords.clear();
197  for (size_t voice = 0; voice < drumVoiceCount; ++voice) {
198  for (auto &chord: chords[voice]) {
199  // correct voice because it can be changed during tuplet detection
200  setChordVoice(chord.second, voice);
201  mtrack.chords.insert({chord.first, chord.second});
202  }
203  }
204  mtrack.updateTupletsFromChords();
205  // note: temporary local tuplets and chords are deleted here
206  }

208 void quantizeAllTracks(std::multimap<int, MTrack> &tracks,
209  TimeSigMap *sigmap,
210  const ReducedFraction &lastTick)
211  {
212  auto &opers = preferences.midiImportOperations;
213 
214  for (auto &track: tracks) {
215  MTrack &mtrack = track.second;
216  if (mtrack.chords.empty())
217  continue;
218  // pass current track index through MidiImportOperations
219  // for further usage
220  MidiOperations::CurrentTrackSetter setCurrentTrack{opers, mtrack.indexOfOperation};
221 
222  if (opers.data()->processingsOfOpenedFile == 0) {
223  opers.data()->trackOpers.isDrumTrack.setValue(
224  opers.currentTrack(), mtrack.mtrack->drumTrack());
225  if (mtrack.mtrack->drumTrack()) {
226  opers.data()->trackOpers.maxVoiceCount.setValue(
227  opers.currentTrack(), MidiOperations::VoiceCount::V_1);
228  }
229  }
230  const auto basicQuant = Quantize::quantValueToFraction(
231  opers.data()->trackOpers.quantValue.value(mtrack.indexOfOperation));
232 
233  Q_ASSERT_X(MChord::isLastTickValid(lastTick, mtrack.chords),
234  "quantizeAllTracks", "Last tick is less than max note off time");
235 
236  MChord::setBarIndexes(mtrack.chords, basicQuant, lastTick, sigmap);
237 
238  if (mtrack.mtrack->drumTrack())
239  findAllTupletsForDrums(mtrack, sigmap, basicQuant);
240  else
241  MidiTuplet::findAllTuplets(mtrack.tuplets, mtrack.chords, sigmap, basicQuant);
242 
243  Q_ASSERT_X(!doNotesOverlap(track.second),
244  "quantizeAllTracks",
245  "There are overlapping notes of the same voice that is incorrect");
246 
247  // (4/3 of the smallest duration) tol is less sensitive
248  // to on time inaccuracies than 1/2 earlier
249  MChord::collectChords(mtrack, {2, 1}, {4, 3});
250  Quantize::quantizeChords(mtrack.chords, sigmap, basicQuant);
251  MidiTuplet::removeEmptyTuplets(mtrack);
252 
253  Q_ASSERT_X(MidiTuplet::areTupletRangesOk(mtrack.chords, mtrack.tuplets),
254  "quantizeAllTracks", "Tuplet chord/note is outside tuplet "
255  "or non-tuplet chord/note is inside tuplet");
256  }

263 void MTrack::processMeta(int tick, const MidiEvent& mm)
264  {
265  if (!staff) {
266  qDebug("processMeta: no staff");
267  return;
268  }
269  const uchar* data = mm.edata();
270  Score* cs = staff->score();
271 
272  switch (mm.metaType()) {
273  case META_TEXT:
274  case META_LYRIC:
275  break; // lyric and text are added in importmidi_lyrics.cpp
276  case META_TRACK_NAME:
277  {
278  const std::string text = MidiCharset::fromUchar(data);
279 
280  auto &opers = preferences.midiImportOperations;
281  if (opers.data()->processingsOfOpenedFile == 0) {
282  const int currentTrack = indexOfOperation;
283  opers.data()->trackOpers.staffName.setValue(currentTrack, text);
284  }
285 
286  if (name.isEmpty())
287  name = MidiCharset::convertToCharset(text);
288  }
289  break;
290  case META_TEMPO: // add later, after adding of notes
291  break;
292  case META_KEY_SIGNATURE:
293  {
294  const signed char key = ((const signed char*)data)[0];
295  if (key < -7 || key > 7) {
296  qDebug("ImportMidi: illegal key %d", key);
297  break;
298  }
299  KeySigEvent ke;
300  ke.setKey(Key(key));
301  staff->setKey(tick, ke);
302  hasKey = true;
303  }
304  break;
305  case META_COMPOSER: // mscore extension
306  case META_POET:
307  case META_TRANSLATOR:
308  case META_SUBTITLE:
309  case META_TITLE:
310  {
311  Text* text = new Text(cs);
312  switch(mm.metaType()) {
313  case META_COMPOSER:
314  text->setSubStyle(SubStyle::COMPOSER);
315  break;
316  case META_TRANSLATOR:
317  text->setSubStyle(SubStyle::TRANSLATOR);
318  break;
319  case META_POET:
320  text->setSubStyle(SubStyle::POET);
321  break;
322  case META_SUBTITLE:
323  text->setSubStyle(SubStyle::SUBTITLE);
324  break;
325  case META_TITLE:
326  text->setSubStyle(SubStyle::TITLE);
327  break;
328  }
329 
330  text->setPlainText((const char*)(mm.edata()));
331 
332  MeasureBase* measure = cs->first();
333  if (!measure->isVBox()) {
334  measure = new VBox(cs);
335  measure->setTick(0);
336  measure->setNext(cs->first());
337  cs->measures()->add(measure);
338  }
339  measure->add(text);
340  }
341  break;
342  case META_COPYRIGHT:
343  cs->setMetaTag("copyright", QString((const char*)(mm.edata())));
344  break;
345  case META_TIME_SIGNATURE:
346  break; // added earlier
347  case META_PORT_CHANGE:
348  staff->part()->setMidiChannel(-1, (int)data[0]);
349  break;
350  default:
351  if (MScore::debugMode)
352  qDebug("unknown meta type 0x%02x", mm.metaType());
353  break;
354  }
355  }

357 QList<std::pair<ReducedFraction, TDuration> >
358 MTrack::toDurationList(const Measure *measure,
359  int voice,
360  const ReducedFraction &startTick,
361  const ReducedFraction &len,
362  Meter::DurationType durationType)
363  {
364  // find tuplets over which duration goes
365  auto barTick = ReducedFraction::fromTicks(measure->tick());
366  auto tupletsData = MidiTuplet::findTupletsInBarForDuration(
367  voice, barTick, startTick, len, tuplets);
368  struct {
369  bool operator()(const MidiTuplet::TupletData &d1,
370  const MidiTuplet::TupletData &d2)
371  {
372  return (d1.len > d2.len);
373  }
374  } comparator;
375  // sort by tuplet length in desc order
376  sort(tupletsData.begin(), tupletsData.end(), comparator);
377 
378  const ReducedFraction startTickInBar = startTick - barTick;
379  const ReducedFraction endTickInBar = startTickInBar + len;
380 
381  const auto &opers = preferences.midiImportOperations;
382  const bool useDots = opers.data()->trackOpers.useDots.value(indexOfOperation);
383  return Meter::toDurationList(startTickInBar, endTickInBar,
384  ReducedFraction(measure->timesig()), tupletsData,
385  durationType, useDots);
386  }

388 ReducedFraction splitDurationOnBarBoundary(const ReducedFraction &len,
389  const ReducedFraction &onTime,
390  const Measure* measure)
391  {
392  const ReducedFraction barLimit = ReducedFraction::fromTicks(measure->tick() + measure->ticks());
393  if (onTime + len > barLimit)
394  return barLimit - onTime;
395  return len;
396  }

400 void MTrack::fillGapWithRests(Score* score,
401  int voice,
402  const ReducedFraction &startChordTickFrac,
403  const ReducedFraction &restLength,
404  int track)
405  {
406  ReducedFraction startChordTick = startChordTickFrac;
407  ReducedFraction restLen = restLength;
408  while (restLen > ReducedFraction(0, 1)) {
409  ReducedFraction len = restLen;
410  Measure* measure = score->tick2measure(startChordTick.ticks());
411  if (startChordTick >= ReducedFraction::fromTicks(measure->tick() + measure->ticks())) {
412  qDebug("tick2measure: %d end of score?", startChordTick.ticks());
413  startChordTick += restLen;
414  restLen = ReducedFraction(0, 1);
415  break;
416  }
417  len = splitDurationOnBarBoundary(len, startChordTick, measure);
418 
419  if (len >= ReducedFraction::fromTicks(measure->ticks())) {
420  // rest to the whole measure
421  len = ReducedFraction::fromTicks(measure->ticks());
422  if (voice == 0) {
423  TDuration duration(TDuration::DurationType::V_MEASURE);
424  Rest* rest = new Rest(score, duration);
425  rest->setDuration(measure->len());
426  rest->setTrack(track);
427  Segment* s = measure->getSegment(SegmentType::ChordRest, startChordTick.ticks());
428  s->add(rest);
429  }
430  restLen -= len;
431  startChordTick += len;
432  }
433  else {
434  const auto dl = toDurationList(measure, voice, startChordTick, len,
435  Meter::DurationType::REST);
436  if (dl.isEmpty()) {
437  qDebug("cannot create duration list for len %d", len.ticks());
438  restLen = ReducedFraction(0, 1); // fake
439  break;
440  }
441  for (const auto &durationPair: dl) {
442  const TDuration &duration = durationPair.second;
443  const ReducedFraction &tupletRatio = durationPair.first;
444  len = ReducedFraction(duration.fraction()) / tupletRatio;
445  Rest* rest = new Rest(score, duration);
446  rest->setDuration(duration.fraction());
447  rest->setTrack(track);
448  Segment* s = measure->getSegment(SegmentType::ChordRest,
449  startChordTick.ticks());
450  s->add(rest);
451  MidiTuplet::addElementToTuplet(voice, startChordTick, len, rest, tuplets);
452  restLen -= len;
453  startChordTick += len;
454  }
455  }
456 
457  }
458  }

460 void setMusicNotesFromMidi(Score *score,
461  const QList<MidiNote> &midiNotes,
462  Chord *chord,
463  const Drumset *drumset,
464  bool useDrumset)
465  {
466  for (int i = 0; i < midiNotes.size(); ++i) {
467  const MidiNote& mn = midiNotes[i];
468  Note* note = new Note(score);
469  note->setTrack(chord->track());
470 
471  NoteVal nval(mn.pitch);
472  note->setNval(nval, chord->tick());
473  // TODO - does this need to be key-aware?
474  //note->setPitch(mn.pitch);
475  //note->setTpcFromPitch();
476 
477  chord->add(note);
478  note->setVeloType(Note::ValueType::USER_VAL);
479  note->setVeloOffset(mn.velo);
480 
481  if (useDrumset) {
482  if (!drumset->isValid(mn.pitch))
483  qDebug("unmapped drum note 0x%02x %d", mn.pitch, mn.pitch);
484  else {
485  Direction sd = drumset->stemDirection(mn.pitch);
486  chord->setStemDirection(sd);
487  }
488  }
489 
490  if (midiNotes[i].tie) {
491  midiNotes[i].tie->setEndNote(note);
492  midiNotes[i].tie->setTrack(note->track());
493  note->setTieBack(midiNotes[i].tie);
494  }
495  }
496  }

514 
515 void MTrack::processPendingNotes(QList<MidiChord> &midiChords,
516  int voice,
517  const ReducedFraction &startChordTickFrac,
518  const ReducedFraction &nextChordTick)
519  {
520  Score* score = staff->score();
521  const int track = staff->idx() * VOICES + voice;
522  const Drumset* drumset = staff->part()->instrument()->drumset();
523  const bool useDrumset = staff->part()->instrument()->useDrumset();
524 
525  const auto& opers = preferences.midiImportOperations.data()->trackOpers;
526  const int currentTrack = indexOfOperation;
527 
528  // all midiChords here should have the same onTime value
529  // and all notes in each midiChord should have the same duration
530  ReducedFraction startChordTick = startChordTickFrac;
531  while (!midiChords.isEmpty()) {
532  const ReducedFraction tick = startChordTick;
533  ReducedFraction len = nextChordTick - tick;
534  if (len <= ReducedFraction(0, 1))
535  break;
536  len = MChord::findMinDuration(tick, midiChords, len);
537  Measure* measure = score->tick2measure(tick.ticks());
538  len = splitDurationOnBarBoundary(len, tick, measure);
539 
540  const auto dl = toDurationList(measure, voice, tick, len, Meter::DurationType::NOTE);
541  if (dl.isEmpty())
542  break;
543  const TDuration &d = dl[0].second;
544  const ReducedFraction &tupletRatio = dl[0].first;
545  len = ReducedFraction(d.fraction()) / tupletRatio;
546 
547  Chord* chord = new Chord(score);
548  chord->setTrack(track);
549  chord->setDurationType(d);
550  chord->setDuration(d.fraction());
551 
552  if (opers.showStaccato.value(currentTrack)
553  && startChordTick == startChordTickFrac // first chord in tied chord sequence
554  && midiChords.begin()->isStaccato()) {
555  Articulation* a = new Articulation(chord->score());
556  a->setSymId(SymId::articStaccatoAbove);
557  chord->add(a);
558  }
559 
560  Segment* s = measure->getSegment(SegmentType::ChordRest, tick.ticks());
561  s->add(chord);
562  MidiTuplet::addElementToTuplet(voice, tick, len, chord, tuplets);
563 
564  for (int k = 0; k < midiChords.size(); ++k) {
565  MidiChord& midiChord = midiChords[k];
566  setMusicNotesFromMidi(score, midiChord.notes, chord, drumset, useDrumset);
567  if (!midiChord.notes.empty() && midiChord.notes.first().offTime - tick <= len) {
568  midiChords.removeAt(k);
569  --k;
570  continue;
571  }
572  setTies(chord, score, midiChord.notes);
573  }
574  startChordTick += len;
575  }
576  fillGapWithRests(score, voice, startChordTick, nextChordTick - startChordTick, track);
577  }

600 void MTrack::createNotes(const ReducedFraction &lastTick)
601  {
602  for (int voice = 0; voice < VOICES; ++voice) {
603  // startChordTick is onTime value of all simultaneous notes
604  // chords here are consist of notes with equal durations
605  // several chords may have the same onTime value
606  ReducedFraction startChordTick;
607  QList<MidiChord> midiChords;
608 
609  for (auto it = chords.begin(); it != chords.end();) {
610  const auto &nextChordTick = it->first;
611  const MidiChord& midiChord = it->second;
612  if (midiChord.voice != voice) {
613  ++it;
614  continue;
615  }
616  processPendingNotes(midiChords, voice, startChordTick, nextChordTick);
617  // now 'midiChords' list is empty
618  // so - fill it:
619  // collect all midiChords on current tick position
620  startChordTick = nextChordTick; // debug
621  for ( ; it != chords.end(); ++it) {
622  const MidiChord& midiChord = it->second;
623  if (it->first != startChordTick)
624  break;
625  if (midiChord.voice != voice)
626  continue;
627  midiChords.append(midiChord);
628  }
629  if (midiChords.isEmpty())
630  break;

774  const ReducedFraction &firstTick,
775  Score *score,
776  int *begBarIndex,
777  int *barCount)
778  {

780  const int secondBarTick = score->sigmap()->bar2tick(1, 0);
781  const Fraction firstTimeSig = score->sigmap()->timesig(firstBarTick).timesig();

788  pickup->setIrregular(true);
789  pickup->setTimesig(secondTimeSig); // nominal time signature
790  pickup->setLen(firstTimeSig); // actual length
791  score->measures()->add(pickup);
792  *begBarIndex = 1;
793  }
794  else if (isPickupWithGreaterTimeSig(firstTimeSig, secondTimeSig, firstTick)) {
795  // split measure into 2 equal measures
796  // but for simplicity don't treat first bar as a pickup measure:
797  // leave its actual length equal to nominal length
798  ++(*barCount);

801 
802  Measure* firstBar = new Measure(score);
803  firstBar->setTick(firstBarTick);
804  firstBar->setNo(0);
805  firstBar->setTimesig(secondTimeSig);
806  firstBar->setLen(secondTimeSig);
807  score->measures()->add(firstBar);
808 
809  Measure* secondBar = new Measure(score);
810  secondBar->setTick(firstBarTick + secondTimeSig.ticks());
811  secondBar->setNo(1);
812  secondBar->setTimesig(secondTimeSig);
813  secondBar->setLen(secondTimeSig);
814  score->measures()->add(secondBar);
