80 int GuitarPro6::readBit() {
81  // calculate the byte index by dividing the position in bits by the bits per byte
82  int byteIndex = this->position / BITS_IN_BYTE;
83  // calculate our offset so we know how much to bit shift
84  int byteOffset = ((BITS_IN_BYTE-1) - (this->position % BITS_IN_BYTE));
85  // calculate the bit which we want to read
86  int bit = ((((*buffer)[byteIndex] & 0xff) >> byteOffset) & 0x01);
87  // increment our curent position so we know this bit has been read
88  this->position++;
89  // return the bit we calculated
90  return bit;
91  }

137 int GuitarPro6::readInteger(QByteArray* buffer, int offset) {
138  // assign four bytes and take them from the buffer
139  char bytes[4];
140  bytes[0] = (*buffer)[offset+0];
141  bytes[1] = (*buffer)[offset+1];
142  bytes[2] = (*buffer)[offset+2];
143  bytes[3] = (*buffer)[offset+3];
144  // increment positioning so we keep track of where we are
145  this->position+=sizeof(int)*BITS_IN_BYTE;
146  // bit shift in order to compute our integer value and return
147  return ((bytes[3] & 0xff) << 24) | ((bytes[2] & 0xff) << 16) | ((bytes[1] & 0xff) << 8) | (bytes[0] & 0xff);
148  }

170 void GuitarPro6::readGPX(QByteArray* buffer) {
171  // start by reading the file header. It will tell us if the byte array is compressed.
172  int fileHeader = readInteger(buffer, 0);
173 
174  if (fileHeader == GPX_HEADER_COMPRESSED) {
175  // this is a compressed file.
176  int length = readInteger(buffer, this->position/BITS_IN_BYTE);
177  QByteArray* bcfsBuffer = new QByteArray();
178  int positionCounter = 0;
179  while(!f->error() && (this->position/this->BITS_IN_BYTE) < length) {
180  // read the bit indicating compression information
181  int flag = this->readBits(1);
182 
183  if (flag) {
184  int bits = this->readBits(4);
185  int offs = this->readBitsReversed(bits);
186  int size = this->readBitsReversed(bits);
187 
188  QByteArray bcfsBufferCopy = *bcfsBuffer;
189  int pos = (bcfsBufferCopy.length() - offs );
190  for( int i = 0; i < (size > offs ? offs : size) ; i ++ ) {
191  bcfsBuffer->insert(positionCounter, bcfsBufferCopy[pos + i] ) ;
192  positionCounter++;
193  }
194  }
195  else {
196  int size = this->readBitsReversed(2);
197  for(int i = 0; i < size; i++) {
198  bcfsBuffer->insert(positionCounter, this->readBits(8));
199  positionCounter++;
200  }
201  }
202  }
203  // recurse on the decompressed file stored as a byte array
204  readGPX(bcfsBuffer);
205  delete bcfsBuffer;
206  }
207  else if (fileHeader == GPX_HEADER_UNCOMPRESSED) {
208  // this is an uncompressed file - strip the header off
209  *buffer = buffer->right(buffer->length()-sizeof(int));
210  int sectorSize = 0x1000;
211  int offset = 0;
212  while ((offset = (offset + sectorSize)) + 3 < buffer->length()) {
213  int newInt = readInteger(buffer,offset);
214  if (newInt == 2) {
215  int indexFileName = (offset + 4);
216  int indexFileSize = (offset + 0x8C);
217  int indexOfBlock = (offset + 0x94);
218 
219  // create a byte array and put information about files found in it
220  int block = 0;
221  int blockCount = 0;
222  QByteArray* fileBytes = new QByteArray();
223  while((block = (readInteger(buffer, (indexOfBlock + (4 * (blockCount ++)))))) != 0 ) {
224  fileBytes->push_back(getBytes(buffer, (offset = (block*sectorSize)), sectorSize));
225  }
226  // get file information and read the file
227  int fileSize = readInteger(buffer, indexFileSize);
228  if (fileBytes->length() >= fileSize) {
229  QByteArray filenameBytes = readString(buffer, indexFileName, 127);
230  char* filename = filenameBytes.data();
231  QByteArray data = getBytes(fileBytes, 0, fileSize);
232  parseFile(filename, &data);
233  }
234  delete fileBytes;
235  }
236  }
237  }
238  }

319 void GuitarPro6::readChord(QDomNode* diagram, int track)
320  {
321  // initialise a new fret diagram for our current track
322  FretDiagram* fretDiagram = new FretDiagram(score);
323  fretDiagram->setTrack(track);
324 
325  // get the identifier to set as the domain in the map
326  int id = diagram->attributes().namedItem("id").toAttr().value().toInt();
327  QDomNode diagramNode = diagram->firstChild();
328 
329  // set the number of strings on this part
330  int stringCount = diagramNode.attributes().namedItem("stringCount").toAttr().value().toInt();
331  fretDiagram->setStrings(stringCount);
332 
333  // set the fret offset
334  int baseFret = diagramNode.attributes().namedItem("baseFret").toAttr().value().toInt();
335  fretDiagram->setOffset(baseFret);
336 
337  QDomNode diagramEntity = diagramNode.firstChild();
338  int counter = 0;
339  while (!diagramEntity.isNull()) {
340  QString nodeName = diagramEntity.nodeName();
341  // new fret
342  if (!nodeName.compare("Fret")) {
343  // get the string and fret numbers from the arguments to the node as integers
344  int string = diagramEntity.attributes().namedItem("string").toAttr().value().toInt();
345  int fret = diagramEntity.attributes().namedItem("fret").toAttr().value().toInt();
346 
347  // if there are unspecified string values, add the X marker to that string
348  while (counter < string) {
349  fretDiagram->setMarker(counter, 'X');
350  counter++;
351  }
352 
353  // look at the specified string/fret and add to diagram
354  if (fret == 0) {
355  fretDiagram->setMarker(string, '0');
356  counter++;
357  }
358  else {
359  fretDiagram->setDot(string, fret);
360  counter++;
361  }
362  }
363  // move to the next string/fret specification
364  diagramEntity = diagramEntity.nextSibling();
365  }
366 
367  // mark any missing strings as 'X'
368  while (counter < stringCount) {
369  fretDiagram->setMarker(counter, 'X');
370  counter++;
371  }
372 
373  // insert the fret diagram into the map of diagrams
374  fretDiagrams.insert(id, fretDiagram);
375  }

381 void GuitarPro6::readTracks(QDomNode* track)
382  {
383  QDomNode nextTrack = track->firstChild();
384  int trackCounter = 0;
385  while (!nextTrack.isNull()) {
386  QDomNode currentNode = nextTrack.firstChild();
387  Part* part = new Part(score);
388  Staff* s = new Staff(score);
389  s->setPart(part);
390  part->insertStaff(s, -1);
391  score->staves().push_back(s);
392  while (!currentNode.isNull()) {
393  QString nodeName = currentNode.nodeName();
394  if (nodeName == "Name")
395  part->setPlainLongName(currentNode.toElement().text());
396  else if (nodeName == "Color") {}
397  else if (nodeName == "SystemsLayout") {}
398  // this is a typo is guitar pro - 'defaut' is correct here
399  else if (nodeName == "SystemsDefautLayout") {}
400  else if (nodeName == "RSE") {}
401  else if (nodeName == "GeneralMidi") {
402  if (currentNode.toElement().hasChildNodes()) {
403  int midiPort = currentNode.firstChildElement("Port").text().toInt();
404  int midiChannel = currentNode.firstChildElement("PrimaryChannel").text().toInt();
405  part->setMidiChannel(midiChannel, midiPort);
406  if (midiChannel == GP_DEFAULT_PERCUSSION_CHANNEL) {
407  part->instrument()->setDrumset(gpDrumset);
408  s->setStaffType(0, StaffType::preset(StaffTypes::PERC_DEFAULT));
409  }
410  }
411  }
412  else if (nodeName == "PlaybackState") {}
413  else if (nodeName == "PlayingStyle") {}
414  else if (nodeName == "PageSetup") {}
415  else if (nodeName == "MultiVoice") {}
416  else if (nodeName == "ShortName")
417  part->setPartName(currentNode.toElement().text());
418  else if (nodeName == "Instrument") {
419  QString ref = currentNode.attributes().namedItem("ref").toAttr().value();
420  auto it = instrumentMapping.find(ref);
421  if (it != instrumentMapping.end()) {
422  part->setInstrument(Instrument::fromTemplate(Ms::searchTemplate(it->second)));
423  }
424  else
425  qDebug() << "Unknown instrument: " << ref;
426  if (ref.endsWith("-gs") || ref.startsWith("2")) { // grand staff
427  Staff* s2 = new Staff(score);
428  s2->setPart(part);
429  part->insertStaff(s2, -1);
430  score->staves().push_back(s2);
431  s->addBracket(new BracketItem(s->score(), BracketType::BRACE, 2));
432  s->setBarLineSpan(2);
433  }
434  }
435  else if (nodeName == "Properties") {
436  QDomNode currentProperty = currentNode.firstChild();
437  while (!currentProperty.isNull()) {
438  QString propertyName = currentProperty.attributes().namedItem("name").toAttr().value();
439  if (!propertyName.compare("Tuning")) {
440  // set up the tuning for the part
441  QString tuningString = currentProperty.firstChild().toElement().text();
442  QStringList tuningStringList = tuningString.split(" ");
443  int strings = 0;
444  int tuning[tuningStringList.length()];
445  int frets = 21;
446  for (auto iter = tuningStringList.begin(); iter != tuningStringList.end(); ++iter) {
447  int currentString = (*iter).toInt();
448  tuning[strings] = currentString;
449  strings++;
450  }
451  StringData* stringData = new StringData(frets, strings, tuning);
452  Instrument* instr = part->instrument();
453  instr->setStringData(*stringData);
454  }
455  else if (!propertyName.compare("DiagramCollection")) {
456  QDomNode items = currentProperty.firstChild();
457  QDomNode currentItem = items.firstChild();
458  while (!currentItem.isNull()) {
459  readChord(&currentItem, trackCounter);
460  currentItem = currentItem.nextSibling();
461  }
462  }
463  currentProperty = currentProperty.nextSibling();
464  }
465  }
466  currentNode = currentNode.nextSibling();
467  }
468 
469  // add in a new part
470  score->appendPart(part);
471  trackCounter++;
472  nextTrack = nextTrack.nextSibling();
473  }
474 
475  previousDynamic = new int[score->staves().length() * VOICES];
476  // initialise the dynamics to 0
477  for (int i = 0; i < score->staves().length() * VOICES; i++)
478  previousDynamic[i] = 0;
479  // set the number of staves we need
480  staves = score->staves().length();
481  }

504 int GuitarPro6::findNumMeasures(GPPartInfo* partInfo)
505  {
506  QDomNode masterBar = partInfo->masterBars.nextSibling();
507  QDomNode masterBarElement;
508  while (!masterBar.isNull()) {
509  GpBar gpBar;
510  gpBar.freeTime = false;
511  gpBar.direction = "";
512  gpBar.directionStyle = "";
513  masterBarElement = masterBar.firstChild();
514  while (!masterBarElement.isNull()) {
515  if (!masterBarElement.nodeName().compare("Key"))
516  gpBar.keysig = masterBarElement.firstChild().toElement().text().toInt();
517  else
518  if (!masterBarElement.nodeName().compare("Time")) {
519  QString timeSignature = masterBarElement.toElement().text();
520  QList<QString> timeSignatureList = timeSignature.split("/");
521  gpBar.timesig = Fraction(timeSignatureList.first().toInt(), timeSignatureList.last().toInt());
522  }
523  else if (!masterBarElement.nodeName().compare("Directions")) {
524  gpBar.direction = masterBarElement.firstChild().toElement().text();
525  gpBar.directionStyle = masterBarElement.firstChild().nodeName();
526  }
527  else if (!masterBarElement.nodeName().compare("FreeTime")) {
528  gpBar.freeTime = true;
529  gpBar.barLine = BarLineType::DOUBLE;
530  }
531  else if (!masterBarElement.nodeName().compare("DoubleBar"))
532  gpBar.barLine = BarLineType::DOUBLE;
533  masterBarElement = masterBarElement.nextSibling();
534  }
535  bars.append(gpBar);
536  if (masterBar.nextSibling().isNull())
537  break;
538  masterBar = masterBar.nextSibling();
539  }
540  QString bars = masterBar.lastChildElement("Bars").toElement().text();
541  //work out the number of measures (add 1 as couning from 0, and divide by number of parts)
542  int numMeasures = (bars.split(" ").last().toInt() + 1) / score->parts().length();
543 
544  if (numMeasures > bars.size()) {
545  qDebug("GuitarPro6:findNumMeasures: bars %d < numMeasures %d\n", bars.size(), numMeasures);
546  // HACK (ws)
547  numMeasures = bars.size();
548  }
549  return numMeasures;
550  }

602 void GuitarPro6::readDrumNote(Note* note, int element, int variation)
603  {
604  int pitch = 44;
605  /* These numbers below were determined by creating all drum
606  * notes in a GPX format file and then analyzing the score.gpif
607  * file which specifies the score and then matching as much
608  * as possible with the gpDrumset... */
609  if (element == 11 && variation == 0) // pedal hihat
610  pitch = 44;
611  else if (element == 0 && variation == 0) // Kick (hit)
612  pitch = 35; // or 36
613  else if (element == 5 && variation == 0) // Tom very low (hit)
614  pitch = 41;
615  else if (element == 6 && variation == 0) // Tom low (hit)
616  pitch = 43;
617  else if (element == 7 && variation == 0) // Tom medium (hit)
618  pitch = 45;
619  else if (element == 1 && variation == 0) // Snare (hit)
620  pitch = 38; //or 40
621  else if (element == 1 && variation == 1) // Snare (rim shot)
622  pitch = 37;
623  else if (element == 1 && variation == 2) // Snare (side stick)
624  pitch = 37;
625  else if (element == 8 && variation == 0) // Tom high (hit)
626  pitch = 48;
627  else if (element == 9 && variation == 0) // Tom very high (hit)
628  pitch = 50;
629  else if (element == 15 && variation == 0) // Ride (middle)
630  pitch = 51;
631  else if (element == 15 && variation == 1) // Ride (edge)
632  pitch = 59;
633  else if (element == 15 && variation == 2) // Ride (bell)
634  pitch = 59;
635  else if (element == 10 && variation == 0) // Hihat (closed)
636  pitch = 42;
637  else if (element == 10 && variation == 1) // Hihat (half)
638  pitch = 46;
639  else if (element == 10 && variation == 2) // Hihat (open)
640  pitch = 46;
641  else if (element == 12 && variation == 0) // Crash medium (hit)
642  pitch = 49;
643  else if (element == 14 && variation == 0) // Splash (hit)
644  pitch = 55;

682  }
