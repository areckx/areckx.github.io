121 static QString createDefaultFileName(QString fn)
122  {
123  //
124  // special characters in filenames are a constant source
125  // of trouble, this replaces some of them common in german:
126  //
127  fn = fn.simplified();
128  fn = fn.replace(QChar(' '), "_");
129  fn = fn.replace(QChar('\n'), "_");
130  fn = fn.replace(QChar(0xe4), "ae"); // &auml;
131  fn = fn.replace(QChar(0xf6), "oe"); // &ouml;
132  fn = fn.replace(QChar(0xfc), "ue"); // &uuml;
133  fn = fn.replace(QChar(0xdf), "ss"); // &szlig;
134  fn = fn.replace(QChar(0xc4), "Ae"); // &Auml;
135  fn = fn.replace(QChar(0xd6), "Oe"); // &Ouml;
136  fn = fn.replace(QChar(0xdc), "Ue"); // &Uuml;
137  fn = fn.replace(QChar(0x266d),"b"); // musical flat sign, happen in instrument names, so can happen in part (file) names
138  fn = fn.replace(QChar(0x266f),"#"); // musical sharp sign, can happen in titles, so can happen in score (file) names
139  fn = fn.replace( QRegExp( "[" + QRegExp::escape( "\\/:*?\"<>|" ) + "]" ), "_" ); //FAT/NTFS special chars
140  return fn;
141  }

150 static bool readScoreError(const QString& name, Score::FileError error, bool ask)
151  {
152  QString msg = QObject::tr("Cannot read file %1:\n").arg(name);
153  QString detailedMsg;
154  bool canIgnore = false;
155  switch(error) {
156  case Score::FileError::FILE_NO_ERROR:
157  return false;
158  case Score::FileError::FILE_BAD_FORMAT:
159  msg += QObject::tr("bad format");
160  detailedMsg = MScore::lastError;
161  break;
162  case Score::FileError::FILE_UNKNOWN_TYPE:
163  msg += QObject::tr("unknown type");
164  break;
165  case Score::FileError::FILE_NO_ROOTFILE:
166  break;
167  case Score::FileError::FILE_TOO_OLD:
168  msg += QObject::tr("It was last saved with a version older than 2.0.0.\n"
169  "You can convert this score by opening and then\n"
170  "saving with MuseScore version 2.x.\n"
171  "Visit the %1MuseScore download page%2 to obtain such a 2.x version.")
172  .arg("<a href=\"https://musescore.org/download#older-versions\">")
173  .arg("</a>");
174  canIgnore = true;
175  break;
176  case Score::FileError::FILE_TOO_NEW:
177  msg += QObject::tr("This score was saved using a newer version of MuseScore.\n"
178  "Visit the %1MuseScore website%2 to obtain the latest version.")
179  .arg("<a href=\"https://musescore.org\">")
180  .arg("</a>");
181  canIgnore = true;
182  break;
183  case Score::FileError::FILE_NOT_FOUND:
184  msg = QObject::tr("File \"%1\" not found.").arg(name);
185  break;
186  case Score::FileError::FILE_CORRUPTED:
187  msg = QObject::tr("File \"%1\" corrupted.").arg(name);
188  detailedMsg = MScore::lastError;
189  canIgnore = true;
190  break;
191  case Score::FileError::FILE_ERROR:
192  case Score::FileError::FILE_OPEN_ERROR:
193  default:
194  msg += MScore::lastError;
195  break;
196  }
197  if (converterMode && canIgnore && ignoreWarnings) {
198  fprintf(stderr, "%s\n\nWarning ignored, forcing score to load\n", qPrintable(msg));
199  return true;
200  }
201  if (converterMode || pluginMode) {
202  fprintf(stderr, "%s\n", qPrintable(msg));
203  return false;
204  }
205  QMessageBox msgBox;
206  msgBox.setWindowTitle(QObject::tr("Load Error"));
207  msgBox.setText(msg.replace("\n", "<br/>"));
208  msgBox.setDetailedText(detailedMsg);
209  msgBox.setTextFormat(Qt::RichText);
210  if (canIgnore && ask) {
211  msgBox.setIcon(QMessageBox::Warning);
212  msgBox.setStandardButtons(
213  QMessageBox::Cancel | QMessageBox::Ignore
214  );
215  return msgBox.exec() == QMessageBox::Ignore;
216  }
217  else {
218  msgBox.setIcon(QMessageBox::Critical);
219  msgBox.setStandardButtons(
220  QMessageBox::Ok
221  );
222  msgBox.exec();
223  }
224  return false;
225  }

322 MasterScore* MuseScore::readScore(const QString& name)
323  {
324  if (name.isEmpty())
325  return 0;
326 
327  MasterScore* score = new MasterScore(MScore::defaultStyle());
328  setMidiReopenInProgress(name);
329  Score::FileError rv = Ms::readScore(score, name, false);
330  if (rv == Score::FileError::FILE_TOO_OLD || rv == Score::FileError::FILE_TOO_NEW || rv == Score::FileError::FILE_CORRUPTED) {
331  if (readScoreError(name, rv, true)) {
332  if (rv != Score::FileError::FILE_CORRUPTED) {
333  // dont read file again if corrupted
334  // the check routine may try to fix it
335  delete score;
336  score = new MasterScore();
337  score->setMovements(new Movements());
338  score->setStyle(MScore::defaultStyle());
339  rv = Ms::readScore(score, name, true);
340  }
341  else
342  rv = Score::FileError::FILE_NO_ERROR;
343  }
344  else {
345  delete score;
346  return 0;
347  }
348  }
349  if (rv != Score::FileError::FILE_NO_ERROR) {
350  // in case of user abort while reading, the error has already been reported
351  // else report it now
352  if (rv != Score::FileError::FILE_USER_ABORT && rv != Score::FileError::FILE_IGNORE_ERROR)
353  readScoreError(name, rv, false);
354  delete score;
355  score = 0;
356  return 0;
357  }
358  allowShowMidiPanel(name);
359  if (score)
360  addRecentScore(score);
361  return score;
362  }

384 bool MuseScore::saveFile(MasterScore* score)
385  {
386  if (score == 0)
387  return false;
388  if (score->created()) {
389  QString fn = score->masterScore()->fileInfo()->fileName();
390  Text* t = score->getText(SubStyle::TITLE);
391  if (t)
392  fn = t->plainText();
393  QString name = createDefaultFileName(fn);
394  QString f1 = tr("MuseScore File") + " (*.mscz)";
395  QString f2 = tr("Uncompressed MuseScore File") + " (*.mscx)";
396 
397  QSettings settings;
398  if (mscore->lastSaveDirectory.isEmpty())
399  mscore->lastSaveDirectory = settings.value("lastSaveDirectory", preferences.myScoresPath).toString();
400  QString saveDirectory = mscore->lastSaveDirectory;
401 
402  if (saveDirectory.isEmpty())
403  saveDirectory = preferences.myScoresPath;
404 
405  QString fname = QString("%1/%2").arg(saveDirectory).arg(name);
406  QString filter = f1 + ";;" + f2;
407  if (QFileInfo(fname).suffix().isEmpty())
408  fname += ".mscz";
409 
410  fn = mscore->getSaveScoreName(tr("Save Score"), fname, filter);
411  if (fn.isEmpty())
412  return false;
413  score->masterScore()->fileInfo()->setFile(fn);
414 
415  mscore->lastSaveDirectory = score->masterScore()->fileInfo()->absolutePath();
416 
417  if (!score->masterScore()->saveFile()) {
418  QMessageBox::critical(mscore, tr("Save File"), MScore::lastError);
419  return false;
420  }
421  addRecentScore(score);
422  writeSessionFile(false);
423  }
424  else if (!score->masterScore()->saveFile()) {
425  QMessageBox::critical(mscore, tr("Save File"), MScore::lastError);
426  return false;
427  }
428  score->setCreated(false);
429  setWindowTitle(QString(MUSESCORE_NAME_VERSION) + ": " + score->fileInfo()->completeBaseName());
430  int idx = scoreList.indexOf(score->masterScore());
431  tab1->setTabText(idx, score->fileInfo()->completeBaseName());
432  if (tab2)
433  tab2->setTabText(idx, score->fileInfo()->completeBaseName());
434  QString tmp = score->tmpName();
435  if (!tmp.isEmpty()) {
436  QFile f(tmp);
437  if (!f.remove())
438  qDebug("cannot remove temporary file <%s>", qPrintable(f.fileName()));
439  score->setTmpName("");
440  }
441  writeSessionFile(false);
442  return true;
443  }

479 MasterScore* MuseScore::getNewFile()
480  {
481  if (!newWizard)
482  newWizard = new NewWizard(this);
483  else
484  newWizard->restart();
485  if (newWizard->exec() != QDialog::Accepted)
486  return 0;
487  int measures = newWizard->measures();
488  Fraction timesig = newWizard->timesig();
489  TimeSigType timesigType = newWizard->timesigType();
490  KeySigEvent ks = newWizard->keysig();
491  VBox* nvb = nullptr;
492 
493  int pickupTimesigZ;
494  int pickupTimesigN;
495  bool pickupMeasure = newWizard->pickupMeasure(&pickupTimesigZ, &pickupTimesigN);
496  if (pickupMeasure)
497  measures += 1;
498 
499  MasterScore* score;
500  QString tp = newWizard->templatePath();
501 
502  QList<Excerpt*> excerpts;
503  if (!newWizard->emptyScore()) {
504  MasterScore* tscore = new MasterScore();
505  tscore->setMovements(new Movements());
506  tscore->setStyle(MScore::defaultStyle());
507  Score::FileError rv = Ms::readScore(tscore, tp, false);
508  if (rv != Score::FileError::FILE_NO_ERROR) {
509  readScoreError(newWizard->templatePath(), rv, false);
510  delete tscore;
511  return 0;
512  }
513  score = new MasterScore();
514  score->setMovements(new Movements());
515  score->setStyle(tscore->style());
516  // create instruments from template
517  for (Part* tpart : tscore->parts()) {
518  Part* part = new Part(score);
519  part->setInstrument(tpart->instrument());
520  part->setPartName(tpart->partName());
521 
522  for (Staff* tstaff : *tpart->staves()) {
523  Staff* staff = new Staff(score);
524  staff->setPart(part);
525  staff->init(tstaff);
526  if (tstaff->linkedStaves() && !part->staves()->isEmpty()) {
527  Staff* linkedStaff = part->staves()->back();
528  staff->linkTo(linkedStaff);
529  }
530  part->insertStaff(staff, -1);
531  score->staves().append(staff);
532  }
533  score->appendPart(part);
534  }
535  for (Excerpt* ex : tscore->excerpts()) {
536  Excerpt* x = new Excerpt(score);
537  x->setTitle(ex->title());
538  for (Part* p : ex->parts()) {
539  int pidx = tscore->parts().indexOf(p);
540  if (pidx == -1)
541  qDebug("newFile: part not found");
542  else
543  x->parts().append(score->parts()[pidx]);
544  }
545  excerpts.append(x);
546  }
547  MeasureBase* mb = tscore->first();
548  if (mb && mb->type() == ElementType::VBOX) {
549  VBox* tvb = static_cast<VBox*>(mb);
550  nvb = new VBox(score);
551  nvb->setBoxHeight(tvb->boxHeight());
552  nvb->setBoxWidth(tvb->boxWidth());
553  nvb->setTopGap(tvb->topGap());
554  nvb->setBottomGap(tvb->bottomGap());
555  nvb->setTopMargin(tvb->topMargin());
556  nvb->setBottomMargin(tvb->bottomMargin());
557  nvb->setLeftMargin(tvb->leftMargin());
558  nvb->setRightMargin(tvb->rightMargin());
559  }
560  delete tscore;
561  }
562  else {
563  score = new MasterScore();
564  score->setMovements(new Movements());
565  score->setStyle(MScore::defaultStyle());
566  newWizard->createInstruments(score);
567  }
568  score->setCreated(true);
569  score->masterScore()->fileInfo()->setFile(createDefaultName());
570 
571  if (!score->style().chordList()->loaded()) {
572  if (score->styleB(StyleIdx::chordsXmlFile))
573  score->style().chordList()->read("chords.xml");
574  score->style().chordList()->read(score->styleSt(StyleIdx::chordDescriptionFile));
575  }
576  if (!newWizard->title().isEmpty())
577  score->masterScore()->fileInfo()->setFile(newWizard->title());
578 
579  score->sigmap()->add(0, timesig);
580 
581  int firstMeasureTicks = pickupMeasure ? Fraction(pickupTimesigZ, pickupTimesigN).ticks() : timesig.ticks();
582 
583  for (int i = 0; i < measures; ++i) {
584  int tick = firstMeasureTicks + timesig.ticks() * (i - 1);
585  if (i == 0)
586  tick = 0;
587  QList<Rest*> puRests;
588  for (Score* _score : score->scoreList()) {
589  Rest* rest = 0;
590  Measure* measure = new Measure(_score);
591  measure->setTimesig(timesig);
592  measure->setLen(timesig);
593  measure->setTick(tick);
594 
595  if (pickupMeasure && tick == 0) {
596  measure->setIrregular(true); // dont count pickup measure
597  measure->setLen(Fraction(pickupTimesigZ, pickupTimesigN));
598  }
599  _score->measures()->add(measure);
600 
601  for (Staff* staff : _score->staves()) {
602  int staffIdx = staff->idx();
603  if (tick == 0) {
604  TimeSig* ts = new TimeSig(_score);
605  ts->setTrack(staffIdx * VOICES);
606  ts->setSig(timesig, timesigType);
607  Measure* m = _score->firstMeasure();
608  Segment* s = m->getSegment(SegmentType::TimeSig, 0);
609  s->add(ts);
610  Part* part = staff->part();
611  if (!part->instrument()->useDrumset()) {
612  //
613  // transpose key
614  //
615  KeySigEvent nKey = ks;
616  if (!nKey.custom() && !nKey.isAtonal() && part->instrument()->transpose().chromatic && !score->styleB(StyleIdx::concertPitch)) {
617  int diff = -part->instrument()->transpose().chromatic;
618  nKey.setKey(transposeKey(nKey.key(), diff));
619  }
620  // do not create empty keysig unless custom or atonal
621  if (nKey.custom() || nKey.isAtonal() || nKey.key() != Key::C) {
622  staff->setKey(0, nKey);
623  KeySig* keysig = new KeySig(score);
624  keysig->setTrack(staffIdx * VOICES);
625  keysig->setKeySigEvent(nKey);
626  Segment* s = measure->getSegment(SegmentType::KeySig, 0);
627  s->add(keysig);
628  }
629  }
630  }
631 
632  // determined if this staff is linked to previous so we can reuse rests
633  bool linkedToPrevious = staffIdx && staff->isLinked(_score->staff(staffIdx - 1));
634  if (measure->timesig() != measure->len()) {
635  if (!linkedToPrevious)
636  puRests.clear();
637  std::vector<TDuration> dList = toDurationList(measure->len(), false);
638  if (!dList.empty()) {
639  int ltick = tick;
640  int k = 0;
641  foreach (TDuration d, dList) {
642  if (k < puRests.count())
643  rest = static_cast<Rest*>(puRests[k]->linkedClone());
644  else {
645  rest = new Rest(score, d);
646  puRests.append(rest);
647  }
648  rest->setScore(_score);
649  rest->setDuration(d.fraction());
650  rest->setTrack(staffIdx * VOICES);
651  Segment* seg = measure->getSegment(SegmentType::ChordRest, ltick);
652  seg->add(rest);
653  ltick += rest->actualTicks();
654  k++;
655  }
656  }
657  }
658  else {
659  if (linkedToPrevious && rest)
660  rest = static_cast<Rest*>(rest->linkedClone());
661  else
662  rest = new Rest(score, TDuration(TDuration::DurationType::V_MEASURE));
663  rest->setScore(_score);
664  rest->setDuration(measure->len());
665  rest->setTrack(staffIdx * VOICES);
666  Segment* seg = measure->getSegment(SegmentType::ChordRest, tick);
667  seg->add(rest);
668  }
669  }
670  }

680  if (s->element(0)) {
681  score->select(s->element(0), SelectType::SINGLE, 0);
682  break;
683  }
684  }
685  }

687  QString title = newWizard->title();
688  QString subtitle = newWizard->subtitle();
689  QString composer = newWizard->composer();
690  QString poet = newWizard->poet();
691  QString copyright = newWizard->copyright();
692 
693  if (!title.isEmpty() || !subtitle.isEmpty() || !composer.isEmpty() || !poet.isEmpty()) {
694  MeasureBase* measure = score->measures()->first();
695  if (measure->type() != ElementType::VBOX) {
696  MeasureBase* nm = nvb ? nvb : new VBox(score);
697  nm->setTick(0);
698  nm->setNext(measure);
699  score->measures()->add(nm);
700  measure = nm;
701  }
702  else if (nvb) {
703  delete nvb;
704  }
705  if (!title.isEmpty()) {
706  Text* s = new Text(SubStyle::TITLE, score);
707  s->setPlainText(title);
708  measure->add(s);
709  score->setMetaTag("workTitle", title);
710  }
711  if (!subtitle.isEmpty()) {
712  Text* s = new Text(SubStyle::SUBTITLE, score);
713  s->setPlainText(subtitle);
714  measure->add(s);
715  }
716  if (!composer.isEmpty()) {
717  Text* s = new Text(SubStyle::COMPOSER, score);
718  s->setPlainText(composer);
719  measure->add(s);
720  score->setMetaTag("composer", composer);
721  }
722  if (!poet.isEmpty()) {

724  s->setPlainText(poet);
725  measure->add(s);
726  // the poet() functions returns data called lyricist in the dialog
727  score->setMetaTag("lyricist", poet);
728  }
729  }
730  else if (nvb) {
731  delete nvb;
732  }
733 
734  if (newWizard->createTempo()) {
735  double tempo = newWizard->tempo();
736  TempoText* tt = new TempoText(score);
737  tt->setXmlText(QString("<sym>metNoteQuarterUp</sym> = %1").arg(tempo));
738  tempo /= 60; // bpm -> bps
739 
740  tt->setTempo(tempo);
741  tt->setFollowText(true);
742  tt->setTrack(0);
743  Segment* seg = score->firstMeasure()->first(SegmentType::ChordRest);
744  seg->add(tt);
745  score->setTempo(0, tempo);
746  }
747  if (!copyright.isEmpty())
748  score->setMetaTag("copyright", copyright);
749 
750  score->rebuildMidiMapping();
751  score->doLayout();
752 
753  for (Excerpt* x : excerpts) {
754  Score* xs = new Score(static_cast<MasterScore*>(score));
755  xs->style().set(StyleIdx::createMultiMeasureRests, true);
756  x->setPartScore(xs);
757  xs->setExcerpt(x);
758  score->excerpts().append(x);
759  Excerpt::createExcerpt(x);
760  }
761  score->setExcerptsChanged(true);
762  return score;
