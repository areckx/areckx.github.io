20 // Anatomy of StaffLines:
21 //
22 // step - The possible vertical positions of a note are counted as steps.
23 // The top staff line is step position zero.
24 // lines - number of visible staff lines
25 // lineDistance - The distance between lines, measured in step units. A standard five line
26 // staff has a line distance of two steps.
27 // stepDistance - The distance between steps measured in scaled spatium/2 units. A standard five
28 // line staff has a step distance of 0.5 which results in a line distance of
29 // one spatium. The spatium unit is scaled by staff size.
30 // yoffset - vertical offset to align with other staves of different height
31 // stepOffset - This value changes the staff line step numbering.
32 //

78 void StaffLines::layout()
79  {
80  Staff* s = staff();
81  qreal _spatium = spatium();
82  qreal dist = _spatium;
83  setPos(QPointF(0.0, 0.0));
84  int _lines;
85  if (s) {
86  setMag(s->mag(measure()->tick()));
87  setColor(s->color());
88  StaffType* st = s->staffType(measure()->tick());
89  dist *= st->lineDistance().val();
90  _lines = st->lines();
91  rypos() = st->yoffset().val() * _spatium;
92  if (_lines == 1)
93  rypos() = 2 * _spatium;
94  }
95  else {
96  _lines = 5;
97  setColor(MScore::defaultColor);
98  }
99  qreal w = measure()->width();
100  lw = score()->styleS(StyleIdx::staffLineWidth).val() * _spatium;
101  qreal x1 = pos().x();
102  qreal x2 = x1 + w;
103  qreal y = pos().y();
104  bbox().setRect(x1, -lw*.5 + y, w, (_lines-1) * _spatium + lw);
105 
106  lines.clear();
107  for (int i = 0; i < _lines; ++i) {
108  lines.push_back(QLineF(x1, y, x2, y));
109  y += dist;
110  }
111  }
