22 void SynthesizerState::write(XmlWriter& xml) const
23  {
24  xml.stag("Synthesizer");
25  for (const SynthesizerGroup& g : *this) {
26  if (!g.name().isEmpty()) {
27  xml.stag(g.name());
28  for (const IdValue& v : g)
29  xml.tag(QString("val id=\"%1\"").arg(v.id), v.data);
30  xml.etag();
31  }
32  }
33  xml.etag();
34  }

40 void SynthesizerState::read(XmlReader& e)
41  {
42  while (e.readNextStartElement()) {
43  SynthesizerGroup group;
44  group.setName(e.name().toString());
45 
46  while (e.readNextStartElement()) {
47  if (e.name() == "val") {
48  int id = e.intAttribute("id");
49  group.push_back(IdValue(id, e.readElementText()));
50  }
51  else
52  e.unknown();
53  }
54  push_back(group);
55  }
56  }
