512 
513  Score* score = readScore(DIR + "empty.mscx");
514  score->doLayout();
515 
516  score->inputState().setTrack(0);
517  score->inputState().setSegment(score->tick2segment(0, false, SegmentType::ChordRest));
518  score->inputState().setDuration(TDuration::DurationType::V_128TH);
519  score->inputState().setNoteEntryMode(true);

522 
523  score->inputState().setDuration(TDuration::DurationType::V_BREVE);

525 
526  Segment* s = score->tick2segment(TDuration(TDuration::DurationType::V_128TH).ticks());
527  QVERIFY(s && s->segmentType() == SegmentType::ChordRest);
528 
529  Element* e = s->firstElement(0);
530  QVERIFY(e && e->isNote());
531 
532  int totalTicks = 0;
533  std::vector<Note*> nl = static_cast<Note*>(e)->tiedNotes();
534  QVERIFY(nl.size() >= 3); // the breve must be divided across at least 3 measures
535  for (Note* n : nl)
536  totalTicks += static_cast<Chord*>(n->parent())->durationTypeTicks();
537  QVERIFY(totalTicks == TDuration(TDuration::DurationType::V_BREVE).ticks()); // total duration same as a breve
538  }
