30 bool ProgressBarEngine::registerWidget( QWidget* widget ) {
31

33  if ( !widget ) return false;
34 
35  // create new data class
36  if ( !data_.contains( widget ) ) data_.insert( widget, new ProgressBarData( this, widget, duration() ), enabled() );

38 
39  // connect destruction signal

41

68 void ProgressBarEngine::timerEvent( QTimerEvent* event ) {
69 
70  // check enability and timer
71  if ( !(busyIndicatorEnabled() && event->timerId() == timer_.timerId() ) ) {
72  return BaseEngine::timerEvent( event );
73  }
74 
75  bool animated( false );
76 
77  // loop over objects in map
78  for ( ProgressBarSet::iterator iter = dataSet_.begin(); iter != dataSet_.end(); ++iter ) {
79 
80  // cast to progressbar
81  QProgressBar* progressBar( qobject_cast<QProgressBar*>( *iter ) );
82 
83  // check cast, visibility and range
84  if ( progressBar && progressBar->isVisible() && progressBar->minimum() == 0 && progressBar->maximum() == 0 ) {
85 
86  // update animation flag
87  animated = true;
88 
89  // update value
90  progressBar->setValue(progressBar->value() + 1);
91  progressBar->update();
92 
93  }
94  }
95 
96  if ( !animated ) timer_.stop();
97 
98  }
