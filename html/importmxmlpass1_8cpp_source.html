48 static Fraction noteTypeToFraction(const QString& type)
49  {
50  if (type == "1024th")
51  return Fraction(1, 1024);
52  else if (type == "512th")
53  return Fraction(1, 512);
54  else if (type == "256th")
55  return Fraction(1, 256);
56  else if (type == "128th")
57  return Fraction(1, 128);
58  else if (type == "64th")
59  return Fraction(1, 64);
60  else if (type == "32nd")
61  return Fraction(1, 32);
62  else if (type == "16th")
63  return Fraction(1, 16);
64  else if (type == "eighth")
65  return Fraction(1, 8);
66  else if (type == "quarter")
67  return Fraction(1, 4);
68  else if (type == "half")
69  return Fraction(1, 2);
70  else if (type == "whole")
71  return Fraction(1, 1);
72  else if (type == "breve")
73  return Fraction(2, 1);
74  else if (type == "long")
75  return Fraction(4, 1);
76  else if (type == "maxima")
77  return Fraction(8, 1);
78  else
79  return Fraction(0, 0);
80  }

90 static Fraction calculateFraction(const QString& type, const int dots, const Fraction timeMod)
91  {
92  // type
93  Fraction f = noteTypeToFraction(type);
94  if (f.isValid()) {
95  // dot(s)
96  Fraction f_no_dots = f;
97  for (int i = 0; i < dots; ++i)
98  f += (f_no_dots / (2 << i));
99  // tuplet
100  if (timeMod.isValid())
101  f *= timeMod;
102  // clean up (just in case)
103  f.reduce();
104  }
105  return f;
106  }

117 // for regular (non-overlapping) voices:
118 // 1) assign voice to a staff (allocateStaves)
119 // 2) assign voice numbers (allocateVoices)
120 // due to cross-staving, it is not a priori clear to which staff
121 // a voice has to be assigned
122 // allocate ordered by number of chordrests in the MusicXML voice
123 //
124 // for overlapping voices:
125 // 1) assign voice to staves it is found in (allocateStaves)
126 // 2) assign voice numbers (allocateVoices)
127 
128 static void allocateStaves(VoiceList& vcLst)
129  {
130  // initialize
131  int voicesAllocated[MAX_STAVES]; // number of voices allocated on each staff
132  for (int i = 0; i < MAX_STAVES; ++i)
133  voicesAllocated[i] = 0;
134 
135  // handle regular (non-overlapping) voices
136  // note: outer loop executed vcLst.size() times, as each inner loop handles exactly one item
137  for (int i = 0; i < vcLst.size(); ++i) {
138  // find the regular voice containing the highest number of chords and rests that has not been handled yet
139  int max = 0;
140  QString key;
141  for (VoiceList::const_iterator j = vcLst.constBegin(); j != vcLst.constEnd(); ++j) {
142  if (!j.value().overlaps() && j.value().numberChordRests() > max && j.value().staff() == -1) {
143  max = j.value().numberChordRests();
144  key = j.key();
145  }
146  }
147  if (key != "") {
148  int prefSt = vcLst.value(key).preferredStaff();
149  if (voicesAllocated[prefSt] < VOICES) {
150  vcLst[key].setStaff(prefSt);
151  voicesAllocated[prefSt]++;
152  }
153  else
154  // out of voices: mark as used but not allocated
155  vcLst[key].setStaff(-2);
156  }
157  }
158 
159  // handle overlapping voices
160  // for every staff allocate remaining voices (if space allows)
161  // the ones with the highest number of chords and rests get allocated first
162  for (int h = 0; h < MAX_STAVES; ++h) {
163  // note: middle loop executed vcLst.size() times, as each inner loop handles exactly one item
164  for (int i = 0; i < vcLst.size(); ++i) {
165  // find the overlapping voice containing the highest number of chords and rests that has not been handled yet
166  int max = 0;
167  QString key;
168  for (VoiceList::const_iterator j = vcLst.constBegin(); j != vcLst.constEnd(); ++j) {
169  if (j.value().overlaps() && j.value().numberChordRests(h) > max && j.value().staffAlloc(h) == -1) {
170  max = j.value().numberChordRests(h);
171  key = j.key();
172  }
173  }
174  if (key != "") {
175  int prefSt = h;
176  if (voicesAllocated[prefSt] < VOICES) {
177  vcLst[key].setStaffAlloc(prefSt, 1);
178  voicesAllocated[prefSt]++;
179  }
180  else
181  // out of voices: mark as used but not allocated
182  vcLst[key].setStaffAlloc(prefSt, -2);
183  }
184  }
185  }
186  }

198 static void allocateVoices(VoiceList& vcLst)
199  {
200  int nextVoice[MAX_STAVES]; // number of voices allocated on each staff
201  for (int i = 0; i < MAX_STAVES; ++i)
202  nextVoice[i] = 0;
203  // handle regular (non-overlapping) voices
204  // a voice is allocated on one specific staff
205  for (VoiceList::const_iterator i = vcLst.constBegin(); i != vcLst.constEnd(); ++i) {
206  int staff = i.value().staff();
207  QString key = i.key();
208  if (staff >= 0) {
209  vcLst[key].setVoice(nextVoice[staff]);
210  nextVoice[staff]++;
211  }
212  }
213  // handle overlapping voices
214  // each voice may be in every staff
215  for (VoiceList::const_iterator i = vcLst.constBegin(); i != vcLst.constEnd(); ++i) {
216  for (int j = 0; j < MAX_STAVES; ++j) {
217  int staffAlloc = i.value().staffAlloc(j);
218  QString key = i.key();
219  if (staffAlloc >= 0) {
220  vcLst[key].setVoice(j, nextVoice[j]);
221  nextVoice[j]++;
222  }
223  }
224  }
225  }

283 bool MusicXMLParserPass1::determineMeasureLength(QVector<Fraction>& ml) const
284  {
285  ml.clear();
286 
287  // determine number of measures: max number of measures in any part
288  int nMeasures = 0;
289  foreach (const MusicXmlPart &part, _parts) {
290  if (part.nMeasures() > nMeasures)
291  nMeasures = part.nMeasures();
292  }
293 
294  // determine max length of a specific measure in all parts
295  for (int i = 0; i < nMeasures; ++i) {
296  Fraction maxMeasDur;
297  foreach (const MusicXmlPart &part, _parts) {
298  if (i < part.nMeasures()) {
299  Fraction measDurPartJ = part.measureDuration(i);
300  if (measDurPartJ > maxMeasDur)
301  maxMeasDur = measDurPartJ;
302  }
303  }
304  //qDebug("determineMeasureLength() measure %d %s (%d)", i, qPrintable(maxMeasDur.print()), maxMeasDur.ticks());
305  ml.append(maxMeasDur);
306  }
307  return true;
308  }

378 bool MusicXMLParserPass1::determineStaffMoveVoice(const QString& id, const int mxStaff, const QString& mxVoice,
379  int& msMove, int& msTrack, int& msVoice) const
380  {
381  VoiceList voicelist = getVoiceList(id);
382  msMove = 0; // TODO
383  msTrack = 0; // TODO
384  msVoice = 0; // TODO
385 
386 
387  // Musicxml voices are counted for all staffs of an
388  // instrument. They are not limited. In mscore voices are associated
389  // with a staff. Every staff can have at most VOICES voices.
390 
391  // The following lines map musicXml voices to mscore voices.
392  // If a voice crosses two staffs, this is expressed with the
393  // "move" parameter in mscore.
394 
395  // Musicxml voices are unique within a part, but not across parts.
396 
397  //qDebug("voice mapper before: voice='%s' staff=%d", qPrintable(mxVoice), mxStaff);
398  int s; // staff mapped by voice mapper
399  int v; // voice mapped by voice mapper
400  if (voicelist.value(mxVoice).overlaps()) {
401  // for overlapping voices, the staff does not change
402  // and the voice is mapped and staff-dependent
403  s = mxStaff;
404  v = voicelist.value(mxVoice).voice(s);
405  }
406  else {
407  // for non-overlapping voices, both staff and voice are
408  // set by the voice mapper
409  s = voicelist.value(mxVoice).staff();
410  v = voicelist.value(mxVoice).voice();
411  }
412 
413  //qDebug("voice mapper mapped: s=%d v=%d", s, v);
414  if (s < 0 || v < 0) {
415  qDebug("too many voices (staff=%d voice='%s' -> s=%d v=%d)",
416  mxStaff + 1, qPrintable(mxVoice), s, v);
417  return false;
418  }
419 
420  msMove = mxStaff - s;
421  msVoice = v;
422 
423  // make score-relative instead on part-relative
424  Part* part = _partMap.value(id);
425  Q_ASSERT(part);
426  int scoreRelStaff = _score->staffIdx(part); // zero-based number of parts first staff in the score
427  msTrack = (scoreRelStaff + s) * VOICES;
428 
429  //qDebug("voice mapper after: scoreRelStaff=%d partRelStaff=%d msMove=%d msTrack=%d msVoice=%d",
430  // scoreRelStaff, s, msMove, msTrack, msVoice);
431  // note: relStaff is the staff number relative to the parts first staff
432  // voice is the voice number in the staff
433 
434  return true;
435  }

1071 static QString text2syms(const QString& t)
1072  {
1073  //QTime time;
1074  //time.start();
1075 
1076  // first create a map from symbol (Unicode) text to symId
1077  // note that this takes about 1 msec on a Core i5,
1078  // caching does not gain much
1079 
1080  ScoreFont* sf = ScoreFont::fallbackFont();
1081  QMap<QString, SymId> map;
1082  int maxStringSize = 0; // maximum string size found
1083 
1084  for (int i = int(SymId::noSym); i < int(SymId::lastSym); ++i) {
1085  SymId id((SymId(i)));
1086  QString string(sf->toString(id));
1087  // insert all syms except space to prevent matching all regular spaces
1088  if (id != SymId::space)
1089  map.insert(string, id);
1090  if (string.size() > maxStringSize)
1091  maxStringSize = string.size();
1092  }
1093  //qDebug("text2syms map count %d maxsz %d filling time elapsed: %d ms",
1094  // map.size(), maxStringSize, time.elapsed());
1095 
1096  // then look for matches
1097  QString in = t;
1098  QString res;
1099 
1100  while (in != "") {
1101  // try to find the largest match possible
1102  int maxMatch = qMin(in.size(), maxStringSize);
1103  QString sym;
1104  while (maxMatch > 0) {
1105  QString toBeMatched = in.left(maxMatch);
1106  if (map.contains(toBeMatched)) {
1107  sym = Sym::id2name(map.value(toBeMatched));
1108  break;
1109  }
1110  maxMatch--;
1111  }
1112  if (maxMatch > 0) {
1113  // found a match, add sym to res and remove match from string in
1114  res += "<sym>";
1115  res += sym;
1116  res += "</sym>";
1117  in.remove(0, maxMatch);
1118  }
1119  else {
1120  // not found, move one char from res to in
1121  res += in.left(1);
1122  in.remove(0, 1);
1123  }
1124  }
1125 
1126  //qDebug("text2syms total time elapsed: %d ms, res '%s'", time.elapsed(), qPrintable(res));
1127  return res;
1128  }
