133 // fretChords
134 // Assigns fretting to all the notes of each chord in the same segment of chord
135 // re-using existing fretting wherever possible
136 //
137 // Minimizes fret conflicts (multiple notes on the same string)
138 // but marks as fretConflict notes which cannot be fretted
139 // (outside tablature range) or which cannot be assigned
140 // a separate string
141 //---------------------------------------------------------
142 
143 void StringData::fretChords(Chord * chord) const
144  {
145  int nFret, minFret, maxFret, nNewFret, nTempFret;
146  int nString, nNewString, nTempString;
147 
148  if(bFretting)
149  return;
150  bFretting = true;
151 
152  // we need to keep track of string allocation
153  int bUsed[strings()]; // initially all strings are available
154  for(nString=0; nString<strings(); nString++)
155  bUsed[nString] = 0;
156  // we also need the notes sorted in order of string (from highest to lowest) and then pitch
157  QMap<int, Note *> sortedNotes;
158  int count = 0;
159  // store staff pitch offset at this tick, to speed up actual note pitch calculations
160  // (ottavas not implemented yet)
161  int transp = chord->staff() ? chord->part()->instrument()->transpose().chromatic : 0; // TODO: tick?
162  int pitchOffset = /*chord->staff()->pitchOffset(chord->segment()->tick())*/ - transp;
163  // if chord parent is not a segment, the chord is special (usually a grace chord):
164  // fret it by itself, ignoring the segment
165  if (chord->parent()->type() != ElementType::SEGMENT)
166  sortChordNotes(sortedNotes, chord, pitchOffset, &count);
167  else {
168  // scan each chord of seg from same staff as 'chord', inserting each of its notes in sortedNotes
169  Segment* seg = chord->segment();
170  int trk;
171  int trkFrom = (chord->track() / VOICES) * VOICES;
172  int trkTo = trkFrom + VOICES;
173  for(trk = trkFrom; trk < trkTo; ++trk) {
174  Element* ch = seg->elist().at(trk);
175  if (ch && ch->type() == ElementType::CHORD)
176  sortChordNotes(sortedNotes, static_cast<Chord*>(ch), pitchOffset, &count);
177  }
178  }
179  // determine used range of frets
180  minFret = INT32_MAX;
181  maxFret = INT32_MIN;
182  foreach(Note* note, sortedNotes) {
183  if (note->string() != STRING_NONE)
184  bUsed[note->string()]++;
185  if (note->fret() != FRET_NONE && note->fret() < minFret)
186  minFret = note->fret();
187  if (note->fret() != FRET_NONE && note->fret() > maxFret)
188  maxFret = note->fret();
189  }
190 
191  // scan chord notes from highest, matching with strings from the highest
192  foreach(Note * note, sortedNotes) {
193  nString = nNewString = note->string();
194  nFret = nNewFret = note->fret();
195  note->setFretConflict(false); // assume no conflicts on this note
196  // if no fretting (any invalid fretting has been erased by sortChordNotes() )
197  if (nString == STRING_NONE /*|| nFret == FRET_NONE || getPitch(nString, nFret) != note->pitch()*/) {
198  // get a new fretting
199  if (!convertPitch(note->pitch(), pitchOffset, &nNewString, &nNewFret) ) {
200  // no way to fit this note in this tab:
201  // mark as fretting conflict
202  note->setFretConflict(true);
203  // store fretting change without affecting chord context
204  if (nFret != nNewFret)
205  note->score()->undoChangeProperty(note, P_ID::FRET, nNewFret);
206  if (nString != nNewString)
207  note->score()->undoChangeProperty(note, P_ID::STRING, nNewString);
208  continue;
209  }
210  // note can be fretted: use string
211  else {
212  bUsed[nNewString]++;
213  }
214  }
215 
216  // if the note string (either original or newly assigned) is also used by another note
217  if (bUsed[nNewString] > 1) {
218  // attempt to find a suitable string, from topmost
219  for (nTempString=0; nTempString < strings(); nTempString++) {
220  if (bUsed[nTempString] < 1
221  && (nTempFret=fret(note->pitch(), nTempString, pitchOffset)) != FRET_NONE) {
222  bUsed[nNewString]--; // free previous string
223  bUsed[nTempString]++; // and occupy new string
224  nNewFret = nTempFret;
225  nNewString = nTempString;
226  break;
227  }
228  }
229  }
230 
231  // TODO : try to optimize used fret range, avoiding eccessively open positions
232 
233  // if fretting did change, store as a fret change
234  if (nFret != nNewFret)
235  note->score()->undoChangeProperty(note, P_ID::FRET, nNewFret);
236  if (nString != nNewString)
237  note->score()->undoChangeProperty(note, P_ID::STRING, nNewString);
238  }
239 
240  // check for any remaining fret conflict
241  foreach(Note * note, sortedNotes)
242  if (bUsed[note->string()] > 1)
243  note->setFretConflict(true);
244 
245  bFretting = false;
246  }

270 // convertPitch
271 // Finds string and fret for a note.
272 //
273 // Fills *string and *fret with suitable values for pitch / pitchOffset,
274 // using the highest possible string.
275 // If note cannot be fretted, uses fret 0 on nearest string and returns false
276 //
277 // Note: Strings are stored internally from lowest (0) to highest (strings()-1),
278 // but the returned *string value references strings in reversed, 'visual', order:
279 // from highest (0) to lowest (strings()-1)
280 //---------------------------------------------------------
281 
282 bool StringData::convertPitch(int pitch, int pitchOffset, int* string, int* fret) const
283  {
284  int strings = stringTable.size();
285  if (strings < 1)
286  return false;
287 
288  pitch += pitchOffset;
289 
290  // if above max fret on highest string, fret on first string, but return failure
291  if(pitch > stringTable.at(strings-1).pitch + _frets) {
292  *string = 0;
293  *fret = 0;
294  return false;
295  }
296 
297  // look for a suitable string, starting from the highest
298  // NOTE: this assumes there are always enough frets to fill
299  // the interval between any fretted string and the next
300  for (int i = strings-1; i >=0; i--) {
301  instrString strg = stringTable.at(i);
302  if(pitch >= strg.pitch) {
303  if (pitch == strg.pitch || !strg.open)
304  *string = strings - i - 1;
305  *fret = pitch - strg.pitch;
306  return true;
307  }
308  }
309 
310  // if no string found, pitch is below lowest string:
311  // fret on last string, but return failure
312  *string = strings-1;
313  *fret = 0;
314  return false;
315  }

324 int StringData::getPitch(int string, int fret, int pitchOffset) const
325  {
326  int strings = stringTable.size();
327  if (string < 0 || string >= strings)
328  return INVALID_PITCH;
329  instrString strg = stringTable.at(strings - string - 1);
330  return strg.pitch - pitchOffset + (strg.open ? 0 : fret);
331  }

339 int StringData::fret(int pitch, int string, int pitchOffset) const
340  {
341  int strings = stringTable.size();
342  if (strings < 1) // no strings at all!
343  return FRET_NONE;
344 
345  if (string < 0 || string >= strings) // no such a string
346  return FRET_NONE;
347 
348  pitch += pitchOffset;
349 
350  int fret = pitch - stringTable[strings - string - 1].pitch;
351  // fret number is invalid or string cannot be fretted
352  if (fret < 0 || fret >= _frets || (fret > 0 && stringTable[strings - string - 1].open))
353  return FRET_NONE;
354  return fret;
355  }

357 //---------------------------------------------------------
358 // sortChordNotes
359 // Adds to sortedNotes the notes of Chord in string/pitch order
360 // Note: notes are sorted first by string (top string being 0),
361 // then by negated pitch (higher pitches resulting in lower key),
362 // then by order of submission to disambiguate notes with the same pitch.
363 // Everything else being equal, this makes notes in higher-numbered voices
364 // to be sorted after notes in lower-numbered voices (voice 2 after voice 1 and so on)
365 // Notes without a string assigned yet, are sorted according to the lowest string which can accommodate them.
366 //---------------------------------------------------------
367 
368 void StringData::sortChordNotes(QMap<int, Note *>& sortedNotes, const Chord *chord, int pitchOffset, int* count) const
369 {
370  int key, string, fret;
371 
372  foreach(Note * note, chord->notes()) {
373  string = note->string();
374  fret = note->fret();
375  // if note not fretted yet or current fretting no longer valid,
376  // use most convenient string as key
377  if (string <= STRING_NONE || fret <= FRET_NONE
378  || getPitch(string, fret, pitchOffset) != note->pitch()) {
379  note->setString(STRING_NONE);
380  note->setFret(FRET_NONE);
381  convertPitch(note->pitch(), pitchOffset, &string, &fret);
382  }
383  key = string * 100000;
384  key += -(note->pitch()+pitchOffset) * 100 + *count; // disambiguate notes of equal pitch
385  sortedNotes.insert(key, note);
386  (*count)++;
387  }
388 }
