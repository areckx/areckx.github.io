124 
125  if ( !enabled() ) return false;
126 
127  // check event type
128  switch ( event->type() ) {
129 
130  case QEvent::Enter: {
131  object->event( event );
132  enterEvent( object );
133  return true;
134  }
135

137  case QEvent::Leave: {

142  }

147  return true;
148  }
149

151  break;
152 
153  }
154 
155  // always forward event
156  return false;

173  animatedRect_.setBottom( previousRect().bottom() + progress() * (currentRect().bottom() - previousRect().bottom()) );
