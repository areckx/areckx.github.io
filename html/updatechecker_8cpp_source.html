38 void UpdateChecker::onRequestFinished(QNetworkReply* reply)
39  {
40  if (reply->error() != QNetworkReply::NoError) {
41  qDebug("Error while checking update [%s]", reply->errorString().toLatin1().constData());
42  return;
43  }
44 
45  QSettings s;
46  s.beginGroup("Update");
47  s.setValue("lastUpdateDate", QDateTime::currentDateTime());
48  s.endGroup();
49 
50  QByteArray data = reply->readAll();
51  QXmlStreamReader reader(data);
52  QString version;
53  QString upgradeRevision;
54  QString downloadUrl;
55  QString infoUrl;
56  QString description;
57  QString releaseType;
58 
59  while (!reader.atEnd() && !reader.hasError()) {
60  QXmlStreamReader::TokenType token = reader.readNext();
61  if(token == QXmlStreamReader::StartDocument) {
62  continue;
63  }
64  if(token == QXmlStreamReader::StartElement) {
65  if(reader.name() == "version")
66  version = parseText(reader);
67  else if (reader.name() == "revision")
68  upgradeRevision = parseText(reader);
69  else if (reader.name() == "downloadUrl")
70  downloadUrl = parseText(reader);
71  else if (reader.name() == "infoUrl")
72  infoUrl = parseText(reader);
73  else if (reader.name() == "description")
74  description = parseText(reader);
75  }
76  }
77 
78  if (reader.error() != QXmlStreamReader::NoError)
79  qDebug() << reader.error() << reader.errorString();
80 
81  QString message = tr("An update for MuseScore is available: %1MuseScore %2 r.%3%4")
82  .arg("<a href=\"" + downloadUrl + "\">")
83  .arg(version)
84  .arg(upgradeRevision)
85  .arg("</a>");
86 // qDebug("revision %s", revision.toLatin1().constData());
87  if (!version.isEmpty() && version > _currentVersion ) {
88  QMessageBox msgBox;
89  msgBox.setWindowTitle(tr("Update Available"));
90  msgBox.setText(message);
91  msgBox.setTextFormat(Qt::RichText);
92  msgBox.exec();
93  }
94  else if (manual) {
95  QMessageBox msgBox;
96  msgBox.setWindowTitle(tr("No Update Available"));
97  msgBox.setText(tr("No Update Available"));
98  msgBox.setTextFormat(Qt::RichText);
99  msgBox.exec();
100  }
101  }

112 void UpdateChecker::check(QString currentVersion, bool m)
113  {
114  manual = m;
115 #if defined(Q_OS_WIN)
116  os = "win";
117 #endif
118 #if defined(Q_OS_MAC)
119  os = "mac";
120 #endif
121  if (qApp->applicationName() == "MuseScore3") { //avoid nightly cymbals
122  if (MuseScore::unstable())
123  release = "pre";
124  else
125  release = "stable";
126  }
127  else {
128  release = "nightly";
129  }
130  if (MScore::debugMode)
131  qDebug("release type: %s", release.toLatin1().constData());
132  if (!os.isEmpty() && !release.isEmpty() && release != "nightly") {
133  _currentVersion = currentVersion;
134  manager->get(QNetworkRequest(QUrl("http://update.musescore.org/update_"+os +"_" + release +".xml")));
135  }
136  }

158 bool UpdateChecker::hasToCheck()
159  {
160  if (!preferences.checkUpdateStartup)
161  return false;
162  QSettings s;
163  s.beginGroup("Update");
164  QDateTime now = QDateTime::currentDateTime();
165  QDateTime lastUpdate = s.value("lastUpdateDate", now).value<QDateTime>();
166 
167  if (MScore::debugMode) {
168  qDebug("preferences.checkUpdateStartup: %d" , preferences.checkUpdateStartup);
169  qDebug("lastupdate: %s", qPrintable(lastUpdate.toString("dd.MM.yyyy hh:mm:ss.zzz")));
170  }
171  s.endGroup();
172  return now == lastUpdate || now > lastUpdate.addSecs(3600 * defaultPeriod()) ;
173  }
