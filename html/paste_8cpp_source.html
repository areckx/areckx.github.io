43 static void transposeChord(Chord* c, Interval srcTranspose, int tick)
44  {
45  // set note track
46  // check if staffMove moves a note to a
47  // nonexistant staff
48  //
49  int track = c->track();
50  int nn = (track / VOICES) + c->staffMove();
51  if (nn < 0 || nn >= c->score()->nstaves())
52  c->setStaffMove(0);
53  Part* part = c->part();
54  Interval dstTranspose = part->instrument(tick)->transpose();
55 
56  if (srcTranspose != dstTranspose) {
57  if (!dstTranspose.isZero()) {
58  dstTranspose.flip();
59  for (Note* n : c->notes()) {
60  int npitch;
61  int ntpc;
62  transposeInterval(n->pitch(), n->tpc1(), &npitch, &ntpc, dstTranspose, true);
63  n->setTpc2(ntpc);
64  }
65  }
66  else {
67  for (Note* n : c->notes())
68  n->setTpc2(n->tpc1());
69  }
70  }
71  }

77 
78 bool Score::pasteStaff(XmlReader& e, Segment* dst, int dstStaff)
79  {
80  Q_ASSERT(dst->segmentType() == SegmentType::ChordRest);
81  QList<Chord*> graceNotes;
82  int dstTick = dst->tick();
83  bool pasted = false;
84  int tickLen = 0;
85  int staves = 0;
86  bool done = false;
87  while (e.readNextStartElement()) {
88  if (done)
89  break;
90  if (e.name() != "StaffList") {
91  e.unknown();
92  break;
93  }
94  QString version = e.attribute("version", "NONE");
95  if (!MScore::testMode) {
96  if (version != MSC_VERSION) {
97  qDebug("pasteStaff: bad version");
98  break;
99  }
100  }
101  int tickStart = e.intAttribute("tick", 0);
102  tickLen = e.intAttribute("len", 0);
103  int staffStart = e.intAttribute("staff", 0);
104  staves = e.intAttribute("staves", 0);
105  int voiceOffset[VOICES];
106  std::fill(voiceOffset,voiceOffset+VOICES,-1);
107 
108  e.setTickOffset(dstTick - tickStart);
109  e.initTick(0);
110 
111  while (e.readNextStartElement()) {
112  if (done)
113  break;
114  if (e.name() != "Staff") {
115  e.unknown();
116  break;
117  }
118  e.setTransposeChromatic(0);
119  e.setTransposeDiatonic(0);
120 
121  int srcStaffIdx = e.attribute("id", "0").toInt();
122  e.setTrack(srcStaffIdx * VOICES);
123  e.setTrackOffset((dstStaff - staffStart) * VOICES);
124  int dstStaffIdx = e.track() / VOICES;
125  if (dstStaffIdx >= dst->score()->nstaves()) {
126  qDebug("paste beyond staves");
127  done = true;
128  break;
129  }
130 
131  e.tuplets().clear();
132  bool makeGap = true;
133  while (e.readNextStartElement()) {
134  pasted = true;
135  const QStringRef& tag(e.name());
136 
137  if (tag == "transposeChromatic")
138  e.setTransposeChromatic(e.readInt());
139  else if (tag == "transposeDiatonic")
140  e.setTransposeDiatonic(e.readInt());
141  else if (tag == "voice") {
142  int voiceId = e.attribute("id", "-1").toInt();
143  Q_ASSERT(voiceId >= 0 && voiceId < VOICES);
144  voiceOffset[voiceId] = e.readInt();
145  }
146  else if (tag == "move" || tag == "tick") {
147  int tick = tag == "move" ? e.readFraction().ticks() : e.readInt();
148  e.initTick(tick);
149  int shift = tick - tickStart;
150  if (makeGap && !makeGap1(dstTick, dstStaffIdx, Fraction::fromTicks(tickLen), voiceOffset)) {
151  qDebug("cannot make gap in staff %d at tick %d", dstStaffIdx, dstTick + shift);
152  done = true; // break main loop, cannot make gap
153  break;
154  }
155  makeGap = false; // create gap only once per staff
156  }
157  else if (tag == "Tuplet") {
158  Tuplet* tuplet = new Tuplet(this);
159  tuplet->setTrack(e.track());
160  tuplet->read(e);
161  int tick = e.tick();
162  // no paste into local time signature
163  if (staff(dstStaffIdx)->isLocalTimeSignature(tick)) {
164  MScore::setError(DEST_LOCAL_TIME_SIGNATURE);
165  return false;
166  }
167  Measure* measure = tick2measure(tick);
168  tuplet->setParent(measure);
169  tuplet->setTick(tick);
170  int ticks = tuplet->actualTicks();
171  int rticks = measure->endTick() - tick;
172  if (rticks < ticks) {
173  delete tuplet;
174  MScore::setError(TUPLET_CROSSES_BAR);
175  return false;
176  }
177  e.addTuplet(tuplet);
178  }
179  else if (tag == "Chord" || tag == "Rest" || tag == "RepeatMeasure") {
180  ChordRest* cr = toChordRest(Element::name2Element(tag, this));
181  cr->setTrack(e.track());
182  cr->read(e);
183  cr->setSelected(false);
184  int tick = e.tick();
185  // no paste into local time signature
186  if (staff(dstStaffIdx)->isLocalTimeSignature(tick)) {
187  MScore::setError(DEST_LOCAL_TIME_SIGNATURE);
188  return false;
189  }
190  if (cr->isGrace())
191  graceNotes.push_back(toChord(cr));
192  else {
193  e.incTick(cr->actualTicks());
194  if (cr->isChord()) {
195  Chord* chord = toChord(cr);
196  // disallow tie across barline within two-note tremolo
197  // tremolos can potentially still straddle the barline if no tie is required
198  // but these will be removed later
199  if (chord->tremolo() && chord->tremolo()->twoNotes()) {
200  Measure* m = tick2measure(tick);
201  int ticks = cr->actualTicks();
202  int rticks = m->endTick() - tick;
203  if (rticks < ticks || (rticks != ticks && rticks < ticks * 2)) {
204  MScore::setError(DEST_TREMOLO);
205  return false;
206  }
207  }
208  for (int i = 0; i < graceNotes.size(); ++i) {
209  Chord* gc = graceNotes[i];
210  gc->setGraceIndex(i);
211  transposeChord(gc, e.transpose(), tick);
212  chord->add(gc);
213  }
214  graceNotes.clear();
215  }
216  // delete pending ties, they are not selected when copy
217  if ((tick - dstTick) + cr->actualTicks() >= tickLen) {
218  if (cr->isChord()) {
219  Chord* c = toChord(cr);
220  for (Note* note: c->notes()) {
221  Tie* tie = note->tieFor();
222  if (tie) {
223  note->setTieFor(0);
224  delete tie;
225  }
226  }
227  }
228  }
229  // shorten last cr to fit in the space made by makeGap
230  if ((tick - dstTick) + cr->actualTicks() > tickLen) {
231  int newLength = tickLen - (tick - dstTick);
232  // check previous CR on same track, if it has tremolo, delete the tremolo
233  // we don't want a tremolo and two different chord durations
234  if (cr->isChord()) {
235  Segment* s = tick2leftSegment(tick - 1);
236  if (s) {
237  ChordRest* crt = toChordRest(s->element(cr->track()));
238  if (!crt)
239  crt = s->nextChordRest(cr->track(), true);
240  if (crt && crt->isChord()) {
241  Chord* chrt = toChord(crt);
242  Tremolo* tr = chrt->tremolo();
243  if (tr) {
244  tr->setChords(chrt, toChord(cr));
245  chrt->remove(tr);
246  delete tr;
247  }
248  }
249  }
250  }
251  if (!cr->tuplet()/*|| cr->actualTicks() - newLength > (cr->tuplet()->ratio().numerator() + 1 ) / 2*/) {
252  // shorten duration
253  // exempt notes in tuplets, since we don't allow copy of partial tuplet anyhow
254  // TODO: figure out a reasonable fudge factor to make sure shorten tuplets appropriately if we do ever copy a partial tuplet
255  cr->setDuration(Fraction::fromTicks(newLength));
256  cr->setDurationType(newLength);
257  }
258  }
259  pasteChordRest(cr, tick, e.transpose());
260  }
261  }
262  else if (tag == "HairPin"
263  || tag == "Pedal"
264  || tag == "Ottava"
265  || tag == "Trill"
266  || tag == "TextLine"
267  || tag == "Volta") {
268  Spanner* sp = static_cast<Spanner*>(Element::name2Element(tag, this));
269  sp->setAnchor(Spanner::Anchor::SEGMENT);
270  sp->read(e);
271  sp->setTrack(e.track());
272  sp->setTrack2(e.track());
273  sp->setTick(e.tick());
274  addSpanner(sp);
275  }
276  else if (tag == "Slur") {
277  Slur* sp = new Slur(this);
278  sp->read(e);
279  sp->setTrack(e.track());
280  sp->setTick(e.tick());
281  // check if we saw endSpanner / stop element already
282  int id = e.spannerId(sp);
283  const SpannerValues* sv = e.spannerValues(id);
284  if (sv) {
285  sp->setTick2(sv->tick2);
286  sp->setTrack2(sv->track2);
287  }
288  undoAddElement(sp);
289  }
290  else if (tag == "endSpanner") {
291  int id = e.intAttribute("id");
292  Spanner* spanner = e.findSpanner(id);
293  if (spanner) {
294  // e.spanner().removeOne(spanner);
295  spanner->setTick2(e.tick());
296  removeSpanner(spanner);
297  undoAddElement(spanner);
298  if (spanner->isOttava())
299  spanner->staff()->updateOttava();
300  }
301  e.readNext();
302  }
303  else if (tag == "Harmony") {
304  Harmony* harmony = new Harmony(this);
305  harmony->setTrack(e.track());
306  harmony->read(e);
307  harmony->setTrack(e.track());
308  // transpose
309  Part* partDest = staff(e.track() / VOICES)->part();
310  Interval interval = partDest->instrument(e.tick())->transpose();
311  if (!styleB(StyleIdx::concertPitch) && !interval.isZero()) {
312  interval.flip();
313  int rootTpc = transposeTpc(harmony->rootTpc(), interval, true);
314  int baseTpc = transposeTpc(harmony->baseTpc(), interval, true);
315  undoTransposeHarmony(harmony, rootTpc, baseTpc);
316  }
317 
318  int tick = e.tick();
319  Measure* m = tick2measure(tick);
320  Segment* seg = m->undoGetSegment(SegmentType::ChordRest, tick);
321  if (seg->findAnnotationOrElement(ElementType::HARMONY, e.track(), e.track())) {
322  QList<Element*> elements;
323  foreach (Element* el, seg->annotations()) {
324  if (el->isHarmony() && el->track() == e.track()) {
325  elements.append(el);
326  }
327  }
328  foreach (Element* el, elements)
329  undoRemoveElement(el);
330  }
331  harmony->setParent(seg);
332  undoAddElement(harmony);
333  }
334  else if (tag == "Dynamic"
335  || tag == "Symbol"
336  || tag == "FretDiagram"
337  || tag == "TremoloBar"
338  || tag == "Marker"
339  || tag == "Jump"
340  || tag == "Image"
341  || tag == "Text"
342  || tag == "StaffText"
343  || tag == "TempoText"
344  || tag == "FiguredBass"
345  ) {
346  Element* el = Element::name2Element(tag, this);
347  el->setTrack(e.track()); // a valid track might be necessary for el->read() to work
348  el->read(e);
349 
350  Measure* m = tick2measure(e.tick());
351  Segment* seg = m->undoGetSegment(SegmentType::ChordRest, e.tick());
352  el->setParent(seg);
353 
354  // be sure to paste the element in the destination track;
355  // setting track needs to be repeated, as it might have been overwritten by el->read()
356  // preserve *voice* from source, though
357  el->setTrack((e.track() / VOICES) * VOICES + el->voice());
358 
359  undoAddElement(el);
360  }
361  else if (tag == "Clef") {
362  Clef* clef = new Clef(this);
363  clef->read(e);
364  clef->setTrack(e.track());
365  int tick = e.tick();
366  Measure* m = tick2measure(tick);
367  if (m->tick() && m->tick() == tick)
368  m = m->prevMeasure();
369  Segment* segment = m->undoGetSegment(SegmentType::Clef, tick);
370  clef->setParent(segment);
371  undoChangeElement(segment->element(e.track()), clef);
372  }
373  else if (tag == "Breath") {
374  Breath* breath = new Breath(this);
375  breath->read(e);
376  breath->setTrack(e.track());
377  int tick = e.tick();
378  Measure* m = tick2measure(tick);
379  Segment* segment = m->undoGetSegment(SegmentType::Breath, tick);
380  breath->setParent(segment);
381  undoChangeElement(segment->element(e.track()), breath);
382  }
383  else if (tag == "Beam") {
384  Beam* beam = new Beam(this);
385  beam->setTrack(e.track());
386  beam->read(e);
387  beam->setParent(0);
388  e.addBeam(beam);
389  }
390  else if (tag == "BarLine") {
391  e.skipCurrentElement(); // ignore bar line
392  }
393  else {
394  qDebug("PasteStaff: element %s not handled", tag.toUtf8().data());
395  e.skipCurrentElement(); // ignore
396  }
397  }
398 
399  for (Tuplet* tuplet : e.tuplets()) {
400  Q_ASSERT(!tuplet->elements().empty());
401  Measure* measure = tick2measure(tuplet->tick());
402  tuplet->setParent(measure);
403  tuplet->sortElements();
404  }
405  }
406  }
407  for (Score* s : scoreList()) // for all parts
408  s->connectTies();
409 
410  if (pasted) { //select only if we pasted something
411 //TODO? if (styleB(StyleIdx::createMultiMeasureRests))
412 // createMMRests();
413  Segment* s1 = tick2segmentMM(dstTick);
414  Segment* s2 = tick2segmentMM(dstTick + tickLen);
415  int endStaff = dstStaff + staves;
416  if (endStaff > nstaves())
417  endStaff = nstaves();
418  //check and add truly invisible rests insted of gaps
419  //TODO: look if this could be done different
420  Measure* dstM = tick2measureMM(dstTick);
421  Measure* endM = tick2measureMM(dstTick + tickLen);
422  for (int i = dstStaff; i < endStaff; i++) {
423  for (Measure* m = dstM; m && m != endM->nextMeasureMM(); m = m->nextMeasureMM())
424  m->checkMeasure(i);
425  }
426  _selection.setRange(s1, s2, dstStaff, endStaff);
427  _selection.updateSelectedElements();
428 
429  //finding the first element that has a track
430  //the canvas position will be set to this element
431  Element* e = 0;
432  Segment* s = s1;
433  bool found = false;
434  if (s2)
435  s2 = s2->next1MM();
436  while (!found && s != s2) {
437  for (int i = dstStaff * VOICES; i < (endStaff + 1) * VOICES; i++) {
438  e = s->element(i);
439  if (e) {
440  found = true;
441  break;
442  }
443  }
444  s = s->next1MM();
445  }
446 
447  for (MuseScoreView* v : viewer)
448  v->adjustCanvasPosition(e, false);
449  if (!selection().isRange())
450  _selection.setState(SelState::RANGE);
451  }
452  return true;
453  }

459 void Score::pasteChordRest(ChordRest* cr, int tick, const Interval& srcTranspose)
460  {
461 // qDebug("pasteChordRest %s at %d, len %d/%d", cr->name(), tick, cr->duration().numerator(), cr->duration().denominator() );
462  if (cr->isChord())
463  transposeChord(toChord(cr), srcTranspose, tick);
464 
465  Measure* measure = tick2measure(tick);
466  if (!measure)
467  return;
468 
469  // we can paste a measure rest as such only at start of measure
470  // and only if the lengths of the rest and measure match
471  // otherwise, we need to convert to duration rest(s)
472  // and potentially split the rest up (eg, 5/4 => whole + quarter)
473  bool convertMeasureRest = cr->isRest() && cr->durationType().type() == TDuration::DurationType::V_MEASURE
474  && (tick != measure->tick() || cr->duration() != measure->len());
475 
476  int measureEnd = measure->endTick();
477  bool isGrace = cr->isChord() && toChord(cr)->noteType() != NoteType::NORMAL;
478 
479  // if note is too long to fit in measure, split it up with a tie across the barline
480  // exclude tuplets from consideration
481  // we have already disallowed a tuplet from crossing the barline, so there is no problem here
482  // but due to rounding, it might appear from actualTicks() that the last note is too long by a couple of ticks
483 
484  if (!isGrace && !cr->tuplet() && (tick + cr->actualTicks() > measureEnd || convertMeasureRest)) {
485  if (cr->isChord()) {
486  // split Chord
487  Chord* c = toChord(cr);
488  int rest = c->actualTicks();
489  bool firstpart = true;
490  while (rest) {
491  measure = tick2measure(tick);
492  Chord* c2 = firstpart ? c : toChord(c->clone());
493  if (!firstpart)
494  c2->removeMarkings(true);
495  int mlen = measure->tick() + measure->ticks() - tick;
496  int len = mlen > rest ? rest : mlen;
497  std::vector<TDuration> dl = toDurationList(Fraction::fromTicks(len), true);
498  TDuration d = dl[0];
499  c2->setDurationType(d);
500  c2->setDuration(d.fraction());
501  rest -= c2->actualTicks();
502  undoAddCR(c2, measure, tick);
503 
504  std::vector<Note*> nl1 = c->notes();
505  std::vector<Note*> nl2 = c2->notes();
506 
507  if (!firstpart)
508  for (unsigned i = 0; i < nl1.size(); ++i) {
509  Tie* tie = new Tie(this);
510  tie->setStartNote(nl1[i]);
511  tie->setEndNote(nl2[i]);
512  tie->setTrack(c->track());
513  Tie* tie2 = nl1[i]->tieFor();
514  if (tie2) {
515  nl2[i]->setTieFor(nl1[i]->tieFor());
516  tie2->setStartNote(nl2[i]);
517  }
518  nl1[i]->setTieFor(tie);
519  nl2[i]->setTieBack(tie);
520  }
521  c = c2;
522  firstpart = false;
523  tick += c->actualTicks();
524  }
525  }
526  else if (cr->isRest()) {
527  // split Rest
528  Rest* r = toRest(cr);
529  Fraction rest = r->duration();
530 
531  bool firstpart = true;
532  while (!rest.isZero()) {
533  Rest* r2 = firstpart ? r : toRest(r->clone());
534  measure = tick2measure(tick);
535  Fraction mlen = Fraction::fromTicks(measure->tick() + measure->ticks() - tick);
536  Fraction len = rest > mlen ? mlen : rest;
537  std::vector<TDuration> dl = toDurationList(len, false);
538  TDuration d = dl[0];
539  r2->setDuration(d.fraction());
540  r2->setDurationType(d);
541  undoAddCR(r2, measure, tick);
542  rest -= d.fraction();
543  tick += r2->actualTicks();
544  firstpart = false;
545  }
546  }
547  else if (cr->isRepeatMeasure()) {
548  RepeatMeasure* rm = toRepeatMeasure(cr);
549  std::vector<TDuration> list = toDurationList(rm->actualDuration(), true);
550  for (auto dur : list) {
551  Rest* r = new Rest(this, dur);
552  r->setTrack(cr->track());
553  Fraction rest = r->duration();
554  while (!rest.isZero()) {
555  Rest* r2 = toRest(r->clone());
556  measure = tick2measure(tick);
557  Fraction mlen = Fraction::fromTicks(measure->tick() + measure->ticks() - tick);
558  Fraction len = rest > mlen ? mlen : rest;
559  std::vector<TDuration> dl = toDurationList(len, false);
560  TDuration d = dl[0];
561  r2->setDuration(d.fraction());
562  r2->setDurationType(d);
563  undoAddCR(r2, measure, tick);
564  rest -= d.fraction();
565  tick += r2->actualTicks();
566  }
567  delete r;
568  }
569  delete cr;
570  }
571  }
572  else {
573  undoAddCR(cr, measure, tick);
574  }
575  }

578 //---------------------------------------------------------
579 // pasteSymbols
580 //
581 // pastes a list of symbols into cr and following ChordRest's
582 //
583 // (Note: info about delta ticks is currently ignored)
584 //---------------------------------------------------------
585 
586 void Score::pasteSymbols(XmlReader& e, ChordRest* dst)
587  {
588  Segment* currSegm = dst->segment();
589  int destTick = 0; // the tick and track to place the pasted element at
590  int destTrack = 0;
591  bool done = false;
592  int segDelta = 0;
593  Segment* startSegm= currSegm;
594  int startTick = dst->tick(); // the initial tick and track where to start pasting
595  int startTrack = dst->track();
596  int maxTrack = ntracks();
597  int lastTick = lastSegment()->tick();
598 
599  while (e.readNextStartElement()) {
600  if (done)
601  break;
602  if (e.name() != "SymbolList") {
603  e.unknown();
604  break;
605  }
606  QString version = e.attribute("version", "NONE");
607  if (version != MSC_VERSION)
608  break;
609 
610  while (e.readNextStartElement()) {
611  if (done)
612  break;
613  const QStringRef& tag(e.name());
614 
615  if (tag == "trackOffset") {
616  destTrack = startTrack + e.readInt();
617  currSegm = startSegm;
618  }
619  else if (tag == "tickOffset")
620  destTick = startTick + e.readInt();
621  else if (tag == "segDelta")
622  segDelta = e.readInt();
623  else {
624 
625  if (tag == "Harmony") {
626  //
627  // Harmony elements (= chord symbols) are positioned respecting
628  // the original tickOffset: advance to destTick (or near)
629  //
630  Segment* harmSegm;
631  for (harmSegm = startSegm; harmSegm && (harmSegm->tick() < destTick);
632  harmSegm = harmSegm->nextCR())
633  ;
634  // if destTick overshot, no dest. segment: create one
635  if (destTick >= lastTick) {
636  harmSegm = nullptr;
637  }
638  else if (!harmSegm || harmSegm->tick() > destTick) {
639  Measure* meas = tick2measure(destTick);
640  harmSegm = meas ? meas->undoGetSegment(SegmentType::ChordRest, destTick) : nullptr;
641  }
642  if (destTrack >= maxTrack || harmSegm == nullptr) {
643  qDebug("PasteSymbols: no track or segment for %s", tag.toUtf8().data());
644  e.skipCurrentElement(); // ignore
645  continue;
646  }
647  Harmony* el = new Harmony(this);
648  el->setTrack(trackZeroVoice(destTrack));
649  el->read(e);

693  else if (tag == "FiguredBass") {

699  el->read(e);

718  }

720  // if there is a f.b. element in the right track,

723  if (a->isFiguredBass() && a->track() == destTrack) {

725  done = true;
726  }
727  }

733  }

749  continue;
750  }
751  undoAddElement(nextSegm);
752  }
753  currSegm = nextSegm;
754  }
755  else
756  // by default, assign to FiguredBass element the duration of the chord it refers to
757  ticks = toChordRest(currSegm->element(destTrack))->duration().ticks();
758  // in both cases, look for an existing f.b. element in segment and remove it, if found
759  FiguredBass* oldFB = nullptr;
760  foreach (Element* a, currSegm->annotations()) {
761  if (a->isFiguredBass() && a->track() == destTrack) {
762  oldFB = toFiguredBass(a);
763  break;
764  }
765  }
766  if (oldFB)

768  el->setParent(currSegm);
769  el->setTicks(ticks);
770  undoAddElement(el);
771  }
772  else if (tag == "Lyrics") {
773  // with lyrics, skip rests
774  while (!cr->isChord() && currSegm) {
775  currSegm = currSegm->nextCR(destTrack);
776  if (currSegm)
777  cr = toChordRest(currSegm->element(destTrack));
778  else
779  break;
780  }
781  if (currSegm == nullptr) {
782  qDebug("PasteSymbols: no segment for Lyrics");
783  e.skipCurrentElement();
784  continue;
785  }
786  if (!cr->isChord()) {
787  qDebug("PasteSymbols: can't paste Lyrics to rest");
788  e.skipCurrentElement();
789  continue;
790  }
791  Lyrics* el = new Lyrics(this);
792  el->setTrack(destTrack);
793  el->read(e);
794  el->setTrack(destTrack);
795  el->setParent(cr);
796  undoAddElement(el);
797  }
798  else {
799  qDebug("PasteSymbols: element %s not handled", tag.toUtf8().data());
800  e.skipCurrentElement(); // ignore
801  }
802  } // if !Harmony
803  } // if element
804  } // outer while readNextstartElement()
805  } // inner while readNextstartElement()
