59 FormField::FormField(FormFieldData &dd)
60  : m_formData(&dd)
61 {
62  const int rotation = m_formData->page->getRotate();
63  // reading the coords
64  double left, top, right, bottom;
65  m_formData->fm->getRect(&left, &bottom, &right, &top);
66  // build a normalized transform matrix for this page at 100% scale
67  GfxState gfxState( 72.0, 72.0, m_formData->page->getCropBox(), rotation, gTrue );
68  double * gfxCTM = gfxState.getCTM();
69  double MTX[6];
70  double pageWidth = m_formData->page->getCropWidth();
71  double pageHeight = m_formData->page->getCropHeight();
72  // landscape and seascape page rotation: be sure to use the correct (== rotated) page size
73  if (((rotation / 90) % 2) == 1)
74  qSwap(pageWidth, pageHeight);
75  for ( int i = 0; i < 6; i+=2 )
76  {
77  MTX[i] = gfxCTM[i] / pageWidth;
78  MTX[i+1] = gfxCTM[i+1] / pageHeight;
79  }
80  QPointF topLeft;
81  XPDFReader::transform( MTX, qMin( left, right ), qMax( top, bottom ), topLeft );
82  QPointF bottomRight;
83  XPDFReader::transform( MTX, qMax( left, right ), qMin( top, bottom ), bottomRight );
84  m_formData->box = QRectF(topLeft, QSizeF(bottomRight.x() - topLeft.x(), bottomRight.y() - topLeft.y()));
85 }

186 QString FormFieldButton::caption() const
187 {
188  FormWidgetButton* fwb = static_cast<FormWidgetButton*>(m_formData->fm);
189  QString ret;
190  if (fwb->getButtonType() == formButtonPush)
191  {
192  Dict *dict = m_formData->fm->getObj()->getDict();
193  Object obj1;
194  if (dict->lookup("MK", &obj1)->isDict())
195  {
196  AnnotAppearanceCharacs appearCharacs(obj1.getDict());
197  if (appearCharacs.getNormalCaption())
198  {
199  ret = UnicodeParsedString(appearCharacs.getNormalCaption());
200  }
201  }
202  obj1.free();
203  }
204  else
205  {
206  if (const char *goo = fwb->getOnStr())
207  {
208  ret = QString::fromUtf8(goo);
209  }
210  }
211  return ret;
212 }

226 QList<int> FormFieldButton::siblings() const
227 {
228  FormWidgetButton* fwb = static_cast<FormWidgetButton*>(m_formData->fm);
229  ::FormFieldButton* ffb = static_cast< ::FormFieldButton* >(fwb->getField());
230  if (fwb->getButtonType() == formButtonPush)
231  return QList<int>();
232 
233  QList<int> ret;
234  for (int i = 0; i < ffb->getNumSiblings(); ++i)
235  {
236  ::FormFieldButton* sibling = static_cast< ::FormFieldButton* >(ffb->getSibling(i));
237  for (int j = 0; j < sibling->getNumWidgets(); ++j)
238  {
239  FormWidget *w = sibling->getWidget(j);
240  if (w) ret.append(w->getID());
241  }
242  }
243 
244  return ret;
245 }
