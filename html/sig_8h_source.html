27 enum class BeatType : char {
28  DOWNBEAT, // 1st beat of measure (rtick == 0)
29  COMPOUND_STRESSED, // e.g. eighth-note number 7 in 12/8
30  SIMPLE_STRESSED, // e.g. beat 3 in 4/4
31  COMPOUND_UNSTRESSED, // e.g. eighth-note numbers 4 or 10 in 12/8
32  SIMPLE_UNSTRESSED, // "offbeat" e.g. beat 2 and 4 in 4/4 (i.e. the denominator unit)
33  COMPOUND_SUBBEAT, // e.g. any other eighth-note in 12/8 (i.e. the denominator unit)
34  SUBBEAT // does not fall on a beat
35  };

41 class TimeSigFrac : public Fraction {
42 
43  public:
44  using Fraction::Fraction;
45  constexpr TimeSigFrac(int n = 0, int d = 1) : Fraction(n, d) {}
46  TimeSigFrac(const Fraction& f) : TimeSigFrac(f.numerator(), f.denominator()) {}
47  TimeSigFrac(const TimeSigFrac& f) : TimeSigFrac(f.numerator(), f.denominator()) {}
48 
49  // isCompound? Note: 3/8, 3/16, ... are NOT considered compound.
50  bool isCompound() const { return numerator() > 3 /*&& denominator() >= 8*/ && numerator() % 3 == 0; }
51 
52  // isBeatedCompound? Note: Conductors will beat the simple unit at slow tempos (<60 compound units per minute)
53  // However, the meter is still considered to be compound (at least for our purposes).
54  bool isBeatedCompound(qreal tempo) const { return tempo2beatsPerMinute(tempo) >= 60.0; }
55 
56  int dUnitTicks() const;
57  int ticksPerMeasure() const { return numerator() * dUnitTicks(); }
58 
59  int dUnitsPerBeat() const { return isCompound() ? 3 : 1; }
60  int beatTicks() const { return dUnitTicks() * dUnitsPerBeat(); }
61  int beatsPerMeasure() const { return numerator() / dUnitsPerBeat(); }
62 
63  int subbeatTicks(int level) const;
64  int maxSubbeatLevel() const;
65 
66  bool isTriple() const { return beatsPerMeasure() % 3 == 0; }
67  bool isDuple() const { Q_ASSERT(!isTriple()); return beatsPerMeasure() % 2 == 0; } // note: always test isTriple() first
68 
69  // MuseScore stores tempos in quarter-notes-per-second, so conversions to conventional beats-per-minute format are provided here:
70  qreal tempo2beatsPerMinute(qreal tempo) const { return tempo * denominator() * 15.0 / dUnitsPerBeat(); }
71  qreal beatsPerMinute2tempo(qreal bpm) const { return bpm * dUnitsPerBeat() / (15.0 * denominator()); }
72 
73  BeatType rtick2beatType(int rtick) const;
74  int rtick2subbeatLevel(int rtick) const; // returns negative value if not on a well-defined subbeat
75 
76  BeatType strongestBeatInRange(int rtick1, int rtick2, int* dUnitsCrossed = 0, int* subbeatTick = 0, bool saveLast = false) const; // range is exclusive
77  int strongestSubbeatLevelInRange(int rtick1, int rtick2, int* subbeatTick = 0) const; // range is exclusive
78 
79  int ticksPastDUnit(int rtick) const { return rtick % dUnitTicks(); } // returns 0 if rtick is exactly on a dUnit
80  int ticksToNextDUnit(int rtick) const { return dUnitTicks() - ticksPastDUnit(rtick); } // returns dUnitTicks() if rtick is on a dUnit
81 
82  int ticksPastBeat(int rtick) const { return rtick % beatTicks(); } // returns 0 if rtick is exactly on a beat
83  int ticksToNextBeat(int rtick) const { return beatTicks() - ticksPastBeat(rtick); } // returns beatTicks() if rtick is on a beat
84 
85  int ticksPastSubbeat(int rtick, int level) const { return rtick % subbeatTicks(level); }
86  int ticksToNextSubbeat(int rtick, int level) const { return subbeatTicks(level) - ticksPastSubbeat(rtick, level); }
87  };
