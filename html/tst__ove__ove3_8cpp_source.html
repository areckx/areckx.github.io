118 void TestOveIO::oveReadTest(const char* file)
119  {

121  MasterScore* score = readScore(DIR + file + ".ove");
122  QVERIFY(score);

126  score->update();
127  QVERIFY(saveCompareScore(score, QString("%1.ove.mscx").arg(file),
128  DIR + QString("%1.ove-ref.mscx").arg(file)));
129  delete score;
130  }
