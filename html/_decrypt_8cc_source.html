73 GBool Decrypt::makeFileKey(int encVersion, int encRevision, int keyLength,
74  GooString *ownerKey, GooString *userKey,
75  GooString *ownerEnc, GooString *userEnc,
76  int permissions, GooString *fileID,
77  GooString *ownerPassword, GooString *userPassword,
78  Guchar *fileKey, GBool encryptMetadata,
79  GBool *ownerPasswordOk) {
80  DecryptAES256State state;
81  Guchar test[127 + 56], test2[32];
82  GooString *userPassword2;
83  Guchar fState[256];
84  Guchar tmpKey[16];
85  Guchar fx, fy;
86  int len, i, j;
87 
88  *ownerPasswordOk = gFalse;
89 
90  if (encRevision == 5 || encRevision == 6) {
91 
92  // check the owner password
93  if (ownerPassword) {
94  //~ this is supposed to convert the password to UTF-8 using "SASLprep"
95  len = ownerPassword->getLength();
96  if (len > 127) {
97  len = 127;
98  }
99  memcpy(test, ownerPassword->getCString(), len);
100  memcpy(test + len, ownerKey->getCString() + 32, 8);
101  memcpy(test + len + 8, userKey->getCString(), 48);
102  sha256(test, len + 56, test);
103  if (encRevision == 6) {
104  //test contains the initial SHA-256 hash as input K.
105  revision6Hash(ownerPassword, test, userKey->getCString());
106  }
107  if (!memcmp(test, ownerKey->getCString(), 32)) {
108 
109  // compute the file key from the owner password
110  memcpy(test, ownerPassword->getCString(), len);
111  memcpy(test + len, ownerKey->getCString() + 40, 8);
112  memcpy(test + len + 8, userKey->getCString(), 48);
113  sha256(test, len + 56, test);
114  if (encRevision == 6) {
115  //test contains the initial SHA-256 hash input K.
116  revision6Hash(ownerPassword, test, userKey->getCString());
117  }
118  aes256KeyExpansion(&state, test, 32, gTrue);
119  for (i = 0; i < 16; ++i) {
120  state.cbc[i] = 0;
121  }
122  aes256DecryptBlock(&state, (Guchar *)ownerEnc->getCString(), gFalse);
123  memcpy(fileKey, state.buf, 16);
124  aes256DecryptBlock(&state, (Guchar *)ownerEnc->getCString() + 16,
125  gFalse);
126  memcpy(fileKey + 16, state.buf, 16);
127 
128  *ownerPasswordOk = gTrue;
129  return gTrue;
130  }
131  }
132 
133  // check the user password
134  if (userPassword) {
135  //~ this is supposed to convert the password to UTF-8 using "SASLprep"
136  len = userPassword->getLength();
137  if (len > 127) {
138  len = 127;
139  }
140  memcpy(test, userPassword->getCString(), len);
141  memcpy(test + len, userKey->getCString() + 32, 8);
142  sha256(test, len + 8, test);
143  if(encRevision == 6) {
144  // test contains the initial SHA-256 hash input K.
145  // user key is not used in checking user password.
146  revision6Hash(userPassword, test, NULL);
147  }
148  if (!memcmp(test, userKey->getCString(), 32)) {
149 
150  // compute the file key from the user password
151  memcpy(test, userPassword->getCString(), len);
152  memcpy(test + len, userKey->getCString() + 40, 8);
153  sha256(test, len + 8, test);
154  if(encRevision == 6) {
155  //test contains the initial SHA-256 hash input K.
156  //user key is not used in computing intermediate user key.
157  revision6Hash(userPassword, test, NULL);
158  }
159  aes256KeyExpansion(&state, test, 32, gTrue);
160  for (i = 0; i < 16; ++i) {
161  state.cbc[i] = 0;
162  }
163  aes256DecryptBlock(&state, (Guchar *)userEnc->getCString(), gFalse);
164  memcpy(fileKey, state.buf, 16);
165  aes256DecryptBlock(&state, (Guchar *)userEnc->getCString() + 16,
166  gFalse);
167  memcpy(fileKey + 16, state.buf, 16);
168 
169  return gTrue;
170  }
171  }
172 
173  return gFalse;
174  } else {
175 
176  // try using the supplied owner password to generate the user password
177  if (ownerPassword) {
178  len = ownerPassword->getLength();
179  if (len < 32) {
180  memcpy(test, ownerPassword->getCString(), len);
181  memcpy(test + len, passwordPad, 32 - len);
182  } else {
183  memcpy(test, ownerPassword->getCString(), 32);
184  }
185  md5(test, 32, test);
186  if (encRevision == 3) {
187  for (i = 0; i < 50; ++i) {
188  md5(test, keyLength, test);
189  }
190  }
191  if (encRevision == 2) {
192  rc4InitKey(test, keyLength, fState);
193  fx = fy = 0;
194  for (i = 0; i < 32; ++i) {
195  test2[i] = rc4DecryptByte(fState, &fx, &fy, ownerKey->getChar(i));
196  }
197  } else {
198  memcpy(test2, ownerKey->getCString(), 32);
199  for (i = 19; i >= 0; --i) {
200  for (j = 0; j < keyLength; ++j) {
201  tmpKey[j] = test[j] ^ i;
202  }
203  rc4InitKey(tmpKey, keyLength, fState);
204  fx = fy = 0;
205  for (j = 0; j < 32; ++j) {
206  test2[j] = rc4DecryptByte(fState, &fx, &fy, test2[j]);
207  }
208  }
209  }
210  userPassword2 = new GooString((char *)test2, 32);
211  if (makeFileKey2(encVersion, encRevision, keyLength, ownerKey, userKey,
212  permissions, fileID, userPassword2, fileKey,
213  encryptMetadata)) {
214  *ownerPasswordOk = gTrue;
215  delete userPassword2;
216  return gTrue;
217  }
218  delete userPassword2;
219  }
220 
221  // try using the supplied user password
222  return makeFileKey2(encVersion, encRevision, keyLength, ownerKey, userKey,
223  permissions, fileID, userPassword, fileKey,
224  encryptMetadata);
225  }
226 }

229  GooString *ownerKey, GooString *userKey,
230  int permissions, GooString *fileID,
231  GooString *userPassword, Guchar *fileKey,
232  GBool encryptMetadata) {
233  Guchar *buf;
234  Guchar test[32];
235  Guchar fState[256];
236  Guchar tmpKey[16];
237  Guchar fx, fy;
238  int len, i, j;
239  GBool ok;
240 
241  // generate file key
242  buf = (Guchar *)gmalloc(72 + fileID->getLength());
243  if (userPassword) {
244  len = userPassword->getLength();
245  if (len < 32) {
246  memcpy(buf, userPassword->getCString(), len);
247  memcpy(buf + len, passwordPad, 32 - len);
248  } else {
249  memcpy(buf, userPassword->getCString(), 32);
250  }
251  } else {
252  memcpy(buf, passwordPad, 32);
253  }
254  memcpy(buf + 32, ownerKey->getCString(), 32);
255  buf[64] = permissions & 0xff;
256  buf[65] = (permissions >> 8) & 0xff;
257  buf[66] = (permissions >> 16) & 0xff;
258  buf[67] = (permissions >> 24) & 0xff;
259  memcpy(buf + 68, fileID->getCString(), fileID->getLength());
260  len = 68 + fileID->getLength();
261  if (!encryptMetadata) {
262  buf[len++] = 0xff;
263  buf[len++] = 0xff;
264  buf[len++] = 0xff;
265  buf[len++] = 0xff;
266  }
267  md5(buf, len, fileKey);
268  if (encRevision == 3) {
269  for (i = 0; i < 50; ++i) {
270  md5(fileKey, keyLength, fileKey);
271  }
272  }
273 
274  // test user password
275  if (encRevision == 2) {
276  rc4InitKey(fileKey, keyLength, fState);
277  fx = fy = 0;
278  for (i = 0; i < 32; ++i) {
279  test[i] = rc4DecryptByte(fState, &fx, &fy, userKey->getChar(i));
280  }
281  ok = memcmp(test, passwordPad, 32) == 0;
282  } else if (encRevision == 3) {
283  memcpy(test, userKey->getCString(), 32);
284  for (i = 19; i >= 0; --i) {
285  for (j = 0; j < keyLength; ++j) {
286  tmpKey[j] = fileKey[j] ^ i;
287  }
288  rc4InitKey(tmpKey, keyLength, fState);
289  fx = fy = 0;
290  for (j = 0; j < 32; ++j) {
291  test[j] = rc4DecryptByte(fState, &fx, &fy, test[j]);
292  }
293  }
294  memcpy(buf, passwordPad, 32);
295  memcpy(buf + 32, fileID->getCString(), fileID->getLength());
296  md5(buf, 32 + fileID->getLength(), buf);
297  ok = memcmp(test, buf, 16) == 0;
298  } else {
299  ok = gFalse;
300  }
301 
302  gfree(buf);
303  return ok;
304 }

310 BaseCryptStream::BaseCryptStream(Stream *strA, Guchar *fileKey, CryptAlgorithm algoA,
311  int keyLength, int objNum, int objGen):
312  FilterStream(strA)
313 {
314  int i;
315 
316  algo = algoA;
317 
318  // construct object key
319  for (i = 0; i < keyLength; ++i) {
320  objKey[i] = fileKey[i];
321  }
322  switch (algo) {
323  case cryptRC4:
324  objKey[keyLength] = objNum & 0xff;
325  objKey[keyLength + 1] = (objNum >> 8) & 0xff;
326  objKey[keyLength + 2] = (objNum >> 16) & 0xff;
327  objKey[keyLength + 3] = objGen & 0xff;
328  objKey[keyLength + 4] = (objGen >> 8) & 0xff;
329  md5(objKey, keyLength + 5, objKey);
330  if ((objKeyLength = keyLength + 5) > 16) {
331  objKeyLength = 16;
332  }
333  break;
334  case cryptAES:
335  objKey[keyLength] = objNum & 0xff;
336  objKey[keyLength + 1] = (objNum >> 8) & 0xff;
337  objKey[keyLength + 2] = (objNum >> 16) & 0xff;
338  objKey[keyLength + 3] = objGen & 0xff;
339  objKey[keyLength + 4] = (objGen >> 8) & 0xff;

343  objKey[keyLength + 8] = 0x54; // 'T'

439 int EncryptStream::lookChar() {
440  Guchar in[16];
441  int c;
442 
443  if (nextCharBuff != EOF)
444  return nextCharBuff;
445 
446  c = EOF; // make gcc happy
447  switch (algo) {
448  case cryptRC4:
449  if ((c = str->getChar()) != EOF) {
450  // RC4 is XOR-based: the decryption algorithm works for encryption too
451  c = rc4DecryptByte(state.rc4.state, &state.rc4.x, &state.rc4.y, (Guchar)c);
452  }
453  break;
454  case cryptAES:
455  if (state.aes.bufIdx == 16 && !state.aes.paddingReached) {
456  state.aes.paddingReached = !aesReadBlock(str, in, gTrue);
457  aesEncryptBlock(&state.aes, in);
458  }
459  if (state.aes.bufIdx == 16) {
460  c = EOF;
461  } else {
462  c = state.aes.buf[state.aes.bufIdx++];
463  }
464  break;
465  case cryptAES256:
466  if (state.aes256.bufIdx == 16 && !state.aes256.paddingReached) {
467  state.aes256.paddingReached = !aesReadBlock(str, in, gTrue);
468  aes256EncryptBlock(&state.aes256, in);
469  }
470  if (state.aes256.bufIdx == 16) {
471  c = EOF;
472  } else {
473  c = state.aes256.buf[state.aes256.bufIdx++];
474  }
475  break;
476  }
477  return (nextCharBuff = c);
478 }

565 static void rc4InitKey(Guchar *key, int keyLen, Guchar *state) {
566  Guchar index1, index2;
567  Guchar t;
568  int i;
569 
570  for (i = 0; i < 256; ++i)
571  state[i] = i;
572 
573  if (unlikely(keyLen == 0))
574  return;
575 
576  index1 = index2 = 0;
577  for (i = 0; i < 256; ++i) {
578  index2 = (key[index1] + state[i] + index2) % 256;
579  t = state[i];
580  state[i] = state[index2];
581  state[index2] = t;
582  index1 = (index1 + 1) % keyLen;
583  }
584 }

586 static Guchar rc4DecryptByte(Guchar *state, Guchar *x, Guchar *y, Guchar c) {
587  Guchar x1, y1, tx, ty;
588 
589  x1 = *x = (*x + 1) % 256;
590  y1 = *y = (state[*x] + *y) % 256;
591  tx = state[x1];
592  ty = state[y1];
593  state[x1] = ty;
594  state[y1] = tx;
595  return c ^ state[(tx + ty) % 256];
596 }

602 // Returns gFalse if EOF was reached, gTrue otherwise
603 static GBool aesReadBlock(Stream *str, Guchar *in, GBool addPadding)
604 {
605  int c, i;
606 
607  for (i = 0; i < 16; ++i) {
608  if ((c = str->getChar()) != EOF) {
609  in[i] = (Guchar)c;
610  } else {
611  break;
612  }
613  }
614 
615  if (i == 16) {
616  return gTrue;
617  } else {
618  if (addPadding) {
619  c = 16 - i;
620  while (i < 16) {
621  in[i++] = (Guchar)c;
622  }
623  }
624  return gFalse;
625  }
626 }
