101  FT_CALLBACK_DEF( FT_UInt )
102  pcf_cmap_char_index( FT_CMap pcfcmap, /* PCF_CMap */
103  FT_UInt32 charcode )
104  {
105  PCF_CMap cmap = (PCF_CMap)pcfcmap;
106  PCF_Encoding encodings = cmap->encodings;
107  FT_ULong min, max, mid;
108  FT_UInt result = 0;
109 
110 
111  min = 0;
112  max = cmap->num_encodings;
113 
114  while ( min < max )
115  {
116  FT_ULong code;
117 
118 
119  mid = ( min + max ) >> 1;
120  code = (FT_ULong)encodings[mid].enc;
121 
122  if ( charcode == code )
123  {
124  result = encodings[mid].glyph + 1;
125  break;
126  }
127 
128  if ( charcode < code )
129  max = mid;
130  else
131  min = mid + 1;
132  }
133 
134  return result;
135  }

480  FT_CALLBACK_DEF( FT_Error )
481  PCF_Glyph_Load( FT_GlyphSlot slot,
482  FT_Size size,
483  FT_UInt glyph_index,
484  FT_Int32 load_flags )
485  {
486  PCF_Face face = (PCF_Face)FT_SIZE_FACE( size );
487  FT_Stream stream;
488  FT_Error error = FT_Err_Ok;
489  FT_Bitmap* bitmap = &slot->bitmap;
490  PCF_Metric metric;
491  FT_ULong bytes;
492 
493  FT_UNUSED( load_flags );
494 
495 
496  FT_TRACE1(( "PCF_Glyph_Load: glyph index %d\n", glyph_index ));
497 
498  if ( !face )
499  {
500  error = FT_THROW( Invalid_Face_Handle );
501  goto Exit;
502  }
503 
504  if ( glyph_index >= (FT_UInt)face->root.num_glyphs )
505  {
506  error = FT_THROW( Invalid_Argument );
507  goto Exit;
508  }
509 
510  stream = face->root.stream;
511 
512  if ( glyph_index > 0 )
513  glyph_index--;
514 
515  metric = face->metrics + glyph_index;
516 
517  bitmap->rows = (unsigned int)( metric->ascent +
518  metric->descent );
519  bitmap->width = (unsigned int)( metric->rightSideBearing -
520  metric->leftSideBearing );
521  bitmap->num_grays = 1;
522  bitmap->pixel_mode = FT_PIXEL_MODE_MONO;
523 
524  FT_TRACE6(( "BIT_ORDER %d ; BYTE_ORDER %d ; GLYPH_PAD %d\n",
525  PCF_BIT_ORDER( face->bitmapsFormat ),
526  PCF_BYTE_ORDER( face->bitmapsFormat ),
527  PCF_GLYPH_PAD( face->bitmapsFormat ) ));
528 
529  switch ( PCF_GLYPH_PAD( face->bitmapsFormat ) )
530  {
531  case 1:
532  bitmap->pitch = (int)( ( bitmap->width + 7 ) >> 3 );
533  break;
534 
535  case 2:
536  bitmap->pitch = (int)( ( ( bitmap->width + 15 ) >> 4 ) << 1 );
537  break;
538 
539  case 4:
540  bitmap->pitch = (int)( ( ( bitmap->width + 31 ) >> 5 ) << 2 );
541  break;
542 
543  case 8:
544  bitmap->pitch = (int)( ( ( bitmap->width + 63 ) >> 6 ) << 3 );
545  break;
546 
547  default:
548  return FT_THROW( Invalid_File_Format );
549  }
550 
551  /* XXX: to do: are there cases that need repadding the bitmap? */
552  bytes = (FT_ULong)bitmap->pitch * bitmap->rows;
553 
554  error = ft_glyphslot_alloc_bitmap( slot, (FT_ULong)bytes );
555  if ( error )
556  goto Exit;
557 
558  if ( FT_STREAM_SEEK( metric->bits ) ||
559  FT_STREAM_READ( bitmap->buffer, bytes ) )
560  goto Exit;
561 
562  if ( PCF_BIT_ORDER( face->bitmapsFormat ) != MSBFirst )
563  BitOrderInvert( bitmap->buffer, bytes );
564 
565  if ( ( PCF_BYTE_ORDER( face->bitmapsFormat ) !=
566  PCF_BIT_ORDER( face->bitmapsFormat ) ) )
567  {
568  switch ( PCF_SCAN_UNIT( face->bitmapsFormat ) )
569  {
570  case 1:
571  break;
572 
573  case 2:
574  TwoByteSwap( bitmap->buffer, bytes );
575  break;
576 
577  case 4:
578  FourByteSwap( bitmap->buffer, bytes );
579  break;
580  }
581  }
582 
583  slot->format = FT_GLYPH_FORMAT_BITMAP;
584  slot->bitmap_left = metric->leftSideBearing;
585  slot->bitmap_top = metric->ascent;
586 
587  slot->metrics.horiAdvance = (FT_Pos)( metric->characterWidth * 64 );
588  slot->metrics.horiBearingX = (FT_Pos)( metric->leftSideBearing * 64 );
589  slot->metrics.horiBearingY = (FT_Pos)( metric->ascent * 64 );
590  slot->metrics.width = (FT_Pos)( ( metric->rightSideBearing -
591  metric->leftSideBearing ) * 64 );
592  slot->metrics.height = (FT_Pos)( bitmap->rows * 64 );
593 
594  ft_synthesize_vertical_metrics( &slot->metrics,
595  ( face->accel.fontAscent +
596  face->accel.fontDescent ) * 64 );
597 
598  Exit:
599  return error;
600  }
