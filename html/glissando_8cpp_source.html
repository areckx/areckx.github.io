65 void GlissandoSegment::draw(QPainter* painter) const
66  {
67  painter->save();
68  //qreal _spatium = spatium();
69 
70  QPen pen(glissando()->curColor());
71  pen.setWidthF(glissando()->lineWidth().val() * spatium());
72  pen.setCapStyle(Qt::RoundCap);
73  painter->setPen(pen);
74 // painter->drawLine(QPointF(), pos2()); // DEBUG
75 
76  // rotate painter so that the line become horizontal
77  qreal w = pos2().x();
78  qreal h = pos2().y();
79  qreal l = sqrt(w * w + h * h);
80  qreal wi = asin(-h / l) * 180.0 / M_PI;
81  qreal scale = painter->worldTransform().m11();
82  painter->rotate(-wi);
83 
84  if (glissando()->glissandoType() == Glissando::Type::STRAIGHT) {
85  painter->drawLine(QLineF(0.0, 0.0, l, 0.0));
86  }
87  else if (glissando()->glissandoType() == Glissando::Type::WAVY) {
88  QRectF b = symBbox(SymId::wiggleTrill);
89  qreal w = symAdvance(SymId::wiggleTrill);
90  int n = (int)(l / w); // always round down (truncate) to avoid overlap
91  qreal x = (l - n*w) * 0.5; // centre line in available space
92  std::vector<SymId> ids;
93  for (int i = 0; i < n; ++i)
94  ids.push_back(SymId::wiggleTrill);
95  // this is very ugly but fix #68846 for now
96 // bool tmp = MScore::pdfPrinting;
97 // MScore::pdfPrinting = true;
98  score()->scoreFont()->draw(ids, painter, magS(), QPointF(x, -(b.y() + b.height()*0.5) ), scale);
99 // MScore::pdfPrinting = tmp;
100  }
101 #if 0 // TODO
102  if (glissando()->showText()) {
103  const TextStyle& st = score()->textStyle(StyledPropertyListIdx::GLISSANDO);
104  QRectF r = st.fontMetrics(_spatium).boundingRect(glissando()->text());
105  // if text longer than available space, skip it
106  if (r.width() < l) {
107  qreal yOffset = r.height() + r.y(); // find text descender height
108  // raise text slightly above line and slightly more with WAVY than with STRAIGHT
109  yOffset += _spatium * (glissando()->glissandoType() == Glissando::Type::WAVY ? 0.4 : 0.1);
110  painter->setFont(st.font(_spatium * MScore::pixelRatio));
111  qreal x = (l - r.width()) * 0.5;
112  painter->drawText(QPointF(x, -yOffset), glissando()->text());
113  }
114  }
115 #endif
116  painter->restore();
117  }

232 
233 void Glissando::layout()
234  {
235  qreal _spatium = spatium();
236 
237  if (score() == gscore || !startElement() || !endElement()) { // for use in palettes or while dragging
238  if (spannerSegments().empty())
239  add(createLineSegment());
240  LineSegment* s = frontSegment();
241  s->setPos(QPointF());
242  s->setPos2(QPointF(_spatium * GLISS_PALETTE_WIDTH, -_spatium * GLISS_PALETTE_HEIGHT));
243  s->layout();
244  return;
245  }
246 
247  SLine::layout();
248  setPos(0.0, 0.0);
249  adjustReadPos();
250 
251  Note* anchor1 = static_cast<Note*>(startElement());
252  Note* anchor2 = static_cast<Note*>(endElement());
253  Chord* cr1 = anchor1->chord();
254  Chord* cr2 = anchor2->chord();
255  GlissandoSegment* segm1 = static_cast<GlissandoSegment*>(frontSegment());
256  GlissandoSegment* segm2 = static_cast<GlissandoSegment*>(backSegment());
257 
258  // Note: line segments are defined by
259  // initial point: ipos() (relative to system origin)
260  // ending point: pos2() (relative to initial point)
261 
262  // LINE ENDING POINTS TO NOTEHEAD CENTRES
263 
264  // assume gliss. line goes from centre of initial note centre to centre of ending note:
265  // move first segment origin and last segment ending point from notehead origin to notehead centre
266  QPointF offs1 = QPointF(anchor1->headWidth() * 0.5, 0.0);
267  QPointF offs2 = QPointF(anchor2->headWidth() * 0.5, 0.0);
268 
269  // AVOID HORIZONTAL LINES
270 
271  int upDown = (0 < (anchor2->pitch() - anchor1->pitch())) - ((anchor2->pitch() - anchor1->pitch()) < 0);
272  // on TAB's, glissando are by necessity on the same string, this gives an horizontal glissando line;
273  // make bottom end point lower and top ending point higher
274  if (cr1->staff()->isTabStaff(cr1->tick())) {
275  qreal yOff = cr1->staff()->lineDistance(cr1->tick()) * 0.4 * _spatium;
276  offs1.ry() += yOff * upDown;
277  offs2.ry() -= yOff * upDown;
278  }
279  // if not TAB, angle glissando between notes on the same line
280  else {
281  if (anchor1->line() == anchor2->line()) {
282  offs1.ry() += _spatium * 0.25 * upDown;
283  offs2.ry() -= _spatium * 0.25 * upDown;
284  }
285  }
286 
287  // move initial point of first segment and adjust its length accordingly
288  segm1->setPos (segm1->ipos() + offs1);
289  segm1->setPos2(segm1->ipos2() - offs1);
290  // adjust ending point of last segment
291  segm2->setPos2(segm2->ipos2() + offs2);
292 
293  // FINAL SYSTEM-INITIAL NOTE
294  // if the last gliss. segment attaches to a system-initial note, some extra width has to be added
295  if (cr2->segment()->measure() == cr2->segment()->system()->firstMeasure() && cr2->rtick() == 0
296  // but ignore graces after, as they are not the first note of the system,
297  // even if their segment is the first segment of the system
298  && !(cr2->noteType() == NoteType::GRACE8_AFTER
299  || cr2->noteType() == NoteType::GRACE16_AFTER || cr2->noteType() == NoteType::GRACE32_AFTER)
300  // also ignore if cr1 is a child of cr2, which means cr1 is a grace-before of cr2
301  && !(cr1->parent() == cr2))
302  {
303  segm2->rxpos() -= GLISS_STARTOFSYSTEM_WIDTH * _spatium;
304  segm2->rxpos2()+= GLISS_STARTOFSYSTEM_WIDTH * _spatium;
305  }
306 
307  // INTERPOLATION OF INTERMEDIATE POINTS
308  // This probably belongs to SLine class itself; currently it does not seem
309  // to be needed for anything else than Glissando, though
310 
311  // get total x-width and total y-height of all segments
312  qreal xTot = 0.0;
313  for (SpannerSegment* segm : spannerSegments())
314  xTot += segm->ipos2().x();
315  qreal y0 = segm1->ipos().y();
316  qreal yTot = segm2->ipos().y() + segm2->ipos2().y() - y0;
317  qreal ratio = yTot / xTot;
318  // interpolate y-coord of intermediate points across total width and height
319  qreal xCurr = 0.0;
320  qreal yCurr;
321  for (int i = 0; i < spannerSegments().count()-1; i++) {
322  SpannerSegment* segm = segmentAt(i);
323  xCurr += segm->ipos2().x();
324  yCurr = y0 + ratio * xCurr;
325  segm->rypos2() = yCurr - segm->ipos().y(); // position segm. end point at yCurr
326  // next segment shall start where this segment stopped
327  segm = segmentAt(i+1);
328  segm->rypos2() += segm->ipos().y() - yCurr; // adjust next segm. vertical length
329  segm->rypos() = yCurr; // position next segm. start point at yCurr
330  }
331 
332  // STAY CLEAR OF NOTE APPENDAGES
333 
334  // initial note dots / ledger line / notehead
335  offs1 *= -1.0; // discount changes already applied
336  int dots = cr1->dots();
337  LedgerLine * ledLin = cr1->ledgerLines();
338  // if dots, start at right of last dot
339  // if no dots, from right of ledger line, if any; from right of notehead, if no ledger line
340  offs1.rx() += (dots && anchor1->dot(dots-1) ? anchor1->dot(dots-1)->pos().x() + anchor1->dot(dots-1)->width()
341  : (ledLin ? ledLin->pos().x() + ledLin->width() : anchor1->headWidth()) );
342 
343  // final note arpeggio / accidental / ledger line / accidental / arpeggio (i.e. from outermost to innermost)
344  offs2 *= -1.0; // discount changes already applied
345  if (Arpeggio* a = cr2->arpeggio())
346  offs2.rx() += a->pos().x() + a->userOff().x();
347  else if (Accidental* a = anchor2->accidental())
348  offs2.rx() += a->pos().x() + a->userOff().x();
349  else if ( (ledLin = cr2->ledgerLines()) != nullptr)
350  offs2.rx() += ledLin->pos().x();
351 
352  // add another a quarter spatium of 'air'
353  offs1.rx() += _spatium * 0.25;
354  offs2.rx() -= _spatium * 0.25;
355 
356  // apply offsets: shorten first segment by x1 (and proportionally y) and adjust its length accordingly
357  offs1.ry() = segm1->ipos2().y() * offs1.x() / segm1->ipos2().x();
358  segm1->setPos(segm1->ipos() + offs1);
359  segm1->setPos2(segm1->ipos2() - offs1);
360  // adjust last segment length by x2 (and proportionally y)
361  offs2.ry() = segm2->ipos2().y() * offs2.x() / segm2->ipos2().x();
362  segm2->setPos2(segm2->ipos2() + offs2);
363 
364  for (SpannerSegment* segm : spannerSegments())
365  static_cast<GlissandoSegment*>(segm)->layout();
366 
367  // compute glissando bbox as the bbox of the last segment, relative to the end anchor note
368  QPointF anchor2PagePos = anchor2->pagePos();
369  QPointF system2PagePos = cr2->segment()->system()->pagePos();
370  QPointF anchor2SystPos = anchor2PagePos - system2PagePos;
371  QRectF r = QRectF(anchor2SystPos - segm2->pos(), anchor2SystPos - segm2->pos() - segm2->pos2()).normalized();
372  qreal lw = _spatium * lineWidth().val() * .5;
373  setbbox(r.adjusted(-lw, -lw, lw, lw));
374  }

398 void Glissando::read(XmlReader& e)
399  {
400  qDeleteAll(spannerSegments());
401  spannerSegments().clear();
402  e.addSpanner(e.intAttribute("id", -1), this);
403 
404  _showText = false;
405  while (e.readNextStartElement()) {
406  const QStringRef& tag = e.name();
407  if (tag == "text") {
408  _showText = true;
409  _text = e.readElementText();
410  }
411  else if (tag == "subtype")
412  _glissandoType = Type(e.readInt());
413  else if (tag == "glissandoStyle") {
414  setProperty(P_ID::GLISSANDO_STYLE, Ms::getProperty(P_ID::GLISSANDO_STYLE, e));
415  } else if ( tag == "play") {
416  setPlayGlissando(e.readBool());
417  }
418  else if (!SLine::readProperties(e))
419  e.unknown();
420  }
421  }

496 //---------------------------------------------------------
497 // STATIC FUNCTIONS: guessInitialNote
498 //
499 // Used while reading old scores (either 1.x or transitional 2.0) to determine (guess!)
500 // the glissando initial note from its final chord. Returns the top note of previous chord
501 // of the same instrument, preferring the chord in the same track as chord, if it exists.
502 //
503 // CANNOT be called if the final chord and/or its segment do not exist yet in the score
504 //
505 // Parameter: chord: the chord this glissando ends into
506 // Returns: the top note in a suitable previous chord or nullptr if none found.
507 //---------------------------------------------------------
508 
509 Note* Glissando::guessInitialNote(Chord* chord)
510  {
511  switch (chord->noteType()) {
512 // case NoteType::INVALID:
513 // return nullptr;
514  // for grace notes before, previous chord is previous chord of parent chord
515  case NoteType::ACCIACCATURA:
516  case NoteType::APPOGGIATURA:
517  case NoteType::GRACE4:
518  case NoteType::GRACE16:
519  case NoteType::GRACE32:
520  // move unto parent chord and proceed to standard case
521  if (chord->parent() && chord->parent()->type() == ElementType::CHORD)
522  chord = static_cast<Chord*>(chord->parent());
523  else
524  return nullptr;
525  break;
526  // for grace notes after, return top note of parent chord
527  case NoteType::GRACE8_AFTER:
528  case NoteType::GRACE16_AFTER:
529  case NoteType::GRACE32_AFTER:
530  if (chord->parent() && chord->parent()->type() == ElementType::CHORD)
531  return static_cast<Chord*>(chord->parent())->upNote();
532  else // no parent or parent is not a chord?
533  return nullptr;
534  case NoteType::NORMAL:
535  {
536  // if chord has grace notes before, the last one is the previous note
537  QVector<Chord*>graces = chord->graceNotesBefore();
538  if (graces.size() > 0)
539  return graces.last()->upNote();
540  }
541  break; // else process to standard case
542  default:
543  break;
544  }
545 
546  // standard case (NORMAL or grace before chord)
547 
548  // if parent not a segment, can't locate a target note
549  if (chord->parent()->type() != ElementType::SEGMENT)
550  return nullptr;
551 
552  int chordTrack = chord->track();
553  Segment* segm = chord->segment();
554  Part* part = chord->part();
555  if (segm != nullptr)
556  segm = segm->prev1();
557  while (segm) {
558  // if previous segment is a ChordRest segment
559  if (segm->segmentType() == SegmentType::ChordRest) {
560  Chord* target = nullptr;
561  // look for a Chord in the same track
562  if (segm->element(chordTrack) && segm->element(chordTrack)->type() == ElementType::CHORD)
563  target = static_cast<Chord*>(segm->element(chordTrack));
564  else // if no same track, look for other chords in the same instrument
565  for (Element* currChord : segm->elist())
566  if (currChord != nullptr && currChord->type() == ElementType::CHORD
567  && static_cast<Chord*>(currChord)->part() == part) {
568  target = static_cast<Chord*>(currChord);
569  break;
570  }
571  // if we found a target previous chord
572  if (target) {
573  // if chord has grace notes after, the last one is the previous note
574  QVector<Chord*>graces = target->graceNotesAfter();
575  if (graces.size() > 0)
576  return graces.last()->upNote();
577  return target->upNote(); // if no grace after, return top note
578  }
579  }
580  segm = segm->prev1();
581  }
582  qDebug("no first note for glissando found");
583  return nullptr;
584  }

586 //---------------------------------------------------------
587 // STATIC FUNCTIONS: guessFinalNote
588 //
589 // Used while dropping a glissando on a note to determine (guess!) the glissando final
590 // note from its initial chord.
591 // Returns the top note of next chord of the same instrument,
592 // preferring the chord in the same track as chord, if it exists.
593 //
594 // Parameter: chord: the chord this glissando start from
595 // Returns: the top note in a suitable following chord or nullptr if none found
596 //---------------------------------------------------------
597 
598 Note* Glissando::guessFinalNote(Chord* chord)
599  {
600  switch (chord->noteType()) {
601 // case NoteType::INVALID:
602 // return nullptr;
603  // for grace notes before, return top note of parent chord
604  // TODO : if the grace-before is not the LAST ONE, this still returns the main note
605  // which is probably not correct; however a glissando between two grace notes
606  // probably makes little sense.
607  case NoteType::ACCIACCATURA:
608  case NoteType::APPOGGIATURA:
609  case NoteType::GRACE4:
610  case NoteType::GRACE16:
611  case NoteType::GRACE32:
612  if (chord->parent() && chord->parent()->type() == ElementType::CHORD)
613  return static_cast<Chord*>(chord->parent())->upNote();
614  else // no parent or parent is not a chord?
615  return nullptr;
616  // for grace notes after, next chord is next chord of parent chord
617  // TODO : same note as case above!
618  case NoteType::GRACE8_AFTER:
619  case NoteType::GRACE16_AFTER:
620  case NoteType::GRACE32_AFTER:
621  // move unto parent chord and proceed to standard case
622  if (chord->parent() && chord->parent()->type() == ElementType::CHORD)
623  chord = static_cast<Chord*>(chord->parent());
624  else
625  return nullptr;
626  break;
627  case NoteType::NORMAL:
628  {
629  // if chord has grace notes after, the first one is the next note
630  QVector<Chord*>graces = chord->graceNotesAfter();
631  if (graces.size() > 0)
632  return graces.first()->upNote();
633  }
634  break;
635  default:
636  break;
637  }
638 
639  // standard case (NORMAL or grace after chord)
640 
641  // if parent not a segment, can't locate a target note
642  if (chord->parent()->type() != ElementType::SEGMENT)
643  return nullptr;
644 
645  // look for first ChordRest segment after initial note is elapsed
646  Segment* segm = chord->score()->tick2rightSegment(chord->tick() + chord->actualTicks());
647  int chordTrack = chord->track();
648  Part* part = chord->part();
649  while (segm) {
650  // if next segment is a ChordRest segment
651  if (segm->segmentType() == SegmentType::ChordRest) {
652  Chord* target = nullptr;
653 
654  // look for a Chord in the same track
655  if (segm->element(chordTrack) && segm->element(chordTrack)->type() == ElementType::CHORD)
656  target = static_cast<Chord*>(segm->element(chordTrack));
657  else // if no same track, look for other chords in the same instrument
658  for (Element* currChord : segm->elist())
659  if (currChord != nullptr && currChord->type() == ElementType::CHORD
660  && static_cast<Chord*>(currChord)->part() == part) {
661  target = static_cast<Chord*>(currChord);
662  break;
663  }
664 
665  // if we found a target next chord
666  if (target) {
667  // if chord has grace notes before, the first one is the next note
668  QVector<Chord*>graces = target->graceNotesBefore();
669  if (graces.size() > 0)
670  return graces.first()->upNote();
671  return target->upNote(); // if no grace before, return top note
672  }
673  }
674  segm = segm->next1();
675  }
676  qDebug("no second note for glissando found");
677  return nullptr;
678  }
