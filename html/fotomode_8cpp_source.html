194 bool ScoreView::fotoEditElementDragTransition(QMouseEvent* ev)
195  {
196  editData.startMove = imatrix.map(QPointF(ev->pos()));
197  int i;
198  for (i = 0; i < editData.grips; ++i) {
199  if (editData.grip[i].contains(editData.startMove)) {
200  editData.curGrip = Grip(i);
201  switch (int(editData.curGrip)) {
202  case 0:
203  case 2:
204  setCursor(Qt::SizeFDiagCursor);
205  break;
206  case 1:
207  case 3:
208  setCursor(Qt::SizeBDiagCursor);
209  break;
210  case 4:
211  case 6:
212  setCursor(Qt::SizeVerCursor);
213  break;
214  case 5:
215  case 7:
216  setCursor(Qt::SizeHorCursor);
217  break;
218  }
219  updateGrips();
220  score()->update();
221  break;
222  }
223  }
224  return i != editData.grips;
225  }

455 
456 bool ScoreView::saveFotoAs(bool printMode, const QRectF& r)
457  {
458  QStringList fl;
459  fl.append(tr("PNG Bitmap Graphic") + " (*.png)");
460  fl.append(tr("PDF File") + " (*.pdf)");
461  fl.append(tr("Scalable Vector Graphics") + " (*.svg)");
462 
463  QString selectedFilter;
464  QString filter = fl.join(";;");
465  QString fn = mscore->getFotoFilename(filter, &selectedFilter);
466 
467  if (fn.isEmpty())
468  return false;
469 
470  QFileInfo fi(fn);
471  mscore->lastSaveDirectory = fi.absolutePath();
472 
473  QString ext;
474  if (selectedFilter.isEmpty()) {
475  ext = fi.suffix();
476  }
477  else {
478  int idx = fl.indexOf(selectedFilter);
479  if (idx != -1) {
480  static const char* extensions[] = {
481  "png",
482  "pdf",
483  "svg"
484  };
485  ext = extensions[idx];
486  }
487  }
488 
489  if (ext.isEmpty()) {
490  QMessageBox::critical(mscore, tr("Save As"), tr("Cannot determine file type"));
491  return false;
492  }
493 
494  ext = ext.toLower();
495  if (fi.suffix().toLower() != ext)
496  fn += "." + ext;
497 
498  bool transparent = preferences.pngTransparent;
499  double convDpi = preferences.pngResolution;
500  double mag = convDpi / DPI;
501 
502  if (ext == "svg")
503  mag = 1; // SVG is not scaled, it's scalable.
504 
505  int w = lrint(r.width() * mag);
506  int h = lrint(r.height() * mag);
507 
508  double pr = MScore::pixelRatio;
509  if (ext == "pdf") {
510  QPdfWriter pdfWriter(fn);
511  mag = pdfWriter.logicalDpiX() / DPI;
512  pdfWriter.setResolution(preferences.exportPdfDpi);
513  QSizeF size(r.width() * mag, r.height() * mag);
514  QPageSize ps(QPageSize::id(size, QPageSize::Inch));
515  pdfWriter.setPageSize(ps);
516  pdfWriter.setCreator("MuseScore Version: " VERSION);
517  pdfWriter.setTitle(fn);
518  MScore::pixelRatio = DPI / pdfWriter.logicalDpiX();
519  QPainter p(&pdfWriter);
520  paintRect(printMode, p, r, mag);
521  }
522  else if (ext == "svg") {
523  // note that clipping is not implemented
524  // (as of 4.8)
525  SvgGenerator printer;
526  printer.setFileName(fn);
527  printer.setTitle(_score->title());
528  printer.setSize(QSize(w, h));
529  printer.setViewBox(QRect(0, 0, w, h));
530  MScore::pixelRatio = DPI / printer.logicalDpiX();
531  QPainter p(&printer);
532  MScore::pdfPrinting = true;
533  paintRect(printMode, p, r, mag);
534  MScore::pdfPrinting = false;
535  }
536  else if (ext == "png") {
537  QImage::Format f = QImage::Format_ARGB32_Premultiplied;
538  QImage printer(w, h, f);
539  printer.setDotsPerMeterX(lrint((convDpi * 1000) / INCH));
540  printer.setDotsPerMeterY(lrint((convDpi * 1000) / INCH));
541  printer.fill(transparent ? 0 : 0xffffffff);
542  MScore::pixelRatio = 1.0 / mag;
543  QPainter p(&printer);
544  paintRect(printMode, p, r, mag);
545  printer.save(fn, "png");
546  }
547  else
548  qDebug("unknown extension <%s>", qPrintable(ext));
549  MScore::pixelRatio = pr;
550  return true;
551  }

557 void ScoreView::paintRect(bool printMode, QPainter& p, const QRectF& r, double mag)
558  {
559  p.scale(mag, mag);
560  p.translate(-r.topLeft());
561  p.setRenderHint(QPainter::Antialiasing, true);
562  p.setRenderHint(QPainter::TextAntialiasing, true);
563 
564  score()->setPrinting(printMode);
565 
566  foreach (Page* page, _score->pages()) {
567  // QRectF pr(page->abbox());
568  QRectF pr(page->canvasBoundingRect());
569  if (pr.right() < r.left())
570  continue;
571  if (pr.left() > r.right())
572  break;
573  p.translate(page->pos());
574  QList<Element*> ell = page->items(r.translated(-page->pos()));
575  qStableSort(ell.begin(), ell.end(), elementLessThan);
576  drawElements(p, ell);
577  p.translate(-page->pos());
578  }
579 
580  score()->setPrinting(false);
581  p.end();
582  }
