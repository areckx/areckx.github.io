79 void TrillSegment::symbolLine(SymId start, SymId fill)
80  {
81  qreal x1 = 0;
82  qreal x2 = pos2().x();
83  qreal w = x2 - x1;
84  qreal mag = magS();
85  ScoreFont* f = score()->scoreFont();
86 
87  _symbols.clear();
88  _symbols.push_back(start);
89  qreal w1 = f->advance(start, mag);
90  qreal w2 = f->advance(fill, mag);
91  int n = lrint((w - w1) / w2);
92  for (int i = 0; i < n; ++i)
93  _symbols.push_back(fill);
94  QRectF r(f->bbox(_symbols, mag));
95  setbbox(r);
96  }

98 void TrillSegment::symbolLine(SymId start, SymId fill, SymId end)
99  {
100  qreal x1 = 0;
101  qreal x2 = pos2().x();
102  qreal w = x2 - x1;
103  qreal mag = magS();
104  ScoreFont* f = score()->scoreFont();
105 
106  _symbols.clear();
107  _symbols.push_back(start);
108  qreal w1 = f->bbox(start, mag).width();
109  qreal w2 = f->width(fill, mag);
110  qreal w3 = f->width(end, mag);
111  int n = lrint((w - w1 - w3) / w2);
112  for (int i = 0; i < n; ++i)
113  _symbols.push_back(fill);
114  _symbols.push_back(end);
115  QRectF r(f->bbox(_symbols, mag));
116  setbbox(r);
117  }

123 void TrillSegment::layout()
124  {
125  if (autoplace())
126  setUserOff(QPointF());
127 
128  if (staff())
129  setMag(staff()->mag(tick()));
130  if (isSingleType() || isBeginType()) {
131  Accidental* a = trill()->accidental();
132  if (a) {
133  a->layout();
134  a->setMag(a->mag() * .6);
135  qreal _spatium = spatium();
136  a->setPos(_spatium * 1.3, -2.2 * _spatium);
137  a->adjustReadPos();
138  a->setParent(this);
139  }
140  switch (trill()->trillType()) {
141  case Trill::Type::TRILL_LINE:
142  symbolLine(SymId::ornamentTrill, SymId::wiggleTrill);
143  break;
144  case Trill::Type::PRALLPRALL_LINE:
145  symbolLine(SymId::wiggleTrill, SymId::wiggleTrill);
146  break;
147  case Trill::Type::UPPRALL_LINE:
148  if (score()->scoreFont()->isValid(SymId::ornamentBottomLeftConcaveStroke))
149  symbolLine(SymId::ornamentBottomLeftConcaveStroke,
150  SymId::ornamentZigZagLineNoRightEnd, SymId::ornamentZigZagLineWithRightEnd);
151  else
152  symbolLine(SymId::ornamentUpPrall,
153  // SymId::ornamentZigZagLineNoRightEnd, SymId::ornamentZigZagLineWithRightEnd);
154  SymId::ornamentZigZagLineNoRightEnd);
155  break;
156  case Trill::Type::DOWNPRALL_LINE:
157  if (score()->scoreFont()->isValid(SymId::ornamentLeftVerticalStroke))
158  symbolLine(SymId::ornamentLeftVerticalStroke,
159  SymId::ornamentZigZagLineNoRightEnd, SymId::ornamentZigZagLineWithRightEnd);
160  else
161  symbolLine(SymId::ornamentDownPrall,
162  // SymId::ornamentZigZagLineNoRightEnd, SymId::ornamentZigZagLineWithRightEnd);
163  SymId::ornamentZigZagLineNoRightEnd);
164  break;
165  }
166  }
167  else
168  symbolLine(SymId::wiggleTrill, SymId::wiggleTrill);
169 
170  if (parent()) {
171  qreal yo = score()->styleP(trill()->placeBelow() ? StyleIdx::trillPosBelow : StyleIdx::trillPosAbove);
172  rypos() = yo;
173  if (autoplace()) {
174  qreal minDistance = spatium();
175  Shape s1 = shape().translated(pos());
176  if (trill()->placeAbove()) {
177  qreal d = system()->topDistance(staffIdx(), s1);
178  if (d > -minDistance)
179  rUserYoffset() = -d - minDistance;
180  }
181  else {
182  qreal d = system()->bottomDistance(staffIdx(), s1);
183  if (d > -minDistance)
184  rUserYoffset() = d + minDistance;
185  }
186  }
187  else
188  adjustReadPos();
189  }
190  }

419 
420 void Trill::read(XmlReader& e)
421  {
422  qDeleteAll(spannerSegments());
423  spannerSegments().clear();
424 
425  e.addSpanner(e.intAttribute("id", -1), this);
426  while (e.readNextStartElement()) {
427  const QStringRef& tag(e.name());
428  if (tag == "subtype")
429  setTrillType(e.readElementText());
430  else if (tag == "Accidental") {
431  _accidental = new Accidental(score());
432  _accidental->read(e);
433  _accidental->setParent(this);
434  }
435  else if ( tag == "ornamentStyle")
436  setProperty(P_ID::ORNAMENT_STYLE, Ms::getProperty(P_ID::ORNAMENT_STYLE, e));
437  else if ( tag == "play")
438  setPlayArticulation(e.readBool());
439  else if (!SLine::readProperties(e))
440  e.unknown();
441  }
442  }
