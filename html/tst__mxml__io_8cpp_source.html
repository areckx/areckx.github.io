239 void TestMxmlIO::mxmlIoTest(const char* file)
240  {
241  MScore::debugMode = true;
242  preferences.musicxmlExportBreaks = MusicxmlExportBreaks::MANUAL;
243  preferences.musicxmlImportBreaks = true;
244  MasterScore* score = readScore(DIR + file + ".xml");
245  QVERIFY(score);
246  fixupScore(score);
247  score->doLayout();
248  QVERIFY(saveMusicXml(score, QString(file) + ".xml"));
249  QVERIFY(saveCompareMusicXmlScore(score, QString(file) + ".xml", DIR + file + ".xml"));
250  delete score;
251  }
