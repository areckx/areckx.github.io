60 bool TestTuplet::createTuplet(int n, ChordRest* cr)
61  {
62  if (cr->durationType() < TDuration(TDuration::DurationType::V_128TH))
63  return false;
64 
65  Fraction f(cr->duration());
66  int tick = cr->tick();
67  Tuplet* ot = cr->tuplet();
68 
69  f.reduce(); //measure duration might not be reduced
70  Fraction ratio(n, f.numerator());
71  Fraction fr(1, f.denominator());
72  while (ratio.numerator() >= ratio.denominator()*2) {
73  ratio /= 2;
74  fr /= 2;
75  }
76 
77  Tuplet* tuplet = new Tuplet(cr->score());
78  tuplet->setRatio(ratio);
79 
80  //
81  // "fr" is the fraction value of one tuple element
82  //
83  // "tuplet time" is "normal time" / tuplet->ratio()
84  // Example: an 1/8 has 240 midi ticks, in an 1/8 triplet the note
85  // has a tick duration of 240 / (3/2) = 160 ticks
86  // (assume tpq = 480)
87  //
88 
89  tuplet->setDuration(f);
90  TDuration baseLen(fr);
91  tuplet->setBaseLen(baseLen);
92 
93  tuplet->setTrack(cr->track());
94  tuplet->setTick(tick);
95  Measure* measure = cr->measure();
96  tuplet->setParent(measure);
97 
98  if (ot)
99  tuplet->setTuplet(ot);
100  cr->score()->cmdCreateTuplet(cr, tuplet);
101  return true;
102  }

108 void TestTuplet::tuplet(const char* p1, const char* p2)
109  {
110  MasterScore* score = readScore(DIR + p1);
111  Measure* m1 = score->firstMeasure();
112  Measure* m2 = m1->nextMeasure();
113 
114  QVERIFY(m1 != 0);
115  QVERIFY(m2 != 0);
116  QVERIFY(m1 != m2);
117 
118  Segment* s = m2->first(SegmentType::ChordRest);
119  QVERIFY(s != 0);
120  Ms::Chord* c = static_cast<Ms::Chord*>(s->element(0));
121  QVERIFY(c != 0);
122 
123  QVERIFY(createTuplet(3, c));
124 
125  QVERIFY(saveCompareScore(score, p1, DIR + p2));
126  delete score;
127  }
