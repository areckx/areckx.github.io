192 bool PSConverter::convert()
193 {
194  Q_D(PSConverter);
195  d->lastError = NoError;
196 
197  Q_ASSERT(!d->pageList.isEmpty());
198  Q_ASSERT(d->paperWidth != -1);
199  Q_ASSERT(d->paperHeight != -1);
200 
201  if (d->document->locked)
202  {
203  d->lastError = FileLockedError;
204  return false;
205  }
206 
207  QIODevice *dev = d->openDevice();
208  if (!dev)
209  {
210  d->lastError = OpenOutputError;
211  return false;
212  }
213 
214  QByteArray pstitle8Bit = d->title.toLocal8Bit();
215  char* pstitlechar;
216  if (!d->title.isEmpty()) pstitlechar = pstitle8Bit.data();
217  else pstitlechar = 0;
218 
219  std::vector<int> pages;
220  foreach(int page, d->pageList)
221  {
222  pages.push_back(page);
223  }
224 
225  PSOutputDev *psOut = new PSOutputDev(outputToQIODevice, dev,
226  pstitlechar,
227  d->document->doc,
228  pages,
229  (d->opts & PrintToEPS) ? psModeEPS : psModePS,
230  d->paperWidth,
231  d->paperHeight,
232  gFalse,
233  gFalse,
234  d->marginLeft,
235  d->marginBottom,
236  d->paperWidth - d->marginRight,
237  d->paperHeight - d->marginTop,
238  (d->opts & ForceRasterization));
239 
240  if (d->opts & StrictMargins)
241  {
242  double xScale = ((double)d->paperWidth - (double)d->marginLeft - (double)d->marginRight) / (double)d->paperWidth;
243  double yScale = ((double)d->paperHeight - (double)d->marginBottom - (double)d->marginTop) / (double)d->paperHeight;
244  psOut->setScale(xScale, yScale);
245  }
246 
247  if (psOut->isOk())
248  {
249  GBool isPrinting = (d->opts & Printing) ? gTrue : gFalse;
250  GBool showAnnotations = (d->opts & HideAnnotations) ? gFalse : gTrue;
251  foreach(int page, d->pageList)
252  {
253  d->document->doc->displayPage(psOut,
254  page,
255  d->hDPI,
256  d->vDPI,
257  d->rotate,
258  gFalse,
259  gTrue,
260  isPrinting,
261  NULL,
262  NULL,
263  annotDisplayDecideCbk,
264  &showAnnotations, gTrue);
265  if (d->pageConvertedCallback)
266  (*d->pageConvertedCallback)(page, d->pageConvertedPayload);
267  }
268  delete psOut;
269  d->closeDevice();
270  return true;
271  }
272  else
273  {
274  delete psOut;
275  d->closeDevice();
276  return false;
277  }
278 }
