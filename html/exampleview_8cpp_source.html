29 ExampleView::ExampleView(QWidget* parent)
30  : QFrame(parent)
31  {
32  _score = 0;
33  setAcceptDrops(true);
34  setFocusPolicy(Qt::StrongFocus);
35  resetMatrix();
36  _fgPixmap = nullptr;
37  if (preferences.fgUseColor)
38  _fgColor = preferences.fgColor;
39  else {
40  _fgPixmap = new QPixmap(preferences.fgWallpaper);
41  if (_fgPixmap == 0 || _fgPixmap->isNull())
42  qDebug("no valid pixmap %s", qPrintable(preferences.fgWallpaper));
43  }
44  // setup drag canvas state
45  sm = new QStateMachine(this);
46  QState* stateActive = new QState;
47 
48  QState* s1 = new QState(stateActive);
49  s1->setObjectName("example-normal");
50  s1->assignProperty(this, "cursor", QCursor(Qt::ArrowCursor));
51 
52  QState* s = new QState(stateActive);
53  s->setObjectName("example-drag");
54  s->assignProperty(this, "cursor", QCursor(Qt::SizeAllCursor));
55  QEventTransition* cl = new QEventTransition(this, QEvent::MouseButtonRelease);
56  cl->setTargetState(s1);
57  s->addTransition(cl);
58  s1->addTransition(new DragTransitionExampleView(this));
59 
60 
61  sm->addState(stateActive);
62  stateActive->setInitialState(s1);
63  sm->setInitialState(stateActive);
64 
65  sm->start();
66  }

84  {
85  double mag = 0.9 * guiScaling * (DPI_DISPLAY / DPI); // 90% of nominal
86  qreal _spatium = SPATIUM20 * mag;
87  // example would normally be 10sp from top of page; this leaves 3sp margin above

89  _matrix = QTransform(mag, 0.0, 0.0, mag, _spatium, -_spatium * 10.0);
90  imatrix = _matrix.inverted();
91  }

282 void ExampleView::dragMoveEvent(QDragMoveEvent* event)
283  {
284  event->acceptProposedAction();
285 
286  if (!dragElement || dragElement->type() != ElementType::ICON)
287  return;
288 
289  QPointF pos(imatrix.map(QPointF(event->pos())));
290  QList<Element*> el = elementsAt(pos);
291  bool found = false;
292  foreach(const Element* e, el) {
293  if (e->type() == ElementType::NOTE) {
294  setDropTarget(const_cast<Element*>(e));
295  found = true;
296  break;
297  }
298  }
299  if (!found)
300  setDropTarget(0);
301  dragElement->scanElements(&pos, moveElement, false);
302  _score->update();
303  return;
304  }

338 void ExampleView::dropEvent(QDropEvent* event)
339  {
340  QPointF pos(imatrix.map(QPointF(event->pos())));
341 
342  if (!dragElement)
343  return;
344  if (dragElement->type() != ElementType::ICON) {
345  delete dragElement;
346  dragElement = 0;
347  return;
348  }
349  foreach (Element* e, elementsAt(pos)) {
350  if (e->type() == ElementType::NOTE) {
351  Icon* icon = static_cast<Icon*>(dragElement);
352  Chord* chord = static_cast<Note*>(e)->chord();
353  emit beamPropertyDropped(chord, icon);
354  switch (icon->iconType()) {
355  case IconType::SBEAM:
356  chord->setBeamMode(Beam::Mode::BEGIN);
357  break;
358  case IconType::MBEAM:
359  chord->setBeamMode(Beam::Mode::AUTO);
360  break;
361  case IconType::BEAM32:
362  chord->setBeamMode(Beam::Mode::BEGIN32);
363  break;
364  case IconType::BEAM64:
365  chord->setBeamMode(Beam::Mode::BEGIN64);
366  break;
367  default:
368  break;
369  }
370  score()->doLayout();
371  break;
372  }
373  }
374  event->acceptProposedAction();
375  delete dragElement;
376  dragElement = 0;
377  setDropTarget(0);
378  }

417 void ExampleView::dragExampleView(QMouseEvent* ev)
418  {
419  QPoint d = ev->pos() - _matrix.map(startMove).toPoint();
420  int dx = d.x();
421  if (dx == 0)
422  return;
423 
424  constraintCanvas(&dx);
425 
426  // Perform the actual scrolling
427  _matrix.setMatrix(_matrix.m11(), _matrix.m12(), _matrix.m13(), _matrix.m21(),
428  _matrix.m22(), _matrix.m23(), _matrix.dx()+dx, _matrix.dy(), _matrix.m33());
429  imatrix = _matrix.inverted();
430  scroll(dx, 0);
431  }

444 void ExampleView::wheelEvent(QWheelEvent* event)
445  {
446  QPoint pixelsScrolled = event->pixelDelta();
447  QPoint stepsScrolled = event->angleDelta();
448  int dx = 0, dy = 0;
449  if (!pixelsScrolled.isNull()) {
450  dx = pixelsScrolled.x();
451  dy = pixelsScrolled.y();
452  }
453  else if (!stepsScrolled.isNull()) {
454  dx = static_cast<qreal>(stepsScrolled.x()) * qMax(2, width() / 10) / 120;
455  dy = static_cast<qreal>(stepsScrolled.y()) * qMax(2, height() / 10) / 120;
456  }
457 
458  if (dx == 0) {
459  if (dy == 0)
460  return;
461  else
462  dx = dy;
463  }
464 
465  constraintCanvas(&dx);
466 
467  _matrix.setMatrix(_matrix.m11(), _matrix.m12(), _matrix.m13(), _matrix.m21(),
468  _matrix.m22(), _matrix.m23(), _matrix.dx()+dx, _matrix.dy(), _matrix.m33());
469  imatrix = _matrix.inverted();
470  scroll(dx, 0);
471  }

476 
477 void ExampleView::constraintCanvas (int* dxx)
478  {
479  int dx = *dxx;
480 
481  Q_ASSERT(_score->pages().front()->system(0)); // should exist if doLayout ran
482 
483  // form rectangle bounding the the system with a spatium margin and translate relative to view space
484  qreal xstart = _score->pages().front()->system(0)->bbox().left() - SPATIUM20;
485  qreal xend = _score->pages().front()->system(0)->bbox().right() + 2.0 * SPATIUM20;
486  QRectF systemScaledViewRect(xstart * _matrix.m11(), 0, xend * _matrix.m11(), 0);
487  systemScaledViewRect.translate(_matrix.dx(), 0);
488 
489  qreal frameWidth = static_cast<QFrame*>(this)->frameRect().width();
490 
491  // constrain the dx of scrolling so that this ExampleView won't be moved past the borders of its QFrame
492  if (dx > 0) {
493  // when moving right, ensure the left edge of systemScaledViewRect won't be right of frame's left edge
494  if (systemScaledViewRect.left() + dx > 0)
495  dx = -systemScaledViewRect.left();
496  }
497  else {
498  // never move left if entire system already fits entirely within the frame
499  if (systemScaledViewRect.width() < frameWidth)
500  dx = 0;
501  // when moving left, ensure the right edge of systemScaledViewRect won't be left of frame's right edge
502  else if (systemScaledViewRect.right() + dx < frameWidth)
503  dx = frameWidth - systemScaledViewRect.right();
504  }
505 
506  *dxx = dx;
507  }
508 }
