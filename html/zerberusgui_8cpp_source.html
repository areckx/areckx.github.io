111 static void collectFiles(QFileInfoList* l, const QString& path)
112  {
113  // printf("collect files <%s>\n", qPrintable(path));
114 
115  QDir dir(path);
116  foreach (const QFileInfo& s, dir.entryInfoList(QDir::Files | QDir::Dirs | QDir::NoDotAndDotDot)) {
117  if (path == s.absoluteFilePath())
118  return;
119 
120  if (s.isDir() && !s.isHidden())
121  collectFiles(l, s.absoluteFilePath());
122  else {
123  if (s.suffix().toLower() == "sfz")
124  l->append(s);
125  }
126  }
127  }

133 QFileInfoList Zerberus::sfzFiles()
134  {
135  QFileInfoList l;
136 
137  QStringList pl = Ms::preferences.mySoundfontsPath.split(";");
138  pl.prepend(QFileInfo(QString("%1%2").arg(Ms::mscoreGlobalShare).arg("sound")).absoluteFilePath());
139 
140  foreach (const QString& s, pl) {
141  QString ss(s);
142  if (!s.isEmpty() && s[0] == '~')
143  ss = QDir::homePath() + s.mid(1);
144  collectFiles(&l, ss);
145  }
146  return l;
147  }

149 void ZerberusGui::loadSfz() {
150 
151  if (_sfzToLoad.empty())
152  return;
153 
154  struct SfzNamePath item = _sfzToLoad.back();
155  QString sfName = item.name;
156  QString sfPath = item.path;
157  _sfzToLoad.pop_back();
158 
159  QStringList sl;
160  for (int i = 0; i < files->count(); ++i) {
161  QListWidgetItem* item = files->item(i);
162  sl.append(item->text());
163  }
164 
165  if (sl.contains(sfPath)) {
166  QMessageBox::warning(this,
167  tr("MuseScore"),
168  tr("SoundFont %1 already loaded").arg(sfPath));
169  }
170  else {
171  _loadedSfName = sfName;
172  _loadedSfPath = sfPath;
173  QFuture<bool> future = QtConcurrent::run(zerberus(), &Zerberus::addSoundFont, sfName);
174  _futureWatcher.setFuture(future);
175  _progressTimer->start(1000);
176  _progressDialog->exec();
177  }
178  }

184 void ZerberusGui::addClicked()
185  {
186  zerberus()->setLoadWasCanceled(false);
187 
188  QFileInfoList l = Zerberus::sfzFiles();
189 
190  SfzListDialog ld(this);
191  foreach (const QFileInfo& fi, l)
192  ld.add(fi.fileName(), fi.absoluteFilePath());
193  if (!ld.exec())
194  return;
195 
196  for (auto item : ld.getNamePaths()) {
197  _sfzToLoad.push_back(item);
198  }
199  loadSfz();
200  }

234 void ZerberusGui::onSoundFontLoaded()
235  {
236  bool loaded = _futureWatcher.result();
237  bool wasNotCanceled = !_progressDialog->wasCanceled();
238  _progressTimer->stop();
239  _progressDialog->reset();
240  if (loaded) {
241  QListWidgetItem* item = new QListWidgetItem;
242  item->setText(_loadedSfName);
243  item->setData(Qt::UserRole, _loadedSfPath);
244  files->insertItem(0, item);
245  emit valueChanged();
246  emit sfChanged();
247  }
248  else if (wasNotCanceled) {
249  QMessageBox::warning(this,
250  tr("MuseScore"),
251  tr("Cannot load SoundFont %1").arg(_loadedSfPath));
252  }
253  loadSfz();
254  }

260 void ZerberusGui::removeClicked()
261  {
262  int row = files->currentRow();
263  if (row >= 0) {
264  QString path(files->item(row)->data(Qt::UserRole).toString());
265  if (!zerberus()->removeSoundFont(path))
266  qDebug("ZerberusGui::removeClicked: cannot remove sf %s", qPrintable(files->item(row)->text()));
267  delete files->takeItem(row);
268  emit valueChanged();
269  emit sfChanged();
270  updateButtons();
271  }
