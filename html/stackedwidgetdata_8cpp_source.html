48 bool StackedWidgetData::initializeAnimation( void ) {
49 
50  // check enability
51  if ( !( target_ && target_.data()->isVisible() ) ) {
52  return false;
53  }
54 
55  // check index
56  if ( target_.data()->currentIndex() == index_ ) {
57  return false;
58  }
59 
60  // do not animate if either index or currentIndex is not valid
61  // but update index_ none the less
62  if ( target_.data()->currentIndex() < 0 || index_ < 0 ) {
63  index_ = target_.data()->currentIndex();
64  return false;
65  }
66 
67  // get old widget (matching index_) and initialize transition
68  if ( QWidget* widget = target_.data()->widget( index_ ) ) {
69 
70  transition().data()->setOpacity( 0 );
71  startClock();
72  transition().data()->setGeometry( widget->geometry() );
73  transition().data()->setStartPixmap( transition().data()->grab( widget ) );
74 
75  index_ = target_.data()->currentIndex();
76  return !slow();
77 
78  }
79  else {
80 
81  index_ = target_.data()->currentIndex();
82  return false;
83 
84  }
85 
86  }

106 void StackedWidgetData::finishAnimation( void ) {

109  target_.data()->currentWidget()->setUpdatesEnabled( false );
110  }
111 
112  // hide transition

115  // reenable updates and repaint

117  target_.data()->currentWidget()->setUpdatesEnabled( true );

119  }
120
