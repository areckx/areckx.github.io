153 SplashState::SplashState(SplashState *state) {
154  memcpy(matrix, state->matrix, 6 * sizeof(SplashCoord));
155  strokePattern = state->strokePattern->copy();
156  fillPattern = state->fillPattern->copy();
157  screen = state->screen->copy();
158  blendFunc = state->blendFunc;
159  strokeAlpha = state->strokeAlpha;
160  fillAlpha = state->fillAlpha;
161  multiplyPatternAlpha = state->multiplyPatternAlpha;
162  patternStrokeAlpha = state->patternStrokeAlpha;
163  patternFillAlpha = state->patternFillAlpha;
164  lineWidth = state->lineWidth;
165  lineCap = state->lineCap;
166  lineJoin = state->lineJoin;
167  miterLimit = state->miterLimit;
168  flatness = state->flatness;
169  if (state->lineDash) {
170  lineDashLength = state->lineDashLength;
171  lineDash = (SplashCoord *)gmallocn(lineDashLength, sizeof(SplashCoord));
172  memcpy(lineDash, state->lineDash, lineDashLength * sizeof(SplashCoord));
173  } else {
174  lineDash = NULL;
175  lineDashLength = 0;
176  }
177  lineDashPhase = state->lineDashPhase;
178  strokeAdjust = state->strokeAdjust;
179  clip = state->clip->copy();
180  softMask = state->softMask;
181  deleteSoftMask = gFalse;
182  inNonIsolatedGroup = state->inNonIsolatedGroup;
183  fillOverprint = state->fillOverprint;
184  strokeOverprint = state->strokeOverprint;
185  overprintMode = state->overprintMode;
186  memcpy(rgbTransferR, state->rgbTransferR, 256);
187  memcpy(rgbTransferG, state->rgbTransferG, 256);
188  memcpy(rgbTransferB, state->rgbTransferB, 256);
189  memcpy(grayTransfer, state->grayTransfer, 256);
190 #if SPLASH_CMYK
191  memcpy(cmykTransferC, state->cmykTransferC, 256);
192  memcpy(cmykTransferM, state->cmykTransferM, 256);
193  memcpy(cmykTransferY, state->cmykTransferY, 256);
194  memcpy(cmykTransferK, state->cmykTransferK, 256);
195  for (int cp = 0; cp < SPOT_NCOMPS+4; cp++)
196  memcpy(deviceNTransfer[cp], state->deviceNTransfer[cp], 256);
197 #endif
198  overprintMask = state->overprintMask;
199  overprintAdditive = state->overprintAdditive;
200  next = NULL;
201 }

250 void SplashState::setTransfer(Guchar *red, Guchar *green, Guchar *blue,
251  Guchar *gray) {
252 #if SPLASH_CMYK
253  int i;
254 
255  for (i = 0; i < 256; ++i) {
256  cmykTransferC[i] = 255 - rgbTransferR[255 - i];
257  cmykTransferM[i] = 255 - rgbTransferG[255 - i];
258  cmykTransferY[i] = 255 - rgbTransferB[255 - i];
259  cmykTransferK[i] = 255 - grayTransfer[255 - i];
260  }
261  for (i = 0; i < 256; ++i) {
262  deviceNTransfer[0][i] = 255 - rgbTransferR[255 - i];
263  deviceNTransfer[1][i] = 255 - rgbTransferG[255 - i];
264  deviceNTransfer[2][i] = 255 - rgbTransferB[255 - i];
265  deviceNTransfer[3][i] = 255 - grayTransfer[255 - i];
266  }
267 #endif
268  memcpy(rgbTransferR, red, 256);
269  memcpy(rgbTransferG, green, 256);
270  memcpy(rgbTransferB, blue, 256);
271  memcpy(grayTransfer, gray, 256);
272 }
