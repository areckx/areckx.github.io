214 void Score::update()
215  {
216  bool updateAll = false;
217  for (MasterScore* ms : *movements()) {
218  CmdState& cs = ms->cmdState();
219  ms->deletePostponed();
220  if (cs.layoutRange()) {
221  for (Score* s : ms->scoreList())
222  s->doLayoutRange(cs.startTick(), cs.endTick());
223  updateAll = true;
224  }
225  }
226 
227  for (MasterScore* ms : *movements()) {
228  CmdState& cs = ms->cmdState();
229  if (updateAll || cs.updateAll()) {
230  for (Score* s : scoreList()) {
231  for (MuseScoreView* v : s->viewer) {
232  v->updateAll();
233  }
234  }
235  }
236  else if (cs.updateRange()) {
237  // updateRange updates only current score
238  qreal d = spatium() * .5;
239  _updateState.refresh.adjust(-d, -d, 2 * d, 2 * d);
240  for (MuseScoreView* v : viewer)
241  v->dataChanged(_updateState.refresh);
242  _updateState.refresh = QRectF();
243  }
244  const InputState& is = inputState();
245  if (is.noteEntryMode() && is.segment()) {
246  setPlayPos(is.segment()->tick());
247  }
248  if (_playlistDirty) {
249  emit playlistChanged();
250  _playlistDirty = false;
251  }
252  cs.reset();
253  }

278  {

280  Segment* segment;

282  // ignore if we do not have a measure
283  if (mb == 0 || mb->type() != ElementType::MEASURE) {
284  qDebug("cmdAddSpanner: cannot put object here");
285  delete spanner;
286  return;
287  }
288 
289  // all spanners live in voice 0 (except slurs/ties)
290  int track = staffIdx == -1 ? -1 : staffIdx * VOICES;
291 
292  spanner->setTrack(track);
293  spanner->setTrack2(track);
294 
295  if (spanner->anchor() == Spanner::Anchor::SEGMENT) {
296  spanner->setTick(segment->tick());
297  int lastTick = lastMeasure()->tick() + lastMeasure()->ticks();
298  int tick2 = qMin(segment->measure()->tick() + segment->measure()->ticks(), lastTick);
299  spanner->setTick2(tick2);
300  }
301  else { // Anchor::MEASURE, Anchor::CHORD, Anchor::NOTE
302  Measure* m = toMeasure(mb);
303  QRectF b(m->canvasBoundingRect());
304 
305  if (pos.x() >= (b.x() + b.width() * .5) && m != lastMeasureMM())
306  m = m->nextMeasure();
307  spanner->setTick(m->tick());
308  spanner->setTick2(m->endTick());
309  }
310  qDeleteAll(spanner->spannerSegments());
311  spanner->spannerSegments().clear();
312 
313  undoAddElement(spanner);
314  select(spanner, SelectType::SINGLE, 0);
315  }

368 // expandVoice
369 // fills gaps in voice with rests,
370 // from previous cr (or beginning of measure) to next cr (or end of measure)
371 //---------------------------------------------------------
372 
373 void Score::expandVoice(Segment* s, int track)
374  {
375  if (!s) {
376  qDebug("expand voice: no segment");
377  return;
378  }
379  if (s->element(track))
380  return;
381 
382  // find previous segment with cr in this track
383  Segment* ps;
384  for (ps = s; ps; ps = ps->prev(SegmentType::ChordRest)) {
385  if (ps->element(track))
386  break;
387  }
388  if (ps) {
389  ChordRest* cr = toChordRest(ps->element(track));
390  int tick = cr->tick() + cr->actualTicks();
391  if (tick > s->tick()) {
392  // previous cr extends past current segment
393  qDebug("expandVoice: cannot insert element here");
394  return;
395  }
396  if (cr->isChord()) {
397  // previous cr ends on or before current segment
398  // for chords, move ps to just after cr ends
399  // so we can fill any gap that might exist
400  // but don't move ps if previous cr is a rest
401  // this will be combined with any new rests needed to fill up to s->tick() below
402  ps = ps->measure()->undoGetSegment(SegmentType::ChordRest, tick);
403  }
404  }
405  //
406  // fill up to s->tick() with rests
407  //
408  Measure* m = s->measure();
409  int stick = ps ? ps->tick() : m->tick();
410  int ticks = s->tick() - stick;
411  if (ticks)
412  setRest(stick, track, Fraction::fromTicks(ticks), false, 0);
413 
414  //
415  // fill from s->tick() until next chord/rest in measure
416  //
417  Segment* ns;
418  for (ns = s->next(SegmentType::ChordRest); ns; ns = ns->next(SegmentType::ChordRest)) {
419  if (ns->element(track))
420  break;
421  }
422  ticks = ns ? (ns->tick() - s->tick()) : (m->ticks() - s->rtick());
423  if (ticks == m->ticks())
424  addRest(s, track, TDuration(TDuration::DurationType::V_MEASURE), 0);
425  else
426  setRest(s->tick(), track, Fraction::fromTicks(ticks), false, 0);
427  }

440 void Score::cmdAddInterval(int val, const std::vector<Note*>& nl)
441  {
442  startCmd();
443  for (Note* on : nl) {
444  Note* note = new Note(this);
445  Chord* chord = on->chord();
446  note->setParent(chord);
447  int valTmp = val < 0 ? val+1 : val-1;
448 
449  int npitch;
450  int ntpc1;
451  int ntpc2;
452  if (abs(valTmp) != 7) {
453  int line = on->line() - valTmp;
454  int tick = chord->tick();
455  Staff* estaff = staff(on->staffIdx() + chord->staffMove());
456  ClefType clef = estaff->clef(tick);
457  Key key = estaff->key(tick);
458  npitch = line2pitch(line, clef, key);
459 
460  int ntpc = pitch2tpc(npitch, key, Prefer::NEAREST);
461  Interval v = on->part()->instrument(tick)->transpose();
462  if (v.isZero())
463  ntpc1 = ntpc2 = ntpc;
464  else {
465  if (styleB(StyleIdx::concertPitch)) {
466  v.flip();
467  ntpc1 = ntpc;
468  ntpc2 = Ms::transposeTpc(ntpc, v, true);
469  }
470  else {
471  npitch += v.chromatic;
472  ntpc2 = ntpc;
473  ntpc1 = Ms::transposeTpc(ntpc, v, true);
474  }
475  }
476  }
477  else { //special case for octave
478  Interval interval(7, 12);
479  if (val < 0)
480  interval.flip();
481  transposeInterval(on->pitch(), on->tpc(), &npitch, &ntpc1, interval, false);
482  ntpc1 = on->tpc1();
483  ntpc2 = on->tpc2();
484  }
485  if (npitch < 0 || npitch > 127) {
486  delete note;
487  endCmd();
488  return;
489  }
490  note->setPitch(npitch, ntpc1, ntpc2);
491 
492  undoAddElement(note);
493  setPlayNote(true);
494 
495  select(note, SelectType::SINGLE, 0);
496  }
497  _is.moveToNextInputPos();
498  endCmd();
499  }

510 void Score::setGraceNote(Chord* ch, int pitch, NoteType type, int len)
511  {
512  Note* note = new Note(this);
513  Chord* chord = new Chord(this);
514 
515  // alow grace notes to be added to other grace notes
516  // by really adding to parent chord
517  if (ch->noteType() != NoteType::NORMAL)
518  ch = toChord(ch->parent());
519 
520  chord->setTrack(ch->track());
521  chord->setParent(ch);
522  chord->add(note);
523 
524  note->setPitch(pitch);
525  // find corresponding note within chord and use its tpc information
526  for (Note* n : ch->notes()) {
527  if (n->pitch() == pitch) {
528  note->setTpc1(n->tpc1());
529  note->setTpc2(n->tpc2());
530  break;
531  }
532  }
533  // note with same pitch not found, derive tpc from pitch / key
534  if (!tpcIsValid(note->tpc1()) || !tpcIsValid(note->tpc2()))
535  note->setTpcFromPitch();
536 
537  TDuration d;
538  d.setVal(len);
539  chord->setDurationType(d);
540  chord->setDuration(d.fraction());
541  chord->setNoteType(type);
542  chord->setMag(ch->staff()->mag(chord->tick()) * styleD(StyleIdx::graceNoteMag));
543 
544  undoAddElement(chord);
545  select(note, SelectType::SINGLE, 0);
546  }

551 // If f is not a basic len, create several rests or

555 // cr prototype CR

559 void Score::createCRSequence(Fraction f, ChordRest* cr, int tick)
560  {

563  for (TDuration d : toDurationList(f, true)) {

566  ncr->setDuration(d.fraction());

569  Chord* nc = toChord(ncr);
570  Chord* oc = toChord(ocr);

583  undoAddCR(ncr, measure, tick);
584  tick += ncr->actualTicks();
585  ocr = ncr;

589 //---------------------------------------------------------
590 // setNoteRest
591 // pitch == -1 -> set rest
592 // return segment of last created note/rest
593 //---------------------------------------------------------
594 
595 Segment* Score::setNoteRest(Segment* segment, int track, NoteVal nval, Fraction sd, Direction stemDirection)
596  {
597  Q_ASSERT(segment->segmentType() == SegmentType::ChordRest);
598 
599  bool isRest = nval.pitch == -1;
600  int tick = segment->tick();
601  Element* nr = 0;
602  Tie* tie = 0;
603  ChordRest* cr = toChordRest(segment->element(track));
604 
605  Measure* measure = 0;
606  for (;;) {
607  if (track % VOICES)
608  expandVoice(segment, track);
609 
610  // the returned gap ends at the measure boundary or at tuplet end
611  Fraction dd = makeGap(segment, track, sd, cr ? cr->tuplet() : 0);
612 
613  if (dd.isZero()) {
614  qDebug("cannot get gap at %d type: %d/%d", tick, sd.numerator(),
615  sd.denominator());
616  break;
617  }
618 
619  measure = segment->measure();
620  std::vector<TDuration> dl = toRhythmicDurationList(dd, isRest, segment->rtick(), sigmap()->timesig(tick).nominal(), measure, 1);
621  int n = dl.size();
622  for (int i = 0; i < n; ++i) {
623  const TDuration& d = dl[i];
624  ChordRest* ncr;
625  Note* note = 0;
626  Tie* addTie = 0;
627  if (isRest) {
628  nr = ncr = new Rest(this);
629  nr->setTrack(track);
630  ncr->setDurationType(d);
631  ncr->setDuration(d == TDuration::DurationType::V_MEASURE ? measure->len() : d.fraction());
632  }
633  else {
634  nr = note = new Note(this);
635 
636  if (tie) {
637  tie->setEndNote(note);
638  note->setTieBack(tie);
639  addTie = tie;
640  }
641  Chord* chord = new Chord(this);
642  chord->setTrack(track);
643  chord->setDurationType(d);
644  chord->setDuration(d.fraction());
645  chord->setStemDirection(stemDirection);
646  chord->add(note);
647  note->setNval(nval, tick);
648  ncr = chord;
649  if (i+1 < n) {
650  tie = new Tie(this);
651  tie->setStartNote(note);
652  tie->setTrack(track);
653  note->setTieFor(tie);
654  }
655  }
656  ncr->setTuplet(cr ? cr->tuplet() : 0);
657  undoAddCR(ncr, measure, tick);
658  if (addTie)
659  undoAddElement(addTie);
660  setPlayNote(true);
661  segment = ncr->segment();
662  tick += ncr->actualTicks();
663  }
664 
665  sd -= dd;
666  if (sd.isZero())
667  break;
668 
669  Segment* nseg = tick2segment(tick, false, SegmentType::ChordRest);
670  if (nseg == 0) {
671  qDebug("reached end of score");
672  break;
673  }
674  segment = nseg;
675 
676  cr = toChordRest(segment->element(track));
677 
678  if (cr == 0) {
679  if (track % VOICES)
680  cr = addRest(segment, track, TDuration(TDuration::DurationType::V_MEASURE), 0);
681  else {
682  qDebug("no rest in voice 0");
683  break;
684  }
685  }
686  //
687  // Note does not fit on current measure, create Tie to
688  // next part of note
689  if (!isRest) {
690  tie = new Tie(this);
691  tie->setStartNote((Note*)nr);
692  tie->setTrack(nr->track());
693  ((Note*)nr)->setTieFor(tie);
694  }
695  }
696  if (tie)
697  connectTies();
698  if (nr) {
699  if (_is.slur() && nr->type() == ElementType::NOTE) {
700  //
701  // extend slur
702  //
703  Chord* chord = toNote(nr)->chord();
704  _is.slur()->undoChangeProperty(P_ID::SPANNER_TICKS, chord->tick() - _is.slur()->tick());
705  for (ScoreElement* e : _is.slur()->linkList()) {
706  Slur* slur = static_cast<Slur*>(e);
707  for (ScoreElement* ee : chord->linkList()) {
708  Element* e = static_cast<Element*>(ee);
709  if (e->score() == slur->score() && e->track() == slur->track2()) {
710  slur->score()->undo(new ChangeSpannerElements(slur, slur->startElement(), e));
711  break;
712  }
713  }
714  }
715  }
716  select(nr, SelectType::SINGLE, 0);
717  }
718  return segment;
719  }

721 //---------------------------------------------------------
722 // makeGap
723 // make time gap at tick by removing/shortening
724 // chord/rest
725 //
726 // if keepChord, the chord at tick is not removed
727 //
728 // gap does not exceed measure or scope of tuplet
729 //
730 // return size of actual gap
731 //---------------------------------------------------------
732 
733 Fraction Score::makeGap(Segment* segment, int track, const Fraction& _sd, Tuplet* tuplet, bool keepChord)
734  {
735  Q_ASSERT(_sd.numerator());
736 
737  Measure* measure = segment->measure();
738  Fraction akkumulated;
739  Fraction sd = _sd;
740 
741  //
742  // remember first segment which should
743  // not be deleted (it may contain other elements we want to preserve)
744  //
745  Segment* firstSegment = segment;
746  int nextTick = segment->tick();
747 
748  for (Segment* seg = firstSegment; seg; seg = seg->next(SegmentType::ChordRest)) {
749  //
750  // voices != 0 may have gaps:
751  //
752  ChordRest* cr = toChordRest(seg->element(track));
753  if (!cr) {
754  if (seg->tick() < nextTick)
755  continue;

1339  case UpDownMode::OCTAVE: // move same note to next string, if possible

1384  }

1389  case UpDownMode::OCTAVE:

1391  if (pitch < 116)
