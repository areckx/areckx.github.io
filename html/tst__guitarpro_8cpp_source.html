161 void TestGuitarPro::gpReadTest(const char* file, const char* ext)
162  {

164  QVERIFY(score);
165 
166  QVERIFY(saveCompareScore(score, QString("%1.%2.mscx").arg(file).arg(ext),
167  DIR + QString("%1.%2-ref.mscx").arg(file).arg(ext)));
168  delete score;
169  }
