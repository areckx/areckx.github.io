62 //------------------------------------------------------------------------
63 // @@ Segment
64 // A segment holds all vertical aligned staff elements.
65 // Segments are typed and contain only Elements of the same type.
66 //
67 // All Elements in a segment start at the same tick. The Segment can store one Element for
68 // each voice in each staff in the score.
69 // Some elements (Clef, KeySig, TimeSig etc.) are assumed to always have voice zero
70 // and can be found in _elist[staffIdx * VOICES];
71 
72 // Segments are children of Measures and store Clefs, KeySigs, TimeSigs,
73 // BarLines and ChordRests.
74 //
75 // @P annotations array[Element] the list of annotations (read only)
76 // @P next Segment the next segment in the whole score; null at last score segment (read-only)
77 // @P nextInMeasure Segment the next segment in measure; null at last measure segment (read-only)
78 // @P prev Segment the previous segment in the whole score; null at first score segment (read-only)
79 // @P prevInMeasure Segment the previous segment in measure; null at first measure segment (read-only)
80 // @P segmentType enum (Segment.All, .Ambitus, .BarLine, .Breath, .ChordRest, .Clef, .EndBarLine, .Invalid, .KeySig, .KeySigAnnounce, .StartRepeatBarLine, .TimeSig, .TimeSigAnnounce)
81 // @P tick int midi tick position (read only)
82 //------------------------------------------------------------------------

261 inline Segment* Segment::nextEnabled() const
262  {
263  Segment* ps = next();
264  while (ps && !ps->enabled())
265  ps = ps->next();
266  return ps;
