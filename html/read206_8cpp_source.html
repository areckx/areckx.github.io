310 static void readTextStyle(MStyle* style, XmlReader& e)
311  {
312  Spatium frameWidth(0.0);
313  QString name = e.attribute("name");
314  QString family = "FreeSerif";
315  double size = 10;
316  bool bold = false;
317  bool italic = false;
318  bool underline = false;
319  Align align = Align::LEFT;
320  bool sizeIsSpatiumDependent = true;
321  bool hasFrame = false;
322  double frameWidthMM = 0.0;
323  double paddingWidthMM = 0.0;
324  Spatium paddingWidth(0.0);
325  int frameRound = 0;
326  QColor frameColor = QColor(0, 0, 0, 255);
327  QColor foregroundColor = QColor(0, 0, 0, 255);
328  QColor backgroundColor = QColor(255, 255, 255, 0);
329  bool circle = false;
330  bool systemFlag = false;
331  QPointF offset;
332  OffsetType offsetType = OffsetType::SPATIUM;
333 
334  while (e.readNextStartElement()) {
335  const QStringRef& tag(e.name());
336 
337  if (tag == "name")
338  name = e.readElementText();
339  else if (tag == "family")
340  family = e.readElementText();
341  else if (tag == "size")
342  size = e.readDouble();
343  else if (tag == "bold")
344  bold = e.readInt();
345  else if (tag == "italic")
346  italic = e.readInt();
347  else if (tag == "underline")
348  underline = e.readInt();
349  else if (tag == "align")
350  align = Align(e.readInt());
351  else if (tag == "anchor") // obsolete
352  e.skipCurrentElement();
353 
354  else if (tag == "halign") {
355  const QString& val(e.readElementText());
356  if (val == "center")
357  align = align | Align::HCENTER;
358  else if (val == "right")
359  align = align | Align::RIGHT;
360  else if (val == "left")
361  ;
362  else
363  qDebug("Text::readProperties: unknown alignment: <%s>", qPrintable(val));
364  }
365  else if (tag == "valign") {
366  const QString& val(e.readElementText());
367  if (val == "center")
368  align = align | Align::VCENTER;
369  else if (val == "bottom")
370  align = align | Align::BOTTOM;
371  else if (val == "baseline")
372  align = align | Align::BASELINE;
373  else if (val == "top")
374  ;
375  else
376  qDebug("Text::readProperties: unknown alignment: <%s>", qPrintable(val));
377  }
378  else if (tag == "xoffset") {
379  qreal xo = e.readDouble();
380  if (offsetType == OffsetType::ABS)
381  xo /= INCH;
382  offset.setX(xo);
383  }
384  else if (tag == "yoffset") {
385  qreal yo = e.readDouble();
386  if (offsetType == OffsetType::ABS)
387  yo /= INCH;
388  offset.setY(yo);
389  }
390  else if (tag == "rxoffset" || tag == "ryoffset") // obsolete
391  e.readDouble();
392  else if (tag == "offsetType") {
393  const QString& val(e.readElementText());
394  OffsetType ot = OffsetType::ABS;
395  if (val == "spatium" || val == "1")
396  ot = OffsetType::SPATIUM;
397  if (ot != offsetType) {
398  offsetType = ot;
399  if (ot == OffsetType::ABS)
400  offset /= INCH; // convert spatium -> inch
401  else
402  offset *= INCH; // convert inch -> spatium
403  }
404  }
405  else if (tag == "sizeIsSpatiumDependent" || tag == "spatiumSizeDependent")
406  sizeIsSpatiumDependent = e.readInt();
407  else if (tag == "frameWidth") { // obsolete
408  hasFrame = true;
409  frameWidthMM = e.readDouble();
410  }
411  else if (tag == "frameWidthS") {
412  hasFrame = true;
413  frameWidth = Spatium(e.readDouble());
414  }
415  else if (tag == "frame")
416  hasFrame = e.readInt();
417  else if (tag == "paddingWidth") // obsolete
418  paddingWidthMM = e.readDouble();
419  else if (tag == "paddingWidthS")
420  paddingWidth = Spatium(e.readDouble());
421  else if (tag == "frameRound")
422  frameRound = e.readInt();
423  else if (tag == "frameColor")
424  frameColor = e.readColor();
425  else if (tag == "foregroundColor")
426  foregroundColor = e.readColor();
427  else if (tag == "backgroundColor")
428  backgroundColor = e.readColor();
429  else if (tag == "circle")
430  circle = e.readInt();
431  else if (tag == "systemFlag")
432  systemFlag = e.readInt();
433  else
434  e.unknown();
435  }
436  if (family == "MuseJazz")
437  family = "MuseJazz Text";

871 static void readNote(Note* note, XmlReader& e)
872  {
873  note->setTpc1(Tpc::TPC_INVALID);
874  note->setTpc2(Tpc::TPC_INVALID);
875 
876  while (e.readNextStartElement()) {
877  const QStringRef& tag(e.name());
878  if (tag == "Accidental") {
879  Accidental* a = new Accidental(note->score());
880  a->setTrack(note->track());
881  readAccidental(a, e);
882  note->add(a);
883  }
884  else if (tag == "head") {
885  int i = e.readInt();
886  NoteHead::Group val = convertHeadGroup(i);
887  note->setHeadGroup(val);
888  }
889  else if (tag == "headType") {
890  int i = e.readInt();
891  NoteHead::Type val = convertHeadType(i);
892  note->setHeadType(val);
893  }
894  else if (note->readProperties(e))
895  ;
896  else
897  e.unknown();
898  }
899  // ensure sane values:
900  note->setPitch(limit(note->pitch(), 0, 127));
901 
902  if (!tpcIsValid(note->tpc1()) && !tpcIsValid(note->tpc2())) {
903  Key key = (note->staff() && note->chord()) ? note->staff()->key(note->chord()->tick()) : Key::C;
904  int tpc = pitch2tpc(note->pitch(), key, Prefer::NEAREST);
905  if (note->concertPitch())
906  note->setTpc1(tpc);
907  else
908  note->setTpc2(tpc);
909  }
910  if (!(tpcIsValid(note->tpc1()) && tpcIsValid(note->tpc2()))) {
911  int tick = note->chord() ? note->chord()->tick() : -1;
912  Interval v = note->staff() ? note->part()->instrument(tick)->transpose() : Interval();
913  if (tpcIsValid(note->tpc1())) {
914  v.flip();
915  if (v.isZero())
916  note->setTpc2(note->tpc1());
917  else
918  note->setTpc2(Ms::transposeTpc(note->tpc1(), v, true));
919  }
920  else {
921  if (v.isZero())
922  note->setTpc1(note->tpc2());
923  else
924  note->setTpc1(Ms::transposeTpc(note->tpc2(), v, true));
925  }
926  }
927 
928  // check consistency of pitch, tpc1, tpc2, and transposition
929  // see note in InstrumentChange::read() about a known case of tpc corruption produced in 2.0.x
930  // but since there are other causes of tpc corruption (eg, https://musescore.org/en/node/74746)
931  // including perhaps some we don't know about yet,
932  // we will attempt to fix some problems here regardless of version
933 
934  if (!e.pasteMode() && !MScore::testMode) {
935  int tpc1Pitch = (tpc2pitch(note->tpc1()) + 12) % 12;
936  int tpc2Pitch = (tpc2pitch(note->tpc2()) + 12) % 12;
937  int concertPitch = note->pitch() % 12;
938  if (tpc1Pitch != concertPitch) {
939  qDebug("bad tpc1 - concertPitch = %d, tpc1 = %d", concertPitch, tpc1Pitch);
940  note->setPitch(note->pitch() + tpc1Pitch - concertPitch);
941  }
942  Interval v = note->staff()->part()->instrument(e.tick())->transpose();
943  int transposedPitch = (note->pitch() - v.chromatic) % 12;
944  if (tpc2Pitch != transposedPitch) {
945  qDebug("bad tpc2 - transposedPitch = %d, tpc2 = %d", transposedPitch, tpc2Pitch);
946  // just in case the staff transposition info is not reliable here,
947  // do not attempt to correct tpc
948  // except for older scores where we know there are tpc problems
949  v.flip();
950  note->setTpc2(Ms::transposeTpc(note->tpc1(), v, true));
951  }
952  }
953  }

1372  e.tuplets().clear();

1374

1377  // tick is obsolete
1378  if (e.hasAttribute("tick"))
1379  e.initTick(score->fileDivision(e.intAttribute("tick")));
1380 
1381  bool irregular;
1382  if (e.hasAttribute("len")) {
1383  QStringList sl = e.attribute("len").split('/');
1384  if (sl.size() == 2)
1385  m->setLen(Fraction(sl[0].toInt(), sl[1].toInt()));
1386  else
1387  qDebug("illegal measure size <%s>", qPrintable(e.attribute("len")));
1388  irregular = true;
1389  score->sigmap()->add(m->tick(), SigEvent(m->len(), m->timesig()));
1390  score->sigmap()->add(m->endTick(), SigEvent(m->timesig()));

1392  else

1396  Fraction timeStretch(staff->timeStretch(m->tick()));

1399  // this allows markings that need to apply to previous element to do so
1400  // even though we may have already advanced to next tick position
1401  int lastTick = e.tick();
1402 
1403  while (e.readNextStartElement()) {
1404  const QStringRef& tag(e.name());
1405

1407  e.initTick(e.readFraction().ticks() + m->tick());

1416  const QStringRef& tag(e.name());

1418  bl->setBarLineType(e.readElementText());

1420  e.readInt();

1423  // bl->setSpanTo(e.intAttribute("to", bl->spanTo())); // obsolete

1426  span--;
1427  bl->setSpanStaff(span);
1428  }

1432  bl->setSpanTo(e.readInt());
1433  else if (tag == "Articulation") {
1434  Articulation* a = new Articulation(score);
1435  a->read(e);
1436  bl->add(a);
1437  }
1438  else if (!bl->Element::readProperties(e))
1439  e.unknown();
1440  }

1443  // BarLine: in the middle of a measure, has no semantic
1444  // EndBarLine: at the end tick of a measure
1445  // BeginBarLine: first segment of a measure

1448  if ((e.tick() != m->tick()) && (e.tick() != m->endTick()))
1449  st = SegmentType::BarLine;
1450  else if (bl->barLineType() == BarLineType::START_REPEAT && e.tick() == m->tick())
1451  st = SegmentType::StartRepeatBarLine;
1452  else if (e.tick() == m->tick() && segment == 0)
1453  st = SegmentType::BeginBarLine;
1454  else
1455  st = SegmentType::EndBarLine;
1456  segment = m->getSegment(st, e.tick());
1457  segment->add(bl);
1458  }
1459  else if (tag == "Chord") {
1460  Chord* chord = new Chord(score);
1461  chord->setTrack(e.track());

1463  segment = m->getSegment(SegmentType::ChordRest, e.tick());
1464  if (chord->noteType() != NoteType::NORMAL)
1465  graceNotes.push_back(chord);
1466  else {

1477  }
1478  }
1479  else if (tag == "Rest") {
