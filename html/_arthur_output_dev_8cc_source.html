269 void ArthurOutputDev::updateFont(GfxState *state)
270 {
271 #ifdef HAVE_SPLASH
272  GfxFont *gfxFont;
273  GfxFontLoc *fontLoc;
274  GfxFontType fontType;
275  SplashOutFontFileID *id;
276  SplashFontFile *fontFile;
277  SplashFontSrc *fontsrc = NULL;
278  FoFiTrueType *ff;
279  Object refObj, strObj;
280  GooString *fileName;
281  char *tmpBuf;
282  int tmpBufLen = 0;
283  int *codeToGID;
284  double *textMat;
285  double m11, m12, m21, m22, fontSize;
286  SplashCoord mat[4];
287  int n;
288  int faceIndex = 0;
289  SplashCoord matrix[6];
290 
291  m_needFontUpdate = false;
292  m_font = NULL;
293  fileName = NULL;
294  tmpBuf = NULL;
295  fontLoc = NULL;
296 
297  if (!(gfxFont = state->getFont())) {
298  goto err1;
299  }
300  fontType = gfxFont->getType();
301  if (fontType == fontType3) {
302  goto err1;
303  }
304 
305  // check the font file cache
306  id = new SplashOutFontFileID(gfxFont->getID());
307  if ((fontFile = m_fontEngine->getFontFile(id))) {
308  delete id;
309 
310  } else {
311 
312  if (!(fontLoc = gfxFont->locateFont(xref, NULL))) {
313  error(errSyntaxError, -1, "Couldn't find a font for '{0:s}'",
314  gfxFont->getName() ? gfxFont->getName()->getCString()
315  : "(unnamed)");
316  goto err2;
317  }
318 
319  // embedded font
320  if (fontLoc->locType == gfxFontLocEmbedded) {
321  // if there is an embedded font, read it to memory
322  tmpBuf = gfxFont->readEmbFontFile(xref, &tmpBufLen);
323  if (! tmpBuf)
324  goto err2;
325 
326  // external font
327  } else { // gfxFontLocExternal
328  fileName = fontLoc->path;
329  fontType = fontLoc->fontType;
330  }
331 
332  fontsrc = new SplashFontSrc;
333  if (fileName)
334  fontsrc->setFile(fileName, gFalse);
335  else
336  fontsrc->setBuf(tmpBuf, tmpBufLen, gTrue);
337 
338  // load the font file
339  switch (fontType) {
340  case fontType1:
341  if (!(fontFile = m_fontEngine->loadType1Font(
342  id,
343  fontsrc,
344  (const char **)((Gfx8BitFont *)gfxFont)->getEncoding()))) {
345  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
346  gfxFont->getName() ? gfxFont->getName()->getCString()
347  : "(unnamed)");
348  goto err2;
349  }
350  break;
351  case fontType1C:
352  if (!(fontFile = m_fontEngine->loadType1CFont(
353  id,
354  fontsrc,
355  (const char **)((Gfx8BitFont *)gfxFont)->getEncoding()))) {
356  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
357  gfxFont->getName() ? gfxFont->getName()->getCString()
358  : "(unnamed)");
359  goto err2;
360  }
361  break;
362  case fontType1COT:
363  if (!(fontFile = m_fontEngine->loadOpenTypeT1CFont(
364  id,
365  fontsrc,
366  (const char **)((Gfx8BitFont *)gfxFont)->getEncoding()))) {
367  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
368  gfxFont->getName() ? gfxFont->getName()->getCString()
369  : "(unnamed)");
370  goto err2;
371  }
372  break;
373  case fontTrueType:
374  case fontTrueTypeOT:
375  if (fileName)
376  ff = FoFiTrueType::load(fileName->getCString());
377  else
378  ff = FoFiTrueType::make(tmpBuf, tmpBufLen);
379  if (ff) {
380  codeToGID = ((Gfx8BitFont *)gfxFont)->getCodeToGIDMap(ff);
381  n = 256;
382  delete ff;
383  } else {
384  codeToGID = NULL;
385  n = 0;
386  }
387  if (!(fontFile = m_fontEngine->loadTrueTypeFont(
388  id,
389  fontsrc,
390  codeToGID, n))) {
391  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
392  gfxFont->getName() ? gfxFont->getName()->getCString()
393  : "(unnamed)");
394  goto err2;
395  }
396  break;
397  case fontCIDType0:
398  case fontCIDType0C:
399  if (!(fontFile = m_fontEngine->loadCIDFont(
400  id,
401  fontsrc))) {
402  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
403  gfxFont->getName() ? gfxFont->getName()->getCString()
404  : "(unnamed)");
405  goto err2;
406  }
407  break;
408  case fontCIDType0COT:
409  if (((GfxCIDFont *)gfxFont)->getCIDToGID()) {
410  n = ((GfxCIDFont *)gfxFont)->getCIDToGIDLen();
411  codeToGID = (int *)gmallocn(n, sizeof(int));
412  memcpy(codeToGID, ((GfxCIDFont *)gfxFont)->getCIDToGID(),
413  n * sizeof(int));
414  } else {
415  codeToGID = NULL;
416  n = 0;
417  }
418  if (!(fontFile = m_fontEngine->loadOpenTypeCFFFont(
419  id,
420  fontsrc,
421  codeToGID, n))) {
422  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
423  gfxFont->getName() ? gfxFont->getName()->getCString()
424  : "(unnamed)");
425  goto err2;
426  }
427  break;
428  case fontCIDType2:
429  case fontCIDType2OT:
430  codeToGID = NULL;
431  n = 0;
432  if (((GfxCIDFont *)gfxFont)->getCIDToGID()) {
433  n = ((GfxCIDFont *)gfxFont)->getCIDToGIDLen();
434  if (n) {
435  codeToGID = (int *)gmallocn(n, sizeof(int));
436  memcpy(codeToGID, ((GfxCIDFont *)gfxFont)->getCIDToGID(),
437  n * sizeof(Gushort));
438  }
439  } else {
440  if (fileName)
441  ff = FoFiTrueType::load(fileName->getCString());
442  else
443  ff = FoFiTrueType::make(tmpBuf, tmpBufLen);
444  if (! ff)
445  goto err2;
446  codeToGID = ((GfxCIDFont *)gfxFont)->getCodeToGIDMap(ff, &n);
447  delete ff;
448  }
449  if (!(fontFile = m_fontEngine->loadTrueTypeFont(
450  id,
451  fontsrc,
452  codeToGID, n, faceIndex))) {
453  error(errSyntaxError, -1, "Couldn't create a font for '{0:s}'",
454  gfxFont->getName() ? gfxFont->getName()->getCString()
455  : "(unnamed)");
456  goto err2;
457  }
458  break;
459  default:
460  // this shouldn't happen
461  goto err2;
462  }
463  }
464 
465  // get the font matrix
466  textMat = state->getTextMat();
467  fontSize = state->getFontSize();
468  m11 = textMat[0] * fontSize * state->getHorizScaling();
469  m12 = textMat[1] * fontSize * state->getHorizScaling();
470  m21 = textMat[2] * fontSize;
471  m22 = textMat[3] * fontSize;
472 
473  {
474  QMatrix painterMatrix = m_painter->worldMatrix();
475  matrix[0] = painterMatrix.m11();
476  matrix[1] = painterMatrix.m12();
477  matrix[2] = painterMatrix.m21();
478  matrix[3] = painterMatrix.m22();
479  matrix[4] = painterMatrix.dx();
480  matrix[5] = painterMatrix.dy();
481  }
482 
483  // create the scaled font
484  mat[0] = m11; mat[1] = -m12;
485  mat[2] = m21; mat[3] = -m22;
486  m_font = m_fontEngine->getFont(fontFile, mat, matrix);
487 
488  delete fontLoc;
489  if (fontsrc && !fontsrc->isFile)
490  fontsrc->unref();
491  return;
492 
493  err2:
494  delete id;
495  delete fontLoc;
496  err1:
497  if (fontsrc && !fontsrc->isFile)
498  fontsrc->unref();
499  return;
500 #endif
501 }

503 static QPainterPath convertPath(GfxState *state, GfxPath *path, Qt::FillRule fillRule)
504 {
505  GfxSubpath *subpath;
506  double x1, y1, x2, y2, x3, y3;
507  int i, j;
508 
509  QPainterPath qPath;
510  qPath.setFillRule(fillRule);
511  for (i = 0; i < path->getNumSubpaths(); ++i) {
512  subpath = path->getSubpath(i);
513  if (subpath->getNumPoints() > 0) {
514  state->transform(subpath->getX(0), subpath->getY(0), &x1, &y1);
515  qPath.moveTo(x1, y1);
516  j = 1;
517  while (j < subpath->getNumPoints()) {
518  if (subpath->getCurve(j)) {
519  state->transform(subpath->getX(j), subpath->getY(j), &x1, &y1);
520  state->transform(subpath->getX(j+1), subpath->getY(j+1), &x2, &y2);
521  state->transform(subpath->getX(j+2), subpath->getY(j+2), &x3, &y3);
522  qPath.cubicTo( x1, y1, x2, y2, x3, y3);
523  j += 3;
524  } else {
525  state->transform(subpath->getX(j), subpath->getY(j), &x1, &y1);
526  qPath.lineTo(x1, y1);
527  ++j;
528  }
529  }
530  if (subpath->isClosed()) {
531  qPath.closeSubpath();
532  }
533  }
534  }
535  return qPath;
536 }

599  qPath.closeSubpath();
600  }
601  if (fontPath->flags[i] & splashPathFirst) {
602  state->transform(fontPath->pts[i].x+x, -fontPath->pts[i].y+y, &x1, &y1);
603  qPath.moveTo(x1,y1);
604  }
605  if (fontPath->flags[i] & splashPathCurve) {
606  state->transform(fontPath->pts[i].x+x, -fontPath->pts[i].y+y, &x1, &y1);
607  state->transform(fontPath->pts[i+1].x+x, -fontPath->pts[i+1].y+y, &x2, &y2);
608  qPath.quadTo(x1,y1,x2,y2);
609  ++i;
610  }
611  // FIXME fix this

613  // qDebug() << "Need to implement arc";
614  // }
615  else {
616  state->transform(fontPath->pts[i].x+x, -fontPath->pts[i].y+y, &x1, &y1);
617  qPath.lineTo(x1,y1);
618  }
619  }
620  GfxRGB rgb;
621  QColor brushColour = m_currentBrush.color();
622  state->getFillRGB(&rgb);
623  brushColour.setRgbF(colToDbl(rgb.r), colToDbl(rgb.g), colToDbl(rgb.b), state->getFillOpacity());
624  m_painter->setBrush(brushColour);
625  m_painter->setPen(Qt::NoPen);
626  m_painter->drawPath(qPath);
627  delete fontPath;
628  }
629  }
630 
631  // stroke
632  if ((render & 3) == 1 || (render & 3) == 2) {
633  qDebug() << "no stroke";
634  /*
635  if ((path = m_font->getGlyphPath(code))) {
636  path->offset((SplashCoord)x1, (SplashCoord)y1);
637  splash->stroke(path);
638  delete path;
639  }
640  */
641  }
642 
643  // clip
644  if (render & 4) {
645  qDebug() << "no clip";
646  /*
647  path = m_font->getGlyphPath(code);
648  path->offset((SplashCoord)x1, (SplashCoord)y1);
649  if (textClipPath) {
650  textClipPath->append(path);
651  delete path;
652  } else {
653  textClipPath = path;
654  }
655  */
656  }
657 #endif
658 }
