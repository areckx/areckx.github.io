56 static void __attribute__((constructor)) tryload_libjack()
57 {
58  if (getenv("SKIP_LIBJACK") == 0) { // just in case libjack is causing troubles..
59  #ifdef __APPLE__
60  libjack_handle = dlopen("libjack.0.dylib", RTLD_LAZY);
61  #elif defined(WIN32)
62  libjack_handle = LoadLibrary("libjack.dll");
63  #else
64  libjack_handle = dlopen("libjack.so.0", RTLD_LAZY);
65  #endif
66 
67  }
68  libjack_is_present = (libjack_handle != 0);
69 }

71 void *load_jack_function(const char *fn_name)
72 {
73  void *fn = 0;
74  if (!libjack_handle) {
75  qDebug("libjack not found, so do not try to load %s ffs !", fn_name);
76  return 0;
77  }
78 #ifdef WIN32
79  fn = (void*)GetProcAddress(libjack_handle, fn_name);
80 #else
81  fn = dlsym(libjack_handle, fn_name);
82 #endif
83  if (!fn) {
84 #ifdef WIN32
85  char* lpMsgBuf;
86  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,NULL,GetLastError(),MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),(LPTSTR) &lpMsgBuf,0,NULL );
87  qDebug("could not GetProcAddress( %s ), %s ", fn_name, lpMsgBuf) ;
88 #else
89  qDebug ("could not dlsym( %s ), %s ", fn_name, dlerror()) ;
90 #endif
91  }
92  return fn;
93 }
